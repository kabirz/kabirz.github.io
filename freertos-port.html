02 上下文切换

// ARM Cortex-M4F 移植中的上下文切换（port.c）

// PendSV Handler — 上下文切换核心
void xPortPendSVHandler( void )
{
    // 保存当前任务上下文
    __asm volatile(
        "mrs r0, psp                     \n"  // 读取进程栈指针
        "ldr r3, pxCurrentTCB_const      \n"  // 加载 TCB 指针
        "ldr r2, [r3]                    \n"
        "stmdb r0!, {r4-r11, r14}        \n"  // 保存寄存器
        "str r0, [r2]                    \n"  // 保存栈顶到 TCB

        "bl vTaskSwitchContext           \n"  // 切换任务

        "ldr r3, pxCurrentTCB_const      \n"
        "ldr r1, [r3]                    \n"
        "ldr r0, [r1]                    \n"  // 加载新任务栈顶
        "ldmia r0!, {r4-r11, r14}        \n"  // 恢复寄存器
        "msr psp, r0                     \n"  // 设置 PSP
        "bx r14                          \n"  // 返回
    );
}

// 禁止中断（通过 PRIMASK）
#define portDISABLE_INTERRUPTS()    __asm volatile( "cpsid i" ::: "memory" )

// 恢复中断
#define portENABLE_INTERRUPTS()     __asm volatile( "cpsie i" ::: "memory" )

// 进入临界区
extern void vPortEnterCritical( void );
extern void vPortExitCritical( void );

#define portSET_INTERRUPT_MASK_FROM_ISR()    __asm volatile( "mrs %0, primask \n cpsid i" : "=r" ( ulOriginalMask ) )

04 Tick 移植

// ARM Cortex-M4F Tick 实现

// 配置 SysTick
void xPortSetupTimerInterrupt( void )
{
    // 设置 SysTick 重载值
    portNVIC_SYSTICK_LOAD_REG = ( configCPU_CLOCK_HZ / configTICK_RATE_HZ ) - 1UL;

    // 启动 SysTick
    portNVIC_SYSTICK_CTRL_REG = portNVIC_SYSTICK_CLK |
                                 portNVIC_SYSTICK_INT |
                                 portNVIC_SYSTICK_ENABLE;
}

// SysTick Handler
void xPortSysTickHandler( void )
{
    // 递增 tick 计数
    if( xTaskIncrementTick() != pdFALSE )
    {
        // 需要上下文切换
        portYIELD_WITHIN_API();
    }
}

06 完整示例

// ===== FreeRTOS 移植完整示例 =====
// 以 ARM Cortex-M4F 为例

// portmacro.h（简化版）
#define portSTACK_TYPE     uint32_t
#define portBASE_TYPE      long
#define portSTACK_GROWTH   (-1)  // 向下增长
#define portBYTE_ALIGNMENT 8
#define portYIELD()        portNVIC_INT_CTRL_REG = portNVIC_PENDSVSET_BIT

// port.c（简化版）
BaseType_t xPortStartScheduler( void )
{
    // 1. 配置 SysTick
    xPortSetupTimerInterrupt();

    // 2. 启动第一个任务
    portRESTORE_INTERRUPTS();

    // 不返回
    return pdTRUE;
}
