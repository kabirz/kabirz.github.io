01 LoRaWAN 概览

LoRaWAN 是基于 LoRa 调制的低功耗广域网（LPWAN）协议，适用于 IoT 远程通信。

LoRaWAN 特点

特性参数
通信距离城市 2-5km，农村 10-15km
数据速率0.3 - 50 kbps
电池寿命数年（Class A）
频段EU868, US915, AS923, CN470 等
网络拓扑星型（设备→网关→网络服务器）

03 OTAA 入网

// OTAA（Over-The-Air Activation）
// 动态入网，安全性更高

// 配置参数
struct lorawan_join_config join_cfg = {
    .mode = LORAWAN_ACT_OTAA,
    .dev_eui = dev_eui,
    .app_eui = app_eui,
    .app_key = app_key,
    .channel = LORAWAN_CHANNEL_AUTO,
};

// 入网
int ret = lorawan_join(&join_cfg);
if (ret) {
    printk("入网失败: %d\n", ret);
}

// ABP（Activation By Personalization）
// 预配置密钥，无需入网服务器
struct lorawan_join_config abp_cfg = {
    .mode = LORAWAN_ACT_ABP,
    .dev_addr = dev_addr,
    .app_s_key = app_s_key,
    .nwk_s_key = nwk_s_key,
};

04 设备类别

类别特点适用场景
Class A最低功耗，双向通信（上行后开窗）传感器、计量表
Class B周期性接收窗，中等功耗需要下行控制的设备
Class C持续监听，最高功耗市电供电设备

08 LoRaWAN API

// 入网
int lorawan_join(const struct lorawan_join_config *cfg);
int lorawan_leave(void);

// 数据传输
int lorawan_send(uint8_t port, const uint8_t *data, uint8_t len,
                 enum lorawan_message_type type);

// 接收回调
typedef void (*lorawan_recv_cb)(uint8_t port, const uint8_t *data, uint8_t len);

// 配置
int lorawan_set_class(enum lorawan_device_class dev_class);
int lorawan_set_region(enum lorawan_region_id region);
int lorawan_set_datarate(uint8_t dr);

// 事件回调
lorawan_recv_cb recv_cb = my_recv_handler;
lorawan_recv_register(recv_cb);
