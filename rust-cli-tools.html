02 clap 命令行解析

// clap 是事实标准的 CLI 库
// features = ["derive"] 提供 derive 宏

use clap::{Parser, Subcommand};

#[derive(Parser)]
#[command(name = "myapp")]
#[command(version = "1.0")]
#[command(about = "Does awesome things", long_about = None)]
struct Cli {
    /// Configuration file
    #[arg(short, long, default_value = "config.toml")]
    config: String,
    
    /// Verbose mode (-v, -vv, -vvv)
    #[arg(short, long, action = clap::ArgAction::Count)]
    verbose: u8,
    
    #[command(subcommand)]
    command: Commands,
}

#[derive(Subcommand)]
enum Commands {
    /// Add a new item
    Add {
        /// Item name
        name: String,
        
        /// Tags
        #[arg(short, long, value_name = "TAG")]
        tags: Option<Vec<String>>,
    },
    
    /// List items
    List {
        /// Filter by tag
        #[arg(short, long)]
        tag: Option<String>,
        
        /// Output as JSON
        #[arg(long)]
        json: bool,
    },
    
    /// Remove an item
    Remove {
        id: u32,
    },
}

fn main() -> anyhow::Result<()> {
    let cli = Cli::parse();
    
    match &cli.command {
        Commands::Add { name, tags } => {
            println!("Adding {}", name);
            if let Some(tags) = tags {
                println!("Tags: {:?}", tags);
            }
        }
        Commands::List { tag, json } => {
            // ...
        }
        Commands::Remove { id } => {
            println!("Removing {}", id);
        }
    }
    
    Ok(())
}

// 命令行：
// myapp add "Task" --tags work,urgent
// myapp list --tag work --json
// myapp remove 5
// myapp --help

03 错误处理

// anyhow：错误"包"，简单应用首选
use anyhow::{Context, Result};

fn main() -> Result<()> {
    let config = read_config().context("failed to read config")?;
    run(&config)?;
    Ok(())
}

fn read_config() -> Result<Config> {
    let s = std::fs::read_to_string("config.toml")
        .context("config file not found")?;
    let config = toml::from_str(&s)
        .context("invalid config format")?;
    Ok(config)
}

// 输出：
// Error: failed to read config
// 
// Caused by:
//     config file not found
// 
// Caused by:
//     No such file or directory (os error 2)

// thiserror：库的 Error derive
use thiserror::Error;

#[derive(Error, Debug)]
pub enum AppError {
    #[error("io error: {0}")]
    Io(#[from] std::io::Error),
    
    #[error("config error: {message}")]
    Config { message: String },
    
    #[error("invalid input: {input}")]
    InvalidInput { input: String },
}

// color-eyre：彩色错误报告
// [dependencies]
// color-eyre = "0.6"

use color_eyre::Report;

fn main() -> Result<(), Report> {
    color_eyre::install()?;
    
    // 配置
    color_eyre::config::HookBuilder::default()
        .display_env_section(false)
        .install()?;
    
    run()?;
    Ok(())
}

// 2. tracing（推荐，结构化）
use tracing::{info, debug, error, trace, warn, span, Level};

fn main() {
    tracing_subscriber::fmt()
        .with_max_level(Level::INFO)
        .with_target(false)
        .compact()
        .init();
    
    info!("starting app");
    
    let span = span!(Level::INFO, "task");
    let _enter = span.enter();
    do_work();
}

#[tracing::instrument]
fn do_work() {
    debug!("working");
}

    .init();

08 配置管理

// config-rs - 多来源配置
// [dependencies]
// config = "0.14"
// serde = { version = "1", features = ["derive"] }

use serde::Deserialize;

#[derive(Deserialize)]
struct Config {
    server: ServerConfig,
    database: DbConfig,
}

#[derive(Deserialize)]
struct ServerConfig {
    host: String,
    port: u16,
}

#[derive(Deserialize)]
struct DbConfig {
    url: String,
}

fn main() -> Result<(), config::ConfigError> {
    let config = config::Config::builder()
        // 默认值
        .set_default("server.host", "0.0.0.0")?
        .set_default("server.port", 8080)?
        // 文件
        .add_source(config::File::with_name("config.toml").required(false))
        .add_source(config::File::with_name("config.local.toml").required(false))
        // 环境变量（前缀 APP_）
        .add_source(config::Environment::with_prefix("APP").separator("__"))
        .build()?;
    
    let config: Config = config.try_deserialize()?;
    
    println!("{}:{}", config.server.host, config.server.port);
    
    Ok(())
}

// 6. 信号处理（Ctrl+C）
use tokio::signal;

#[tokio::main]
async fn main() {
    tokio::spawn(async {
        // 主工作
    });
    
    signal::ctrl_c().await.unwrap();
    println!("Ctrl+C received, shutting down...");
}

// 7. 退出码

    }
}
