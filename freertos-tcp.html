02 Socket API

// 创建 socket
Socket_t xSocket = FreeRTOS_socket( FREERTOS_AF_INET,
                                     FREERTOS_SOCK_STREAM,
                                     FREERTOS_IPPROTO_TCP );

// 绑定
struct freertos_sockaddr xAddress;
xAddress.sin_port = FreeRTOS_htons( 80 );
FreeRTOS_bind( xSocket, &xAddress, sizeof( xAddress ) );

// 监听
FreeRTOS_listen( xSocket, 20 );

// 接受连接
Socket_t xClientSocket = FreeRTOS_accept( xSocket, NULL, NULL );

// 发送数据
FreeRTOS_send( xClientSocket, "Hello", 5, 0 );

// 接收数据
char cBuffer[ 100 ];
FreeRTOS_recv( xClientSocket, cBuffer, sizeof( cBuffer ), 0 );

04 完整示例

// ===== FreeRTOS TCP/IP 完整示例 =====

#include "FreeRTOS.h"
#include "FreeRTOS_IP.h"
#include "FreeRTOS_Sockets.h"
#include <stdio.h>

// HTTP 服务器任务
void vHTTPServerTask( void * pvParameters )
{
    Socket_t xServerSocket, xClientSocket;
    struct freertos_sockaddr xAddress;

    // 创建 socket
    xServerSocket = FreeRTOS_socket( FREERTOS_AF_INET,
                                      FREERTOS_SOCK_STREAM,
                                      FREERTOS_IPPROTO_TCP );

    // 绑定
    xAddress.sin_port = FreeRTOS_htons( 80 );
    FreeRTOS_bind( xServerSocket, &xAddress, sizeof( xAddress ) );

    // 监听
    FreeRTOS_listen( xServerSocket, 20 );

    for ( ;; )
    {
        // 接受连接
        xClientSocket = FreeRTOS_accept( xServerSocket, NULL, NULL );

        // 处理请求
        char cRequest[ 100 ];
        FreeRTOS_recv( xClientSocket, cRequest, sizeof( cRequest ), 0 );

        // 发送响应
        char *pcResponse = "HTTP/1.1 200 OK\\r\\nContent-Length: 13\\r\\n\\r\\nHello, World!";
        FreeRTOS_send( xClientSocket, pcResponse, strlen( pcResponse ), 0 );

        // 关闭连接
        FreeRTOS_closesocket( xClientSocket );
    }
}

int main( void )
{
    // 初始化网络
    uint8_t ucIPAddress[ 4 ] = { 192, 168, 1, 100 };
    uint8_t ucNetMask[ 4 ] = { 255, 255, 255, 0 };
    uint8_t ucGateway[ 4 ] = { 192, 168, 1, 1 };
    uint8_t ucDNS[ 4 ] = { 8, 8, 8, 8 };

    FreeRTOS_IPInit( ucIPAddress, ucNetMask, ucGateway, ucDNS, 80 );

    // 创建 HTTP 服务器任务
    xTaskCreate( vHTTPServerTask, "HTTP", 1024, NULL, 3, NULL );

    vTaskStartScheduler();

    for ( ;; ) {}
}
