01 统计子系统概览

Zephyr 统计子系统提供运行时数据收集和分析能力。

统计类型

类型说明
计数器统计简单事件计数
网络统计收发包计数、错误计数
内存统计分配/释放/使用率
自定义统计应用层自定义指标

02 计数器统计

// 计数器统计
#include <zephyr/stats/stats.h>

// 定义统计结构
STATS_SECT_START(my_stats)
    STATS_SECT_ENTRY(packet_count)
    STATS_SECT_ENTRY(error_count)
    STATS_SECT_ENTRY(byte_count)
STATS_SECT_END;

STATS_SECT_DECL(my_stats) my_stats;

// 注册统计
STATS_INIT(my_stats, packet_count, 0);

// 更新计数
STATS_INC(my_stats, packet_count);
STATS_ADD(my_stats, byte_count, 1024);

// 获取值
printk("数据包数: %d\n", my_stats.packet_count);
printk("错误数: %d\n", my_stats.error_count);
printk("字节数: %d\n", my_stats.byte_count);

03 网络统计

// 网络接口统计
struct net_stats stats;
net_mgmt(NET_REQUEST_STATS_GET, iface, &stats, sizeof(stats));

// 统计项
printk("接收包: %d\n", stats.rx.pkts);
printk("发送包: %d\n", stats.tx.pkts);
printk("接收字节: %d\n", stats.rx.bytes);
printk("发送字节: %d\n", stats.tx.bytes);
printk("接收错误: %d\n", stats.rx.error);
printk("发送错误: %d\n", stats.tx.error);
printk("丢弃包: %d\n", stats.rx.drop);

04 内存统计

// 内存统计
// CONFIG_MEM_STATS=y

// 堆统计
struct sys_memory_stats heap_stats;
k_heap_stats(&my_heap, &heap_stats);

printk("总大小: %d\n", heap_stats.total_bytes);
printk("已分配: %d\n", heap_stats.allocated_bytes);
printk("最大分配: %d\n", heap_stats.max_allocated_bytes);
printk("空闲: %d\n", heap_stats.free_bytes);

// Slab 统计
printk("Slab 块数: %d\n", my_slab.info.num_blocks);
printk("Slab 已用: %d\n", my_slab.info.num_used);
printk("Slab 最大: %d\n", my_slab.info.max_used);

05 自定义统计

// 自定义统计指标
STATS_SECT_START(app_stats)
    STATS_SECT_ENTRY(api_calls)
    STATS_SECT_ENTRY(api_errors)
    STATS_SECT_ENTRY(latency_sum)
    STATS_SECT_ENTRY(latency_max)
STATS_SECT_END;

// 记录延迟
timing_t start = timing_counter_get();
do_api_call();
timing_t end = timing_counter_get();
uint32_t latency = timing_cycles_to_ns(end - start);

STATS_INC(app_stats, api_calls);
STATS_ADD(app_stats, latency_sum, latency);
if (latency > app_stats.latency_max) {
    app_stats.latency_max = latency;
}

06 Shell 集成

// CONFIG_STATS_SHELL=y

// Shell 命令：
// stats list           - 列出所有统计模块
// stats print <module> - 打印统计信息
// stats reset <module> - 重置统计

// uart:~$ stats list
// my_stats
// net_stats

// uart:~$ stats print my_stats
// packet_count: 1024
// error_count: 3
// byte_count: 65536

07 ObjCore 统计

// ObjCore 提供统一的对象统计框架
// CONFIG_OBJ_CORE=y

// 支持 ObjCore 的对象类型：
// - Thread（线程）
// - Mutex（互斥锁）
// - Semaphore（信号量）
// - Memory Slab
// - Device（设备）

// 查询对象统计
struct k_thread_runtime_stats stats;
k_thread_runtime_stats_get(thread, &stats);
printk("线程运行时间: %d cycles\n", stats.execution_cycles);

08 统计 API
