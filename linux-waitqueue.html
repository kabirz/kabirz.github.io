01 概述

等待队列（Wait Queue）是 Linux 内核中最灵活的等待/通知机制。与 completion 的"一次性通知"不同，wait_queue 是基于条件的等待——等待某个条件变为真。它被广泛用于 I/O 完成、网络数据到达、信号量实现等场景。

核心特性
  • 条件等待：等待任意布尔条件
  • 可睡眠：等待时不消耗 CPU
  • 多种状态：INTERRUPTIBLE / KILLABLE / UNINTERRUPTIBLE
  • 超时支持：可设置最大等待时间
  • Exclusive：排他唤醒，避免惊群效应
  • 自定义回调：可替换默认唤醒函数
vs Completion

struct wait_queue_entry {
    unsigned int flags;        // WQ_FLAG_EXCLUSIVE 等
    void *private;             // 私有数据（通常为 task_struct）
    wait_queue_func_t func;    // 唤醒回调函数

struct wait_queue_head {
    spinlock_t lock;           // 保护链表的自旋锁
    struct list_head head;     // 等待者链表
};
typedef struct wait_queue_head wait_queue_head_t;

04 wait_event 宏

wait_event 是最常用的等待队列接口，封装了完整的等待逻辑。

wait_event 实现逻辑

// wait_event(wq_head, condition) 的等价展开：
for (;;) {
    prepare_to_wait(&wq_head, &wait, TASK_UNINTERRUPTIBLE);
    if (condition)  // 检查条件
        break;
    schedule();    // 让出 CPU，等待唤醒
}
finish_wait(&wq_head, &wait);

// interruptible 版本使用 TASK_INTERRUPTIBLE 状态
// timeout 版本增加 jiffies 超时检查

05 内部实现

// 唤醒函数
void __wake_up(struct wait_queue_head *wq_head, unsigned int mode,
               int nr_exclusive, void *key)
{
    unsigned long flags;
    spin_lock_irqsave(&wq_head->lock, flags);
    __wake_up_common_lock(wq_head, mode, nr_exclusive, 0, key);
    spin_unlock_irqrestore(&wq_head->lock, flags);
}

// WQ_FLAG_EXCLUSIVE 的作用：
// 当设置 nr_exclusive 时，只唤醒队列尾部的 N 个排他等待者
// 避免惊群效应（thundering herd）
Exclusive 等待者：当多个线程等待同一事件时，wake_up() 默认唤醒所有等待者（惊群）。使用 add_wait_queue_exclusive() 可以让等待者成为排他等待者，配合 __wake_up(wq, mode, 1, key) 只唤醒一个。这在网络等高负载场景中很重要。

07 使用模式

// 模式一：等待事件条件
DECLARE_WAIT_QUEUE_HEAD(wq);
int data_ready = 0;

// 生产者
data = receive_data();
data_ready = 1;
wake_up(&wq);

// 消费者
wait_event(wq, data_ready);
process(data);
// 模式二：I/O 完成等待
struct my_request {
    struct wait_queue_head wq;
    int completed;
    int result;
};

// 启动 I/O
req.completed = 0;
submit_io(req);

// 等待完成
wait_event(req.wq, req.completed);

// I/O 完成回调
req.result = 0;
req.completed = 1;
wake_up(&req.wq);
// 模式三：网络数据到达
static DECLARE_WAIT_QUEUE_HEAD(rx_wq);

// 中断处理：数据到达
__wake_up_rx(&rx_wq);  // 唤醒接收线程

// 接收线程
while (!kthread_should_stop()) {
    wait_event(rx_wq, has_data(&rx_queue));
    process_packets(&rx_queue);
}
// 模式四：超时等待
int ret;
ret = wait_event_timeout(wq, condition, msecs_to_jiffies(5000));
if (ret == 0) {
    // 超时
    pr_err("timeout waiting for condition\n");
} else {
    // 条件满足
}
