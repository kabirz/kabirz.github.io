TaskStatus_t 结构体

// include/task.h
typedef struct xTASK_STATUS
{
    TaskHandle_t xHandle;            // 任务句柄
    const char *pcTaskName;          // 任务名
    UBaseType_t xTaskNumber;         // 任务编号
    eTaskState eCurrentState;        // 当前状态
    UBaseType_t uxCurrentPriority;   // 当前优先级
    UBaseType_t uxBasePriority;      // 基础优先级
    uint32_t ulRunTimeCounter;       // CPU 运行时间
    StackType_t *pxStackBase;        // 栈基地址
    configSTACK_DEPTH_TYPE usStackDepth; // 栈深度
} TaskStatus_t;

04 完整示例

// ===== FreeRTOS 追踪完整示例 =====

#include "FreeRTOS.h"
#include "task.h"
#include <stdio.h>

// 配置
// configUSE_TRACE_FACILITY = 1
// configUSE_STATS_FORMATTING_FUNCTIONS = 1

// 统计任务
void vStatsTask( void * pvParameters )
{
    char pcWriteBuffer[ 512 ];

    for ( ;; )
    {
        // 每 5 秒打印一次统计
        vTaskDelay( pdMS_TO_TICKS( 5000 ) );

        // 打印任务列表
        vTaskList( pcWriteBuffer );
        printf( "Task Name  State  Prio  Stack  Num\\n%s", pcWriteBuffer );

        // 打印运行时间统计
        vTaskGetRunTimeStats( pcWriteBuffer );
        printf( "Task Name  Run Time  %%\\n%s", pcWriteBuffer );
    }
}

int main( void )
{
    xTaskCreate( vStatsTask, "Stats", 512, NULL, 1, NULL );

    vTaskStartScheduler();

    for ( ;; ) {}
}
