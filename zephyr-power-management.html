02 休眠状态

PM 状态定义

// include/zephyr/pm/state.h
enum pm_state {
    PM_STATE_ACTIVE,         // 活跃状态（正常运行）
    PM_STATE_RUNTIME_IDLE,   // 运行时空闲（CPU 停止但外设运行）
    PM_STATE_STANDBY,        // 待机（保留 RAM）
    PM_STATE_SUSPEND,        // 挂起（大部分关闭）
    PM_STATE_SOFT_OFF,       // 软关闭（需要唤醒源）
    PM_STATE_COUNT           // 状态数量
};

// PM 状态信息
struct pm_state_info {
    enum pm_state state;         // 状态类型
    uint8_t substate_id;         // 子状态 ID
    uint32_t min_residency_us;   // 最小停留时间（微秒）
    uint32_t exit_latency_us;    // 退出延迟（微秒）
    bool pm_device_disabled;     // 是否禁用设备 PM
};

状态功耗对比

状态功耗唤醒时间保留内容说明
Active最高0全部正常运行
Runtime Idle低极短全部CPU 停止，外设运行
Standby更低短RAM关闭大部分外设
Suspend很低中等部分 RAM大部分关闭
Soft Off最低长无需要硬件唤醒源

03 PM 策略

PM 策略决定系统在空闲时进入哪种休眠状态。

默认策略

// policy_default.c 中的默认策略
const struct pm_state_info *pm_policy_next_state(uint8_t cpu, int32_t ticks)
{
    uint8_t num_cpu_states;
    const struct pm_state_info *cpu_states;
    const struct pm_state_info *out_state = NULL;

    // 获取 CPU 的所有可用状态
    num_cpu_states = pm_state_cpu_get_all(cpu, &cpu_states);

    // 遍历状态，选择最深的可用状态
    for (uint32_t i = 0; i < num_cpu_states; i++) {
        const struct pm_state_info *state = &cpu_states[i];
        uint32_t min_residency_us = state->min_residency_us + state->exit_latency_us;
        uint32_t min_residency_ticks = k_us_to_ticks_ceil32(min_residency_us);

        // 如果空闲时间不足以进入此状态，停止搜索
        if (ticks < min_residency_ticks) {
            break;
        }

        // 检查状态是否可用
        if (!pm_policy_state_is_available(state->state, state->substate_id)) {
            continue;
        }

        out_state = state;
    }

    return out_state;
}

设备 PM 实现

// device.c 中的设备 PM 控制
int pm_device_action_run(const struct device *dev, enum pm_device_action action)
{
    struct pm_device_base *pm = dev->pm_base;

    // 检查状态转换是否有效
    if (pm->state == action_target_state[action]) {
        return -EALREADY;
    }
    if (pm->state != action_expected_state[action]) {
        return -ENOTSUP;
    }

    // 调用驱动的 PM 回调
    ret = pm->action_cb(dev, action);

    return ret;
}

驱动实现 PM 回调

// 驱动实现 PM 回调
static int my_driver_pm_action(const struct device *dev,
                               enum pm_device_action action)
{
    switch (action) {
    case PM_DEVICE_ACTION_RESUME:
        // 恢复设备
        enable_clocks(dev);
        configure_pins(dev);
        break;
    case PM_DEVICE_ACTION_SUSPEND:
        // 挂起设备
        disable_clocks(dev);
        break;
    case PM_DEVICE_ACTION_TURN_OFF:
        // 关闭设备
        power_off(dev);
        break;
    case PM_DEVICE_ACTION_TURN_ON:
        // 开启设备
        power_on(dev);
        break;
    default:
        return -ENOTSUP;
    }
    return 0;
}

05 运行时 PM

运行时 PM 允许设备在空闲时自动休眠，有使用时自动唤醒。

// device_runtime.c 中的运行时 PM 实现

// 获取设备（增加引用计数）
int pm_device_runtime_get(const struct device *dev)
{
    struct pm_device *pm = dev->pm;
    int ret = 0;

    K_SPINLOCK(&lock) {
        pm->base.usage++;

        // 如果设备已挂起，唤醒它
        if (pm->base.state == PM_DEVICE_STATE_SUSPENDED) {
            ret = runtime_resume(dev, false, K_NO_WAIT);
        }
    }

    return ret;
}

// 释放设备（减少引用计数）
int pm_device_runtime_put(const struct device *dev)
{
    struct pm_device *pm = dev->pm;
    int ret = 0;

    K_SPINLOCK(&lock) {
        if (pm->base.usage > 1U) {
            pm->base.usage--;
            early_exit = true;
        }
    }

    if (!early_exit) {
        // 最后一个用户，可以挂起设备
        ret = runtime_suspend(dev, false, K_NO_WAIT);
    }

    return ret;
}

// 异步释放
int pm_device_runtime_put_async(const struct device *dev, k_timeout_t delay)
{
    // 异步挂起设备
    return runtime_suspend(dev, true, delay);
}

07 空闲管理与 Tickless

// pm.c 中的系统休眠入口
bool pm_system_suspend(int32_t kernel_ticks)
{
    uint8_t id = CPU_ID;

    // 1. 计算下一个事件的时间
    events_ticks = pm_policy_next_event_ticks();
    ticks = ticks_expiring_sooner(kernel_ticks, events_ticks);

    // 2. 选择最佳休眠状态
    pm_state = pm_policy_next_state(id, ticks);

    // 3. 如果选择了休眠状态
    if (pm_state != NULL) {
        // 4. 通知系统即将休眠
        pm_state_notify(true);

        // 5. 执行休眠前操作
        pm_state_enter_post_ops(pm_state->state, pm_state->substate_id);

        // 6. 进入休眠
        pm_state_enter(pm_state->state, pm_state->substate_id);
    }

    return true;
}

09 完整示例

// ===== Zephyr 电源管理完整示例 =====
// 演示：设备 PM、运行时 PM、PM 通知器

#include <zephyr/kernel.h>
#include <zephyr/pm/pm.h>
#include <zephyr/pm/device.h>
#include <zephyr/pm/device_runtime.h>

// PM 通知器
static void pm_state_entry(enum pm_state state)
{
    printk("Entering power state %d\\n", state);
}

static void pm_state_exit(enum pm_state state)
{
    printk("Exited power state %d\\n", state);
}

struct pm_notifier pm_notifier = {
    .state_entry = pm_state_entry,
    .state_exit = pm_state_exit,
};

// 设备 PM 回调
static int my_sensor_pm_action(const struct device *dev,
                               enum pm_device_action action)
{
    switch (action) {
    case PM_DEVICE_ACTION_RESUME:
        printk("Sensor: resuming\\n");
        // 恢复传感器时钟和配置
        break;
    case PM_DEVICE_ACTION_SUSPEND:
        printk("Sensor: suspending\\n");
        // 关闭传感器时钟
        break;
    default:
        return -ENOTSUP;
    }
    return 0;
}

// 应用任务
void vAppTask(void *pvParameters)
{
    // 注册 PM 通知器
    pm_notifier_register(&pm_notifier);

    for (;;) {
        // 使用传感器（运行时 PM 自动管理）
        pm_device_runtime_get(sensor_dev);

        // 读取传感器数据
        sensor_sample_fetch(sensor_dev);
        struct sensor_value temp;
        sensor_channel_get(sensor_dev, SENSOR_CHAN_AMBIENT_TEMP, &temp);
        printk("Temperature: %.2f\\n", sensor_value_to_double(&temp));

        // 释放设备（自动挂起）
        pm_device_runtime_put(sensor_dev);

        // 等待一段时间
        k_msleep(5000);
    }
}
