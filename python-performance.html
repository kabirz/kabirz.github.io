01 Profiling 性能分析

性能优化始于测量。没有数据支撑的优化是盲目优化。Python 生态提供了从宏观到微观的完整 profiling 工具链。

cProfile 与 pprint

cProfile 是 Python 内置的性能分析器，开销低，适合快速定位热点函数。

import cProfile, pstats

def slow_function():
    total = 0
    for i in range(10_000_000):
        total += i ** 2
    return total

# 方式 1: 命令行
# python -m cProfile -s cumulative my_script.py

# 方式 2: 代码内
profiler = cProfile.Profile()
profiler.enable()
result = slow_function()
profiler.disable()

# 统计分析
stats = pstats.Stats(profiler)
stats.sort_stats(pstats.SortKey.CUMULATIVE)
stats.print_stats(20)  # 显示前 20 行

# 保存结果
profiler.dump_stats("profile.prof")

# 使用 snakeviz 可视化
# pip install snakeviz
# snakeviz profile.prof

02 Cython 编译加速

Cython 是 Python 的超集，允许将 Python 代码编译为 C 扩展。通过添加静态类型声明，可以将数值密集型代码的性能提升到接近原生 C 的水平。

基本类型声明

# .pyx 文件 — Cython 源代码

# Python 风格（无加速）
def sum_squares(n):
    total = 0
    for i in range(n):
        total += i * i
    return total

# Cython 风格（有类型声明）
def sum_squares_fast(int n):
    cdef int i
    cdef long long total = 0
    for i in range(n):
        total += i * i
    return total

# 使用 C 类型
cdef double _compute(double x, int n):
    cdef int i
    cdef double result = 1.0
    for i in range(n):
        result *= x
    return result

# 性能对比（n=10_000_000）:
# Python:       ~2.8 秒
# Cython (无类型): ~2.5 秒
# Cython (有类型): ~0.08 秒

调用 C 库

# 声明 C 函数并直接调用
cdef extern from "math.h":
    double sin(double x)
    double cos(double x)
    double sqrt(double x)

def compute_trig(double x):
    return sin(x) + cos(x)

# 调用 libc stdio
cdef extern from "stdio.h":
    int printf(const char *format, ...)

def c_print():
    printf(b"Hello from C!\n")

# 自定义 C 结构体
cdef extern from "point.h":
    struct Point:
        double x
        double y

def distance(Point p1, Point p2):
    return sqrt((p1.x - p2.x) ** 2 + (p1.y - p2.y) ** 2)

03 Numba JIT

Numba 是一个 JIT（Just-In-Time）编译器，它将 Python 函数编译为机器码。与 Cython 不同，Numba 不需要学习新的语法——只需要添加装饰器。

@jit 与 @njit

from numba import jit, njit
import numpy as np
import time

# @jit — 基本 JIT 编译
@jit(nopython=True)  # nopython=True = @njit
def monte_carlo_pi(n):
    count = 0
    for i in range(n):
        x = np.random.random()
        y = np.random.random()
        if x * x + y * y <= 1.0:
            count += 1
    return 4.0 * count / n

# 性能: Python ~45秒 vs Numba ~0.3秒 (n=10_000_000)

# @njit — 简写（强制 nopython 模式）
@njit
def black_scholes(S, K, T, r, sigma):
    """计算期权价格的向量化函数"""
    d1 = (np.log(S / K) + (r + 0.5 * sigma ** 2) * T) / (sigma * np.sqrt(T))
    d2 = d1 - sigma * np.sqrt(T)
    call = S * _norm_cdf(d1) - K * np.exp(-r * T) * _norm_cdf(d2)
    put = K * np.exp(-r * T) * _norm_cdf(-d2) - S * _norm_cdf(-d1)
    return call, put

@njit
def _norm_cdf(x):
    """标准正态分布 CDF"""
    return 0.5 * (1 + _erf(x / np.sqrt(2)))

@njit
def _erf(x):
    """误差函数近似"""
    a1, a2, a3, a4, a5 = 0.254829592, -0.284496736, 1.421413741, -1.453152027, 1.061405429
    p = 0.3275911
    sign = 1 if x >= 0 else -1
    x = abs(x)
    t = 1.0 / (1.0 + p * x)
    y = 1.0 - (((((a5 * t + a4) * t) + a3) * t + a2) * t + a1) * t * np.exp(-x * x)
    return sign * y

@vectorize — 通用函数

from numba import vectorize, float64
import numpy as np

# 创建通用函数（ufunc），可自动广播
@vectorize([float64(float64, float64)])
def fast_multiply(x, y):
    return x * y + x / (y + 1.0)

# 使用方式与 NumPy ufunc 相同
a = np.random.rand(1_000_000).astype(np.float64)
b = np.random.rand(1_000_000).astype(np.float64)
result = fast_multiply(a, b)  # 自动并行

# 目标参数控制并行
@vectorize([float64(float64, float64)], target="parallel")
def fast_add(x, y):
    return x + y

# target 可选: "cpu", "parallel", "cuda"

GPU 加速 (CUDA)

from numba import cuda
import numpy as np

@cuda.jit
def vector_add_gpu(a, b, result):
    idx = cuda.grid(1)
    if idx < a.size:
        result[idx] = a[idx] + b[idx]

# 准备数据
n = 1_000_000
a = np.random.rand(n).astype(np.float32)
b = np.random.rand(n).astype(np.float32)
result = np.zeros(n, dtype=np.float32)

# 配置 GPU 线程
threads_per_block = 256
blocks_per_grid = (n + threads_per_block - 1) // threads_per_block

# 在 GPU 上执行
vector_add_gpu[blocks_per_grid, threads_per_block](a, b, result)

# 复杂 GPU 计算
@cuda.jit
def matrix_multiply_gpu(A, B, C):
    """矩阵乘法: C = A @ B"""
    row, col = cuda.grid(2)
    if row < A.shape[0] and col < B.shape[1]:
        total = 0.0
        for k in range(A.shape[1]):
            total += A[row, k] * B[k, col]
        C[row, col] = total

Numba 与 NumPy 集成

import numpy as np
from numba import njit

# Numba 原生支持大部分 NumPy 函数
@njit
def image_process(img):
    """图像处理 — 完全在 Numba 中编译"""
    h, w = img.shape[:2]
    result = np.zeros((h, w), dtype=np.float64)
    for i in range(1, h - 1):
        for j in range(1, w - 1):
            # Sobel 边缘检测
            gx = (img[i+1, j-1] - img[i-1, j-1] +
                  2 * img[i+1, j] - 2 * img[i-1, j] +
                  img[i+1, j+1] - img[i-1, j+1])
            gy = (img[i-1, j+1] - img[i-1, j-1] +
                  2 * img[i, j+1] - 2 * img[i, j-1] +
                  img[i+1, j+1] - img[i+1, j-1])
            result[i, j] = np.sqrt(gx * gx + gy * gy)
    return result

# 与 NumPy 完全互操作
@njit
def stats_analysis(data):
    mean = np.mean(data)
    std = np.std(data)
    skew = np.mean((data - mean) ** 3) / (std ** 3)
    kurt = np.mean((data - mean) ** 4) / (std ** 4) - 3
    return {"mean": mean, "std": std, "skew": skew, "kurt": kurt}

04 多进程与向量化

Python 的 GIL（全局解释器锁）限制了多线程的并行计算能力。多进程是绕过 GIL 的主要方法，配合 NumPy 向量化可以实现接近线性的加速。

multiprocessing 基础

from multiprocessing import Pool, cpu_count, Process, Queue
import time

def cpu_intensive(n):
    """CPU 密集型任务"""
    total = 0
    for i in range(n):
        total += i ** 2
    return total

if __name__ == "__main__":
    # 获取 CPU 核心数
    cores = cpu_count()
    print(f"CPU 核心数: {cores}")

    # 进程池
    with Pool(processes=cores) as pool:
        results = pool.map(cpu_intensive, [10_000_000] * cores)

    # 异步提交
    with Pool(processes=cores) as pool:
        async_results = [
            pool.apply_async(cpu_intensive, (10_000_000,))
            for _ in range(cores)
        ]
        results = [r.get() for r in async_results]

    # 分段处理大数据
    data = list(range(100_000_000))
    chunk_size = len(data) // cores
    chunks = [data[i:i+chunk_size] for i in range(0, len(data), chunk_size)]

    with Pool(cores) as pool:
        chunk_results = pool.map(process_chunk, chunks)
    final_result = sum(chunk_results)

import numpy as np

def worker(shm_name, shape, dtype):

p.join()

shm.close()
shm.unlink()

05 异步 I/O 加速

I/O 密集型应用（Web 服务、数据库查询、文件操作）的瓶颈不是 CPU，而是等待 I/O 完成的时间。asyncio 通过事件循环实现单线程并发，消除 I/O 等待浪费。

asyncio 事件循环

import asyncio
import aiohttp
import time

async def fetch_url(session, url):
    """异步 HTTP 请求"""
    async with session.get(url) as response:
        return await response.text()

async def main():
    urls = [
        "https://api.example.com/data/1",
        "https://api.example.com/data/2",
        "https://api.example.com/data/3",
    ] * 10  # 30 个请求

    async with aiohttp.ClientSession() as session:
        tasks = [fetch_url(session, url) for url in urls]
        results = await asyncio.gather(*tasks)

    return results

# 性能对比（30 个 HTTP 请求，RTT=100ms）:
# 同步: 30 * 100ms = ~3 秒
# asyncio: ~100ms（全部并发）
# 多线程: ~100ms + 线程开销

# 控制并发数
semaphore = asyncio.Semaphore(10)

async def rate_limited_fetch(session, url):
    async with semaphore:
        return await fetch_url(session, url)

import asyncio
import aiomysql
import asyncpg

# asyncpg — PostgreSQL
async def query_postgres():
    conn = await asyncpg.connect(
        user="user", password="pass",
        database="db", host="localhost"
    )
    rows = await conn.fetch(
        "SELECT id, name FROM users WHERE is_active = $1",
        True
    )
    await conn.close()
    return rows

# aiomysql — MySQL
async def query_mysql():
    pool = await aiomysql.create_pool(
        host="localhost", user="user",
        password="pass", db="db", minsize=5, maxsize=20
    )
    async with pool.acquire() as conn:
        async with conn.cursor() as cur:
            await cur.execute(
                "SELECT id, name FROM users WHERE is_active = %s",
                (True,)
            )
            rows = await cur.fetchall()
    pool.close()
    await pool.wait_closed()
    return rows

# 连接池管理
class AsyncDBPool:
    def __init__(self, dsn: str, pool_size: int = 10):
        self.dsn = dsn
        self.pool_size = pool_size
        self.pool = None

    async def connect(self):
        self.pool = await asyncpg.create_pool(
            dsn=self.dsn,
            min_size=self.pool_size,
            max_size=self.pool_size * 2,
            max_inactive_connection_lifetime=300
        )

    async def execute(self, query: str, *args):
        async with self.pool.acquire() as conn:
            return await conn.fetch(query, *args)

    async def close(self):
        if self.pool:
            await self.pool.close()

提示:

06 内存优化

Python 的内存效率在某些场景下可以大幅优化。从对象结构到数据结构选择，每一层都有优化空间。

__slots__

__slots__ 告诉 Python 不为实例创建 __dict__，而是使用固定大小的 C 结构体。每个对象节省约 64 字节。

# 普通类 — 每个实例 56+ 字节的 __dict__
class User:
    def __init__(self, name, email, age):
        self.name = name
        self.email = email
        self.age = age

# 使用 __slots__ — 每个实例固定 40 字节
class UserOptimized:
    __slots__ = ("name", "email", "age")

    def __init__(self, name, email, age):
        self.name = name
        self.email = email
        self.age = age

# 内存对比（100 万个实例）
# User: ~380 MB
# UserOptimized: ~50 MB

# 继承中的 __slots__
class Base:
    __slots__ = ("id",)

class Derived(Base):
    __slots__ = ("name",)  # 必须包含父类的 __slots__

# __slots__ 的限制:
# - 不能有 __dict__
# - 不能有 __weakref__（除非显式声明）
# - 不能多继承有不同 __slots__ 的类

weakref

import weakref

class Cache:
    """不会阻止对象被 GC 回收的缓存"""
    def __init__(self):
        self._data = weakref.WeakValueDictionary()

    def set(self, key, value):
        self._data[key] = value

    def get(self, key):
        return self._data.get(key)

# 应用: 对象缓存、事件监听、树形结构的父引用
class TreeNode:
    def __init__(self, value):
        self.value = value
        self.children = []
        self._parent = None

    @property
    def parent(self):
        return self._parent() if self._parent else None

    @parent.setter
    def parent(self, node):
        if node is not None:
            self._parent = weakref.ref(node)
        else:
            self._parent = None

# WeakSet — 不阻止 GC 的集合
seen_objects = weakref.WeakSet()

# WeakKeyDictionary — 键为弱引用
callbacks = weakref.WeakKeyDictionary()

generator vs list

import sys

# list — 所有数据在内存中
def load_all_data():
    return [i ** 2 for i in range(10_000_000)]
# 内存: ~350 MB

# generator — 懒加载，每次生成一个
def stream_data():
    for i in range(10_000_000):
        yield i ** 2
# 内存: ~200 字节

# 文件处理
def read_large_file(path):
    with open(path) as f:
        return f.readlines()  # 全部读入内存

def read_large_file_lazy(path):
    with open(path) as f:
        for line in f:  # 逐行读取
            yield line

# 应用场景:
# - 列表: 需要随机访问、多次遍历、数据量小
# - 生成器: 数据量大、只需遍历一次、管道式处理

对象缓存与数据去重

import sys

# sys.intern — 字符串驻留（节省重复字符串内存）
names = ["Alice", "Bob", "Alice", "Charlie", "Bob"]
interned_names = [sys.intern(n) for n in names]
print(interned_names[0] is interned_names[2])  # True

# 小整数缓存 — Python 默认缓存 -5 到 256
a, b = 100, 100
print(a is b)  # True
c, d = 1000, 1000
print(c is d)  # False（取决于实现）

07 PyPy 及其他 JIT

PyPy 是 Python 的替代运行时，内置 JIT 编译器。对于纯 Python 代码，PyPy 通常比 CPython 快 4-5 倍。但 PyPy 与 C 扩展的兼容性有限。

PyPy JIT 编译

PyPy 使用 JIT（Just-In-Time）编译技术。它分析程序的热点路径，将字节码编译为机器码，并在运行时进行类型特化（type specialization）。循环次数越多，性能优势越明显。

# PyPy vs CPython 性能对比
def heavy_loop(n):
    total = 0
    for i in range(n):
        total += i * i
    return total

# CPython 3.12:      ~3.2 秒 (n=50_000_000)
# PyPy 7.3:          ~0.4 秒 (n=50_000_000)
# 加速比: ~8x

# PyPy 的 warmup 效应
# 第一次调用 JIT 编译时较慢
# 后续调用直接使用编译后的机器码
# 适用于长时间运行的应用

PyPy 兼容性限制

特性兼容性说明
纯 Python 代码完全兼容大部分速度提升 4-5x
C 扩展 (CPython API)有限numpy 通过 PyPy 的 numpy 替代
Cython 编译的 .so不兼容需要 PyPy 重新编译
ctypes/cffi完全兼容PyPy 推荐使用 cffi
cffi推荐PyPy 原生支持，性能更好
asyncio完全兼容性能与 CPython 持平
multiprocessing支持略有差异
numpy有限支持PyPy 的 numpy 实现不完整

PyPy 适用场景

# PyPy 最适合:
# 1. 纯 Python 的 CPU 密集型计算
# 2. 长时间运行的服务
# 3. 大量字符串处理和正则匹配
# 4. 解释器密集型代码（元类、装饰器、eval）

# PyPy 不适合:
# 1. 依赖 C 扩展的项目（numpy, pandas, scipy 等）
# 2. 与 Cython 编译的库交互
# 3. 对内存使用敏感的场景（PyPy 内存占用更高）
# 4. 嵌入式或资源受限环境

# 其他 JIT 选项:
# - Pyston: Fork of CPython + JIT（兼容性更好）
# - Codon: Python 到 C++ 的转译器（学术项目）
# - GraalPy: GraalVM 上的 Python 实现
提示:
不要假设 PyPy 一定能加速。测试是关键——在切换到 PyPy 之前，在目标负载上做基准测试。PyPy 对纯 Python 循环密集的代码加速最明显，但对大量 C 扩展调用的代码可能比 CPython 更慢。
