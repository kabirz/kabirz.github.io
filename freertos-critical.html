01 临界区实现

// tasks.c 中的临界区实现

// 方法 1：taskENTER_CRITICAL / taskEXIT_CRITICAL
// 禁止中断 + 禁止调度

void vPortEnterCritical( void )
{
    portDISABLE_INTERRUPTS();
    uxSavedCriticalNesting++;
    __asm volatile( "dsb" ::: "memory" );
    __asm volatile( "isb" );
}

void vPortExitCritical( void )
{
    configASSERT( uxSavedCriticalNesting > 0 );
    --uxSavedCriticalNesting;
    if( uxSavedCriticalNesting == 0 )
    {
        portENABLE_INTERRUPTS();
    }
}

// 方法 2：ISR 中使用
BaseType_t taskENTER_CRITICAL_FROM_ISR( void )
{
    UBaseType_t uxSavedInterruptStatus = portSET_INTERRUPT_MASK_FROM_ISR();
    uxSavedCriticalNesting++;
    return uxSavedInterruptStatus;
}

void taskEXIT_CRITICAL_FROM_ISR( UBaseType_t uxSavedInterruptStatus )
{
    --uxSavedCriticalNesting;
    if( uxSavedCriticalNesting == 0 )
    {
        portCLEAR_INTERRUPT_MASK_FROM_ISR( uxSavedInterruptStatus );
    }
}

05 ISR 中通信

// ISR 中使用 FromISR 版本的 API
void vISR Handler( void )
{
    BaseType_t xHigherPriorityTaskWoken = pdFALSE;
    uint32_t ulValue = 42;

    // 发送队列
    xQueueSendFromISR( xQueue, &ulValue, &xHigherPriorityTaskWoken );

    // 释放信号量
    xSemaphoreGiveFromISR( xSemaphore, &xHigherPriorityTaskWoken );

    // 设置事件位
    xEventGroupSetBitsFromISR( xEventGroup, BIT_0, &xHigherPriorityTaskWoken );

    // 发送任务通知
    xTaskNotifyFromISR( xTaskHandle, ulValue, eSetValueWithOverwrite, &xHigherPriorityTaskWoken );

    // 必须调用此宏触发上下文切换
    portYIELD_FROM_ISR( xHigherPriorityTaskWoken );
}

06 完整示例

// ===== FreeRTOS 临界区完整示例 =====

#include "FreeRTOS.h"
#include "task.h"
#include "queue.h"
#include <stdio.h>

// 共享资源
volatile uint32_t ulSharedCounter = 0;
QueueHandle_t xISRQueue;

// ISR：按键中断
void EXTI0_IRQHandler( void )
{
    BaseType_t xHigherPriorityTaskWoken = pdFALSE;

    // 清除中断标志
    // EXTI->PR = BIT_0;

    // 发送按键值到队列
    uint32_t ulButton = 1;
    xQueueSendFromISR( xISRQueue, &ulButton, &xHigherPriorityTaskWoken );

    // 触发上下文切换
    portYIELD_FROM_ISR( xHigherPriorityTaskWoken );
}

// 任务：处理按键
void vButtonTask( void * pvParameters )
{
    uint32_t ulButton;

    for ( ;; )
    {
        // 等待 ISR 发送的消息
        if( xQueueReceive( xISRQueue, &ulButton, portMAX_DELAY ) == pdTRUE )
        {
            // 在临界区中更新共享资源
            taskENTER_CRITICAL();
            ulSharedCounter++;
            taskEXIT_CRITICAL();

            printf( "Button pressed, counter: %lu\\n", ulSharedCounter );
        }
    }
}

// 任务：读取共享资源
void vReaderTask( void * pvParameters )
{
    for ( ;; )
    {
        // 在临界区中读取共享资源
        taskENTER_CRITICAL();
        uint32_t ulValue = ulSharedCounter;
        taskEXIT_CRITICAL();

        printf( "Counter value: %lu\\n", ulValue );
        vTaskDelay( pdMS_TO_TICKS( 1000 ) );
    }
}

int main( void )
{
    // 创建 ISR 通信队列
    xISRQueue = xQueueCreate( 10, sizeof( uint32_t ) );

    // 创建任务
    xTaskCreate( vButtonTask, "Button", 128, NULL, 3, NULL );
    xTaskCreate( vReaderTask, "Reader", 128, NULL, 1, NULL );

    vTaskStartScheduler();

    for ( ;; ) {}
}
