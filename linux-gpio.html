01 概述

GPIO（General Purpose Input/Output）是 SoC 上最基础的外设接口。Linux GPIO 子系统通过 gpiolib 框架统一管理数百种 GPIO 控制器，提供标准化的控制器驱动和消费者 API。

核心职责
  • 控制器抽象：统一的 gpio_chip 接口
  • 引脚复用：与 pinctrl 子系统协作
  • 中断支持：GPIO 线作为中断源
  • 消费者 API：设备驱动获取 GPIO
  • sysfs/chardev：用户空间访问

03 gpio_chip

struct gpio_chip

include/linux/gpio/driver.h:402-443

struct gpio_chip {
    const char        *label;      // 芯片名（如 "gpio-pca953x"）
    struct device     *parent;     // 父设备
    struct fwnode_handle *fwnode;
    struct module     *owner;

    // 必须实现的回调
    int (*request)(struct gpio_chip *gc, unsigned int offset);
    void (*free)(struct gpio_chip *gc, unsigned int offset);
    int (*get_direction)(struct gpio_chip *gc, unsigned int offset);
    int (*direction_input)(struct gpio_chip *gc, unsigned int offset);
    int (*direction_output)(struct gpio_chip *gc, unsigned int offset, int value);
    int (*get)(struct gpio_chip *gc, unsigned int offset);
    int (*set)(struct gpio_chip *gc, unsigned int offset, int value);

    // 可选回调
    int (*get_multiple)(...);
    int (*set_multiple)(...);
    int (*to_irq)(struct gpio_chip *gc, unsigned int offset);
    int (*set_config)(...);

    // 芯片属性
    int base;           // -1 表示动态分配
    unsigned int ngpio; // GPIO 数量
    bool can_sleep;     // get/set 是否可能睡眠（I2C/SPI 扩展）
};

关键回调

回调说明必须
request()请求引脚时调用（使能时钟/电源）可选
free()释放引脚时调用可选
get_direction()返回方向：0=输出，1=输入推荐
direction_input()设置为输入双向芯片必须
direction_output()设置为输出双向芯片必须
get()读取引脚值必须
set()设置引脚值必须
to_irq()GPIO 编号转 IRQ 编号支持中断时必须

05 消费者 API

消费者 API（gpiod_*）是设备驱动获取 GPIO 的标准方式，基于设备树/ACPI/固件属性。

获取 GPIO

// 获取单个 GPIO
struct gpio_desc *desc;
desc = devm_gpiod_get(&dev, "enable", GPIOD_OUT_LOW);

// 获取多个 GPIO
struct gpio_descs *descs;
descs = devm_gpiod_get_array(&dev, "data", GPIOD_OUT_LOW);

// 可选获取（不存在返回 NULL 而非错误）
desc = devm_gpiod_get_optional(&dev, "reset", GPIOD_OUT_HIGH);

方向与读写

// 方向设置
gpiod_direction_input(desc);
gpiod_direction_output(desc, value);

// 读写（可能睡眠）
int val = gpiod_get_value(desc);
gpiod_set_value(desc, 1);

// 原始读写（忽略 active-low 翻转）
gpiod_get_raw_value(desc);
gpiod_set_raw_value(desc, 1);

// 批量操作
gpiod_set_array_value(descs->ndescs, descs->desc, values);

// 可睡眠版本（用于 can_sleep 芯片）
gpiod_set_value_cansleep(desc, 0);

gpiod_flags

标志说明
GPIOD_ASIS不改变方向
GPIOD_IN设置为输入
GPIOD_OUT_LOW输出低电平
GPIOD_OUT_HIGH输出高电平
GPIOD_OUT_LOW_OPEN_DRAIN开漏输出低
GPIOD_OUT_HIGH_OPEN_DRAIN开漏输出高

Active-Low 处理

active-low：gpiod API 自动处理 active-low 标志。当设备树中设置了 active-low 属性，gpiod_set_value(desc, 1) 会自动设置物理低电平。使用 gpiod_get_raw_value() 读取物理电平。

GPIO 线名称

GPIO 线名称：在设备节点中使用 name-gpios 属性（如 reset-gpios），消费者调用 devm_gpiod_get(dev, "reset", ...) 时自动匹配。gpiolib 也支持 gpio-line-names 属性为每个 GPIO 指定名称。

libgpiod：用户空间推荐使用 libgpiod 库操作 GPIO，而非 sysfs。sysfs 接口已不推荐用于新代码。字符设备接口支持所有高级功能（中断、事件时间戳等）。

09 编写驱动

GPIO 控制器驱动模板

#include <linux/gpio/driver.h>
#include <linux/module.h>
#include <linux/platform_device.h>

static int my_gpio_get(struct gpio_chip *gc, unsigned int offset)
{
    struct my_gpio_priv *priv = gpiochip_get_data(gc);
    return !!(readl(priv->base + GPIO_IN) & BIT(offset));
}

static int my_gpio_set(struct gpio_chip *gc,
                       unsigned int offset, int value)
{
    struct my_gpio_priv *priv = gpiochip_get_data(gc);
    u32 val = readl(priv->base + GPIO_OUT);
    val = value ? (val | BIT(offset)) : (val & ~BIT(offset));
    writel(val, priv->base + GPIO_OUT);
    return 0;
}

static int my_gpio_direction_input(struct gpio_chip *gc,
                                    unsigned int offset)
{
    struct my_gpio_priv *priv = gpiochip_get_data(gc);
    u32 val = readl(priv->base + GPIO_DIR);
    val &= ~BIT(offset);  // 0 = input
    writel(val, priv->base + GPIO_DIR);
    return 0;
}

static int my_gpio_direction_output(struct gpio_chip *gc,
                                     unsigned int offset, int value)
{
    struct my_gpio_priv *priv = gpiochip_get_data(gc);
    u32 val = readl(priv->base + GPIO_DIR);
    val |= BIT(offset);   // 1 = output
    writel(val, priv->base + GPIO_DIR);
    my_gpio_set(gc, offset, value);
    return 0;
}

static int my_gpio_probe(struct platform_device *pdev)
{
    struct my_gpio_priv *priv;
    struct gpio_chip *gc;

    priv = devm_kzalloc(&pdev->dev, sizeof(*priv), GFP_KERNEL);
    gc = &priv->gc;

    gc->label = "my-gpio";
    gc->parent = &pdev->dev;
    gc->owner = THIS_MODULE;
    gc->base = -1;           // 动态分配
    gc->ngpio = 32;
    gc->can_sleep = 0;       // MMIO 访问，不会睡眠

    gc->get_direction = my_gpio_direction_input;
    gc->direction_input = my_gpio_direction_input;
    gc->direction_output = my_gpio_direction_output;
    gc->get = my_gpio_get;
    gc->set = my_gpio_set;

    return devm_gpiochip_add_data(&pdev->dev, gc, priv);
}

static const struct of_device_id my_gpio_of_match[] = {
    { .compatible = "vendor,my-gpio" },
    { }
};
MODULE_DEVICE_TABLE(of, my_gpio_of_match);

static struct platform_driver my_gpio_driver = {
    .probe = my_gpio_probe,
    .driver = {
        .name = "my-gpio",
        .of_match_table = my_gpio_of_match,
    },
};
module_platform_driver(my_gpio_driver);
MODULE_LICENSE("GPL");
