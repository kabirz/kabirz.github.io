01 概述

PWM（Pulse Width Modulation，脉宽调制）子系统提供了统一的 PWM 输出控制框架，广泛用于 LED 调光、电机控制、风扇转速、电源管理、蜂鸣器等场景。

核心概念
  • 周期（Period）：一个完整 PWM 波形的时间
  • 占空比（Duty Cycle）：高电平持续时间
  • 极性（Polarity）：正常（高→低）或反转（低→高）
  • 使能（Enable）：PWM 输出开关
PWM 波形
  ┌────┐      ┌────┐      ┌────┐
  │    │      │    │      │    │
──┘    └──────┘    └──────┘    └──
  |<->||<----->|<->|
  duty  period
  cycle
  占空比 = duty_cycle / period

与 GPIO/LED 的关系

PWM 与 GPIO：某些 PWM 控制器同时支持 GPIO 功能。Linux PWM 框架集成了 GPIO 芯片支持（gpio_chip），允许将 PWM 通道作为普通 GPIO 使用。LED 子系统中的 PWM LED（leds-pwm.c）就是 PWM 的消费者。

};

配置与应用

struct pwm_state state;

// 初始化状态
pwm_init_state(pwm, &state);

// 配置参数
state.period = 20000000;      // 20ms (50 Hz)
state.duty_cycle = 1000000;   // 1ms (5% 占空比)
state.polarity = PWM_POLARITY_NORMAL;
state.enabled = true;

// 应用到硬件
pwm_apply_might_sleep(pwm, &state);

// 相对占空比
pwm_set_relative_duty_cycle(&state, 50, 100);  // 50%

pwm0: pwm@e0000000 {
    compatible = "vendor,soc-pwm";
    reg = <0xe0000000 0x1000>;
    interrupts = <GIC_SPI 30 IRQ_TYPE_LEVEL_HIGH>;

    compatible = "pwm-backlight";
    pwms = <&pwm0 0 5000000 PWM_POLARITY_NORMAL>;

    brightness-levels = <0 25 50 75 100>;
    default-brightness-level = 75;
};

07 编写驱动

PWM 控制器驱动模板

#include <linux/pwm.h>
#include <linux/module.h>
#include <linux/platform_device.h>

static int my_pwm_apply(struct pwm_chip *chip,
                        struct pwm_device *pwm,
                        const struct pwm_state *state)
{
    u64 duty_ns = state->duty_cycle;
    u64 period_ns = state->period;
    u32 reg_val;

    // 计算占空比寄存器值
    reg_val = (u32)div64_u64(duty_ns * 1024, period_ns);

    // 写入硬件寄存器
    writel(reg_val, chip->base + MY_PWM_DUTY);
    writel(1024, chip->base + MY_PWM_PERIOD);

    // 设置极性
    if (state->polarity == PWM_POLARITY_INVERSED)
        writel(1, chip->base + MY_PWM_INVERT);
    else
        writel(0, chip->base + MY_PWM_INVERT);

    // 使能/禁用
    writel(state->enabled ? 1 : 0, chip->base + MY_PWM_ENABLE);

    return 0;
}

static int my_pwm_get_state(struct pwm_chip *chip,
                            struct pwm_device *pwm,
                            struct pwm_state *state)
{
    u32 duty = readl(chip->base + MY_PWM_DUTY);
    u32 period = readl(chip->base + MY_PWM_PERIOD);
    u32 enable = readl(chip->base + MY_PWM_ENABLE);

    state->period = period;
    state->duty_cycle = duty;
    state->enabled = !!enable;
    state->polarity = PWM_POLARITY_NORMAL;
    return 0;
}

static const struct pwm_ops my_pwm_ops = {
    .apply    = my_pwm_apply,
    .get_state = my_pwm_get_state,
};

static int my_pwm_probe(struct platform_device *pdev)
{
    struct pwm_chip *chip;

    chip = devm_pwmchip_alloc(&pdev->dev, &my_pwm_ops,
                                 4, struct my_pwm_priv);
    chip->base = devm_ioremap_resource(&pdev->dev, res);

    return pwmchip_add(chip);
}

static int my_pwm_remove(struct platform_device *pdev)
{
    return pwmchip_remove(platform_get_drvdata(pdev));
}

static const struct of_device_id my_pwm_of_match[] = {
    { .compatible = "vendor,my-pwm" },
    { }
};
module_platform_driver(my_pwm_driver);
