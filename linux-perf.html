01 概述

Linux 内核的可观测性（observability）由多个子系统组成。本文档聚焦性能监控层，与已有的 ftrace 深度解析 互补。

perf_event_open 系统调用：

// kernel/events/core.c
SYSCALL_DEFINE5(perf_event_open,
                struct perf_event_attr __user *, attr_uptr,
                pid_t, pid, int, cpu, int, group_fd, unsigned long, flags)
{
    // 1. 解析 attr
    struct perf_event_attr attr;
    if (copy_from_user(&attr, attr_uptr, sizeof(attr)))
        return -EFAULT;
    
    // 2. 查找 PMU（根据 attr.type）
    pmu = perf_init_event(&attr);
    
    // 3. 分配 perf_event
    event = perf_event_alloc(&attr, cpu, task, group_leader, NULL,
                              NULL, NULL, flags);
    event->pmu = pmu;
    
    // 4. 调用 pmu->event_init(event)
    err = pmu->event_init(event);
    
    // 5. 添加到 ctx（每 task 或每 CPU）
    perf_install_in_context(ctx, event, event->cpu);
    
    // 6. 创建 fd
    fd = get_unused_fd_flags(O_RDWR);
    file = anon_inode_getfile("[perf_event]", &perf_fops, event, O_RDWR);
    fd_install(fd, file);
    
    return fd;
}

//   展开为：

06 Tracepoint 与 perf 的连接

// kernel/trace/trace_event_perf.c
int perf_trace_init(struct perf_event *perf_event)
{
    struct trace_event_call *tp_event;
    
    // 根据 perf_event->attr.config 找 tracepoint
    // config 是 tracepoint 的 ID（来自 /sys/kernel/debug/tracing/events）
    list_for_each_entry(tp_event, &ftrace_events, list) {
        if (tp_event->event.type == perf_event->attr.config) {
            // 找到，注册 perf callback
            return perf_trace_event_init(tp_event, perf_event);
        }
    }
    return -ENOENT;
}

int perf_trace_event_init(struct trace_event_call *tp_event,
                            struct perf_event *perf_event)
{
    int ret;
    
    // 1. 注册 perf callback 到 tracepoint
    ret = tracepoint_probe_register(tp_event->tp,
                                     tp_event->class->perf_probe,
                                     perf_event);
    
    // 2. 增加 tracepoint refcount
    perf_event->tp_event = tp_event;
    
    return ret;
}

tracepoint 的核心优势:
静态插桩：在编译时确定位置，开销稳定。
共享消费者：ftrace、perf、eBPF 可同时使用一个 tracepoint。
类型安全：通过 DECLARE_TRACE/TRACE_EVENT 宏自动生成。
关闭时零开销：static_call 优化为 nop。
但只能用于预先定义的位置，无灵活性（这是 kprobe 的用武之地）。

07 kprobe 内核探针

kprobe 允许在任意内核指令处动态插入探针。

// include/linux/kprobes.h
struct kprobe {
    struct hlist_node     hlist;
    struct list_head      list;
    unsigned long         nmissed;
    kprobe_opcode_t      *addr;                // 目标地址
    const char           *symbol_name;          // 符号名（替代 addr）
    unsigned int          offset;
    kprobe_pre_handler_t  pre_handler;          // 入口回调
    kprobe_post_handler_t post_handler;         // 出口回调
    kprobe_opcode_t       opcode;               // 原始 opcode
    kprobe_opcode_t      *ainsn;                // 架构特定副本
    u32                   flags;
    bool                  enabled;
};

// 注册：
int register_kprobe(struct kprobe *p);
int register_kprobes(struct kprobe **pps, int num);
void unregister_kprobe(struct kprobe *p);

// 实现机制（x86）：
// 1. register_kprobe()
//    - 找到目标地址 p->addr
//    - 备份原始字节（通常 1 字节）
//    - 写入 int3 (0xCC)
//    
// 2. 内核执行到 int3
//    - 触发 #BP 异常
//    - 进入 do_int3() → kprobe_int3_handler()
//    - 查找 kprobe hash table
//    - 调用 p->pre_handler(p, regs)
//    - single-step 执行原始指令
//    - 调用 p->post_handler(p, regs, flags)
//    - 返回到下一条指令
//
// 3. unregister: 恢复原始字节

// ftrace-based kprobe（优化）：
// 如果目标函数已被 ftrace 插桩（mcount）
// 则不用 int3，而是利用 ftrace hook
// 性能开销从 ~100ns (int3) 降到 ~10ns
// 内核启动时若 CONFIG_KPROBE_EVENTS_ON_NOTRACE=y 启用此优化

// kretprobe — 函数返回探针
struct kretprobe {
    struct kprobe          kp;
    kretprobe_handler_t    handler;
    int                    maxactive;
    int                    nmissed;
    size_t                 data_size;
    struct list_head        free_instances;
    raw_spinlock_t          lock;
};

// 实现：
// 1. kprobe 在函数入口拦截
// 2. 保存返回地址，替换为 trampoline
// 3. 函数返回时跳到 trampoline
// 4. trampoline 调用 handler，然后跳回原始返回地址

通过 tracefs 使用 kprobe：

// /sys/kernel/debug/tracing/kprobe_events
// 添加：
echo 'p:my_probe vfs_read' > kprobe_events
echo 'r:my_ret vfs_read $retval' >> kprobe_events

// 启用：
echo 1 > events/kprobes/my_probe/enable

// 输出格式在 events/kprobes/my_probe/format

// 实现路径：
// 1. 文件写入 → kprobe_events 命令解析
// 2. 解析器 (kernel/trace/trace_kprobe.c)
//    - 创建 struct trace_kprobe
//    - register_kprobe(&tk->rp.kp)
// 3. 注册为 trace event call
// 4. 可被 perf / ftrace / eBPF 共同消费

08 uprobe 用户探针

uprobe 是 kprobe 的用户态对应物，可以追踪用户态程序的函数入口。

// kernel/trace/uprobe.c
struct uprobe {
    struct rb_node         rb_node;          // 全局红黑树节点
    refcount_t             ref;
    struct rw_semaphore    register_rwsem;
    struct rw_semaphore    consumer_rwsem;
    struct list_head        pending_list;
    struct uprobe_consumer *consumers;
    struct inode           *inode;            // 用户 ELF 文件的 inode
    loff_t                  offset;           // 文件偏移
    struct delayed_work     work;
    struct rcu_head         rcu;
    bool                    text_dwarf;
};

struct uprobe_consumer {
    int (*handler)(struct uprobe_consumer *self, struct pt_regs *regs);
    int (*ret_handler)(struct uprobe_consumer *self,
                       unsigned long func, struct pt_regs *regs);
    bool (*filter)(struct uprobe_consumer *self, enum uprobe_filter_ctx ctx,
                    struct mm_struct *mm);
    struct list_head        cons_node;
    unsigned long           flags;
};

// 注册：
int uprobe_register(struct inode *inode, loff_t offset,
                     struct uprobe_consumer *uc);
int uprobe_register_refctr(struct inode *inode, loff_t offset,
                            loff_t ref_ctr_offset,
                            struct uprobe_consumer *uc);
void uprobe_unregister(struct inode *inode, loff_t offset,
                        struct uprobe_consumer *uc);

// 实现：
// 1. uprobe_register
//    - 在 inode+offset 处插入断点
//    - 实现是 page-precise 的：内核替换该页的内容
//    - 写入 arch 的 breakpoint 指令（x86: 0xCC）
//    
// 2. 程序执行到断点
//    - 触发 SIGTRAP
//    - uprobe_notify() 检查是否为 uprobe
//    - 调用 handler(consumer, regs)
//
// 3. uretprobe
//    - 在入口替换返回地址为 trampoline
//    - 函数返回时跳到 trampoline 调用 ret_handler
//    - 然后返回到原始地址

// 通过 tracefs 使用：
// /sys/kernel/debug/tracing/uprobe_events
echo 'p:my_probe /bin/bash:0x42450' > uprobe_events
echo 'r:my_ret /bin/bash:0x42450' >> uprobe_events

// 应用场景：
// - strace 替代（更高性能）
// - 用户态程序 hot path 分析
// - 函数调用统计
// - eBPF uprobe 程序

uprobe vs kprobe 对比：

kprobeuprobe
地址空间内核态用户态
断点指令int3 (内核代码)int3 (用户代码)
处理器上下文内核 IRQ 上下文进程上下文
可观察对象所有内核函数用户 ELF 文件任意地址
典型开销~100ns~1μs (mm 切换)
