01 概述

传统 UNIX 权限模型：root（UID 0）能做任何事，普通用户什么都做不了。这导致一个程序要么不需要 root，要么 root 全开。

Linux Capabilities 把 root 权限拆分成约 40 个独立的能力。每个能力代表一类特权操作，进程可以单独拥有。

传统模型
  • UID 0 = 越过所有检查
  • 非 0 = 检查每个 file mode
  • 过度授权（setuid root 全开）
  • 权限提升不可控
  • bug = 整个系统危险
Capabilities 模型
  • 权限拆分细粒度
  • 每个能力独立检查
  • 最小权限原则
  • 能力可在 exec 时调整
  • 容器仅持必要能力

核心思想：

// 检查 root 改为检查特定能力
// 旧代码：
if (capable(CAP_SYS_ADMIN)) {
    // 允许管理操作
}

// 能力检查宏：
// capable(XXX_CAP) - 是否有该能力
// ns_capable(ns, XXX_CAP) - 在指定 namespace 中
// has_ns_capability_noaudit(ns, ..., cap) - 不审计版本

02 能力分类

Linux 7.0.12 有 41 个能力。按用途分类：

分类能力作用
文件系统CAP_CHOWN修改文件所有者
CAP_DAC_OVERRIDE忽略 DAC（文件权限）
CAP_DAC_READ_SEARCH忽略 read/search 限制
进程管理CAP_KILL发信号给非自有进程
CAP_SETGIDsetgid/setgroups
CAP_SETUIDsetuid/setreuid
网络CAP_NET_BIND_SERVICE绑定 < 1024 端口
CAP_NET_RAWRAW socket（ping、tcpdump）
CAP_NET_ADMIN网卡配置、iptables、路由
系统CAP_SYS_ADMIN"万能能力" — mount、chroot、io等
CAP_SYS_BOOTreboot
CAP_SYS_MODULE加载/卸载内核模块
CAP_SYS_TIMEsettimeofday, stime
硬件CAP_SYS_RAWIOraw I/O、ioperm、iopl
CAP_SYS_PTRACEptrace 任意进程
调度CAP_SYS_NICE修改 nice、CPU affinity
IPCCAP_IPC_LOCKmlock, shmctl SHM_LOCK
性能CAP_PERFMONperf_event_open、eBPF
BPFCAP_BPFBPF 程序加载、map 操作
检查点CAP_CHECKPOINT_RESTORECRIU 检查点恢复
CAP_SYS_ADMIN 是新 root:
CAP_SYS_ADMIN 被用于太多场景：mount、chroot、quotactl、io_setup、各种 namespace 操作、USB 设备配置等。
实践中很难绕开它，被称为"new root"。
Docker 默认不授予此能力，需要 --privileged 或 --cap-add=SYS_ADMIN。

03 数据结构

// include/uapi/linux/capability.h
#define _LINUX_CAPABILITY_VERSION_3  0x20080522

// 用户态结构（向后兼容）
struct user_cap_header {
    __u32              version;          // 版本号
    int                pid;              // 0 = 当前进程
};

struct user_cap_data {
    __u32              effective;        // 当前有效
    __u32              permitted;        // 允许（受 inheritable/bounding 限制）
    __u32              inheritable;      // 可继承
};

// 内核态：
// include/linux/cred.h
struct cred {
    kuid_t               uid;
    kgid_t               gid;
    kuid_t               suid;
    kgid_t               sgid;
    kuid_t               euid;
    kgid_t               egid;
    kuid_t               fsuid;
    kgid_t               fsgid;
    unsigned             securebits;
    kernel_cap_t         cap_inheritable;   // 可继承
    kernel_cap_t         cap_permitted;      // 允许
    kernel_cap_t         cap_effective;      // 有效
    kernel_cap_t         cap_bset;            // bounding set
    kernel_cap_t         cap_ambient;         // ambient 集
    // ...
};

// kernel_cap_t 实际上是 bitmap
// include/uapi/linux/capability.h
#define CAP_LAST_CAP   CAP_CHECKPOINT_RESTORE
#define _KERNEL_CAP_T_BIT (CAP_LAST_CAP + 1)

typedef struct {
    unsigned long     cap[_KERNEL_CAP_WORDS];
} kernel_cap_t;

5 个能力集：

集用途影响范围
cap_effective当前可用的能力每次 capability 检查
cap_permitted允许持有的能力上限effective 的超集
cap_inheritableexec 时可保留跨 exec 不丢
cap_bsetbounding set（永久上限）permitted 的硬上限
cap_ambient隐式能力（exec 不丢）无 file cap 也能保留

04 能力继承规则

能力在 execve() 时的转移规则是 Linux capabilities 最复杂的部分：

// security/commoncap.c - cap_bprm_set_creds()
// 计算 exec 后的能力集

// 设：
//   P_in_*  = 父进程（exec 前）
//   F_*     = 可执行文件的 file capabilities
//   P_out_* = exec 后的能力

// 1. 计算 effective：
P_out_effective = (F_inheritable & P_in_inheritable) |
                   (F_permitted & P_in_permitted) |
                   P_in_ambient;
P_out_effective &= P_out_permitted;
if (F_effective)
    P_out_effective |= F_permitted & P_out_permitted;

// 2. permitted 集计算：
P_out_permitted = (P_in_inheritable & F_inheritable) |
                   (F_permitted & P_in_bounding) |
                   P_in_ambient;
P_out_permitted &= P_in_bounding;       // bounding 始终限制

// 3. inheritable 集：
P_out_inheritable = P_in_inheritable;   // 直接保留

// 4. ambient 集：
P_out_ambient = P_in_ambient;            // 保留（除非 permitted 失去某能力）

// 5. 特殊：root (uid==0) 路径有 fallback

三种 file capability：

# getcap 查看文件能力
$ getcap /usr/bin/ping
/usr/bin/ping = cap_net_raw+ep
#                ↑       ↑ ↑
#                能力名  e=effective, p=permitted, i=inheritable

# setcap 设置
$ setcap cap_net_raw+ep /usr/bin/myprogram
$ setcap -r /usr/bin/myprogram   # 移除

# 文件能力的存储：
# security/xattr.h
#define XATTR_NAME_CAPS "security.capability"
// value 是 struct vfs_cap_data
struct vfs_cap_data_v2 {
    __le32 magic_etc;          // VFS_CAP_REVISION_3
    __le32 data[VFS_CAP_U32].permitted;
    __le32 data[VFS_CAP_U32].inheritable;
    __le32 data[VFS_CAP_U32-1].magic_id;  // 唯一 ID
};

setuid root vs file caps：

// 传统 setuid root（如 ping）
$ ls -l /usr/bin/ping
-rwsr-xr-x 1 root root ... ping
// ping 进程 exec 后 UID=0 → 拥有全部能力

// 现代用 file caps
$ getcap /usr/bin/ping
ping = cap_net_raw+ep
// ping 进程 exec 后是非 root 但拥有 CAP_NET_RAW
// 更细粒度，更安全

05 execve 的能力计算

完整的 execve 能力计算（包含 securebits）：

// security/commoncap.c

static int cap_bprm_creds_from_file(struct linux_binprm *bprm,
                                      struct file *file)
{
    // 1. 读取 file capabilities
    struct vfs_cap_data vcap;
    int ret = bprm_caps_from_vfs_caps(&vcap, bprm, &effective, &has_fcap);
    
    // 2. 处理 root 转 non-root 的特殊情况
    if (!uid_eq(new->euid, new->uid) && ...)
        // SECBIT_NOROOT 检查
        if (issecure(SECURE_NOROOT))
            // 即使 root 也不自动获得能力
            ...
    
    // 3. 处理 setuid root 的默认行为
    if (uid_eq(new->euid, root_uid)) {
        // SECBIT_NO_SETUID_FIXUP 检查
        if (!issecure(SECURE_NO_SETUID_FIXUP)) {
            // 标准行为：root 完全继承
            new->cap_effective = new->cap_permitted;
        }
    }
    
    // 4. ambient 集处理
    // 如果新 effective 不包含 ambient 中的某能力
    // 该 ambient 能力被丢弃
    
    // 5. keep_caps flag
    if (issecure(SECURE_KEEP_CAPS))
        // 即使 setuid 改变 uid 也不丢失能力
    
    return 0;
}

// securebits 是一组可配置的安全策略标志
#define SECUREBITS_DEFAULT 0x00000000
#define SECBIT_NOROOT        1  // root 不自动获能力
#define SECBIT_NOROOT_LOCKED 2  // 锁定
#define SECBIT_NO_SETUID_FIXUP       4
#define SECBIT_NO_SETUID_FIXUP_LOCKED 8
#define SECBIT_KEEP_CAPS       16
#define SECBIT_KEEP_CAPS_LOCKED 32
#define SECBIT_NO_CAP_AMBIENT_RAISE 64
#define SECBIT_NO_CAP_AMBIENT_RAISE_LOCKED 128

06 Ambient 集

Ambient 集（Linux 4.3 引入）是为shell 脚本和解释型程序设计的。它允许在没有 file capabilities 的情况下保留能力跨 exec。

// 添加能力到 ambient
prctl(PR_CAP_AMBIENT, PR_CAP_AMBIENT_RAISE, CAP_NET_BIND_SERVICE, 0, 0);
// 现在 execve 后 CAP_NET_BIND_SERVICE 仍有效

// 使用场景：
//   - 普通用户脚本需要绑定低端口
//   - Python/Node 程序没有 file caps（解释器有）
//   - 容器内 daemon 启动子进程时保留能力

// 移除：
prctl(PR_CAP_AMBIENT, PR_CAP_AMBIENT_LOWER, CAP_NET_BIND_SERVICE, 0, 0);
// 清空：
prctl(PR_CAP_AMBIENT, PR_CAP_AMBIENT_CLEAR_ALL, 0, 0, 0);

// 规则：
//   1. Ambient 必须同时是 permitted 和 inheritable
//   2. exec 后 effective 自动包含 ambient
//   3. 如果 exec 后某能力不在 permitted，ambient 自动清除

举例：

# 在容器内，给 Python 程序 CAP_NET_BIND_SERVICE：
# 传统：setcap cap_net_bind_service+ep /usr/bin/python3
#   问题：所有 python 脚本都有这能力（不安全）

# ambient 方案：
# 1. 启动脚本获得 CAP_NET_BIND_SERVICE
# 2. 添加到 ambient
# 3. exec python script.py
#    → python 进程继承 ambient
#    → script.py 能绑定 80 端口
# 4. 其他 python 调用没有这能力

07 BPF 能力与 cgroup

较新的能力与 cgroup 集成：

// CAP_BPF (5.8+)
// 替代了之前的 CAP_SYS_ADMIN 用于 BPF 操作
// 包括：加载 BPF 程序、创建 map、attach

// CAP_PERFMON (5.8+)
// 用于 perf_event_open、kprobe、uprobe
// 替代 CAP_SYS_ADMIN 中的 perf 部分

// CAP_CHECKPOINT_RESTORE (5.9+)
// CRIU 检查点/恢复进程
// 允许修改 /proc/self/exe、设置克隆标记等

// CAP_BPF 不是完全独立的：
//   - 内核网络操作还需要 CAP_NET_ADMIN
//   - perf 还需要 CAP_PERFMON
//   - 这是为了保留审计层次

// namespace 与 capability：
//   ns_capable(ns, cap) 检查能力在指定 namespace 内
//   每个非 user namespace 都有 owner user namespace
//   容器内的"假"能力通过 user_ns 实现

// cgroup v2 的 BPF:
//   BPF_CGROUP_* 程序由 cgroup owner 加载
//   需要 CAP_NET_ADMIN（在 host ns）或容器 user_ns 内

08 Docker 默认能力集

Docker 默认授予容器一组有限的"安全"能力：

// moby/oci/defaults.go
var defaultCapabilities = []string{
    "CAP_AUDIT_WRITE",            // 写 audit 日志
    "CAP_CHOWN",                  // chown
    "CAP_DAC_OVERRIDE",           // 忽略 DAC
    "CAP_FOWNER",                 // 文件 owner 特权
    "CAP_FSETID",                 // setuid bit
    "CAP_KILL",                   // 发信号
    "CAP_MKNOD",                  // mknod
    "CAP_NET_BIND_SERVICE",       // 绑定 < 1024 端口
    "CAP_NET_RAW",                // RAW socket
    "CAP_SETFCAP",                // 设置 file caps
    "CAP_SETGID",                 // setgid
    "CAP_SETPCAP",                // 修改 cap 集
    "CAP_SETUID",                 // setuid
    "CAP_SYS_CHROOT",             // chroot
    "CAP_NET_BIND_SERVICE",
}

// 默认移除的危险能力（如 CAP_SYS_ADMIN）：
//   CAP_SYS_ADMIN       - 太强大（mount、各种 ns 操作）
//   CAP_SYS_MODULE      - 加载模块
//   CAP_SYS_RAWIO       - I/O 端口
//   CAP_SYS_PTRACE      - 调试任何进程
//   CAP_LINUX_IMMUTABLE - 文件 immutable
//   CAP_MAC_ADMIN       - MAC 配置
//   CAP_MAC_OVERRIDE    - 跳过 MAC
//   ...

用户控制：

# 添加能力
docker run --cap-add=NET_ADMIN alpine ...

# 移除能力
docker run --cap-drop=NET_RAW alpine ...

# 全部移除（最小权限）
docker run --cap-drop=ALL --cap-add=NET_BIND_SERVICE nginx

# 完全授权（危险！）
docker run --privileged alpine ...
//   - 所有能力启用
//   - 所有 device cgroup 限制移除
//   - sysfs / proc 完全可写
//   - 不再隔离

# 单独能力最佳实践：
docker run \
  --cap-drop=ALL \
  --cap-add=NET_BIND_SERVICE \
  nginx
//   只让 nginx 绑定 80/443
//   没有任何其他特权
Rootless 模式下的能力:
Rootless Docker 中，容器内的能力是"假"能力：只对容器内的资源有效。
例如 CAP_NET_ADMIN 只能管理容器内的 veth，不能动宿主 docker0。
这通过 user namespace 实现 — 容器 root 在宿主其实是普通用户。
受限能力列表与特权模式相同。
