01 概述

SPI（Serial Peripheral Interface）是 Motorola 开发的高速同步全双工串行总线，广泛用于连接 Flash、显示屏、传感器、ADC/DAC 等高速外设。

核心特性
  • 全双工：同时收发数据
  • 主从架构：一个主设备，多个从设备
  • 片选：每个从设备独立 CS 线
  • 高速：最高可达 100+ MHz
  • 四线制：SCLK + MOSI + MISO + CS
  • 多模式：Dual/Quad/Octal SPI
信号线
信号方向说明
SCLK主→从时钟信号
MOSI主→从主出从入
MISO从→主主入从出
CS/SS主→从片选（低有效）

SPI vs I2C vs UART

与 I2C 的区别

  • 没有从设备地址：每个从设备需要独立的 CS 线
  • 没有 ACK/NACK：主机无法确认从设备是否存在
  • 全双工：同时收发，MOSI 和 MISO 同时工作
  • 无总线仲裁：只有一个主设备
  • 时钟由主机控制：从设备无法延展时钟

04 传输机制

同步传输（spi_sync）

// 最简单的传输方式：阻塞等待完成
u8 tx_buf[] = { 0x9F };
u8 rx_buf[3];

struct spi_transfer xfer = {
    .tx_buf = tx_buf,
    .rx_buf = rx_buf,
    .len    = 4,
};

struct spi_message msg;
spi_message_init_with_transfers(&msg, &xfer, 1);
spi_sync(spi, &msg);  // 阻塞直到完成

便捷 API

函数说明
spi_write(spi, buf, len)同步写入
spi_read(spi, buf, len)同步读取
spi_write_then_read(spi, tx, tx_len, rx, rx_len)先写后读（全双工模拟）
spi_write_two_msgs()写入两个消息

异步传输（spi_async）

异步传输：通过 spi_async() 提交传输后立即返回，通过 spi_message.complete 回调通知完成。适用于高吞吐量场景。需要控制器支持中断或 DMA。

05 SPI 模式

SPI 模式由 CPOL（时钟极性）和 CPHA（时钟相位）两个参数决定。

模式CPOLCPHA说明
Mode 000空闲低电平，第一个边沿采样
Mode 101空闲低电平，第二个边沿采样
Mode 210空闲高电平，第一个边沿采样
Mode 311空闲高电平，第二个边沿采样

06 SPI-MEM

SPI-MEM 是 Linux 内核中专门用于 SPI Flash/NOR/NAND 的统一接口，支持多种 SPI 传输模式。

struct spi_mem_op

// spi-mem.h — SPI Flash 操作
struct spi_mem_op {
    struct spi_mem_op_cmd cmd;     // 命令（1-4 字节）
    struct spi_mem_op_addr addr;   // 地址（0-4 字节）
    struct spi_mem_op_dummy dummy; // 虚拟字节（等待）
    struct spi_mem_op_data data;   // 数据
};

// 典型操作：读取 JEDEC ID
struct spi_mem_op op = SPI_MEM_OP(
    SPI_MEM_OP_CMD(0x9F, 0),       // 命令：Read ID
    SPI_MEM_OP_NO_ADDR(),             // 无地址
    SPI_MEM_OP_NO_DUMMY(),            // 无虚拟字节
    SPI_MEM_OP_DATA_IN(3, buf, 1)   // 读取 3 字节
);
spi_mem_exec_op(mem, &op);
SPI-MEM 优势：自动处理 Dual/Quad/Octal/DTR 传输模式。驱动只需指定命令/地址/数据，框架根据控制器能力自动选择最优传输方式。

08 编写驱动

SPI 从设备驱动模板

#include <linux/spi/spi.h>
#include <linux/module.h>

struct my_spi_data {
    struct spi_device *spi;
};

static int my_spi_read_reg(struct spi_device *spi, u8 reg, u8 *val)
{
    u8 tx[2] = { reg | 0x80, 0 };
    return spi_write_then_read(spi, tx, 1, val, 1);
}

static ssize_t id_show(struct device *dev,
                   struct device_attribute *attr, char *buf)
{
    struct spi_device *spi = to_spi_device(dev);
    u8 id;
    my_spi_read_reg(spi, 0x00, &id);
    return sysfs_emit(buf, "0x%02x\n", id);
}
DEVICE_ATTR_RO(id);

static int my_spi_probe(struct spi_device *spi)
{
    struct my_spi_data *data;
    u8 id;

    data = devm_kzalloc(&spi->dev, sizeof(*data), GFP_KERNEL);
    data->spi = spi;
    spi_set_drvdata(spi, data);

    // 验证设备存在
    my_spi_read_reg(spi, 0x00, &id);
    if (id != 0xA5) {
        dev_err(&spi->dev, "Chip ID mismatch\n");
        return -ENODEV;
    }

    device_create_file(&spi->dev, &dev_attr_id);
    return 0;
}

static void my_spi_remove(struct spi_device *spi)
{
    device_remove_file(&spi->dev, &dev_attr_id);
}

static const struct spi_device_id my_spi_id[] = {
    { "my-device", 0 },
    { }
};
MODULE_DEVICE_TABLE(spi, my_spi_id);

static const struct of_device_id my_spi_of_match[] = {
    { .compatible = "vendor,my-device" },
    { }
};
MODULE_DEVICE_TABLE(of, my_spi_of_match);

static struct spi_driver my_spi_driver = {
    .probe    = my_spi_probe,
    .remove   = my_spi_remove,
    .id_table = my_spi_id,
    .driver = {
        .name = "my-device",
        .of_match_table = my_spi_of_match,
    },
};
module_spi_driver(my_spi_driver);
MODULE_LICENSE("GPL");
