02 PendSV 上下文切换

// ARM Cortex-M4F PendSV Handler（汇编实现）
// portable/GCC/ARM_CM4F/port.c

void xPortPendSVHandler( void )
{
    // 1. 读取 PSP（进程栈指针）
    __asm volatile(
        "mrs r0, psp                     \n"
    );

    // 2. 保存当前任务寄存器
    // r4-r11 是被调用者保存寄存器
    __asm volatile(
        "stmdb r0!, {r4-r11, r14}        \n"
    );

    // 3. 保存栈顶到 TCB
    __asm volatile(
        "ldr r3, pxCurrentTCB_const      \n"
        "ldr r2, [r3]                    \n"
        "str r0, [r2]                    \n"
    );

    // 4. 切换任务
    __asm volatile(
        "bl vTaskSwitchContext           \n"
    );

    // 5. 加载新任务栈顶
    __asm volatile(
        "ldr r3, pxCurrentTCB_const      \n"
        "ldr r1, [r3]                    \n"
        "ldr r0, [r1]                    \n"
    );

    // 6. 恢复新任务寄存器
    __asm volatile(
        "ldmia r0!, {r4-r11, r14}        \n"
    );

    // 7. 设置 PSP 并返回
    __asm volatile(
        "msr psp, r0                     \n"
        "bx r14                          \n"
    );
}

03 SysTick 定时器

// SysTick 配置（port.c）
void xPortSetupTimerInterrupt( void )
{
    // 设置重载值 = 时钟频率 / tick 频率 - 1
    portNVIC_SYSTICK_LOAD_REG = ( configCPU_CLOCK_HZ / configTICK_RATE_HZ ) - 1UL;

    // 启动 SysTick
    portNVIC_SYSTICK_CTRL_REG = portNVIC_SYSTICK_CLK |    // 使用处理器时钟
                                 portNVIC_SYSTICK_INT |    // 使能中断
                                 portNVIC_SYSTICK_ENABLE;  // 启动计数
}

// SysTick Handler
void xPortSysTickHandler( void )
{
    // 递增 tick 计数并检查是否需要调度
    if( xTaskIncrementTick() != pdFALSE )
    {
        // 需要上下文切换
        portYIELD_WITHIN_API();
    }
}
