01 概述

BuildKit 是 Docker 推出的下一代镜像构建引擎，从 Docker 18.09 开始作为默认 builder。替代了原 Docker 内置的 builder（"legacy builder"）。

核心改进：

特性Legacy BuilderBuildKit
构建顺序严格串行并行（依赖分析）
缓存本地本地 + 远程（registry/OCI）
中间镜像每步产生仅最终
--mount=type不支持支持（cache/secret/ssh）
多平台需多次一次 build
Frontend固定 Dockerfile可插拔（frontend image）
输出imageimage/local/oci/tar
secret明文 ARG/ENVsecure mount

启动方式：

# 1. dockerd 内置（默认）
# dockerd 配置：features: { buildkit: true }
# 或 daemon.json:
{
  "features": {"buildkit": true}
}

# 2. 独立 daemon（standalone）
docker run -d --name buildkit \
  --privileged \
  -v /var/lib/buildkit:/var/lib/buildkit \
  moby/buildkit

# 3. buildctl CLI（独立模式）
buildctl build \
  --frontend dockerfile.v0 \
  --local context=. \
  --local dockerfile=. \
  --output type=image,name=myimage

# 4. Kubernetes 部署（BuildKitd as Deployment）
# 配合 BuildCluster 做分布式构建

03 Dockerfile 解析

// 经典 Dockerfile:
FROM debian:12 AS base
RUN apt-get update && apt-get install -y nginx
COPY nginx.conf /etc/nginx/nginx.conf
EXPOSE 80
CMD ["nginx", "-g", "daemon off;"]

// BuildKit 解析步骤：
// 1. frontend/dockerfile/builder/build.go - Build()
// 2. dockerfile2llb.Convert() — Dockerfile → LLB
// 3. 每个 Dockerfile 指令 → 1 个或多个 LLB Op

// 关键指令映射：
//   FROM          → SourceOp (拉取 image)
//   RUN           → ExecOp (在容器内运行)
//   COPY          → FileOp (从上下文复制)
//   ADD           → FileOp (同 COPY + 解压)
//   ENV/LABEL/EXPOSE/USER/WORKDIR → 修改 metadata
//   VOLUME        → metadata (不实际创建)
//   CMD/ENTRYPOINT → metadata
//   ARG           → 编译时变量

// Dockerfile 解析器：
// frontend/dockerfile/parser/parser.go
type Node struct {
    Value      string
    Original   string
    Children   []*Node
    Next       *Node
    Attributes map[string]bool
    Flags      []string
    StartLine  int
    EndLine    int
}

// 解析支持：
//   - 转义字符（\ 续行）
//   - 注释（#）
//   - Parser directives（# syntax=...）
//   - HEREDOC（Dockerfile 1.4+）
//   - multi-line JSON

Dockerfile 语法版本（parser directives）：

# syntax=docker/dockerfile:1.7
# 顶部第一行声明语法版本
# 这是 frontend image

FROM debian:12
...

// 默认 builtin frontend:
//   dockerfile.v0
// 用户可指定任意 frontend:
//   # syntax=example.com/my-frontend:latest

04 LLB DAG

LLB (Low-Level Builder) 是 BuildKit 内部的有向无环图 (DAG) 表示，是并行调度的核心。

// buildkit/client/llb/llb.go

type State struct {
    output Output
}

// 创建 LLB DAG 示例：
src1 := llb.Image("debian:12")
src2 := llb.Image("alpine:3.18")
copy1 := src1.Dir("/app").Run(llb.Shlex("apt-get update"))
copy2 := copy1.Run(llb.Shlex("apt-get install -y nginx"))

// 用户不直接写 LLB Go 代码
// BuildKit 把 Dockerfile 转换为 LLB

// LLB Op 类型（protobuf）：
// frontend/dockerfile/dockerfile2llb/image/.../pb/ops.proto
message Op {
    OpIdentifier    inputs = 1;
    oneof op {
        SourceOp   source = 2;
        ExecOp     exec = 3;
        FileOp     file = 4;
        BuildOp    build = 5;
        MergeOp    merge = 6;
        DiffOp     diff = 7;
    }
    WorkerConstraints worker = 9;
}

// SourceOp - 输入（FROM）
message SourceOp {
    string identifier = 1;  // docker.io/library/debian:12
    map attrs = 2;
}

// ExecOp - 执行（RUN）
message ExecOp {
    MetaExec meta = 1;
    repeated Mount mounts = 2;
    NetMode network = 3;
    SecurityMode security = 4;
    repeated SecretMount secrets = 5;
    repeated SSHMount ssh = 6;
}

// FileOp - 文件操作（COPY）
message FileOp {
    repeated FileAction actions = 1;
}

message FileAction {
    FileActionCopy copy = 1;
    FileActionMkFile mkfile = 2;
    FileActionMkdir mkdir = 3;
    FileActionRm rm = 4;
}

示例：Dockerfile 转 LLB：

// Dockerfile:
// FROM debian:12 AS builder
// RUN apt-get update
// RUN apt-get install -y nginx
// FROM debian:12
// COPY --from=builder /usr/sbin/nginx /usr/sbin/

// LLB DAG:
//         ┌─ Source(debian:12) ────┐
//         │                         │
//         ▼                         ▼
//   Exec(apt-get update)      Source(debian:12)
//         │                         │
//         ▼                         │
//   Exec(apt-get install)           │
//         │                         │
//         ▼                         │
//   (snapshot 1)                    │
//         │                         │
//         └─────────────┬───────────┘
//                       ▼
//                  File(COPY)
//                       │
//                       ▼
//                  (snapshot 2)
//                       │
//                       ▼
//                  Final Image

// 并行机会：
//   - 两个 Source 可以并行
//   - builder 系列依赖链可以并行
//   - COPY 依赖两个 input，最后执行

func (s *Solver) Solve(ctx, id, g, opt SolveOpt) (*Result, error) {
    // 1. 解析 LLB
    // 2. 构建依赖图
    // 3. 找到所有可并行的节点
    // 4. 提交给 scheduler
}

// scheduler 维护 ready 队列
// 每次 ready 后 dispatch 给 worker
// 等待结果，触发后续节点 ready

// buildkit/solver/edge.go
type edge struct {
    dependsOn   []*state // 入边
    state       EdgeState
    cache       CacheKey
    scheduler   *scheduler
    keys        []*CacheKey
    ops         []*op
    vertex      Vertex
}

// Scheduler 在 buildkit/solver/scheduler.go
type Scheduler struct {
    active   map[*edge]struct{}
    polling  bool
    rebuildCount int
}

// 关键算法：
// 1. 拓扑排序所有 Op
// 2. 无依赖的 Op 立即就绪
// 3. 一个 Op 完成 → 检查所有依赖它的 Op 是否所有依赖完成
// 4. 都完成 → 入 ready 队列
// 5. Worker 从队列取 → 执行

// 并行度控制：
// - Worker 数量限制
// - 同 Op 多次执行（如多平台）共享缓存
// - 死锁检测

06 缓存系统

BuildKit 缓存是内容寻址的：相同输入 → 相同缓存。

// 缓存键计算：
// buildkit/solver/llbsolver/vertex.go
type CacheKey struct {
    digest.Digest
    Inputs []*CacheKey
    Output Index
}

// 1. SourceOp 缓存键 = digest(image_name + platform)
// 2. ExecOp 缓存键 = digest(inputs + cmd + env + cwd)
// 3. FileOp 缓存键 = digest(inputs + src/dst + mode)

// 缓存命中：
//   - 计算当前 Op 的 cache key
//   - 在 cache manifest 查找
//   - 命中 → 跳过执行，复用 snapshot

// 缓存存储：
//   本地：/var/lib/buildkit/cache
//   远程：push/pull from registry
//         - layer digest + snapshot digest
//         - 通过 content store

// 远程缓存配置：
docker build --cache-from=type=registry,ref=user/myimage:cache \
             --cache-to=type=inline,mode=max \
             -t myimage:latest .

// 缓存类型：
//   type=inline      - 元数据嵌在 image config
//   type=registry    - 单独镜像，存 layer
//   type=local       - 本地目录
//   type=gha         - GitHub Actions cache
//   type=s3, azblob  - 云存储

缓存层（layer）的复用：

// Dockerfile:
FROM ubuntu
RUN apt update        # layer A
RUN apt install nginx # layer B

// 首次构建：
//   pull ubuntu image
//   exec RUN apt update   → snapshot A
//   exec RUN apt install  → snapshot B

// 第二次（修改 RUN apt install）：
//   pull ubuntu image     → cache hit
//   RUN apt update        → cache hit (input 不变)
//   RUN apt install nginx → cache miss (input 变了)
//                          → 重新执行
缓存最佳实践:
1. 顺序：把不变层放前面（如系统包），频繁变化层放后面（如源码）
2. 合并：相关 RUN 合并一条（用 &&），减少中间层
3. --mount=cache：包管理缓存（apt/dnpm/pip）跨 build 复用
4. BUILDKIT_INLINE_CACHE=1：push 时附带缓存元数据
5. --cache-from：pull 缓存加速 CI
6. .dockerignore：避免本地文件改动让缓存失效

08 高级语法

BuildKit 独有的 Dockerfile 扩展（需要 # syntax=docker/dockerfile:1.4+）：

# 1. Build secrets（安全密钥）
# syntax=docker/dockerfile:1.4
FROM alpine
RUN --mount=type=secret,id=netrc,target=/root/.netrc \
    cat /root/.netrc && \
    do_stuff_with_credentials

# 构建：
docker build --secret id=netrc,src=$HOME/.netrc .

// 优点：secret 不出现在 image layer
// legacy：通过 ARG/ENV，会留在 layer 中
//       需要额外 squash 才能清除

# 2. SSH agent forwarding
RUN --mount=type=ssh ssh -T git@github.com

# 3. Cache mount（缓存目录）
RUN --mount=type=cache,target=/var/cache/apt,sharing=locked \
    --mount=type=cache,target=/var/lib/apt,sharing=locked \
    apt update && apt install -y curl

// 优势：
//   - 多次 build 复用 apt cache
//   - 多个 build 共享 cache（同一台机器）
//   - 显著加速构建

# 4. HEREDOC（Dockerfile 1.4+）
RUN <<EOF
apt update
apt install -y nginx
EOF

COPY <<EOF /etc/nginx/nginx.conf
server {
    listen 80;
}
EOF

# 5. 多平台（一次 build，多平台镜像）
FROM --platform=$BUILDPLATFORM golang:1.21 AS builder
ARG TARGETOS TARGETARCH
RUN GOOS=$TARGETOS GOARCH=$TARGETARCH go build ...

# 构建：
docker buildx build --platform=linux/amd64,linux/arm64 .

# 6. --link（新 layer 链接）
COPY --link file dest
// 优化：不会触发前一层 copy-up
// 仅产生新 layer 引用前一层

# 7. --parents（保留目录结构）
COPY --parents src/foo src/bar dest/
// dest/foo dest/bar 而不是 dest/* 混合

09 导出与多平台

// Exporter 类型：
// buildkit/exporter/

// 1. image exporter
docker build -o type=image,name=myapp:latest .
//   push 到 registry 或本地 daemon

// 2. oci exporter
docker build -o type=oci,dest=image.tar .
//   导出为 OCI tarball
//   标准 OCI 镜像 layout

// 3. docker exporter
docker build -o type=docker,dest=image.tar .
//   导出为 Docker v2 image tarball

// 4. local exporter
docker build -o type=local,dest=./out .
//   导出 filesystem 到本地目录
//   适合：提取构建产物

// 5. tar exporter
docker build -o type=tar,dest=fs.tar .
//   仅导出 filesystem tarball

// 6. cache exporter
docker build -o type=registry,ref=mycache .

// 多平台输出：
docker buildx build \
  --platform=linux/amd64,linux/arm64 \
  -t myapp:latest \
  --push .

// 输出策略：
//   - --push：直接 push 到 registry
//   - --load：加载到本地 dockerd（仅单平台）
//   - --output：导出到文件
//   - --cache-to：缓存导出
