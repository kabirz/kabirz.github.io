01 概述

内核需要向用户态导出大量数据 — /proc/cpuinfo、/proc/meminfo、/proc/slabinfo…… 传统的 read() + mmap() 接口需要开发者自己管理缓冲区、处理跨多次 read 调用的位置跟踪。

seq_file 将"遍历数据"和"格式化输出"完全解耦。开发者只需实现 4 个回调，框架处理所有缓冲、分页、seek 逻辑。

struct seq_operations
struct seq_operations {
    void * (*start)(struct seq_file *m,
                    loff_t *pos);
    void   (*stop) (struct seq_file *m,
                    void *v);
    void * (*next) (struct seq_file *m,
                    void *v, loff_t *pos);
    int    (*show) (struct seq_file *m,
                    void *v);
};
struct seq_file
struct seq_file {
    char *buf;             // 内部缓冲区
    size_t size;           // 缓冲区大小
    size_t count;          // 已写入字节数
    size_t from;           // 已读取偏移
    loff_t index;          // 当前序号
    loff_t read_pos;       // 当前 read 位置
    u64 version;           // 版本号
    const struct seq_operations *op;
    void *private;
};

核心思想：start → show(item1) → next → show(item2) → … → stop。框架自动在后台管理缓冲区溢出、重分配和迭代恢复。

02 核心回调

回调功能返回
start(m, pos) 开始或继续迭代。定位到 pos 指向的位置，返回第一个对象。 NULL = 结束；非 NULL = 当前对象
show(m, v) 打印对象 v。使用 seq_printf 等输出函数向缓冲区追加内容。 0 = 成功；SEQ_SKIP = 跳过该对象
next(m, v, pos) 前进到下一个对象。更新 *pos 指向新位置。 NULL = 没有更多；非 NULL = 下一个对象
stop(m, v) 结束迭代。释放 start/next 中获取的资源（如锁）。 void

pos 语义：

  • pos = 0：开始（首次打开或 seek 到开头）
  • pos = 0xffffffff：结束
  • 0 < pos < EOF：继续迭代
  • start() 返回 NULL 且 pos > 0：停止迭代
注意:
start() 中获取的锁（如 RCU 读锁）必须在 stop() 中释放。
seq_file 本身不管理锁——这完全由回调负责。

缓冲区溢出后，之前输出被丢弃。框架分配更大的缓冲区，从 pos = 0 重新迭代。

04 遍历模式

seq_file 的核心伪代码：

// seq_read() 内部逻辑
ssize_t seq_read(struct file *file, char __user *buf,
                 size_t size, loff_t *ppos)
{
    // 1. 调用 start 开始迭代
    void *v = m->op->start(m, &m->index);
    while (v) {
        // 2. 调用 show 输出
        int err = m->op->show(m, v);
        if (err || seq_has_overflowed(m))
            break;  // 缓冲区满了
        // 3. 前进到下一个
        v = m->op->next(m, v, &m->index);
    }
    // 4. 结束迭代
    m->op->stop(m, v);

    // 5. 如果溢出，扩大缓冲区重新开始
    if (seq_has_overflowed(m)) {
        kvfree(m->buf);
        m->buf = kvmalloc(m->size << 1, GFP_KERNEL);
        m->size <<= 1;
        m->count = 0;
        m->index = 0;  // 从头开始
        goto restart;
    }

    // 6. 拷贝到用户态
    return copy_to_user(buf, m->buf + m->from, ...);
}

缓冲区的增长策略：

  • 初始大小：1 页（PAGE_SIZE）
  • 溢出后翻倍增长：1P → 2P → 4P → 8P
  • 最大限制：通常 4 × PAGE_SIZE（可通过 seq_buf_alloc 调整）
  • seek 操作重置迭代位置

05 single_open 快捷方式

对于"一次性输出"的单条目文件，不需要 start/next/stop 的完整迭代逻辑：

// include/linux/seq_file.h
int single_open(struct file *file,
                int (*show)(struct seq_file *, void *),
                void *data);
void single_release(struct inode *inode, struct file *file);

// 典型用法：/proc/version
static int version_proc_show(struct seq_file *m, void *v)
{
    seq_printf(m, "%s version %s\n",
               utsname()->sysname, utsname()->release);
    return 0;
}

static int __init proc_version_init(void)
{
    proc_create_single("version", 0, NULL, version_proc_show);
    return 0;
}

对于驱动程序，还有更方便的 DEFINE_SHOW_ATTRIBUTE 宏：

// include/linux/seq_file.h
#define DEFINE_SHOW_ATTRIBUTE(__name)                   \
static int __name ## _open(struct inode *inode,          \
                           struct file *file)            \
{                                                        \
    return single_open(file, __name ## _show,            \
                       inode->i_private);                \
}                                                        \
static const struct proc_ops __name ## _proc_ops = {     \
    .proc_open    = __name ## _open,                     \
    .proc_read    = seq_read,                            \
    .proc_lseek   = seq_lseek,                           \
    .proc_release = single_release,                      \
}

06 procfs 实战

/proc/cpuinfo — 完整迭代模式

// arch/x86/kernel/cpu/proc.c
static struct seq_operations cpuinfo_op = {
    .start = c_start,    // 定位到第 n 个 CPU
    .next  = c_next,     // 移动到下一个在线 CPU
    .stop  = c_stop,     // 释放 RCU 读锁
    .show  = show_cpuinfo, // 打印这个 CPU 的详细参数
};

static void *c_start(struct seq_file *m, loff_t *pos)
{
    // pos 从 0 开始，找到第 *pos 个在线 CPU
    // 获取 RCU 读锁（需要在 stop 中释放）
    return *pos < nr_cpu_ids ? (void *)(*pos + 1) : NULL;
}

static int show_cpuinfo(struct seq_file *m, void *v)
{
    int cpu = (unsigned long)v - 1;
    struct cpuinfo_x86 *c = &cpu_data(cpu);
    seq_printf(m, "processor\t: %u\n"
                   "vendor_id\t: %s\n"
                   "model name\t: %s\n"
                   "cpu MHz\t\t: %d\n"
                   "...",
               cpu, c->x86_vendor_id, c->x86_model_name,
               cpu_khz / 1000);
    return 0;
}

07 实现原理

seq_read — 读取循环

ssize_t seq_read(struct file *file, char __user *buf,
                 size_t size, loff_t *ppos)
{
    struct seq_file *m = file->private_data;
    size_t copied = 0;

    // 处理从上次 read 未读完的数据
    if (m->count) {
        copied =min(m->count - m->from, size);
        if (copy_to_user(buf, m->buf + m->from, copied))
            return -EFAULT;
        m->from += copied;
        return copied;
    }

    // 重新填充缓冲区
    while (1) {
        v = m->op->start(m, &m->index);
        while (v) {
            if (m->op->show(m, v) < 0)
                break;
            if (seq_has_overflowed(m))
                break;
            v = m->op->next(m, v, &m->index);
        }
        m->op->stop(m, v);
        if (!seq_has_overflowed(m))
            break;
        // 缓冲区溢出：翻倍后重试
        kvfree(m->buf);
        m->size <<= 1;
        m->buf = kvmalloc(m->size, GFP_KERNEL_ACCOUNT);
        m->count = 0;
        m->index = 0;
    }
    // 拷贝到用户态
}

seq_lseek — 重置迭代

每次 seek 操作会重置内部状态（index = 0, count = 0），下次 read 从 start 开始重新遍历。

SEQ_SKIP 返回值

// 在 show 回调中返回 SEQ_SKIP 可以跳过当前对象
// 框架会立即调用 next() 前进到下一个
#define SEQ_SKIP 1
关键设计:
seq_file 的缓冲区可能在多次 read() 调用之间累积数据。
这意味着 start() 可能被调用多次（每次 read 都从上次的 pos 开始），
但 show() 只对每个对象调用一次——前提是缓冲区够用。
