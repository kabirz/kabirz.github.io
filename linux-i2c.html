01 概述

I2C（Inter-Integrated Circuit）是 Philips（现 NXP）开发的两线式串行总线，广泛用于连接 SoC 与低速外设（传感器、EEPROM、RTC、PMIC 等）。Linux I2C 子系统通过 i2c-core 统一管理数百种 I2C 控制器和从设备。

核心特性
  • 两线制：SCL（时钟）+ SDA（数据）
  • 主从架构：一个主设备，多个从设备
  • 地址寻址：7 位（128 个）或 10 位地址
  • 多速率：100K/400K/1M/3.4M/5MHz
  • SMBus 兼容：Intel SMBus 协议

I2C 消息结构

struct i2c_msg {
    __u16 addr;      // 从设备地址
    __u16 flags;     // I2C_M_RD / I2C_M_TEN / I2C_M_STOP 等
    __u16 len;       // 数据长度
    __u8 *buf;       // 数据缓冲区
};

传输模式

发送 (Write):
  [START] [从设备地址+W] [ACK] [数据1] [ACK] [数据2] [ACK] ... [STOP]

接收 (Read):
  [START] [从设备地址+R] [ACK] [数据1] [ACK] [数据2] [ACK] ... [NACK] [STOP]

写后读 (Combined):
  [START] [从设备地址+W] [ACK] [寄存器] [RESTART] [从设备地址+R] [ACK] [数据] [NACK] [STOP]

05 SMBus

SMBus（System Management Bus）是基于 I2C 的子集协议，定义了更严格的消息格式。Linux I2C 核心自动在 I2C 上模拟 SMBus 协议。

SMBus 操作

操作函数说明
读字节i2c_smbus_read_byte_data()读取指定寄存器的 8 位值
写字节i2c_smbus_write_byte_data()写入指定寄存器的 8 位值
读字i2c_smbus_read_word_data()读取 16 位值
写字i2c_smbus_write_word_data()写入 16 位值
读块i2c_smbus_read_block_data()读取多个字节
写块i2c_smbus_write_block_data()写入多个字节
快速i2c_smbus_xfer()仅地址，无数据

SMBus vs I2C

// SMBus 示例：读取温度传感器
s32 temp = i2c_smbus_read_word_data(client, 0x00);

// 等价的 I2C 原始操作
struct i2c_msg msgs[2] = {
    { .addr = client->addr, .len = 1, .buf = ®, .flags = 0 },
    { .addr = client->addr, .len = 2, .buf = buf, .flags = I2C_M_RD },
};
i2c_transfer(client->adapter, msgs, 2);
SMBus Alert：SMBus 定义了 Alert 协议，允许从设备通过拉低 SDA 线主动通知主机。Linux 通过 i2c_driver.alert 回调实现。

07 字符设备

i2c-dev.c 为每个 I2C 适配器创建 /dev/i2c-N 字符设备，允许用户空间直接访问 I2C 总线。

用户空间接口

// 打开 I2C 总线
$ ls -la /dev/i2c-*
crw-rw---- 1 root i2c 89, 0 Jan  1 00:00 /dev/i2c-0
crw-rw---- 1 root i2c 89, 1 Jan  1 00:00 /dev/i2c-1

// 使用 i2c-tools
$ i2cdetect -y 0          // 扫描总线上的设备
$ i2cget -y 0 0x50 0x00   // 读取 EEPROM 地址 0x00
$ i2cset -y 0 0x50 0x00 0x42  // 写入值 0x42
$ i2cdump -y 0 0x50       // 转储设备寄存器

// 编程（C 语言）
$ fd = open("/dev/i2c-0", O_RDWR);
ioctl(fd, I2C_SLAVE, 0x50);
write(fd, buf, 2);  // 发送数据
read(fd, buf, 32);  // 接收数据

ioctl 命令

命令说明
I2C_SLAVE设置从设备地址
I2C_SLAVE_FORCE强制设置（即使已有驱动）
I2C_RDWR执行多消息传输
I2C_SMBUS执行 SMBus 操作

08 编写驱动

I2C 从设备驱动模板

#include <linux/i2c.h>
#include <linux/module.h>
#include <linux/delay.h>

struct my_sensor_data {
    struct i2c_client *client;
    int scale;
};

static int my_sensor_read_raw(struct i2c_client *client, u8 reg)
{
    return i2c_smbus_read_byte_data(client, reg);
}

static ssize_t temp_show(struct device *dev,
                     struct device_attribute *attr, char *buf)
{
    struct i2c_client *client = to_i2c_client(dev);
    int raw = i2c_smbus_read_word_data(client, 0x00);
    int temp = (raw >> 4) * 625 / 100;  // 转换为 millidegrees
    return sysfs_emit(buf, "%d\n", temp);
}
DEVICE_ATTR_RO(temp);

static int my_sensor_probe(struct i2c_client *client)
{
    struct my_sensor_data *data;
    int id;

    // 验证设备存在
    id = i2c_smbus_read_byte_data(client, 0x0D);
    if (id != 0xA5) {
        dev_err(&client->dev, "Chip ID mismatch: 0x%02x\n", id);
        return -ENODEV;
    }

    data = devm_kzalloc(&client->dev, sizeof(*data), GFP_KERNEL);
    data->client = client;
    i2c_set_clientdata(client, data);

    device_create_file(&client->dev, &dev_attr_temp);
    dev_info(&client->dev, "sensor probed at 0x%02x\n", client->addr);
    return 0;
}

static void my_sensor_remove(struct i2c_client *client)
{
    device_remove_file(&client->dev, &dev_attr_temp);
    dev_info(&client->dev, "sensor removed\n");
}

static const struct i2c_device_id my_sensor_id[] = {
    { "my-sensor", 0 },
    { }
};
MODULE_DEVICE_TABLE(i2c, my_sensor_id);

static const struct of_device_id my_sensor_of_match[] = {
    { .compatible = "vendor,my-sensor" },
    { }
};
MODULE_DEVICE_TABLE(of, my_sensor_of_match);

static struct i2c_driver my_sensor_driver = {
    .probe    = my_sensor_probe,
    .remove   = my_sensor_remove,
    .id_table = my_sensor_id,
    .driver = {
        .name = "my-sensor",
        .of_match_table = my_sensor_of_match,
    },
};
module_i2c_driver(my_sensor_driver);
MODULE_LICENSE("GPL");
