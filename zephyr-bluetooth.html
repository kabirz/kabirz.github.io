01 蓝牙协议栈概览

Zephyr 内置完整的 Bluetooth Low Energy（BLE）协议栈，支持 Host + Controller 一体化或分离架构。

BLE Central/Peripheral中心设备和外围设备角色
GATT Client/Server通用属性协议
LE Secure ConnectionsLE 安全连接（LESC）

Bluetooth AudioLE Audio / LC3 编解码
ISO Channels同步数据通道（Isochronous）
Direction Finding蓝牙方向查找（AoA/AoD）

05 GATT / ATT

GATT（Generic Attribute Profile）定义了数据组织和交换的框架。

// GATT 服务定义
BT_GATT_SERVICE_DEFINE(my_service,
    BT_GATT_PRIMARY_SERVICE(&my_uuid),
    BT_GATT_CHARACTERISTIC(&my_char_uuid.uuid,
                           BT_GATT_CHRC_READ | BT_GATT_CHRC_WRITE,
                           BT_GATT_PERM_READ | BT_GATT_PERM_WRITE,
                           read_func, write_func, &my_data),
    BT_GATT_CCC(my_ccc_changed, BT_GATT_PERM_READ | BT_GATT_PERM_WRITE),
);

// GATT 读回调
static ssize_t read_func(struct bt_conn *conn,
                         const struct bt_gatt_attr *attr,
                         void *buf, uint16_t len, uint16_t offset)
{
    return bt_gatt_attr_read(conn, attr, buf, len, offset,
                             &my_data, sizeof(my_data));
}

// GATT 写回调
static ssize_t write_func(struct bt_conn *conn,
                          const struct bt_gatt_attr *attr,
                          const void *buf, uint16_t len,
                          uint16_t offset, uint8_t flags)
{
    memcpy(&my_data, buf, len);
    return len;
}

06 SMP 安全管理

SMP（Security Manager Protocol）负责配对、密钥分发和加密。

// 安全管理配置
// CONFIG_BT_SMP=y
// CONFIG_BT_SMP_SC_ONLY=y      ← 仅安全连接
// CONFIG_BT_SMP_MAX_KEY_LEN=16

// 配对回调
static struct bt_conn_auth_cb auth_cb = {
    .passkey_display = passkey_display,
    .passkey_confirm = passkey_confirm,
    .cancel = auth_cancel,
};

bt_conn_auth_cb_register(&auth_cb);

// 安全级别
// BT_SECURITY_LOW    - 无安全
// BT_SECURITY_MEDIUM  - 配对
// BT_SECURITY_HIGH    - 加密
// BT_SECURITY_FIPS    - FIPS 合规

bt_conn_set_security(conn, BT_SECURITY_HIGH);

07 Bluetooth Mesh

Bluetooth Mesh 实现多对多通信，适用于楼宇自动化、传感器网络等场景。

// Mesh 初始化
// CONFIG_BT_MESH=y
bt_mesh_init(&prov, &comp);

// 配网
bt_mesh_prov_enable(BT_MESH_PROV_ADV);

// 模型定义
BT_MESH_MODEL_DEFINE(onoff_model,
                      BT_MESH_MODEL_ID_GEN_ONOFF_SRV,
                      model_ops, &onoff_state, NULL);

// 消息发布
struct bt_mesh_msg_ctx ctx = {
    .app_idx = app_key_idx,
    .addr = target_addr,
    .send_ttl = BT_MESH_TTL_DEFAULT,
};

bt_mesh_model_msg_init(&msg, BT_MESH_MODEL_OP_GEN_ONOFF_SET);
net_buf_simple_add_u8(&msg, onoff);

bt_mesh_model_send(&onoff_model, &ctx, &msg, NULL, NULL);
