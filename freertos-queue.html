01 队列结构体

FreeRTOS 的队列、信号量、互斥锁都基于 queue.c 中的 xQUEUE 结构体实现。

xQUEUE 完整结构

// queue.c 第 103-136 行
typedef struct QueueDefinition
{
    // ===== 环形缓冲区指针 =====
    int8_t * pcHead;        // 缓冲区起始地址
    int8_t * pcWriteTo;     // 下一个写入位置

    // ===== 联合体（队列/信号量二选一） =====
    union
    {
        QueuePointers_t xQueue;     // 队列：pcTail + pcReadFrom
        SemaphoreData_t xSemaphore; // 信号量：持有者 + 递归计数
    } u;

    // ===== 等待列表 =====
    List_t xTasksWaitingToSend;     // 等待发送的任务（阻塞在满队列）
    List_t xTasksWaitingToReceive;  // 等待接收的任务（阻塞在空队列）

    // ===== 状态信息 =====
    volatile UBaseType_t uxMessagesWaiting;  // 当前消息数
    UBaseType_t uxLength;                    // 队列容量
    UBaseType_t uxItemSize;                  // 每条消息大小（字节）

    // ===== 锁定计数 =====
    volatile int8_t cRxLock;   // 接收锁计数（ISR 中接收时递增）
    volatile int8_t cTxLock;   // 发送锁计数（ISR 中发送时递增）

    // ===== 分配标记 =====
#if ( ( configSUPPORT_STATIC_ALLOCATION == 1 ) && ( configSUPPORT_DYNAMIC_ALLOCATION == 1 ) )
    uint8_t ucStaticallyAllocated;
#endif
} xQUEUE;

02 队列创建

// queue.c 中的 xQueueGenericCreate()
// 所有队列/信号量创建最终调用此函数

QueueHandle_t xQueueGenericCreate( const UBaseType_t uxQueueLength,
                                    const UBaseType_t uxItemSize,
                                    const uint8_t ucQueueType )
{
    Queue_t * pxNewQueue;
    size_t xQueueSizeInBytes;

    // 1. 计算缓冲区大小
    xQueueSizeInBytes = ( size_t ) ( uxQueueLength * uxItemSize )
                        + ( size_t ) xQueueStructSize;

    // 2. 分配内存
    pxNewQueue = ( Queue_t * ) pvPortMalloc( xQueueSizeInBytes );

    // 3. 初始化队列
    prvInitialiseNewQueue( uxQueueLength, uxItemSize,
                           ( uint8_t * ) pxNewQueue + xQueueStructSize,
                           ucQueueType, pxNewQueue );

    return pxNewQueue;
}

队列初始化

// queue.c 中的 prvInitialiseNewQueue()
static void prvInitialiseNewQueue( const UBaseType_t uxQueueLength,
                                   const UBaseType_t uxItemSize,
                                   uint8_t * pucQueueStorage,
                                   const uint8_t ucQueueType,
                                   Queue_t * pxNewQueue )
{
    // 1. 设置缓冲区指针
    pxNewQueue->pcHead = pucQueueStorage;

    // 2. 初始化队列状态
    xQueueGenericReset( pxNewQueue, pdTRUE );

    // 3. 如果是互斥锁，特殊初始化
#if ( configUSE_MUTEXES == 1 )
    {
        if( ucQueueType == queueQUEUE_TYPE_MUTEX )
        {
            prvInitialiseMutex( pxNewQueue );
        }
    }
#endif
}

03 发送操作

// queue.c 中的 xQueueGenericSend() — 发送核心逻辑
BaseType_t xQueueGenericSend( QueueHandle_t xQueue,
                               const void * pvItemToQueue,
                               TickType_t xTicksToWait,
                               BaseType_t xCopyPosition )
{
    Queue_t * pxQueue = xQueue;
    BaseType_t xEntryTimeCreated = pdFALSE;

    // 1. 挂起调度器（保护队列操作）
    vTaskSuspendAll();
    {
        // 2. 检查队列是否有空间
        if( pxQueue->uxMessagesWaiting < pxQueue->uxLength )
        {
            // 3. 有空间：拷贝数据到队列
            prvCopyDataToQueue( pxQueue, pvItemToQueue, xCopyPosition );
            pxQueue->uxMessagesWaiting++;

            // 4. 检查是否有任务在等待接收
            if( listLIST_IS_EMPTY( &( pxQueue->xTasksWaitingToReceive ) ) == pdFALSE )
            {
                // 5. 唤醒最高优先级等待任务
                if( xTaskRemoveFromEventList( &( pxQueue->xTasksWaitingToReceive ) ) != pdFALSE )
                {
                    // 唤醒的任务优先级更高，触发调度
                    queueYIELD_IF_USING_PREEMPTION();
                }
            }
        }
        else
        {
            // 6. 队列满：如果等待时间为 0，立即返回
            if( xTicksToWait == ( TickType_t ) 0 )
            {
                xQueueGenericReset( pxQueue, pdFALSE );
            }
            else
            {
                // 7. 队列满：将当前任务加入等待发送列表
                vTaskPlaceOnEventList( &( pxQueue->xTasksWaitingToSend ), xTicksToWait );
                // 挂起调度器后恢复，任务被唤醒
            }
        }
    }
    xTaskResumeAll();

    return pdPASS;
}

数据拷贝函数

// queue.c 中的 prvCopyDataToQueue()
static BaseType_t prvCopyDataToQueue( Queue_t * const pxQueue,
                                       const void * pvItemToQueue,
                                       const BaseType_t xPosition )
{
    if( pxQueue->uxItemSize == ( UBaseType_t ) 0 )
    {
#if ( configUSE_MUTEXES == 1 )
        // 信号量/互斥锁：不拷贝数据，只记录持有者
        if( pxQueue->u.xSemaphore.xMutexHolder == NULL )
        {
            pxQueue->u.xSemaphore.xMutexHolder = xTaskGetCurrentTaskHandle();
        }
        else
        {
            // 递归互斥锁：递增计数
            pxQueue->u.xSemaphore.uxRecursiveCallCount++;
        }
#endif
    }
    else if( xPosition == queueSEND_TO_BACK )
    {
        // 向队尾写入（标准 FIFO）
        ( void ) memcpy( ( void * ) pxQueue->pcWriteTo, pvItemToQueue, ( size_t ) pxQueue->uxItemSize );
        pxQueue->pcWriteTo += pxQueue->uxItemSize;
        if( pxQueue->pcWriteTo >= pxQueue->u.xQueue.pcTail )
        {
            pxQueue->pcWriteTo = pxQueue->pcHead;  // 环形回绕
        }
    }
    else
    {
        // 向队首写入（xQueueSendToFront）
        ( void ) memcpy( ( void * ) pxQueue->u.xQueue.pcReadFrom, pvItemToQueue, ( size_t ) pxQueue->uxItemSize );
        pxQueue->u.xQueue.pcReadFrom -= pxQueue->uxItemSize;
        if( pxQueue->u.xQueue.pcReadFrom < pxQueue->pcHead )
        {
            pxQueue->u.xQueue.pcReadFrom = ( pxQueue->u.xQueue.pcTail - pxQueue->uxItemSize );
        }
    }

    return pdFALSE;
}

04 接收操作

// queue.c 中的 xQueueGenericReceive() — 接收核心逻辑
BaseType_t xQueueGenericReceive( QueueHandle_t xQueue,
                                  void * pvBuffer,
                                  TickType_t xTicksToWait,
                                  BaseType_t xJustPeeking )
{
    Queue_t * pxQueue = xQueue;

    vTaskSuspendAll();
    {
        // 1. 检查队列是否有数据
        if( pxQueue->uxMessagesWaiting > ( UBaseType_t ) 0 )
        {
            // 2. 有数据：从队列拷贝数据
            if( xJustPeeking == pdFALSE )
            {
                prvCopyDataFromQueue( pxQueue, pvBuffer );
                pxQueue->uxMessagesWaiting--;
            }

            // 3. 检查是否有任务在等待发送
            if( listLIST_IS_EMPTY( &( pxQueue->xTasksWaitingToSend ) ) == pdFALSE )
            {
                if( xTaskRemoveFromEventList( &( pxQueue->xTasksWaitingToSend ) ) != pdFALSE )
                {
                    queueYIELD_IF_USING_PREEMPTION();
                }
            }
        }
        else
        {
            // 4. 队列空：等待或返回
            if( xTicksToWait == ( TickType_t ) 0 )
            {
                // 不等待，立即返回
            }
            else
            {
                // 将任务加入等待接收列表
                vTaskPlaceOnEventList( &( pxQueue->xTasksWaitingToReceive ), xTicksToWait );
            }
        }
    }
    xTaskResumeAll();

    return pdPASS;
}

05 信号量实现

信号量是队列的特例：长度为 1，不传递数据。

// queue.c 中的二值信号量创建
SemaphoreHandle_t xSemaphoreCreateBinary( void )
{
    QueueHandle_t xNewSemaphore;

    // 创建长度为 1、大小为 0 的队列
    // uxItemSize = 0 表示不传递数据
    xNewSemaphore = xQueueGenericCreate( ( UBaseType_t ) 1,
                                          ( UBaseType_t ) 0,
                                          queueQUEUE_TYPE_BINARY_SEMAPHORE );

    // 初始状态：无信号（uxMessagesWaiting = 0）
    // 需要 xSemaphoreGive() 才能"填充"信号
    return xNewSemaphore;
}

06 互斥锁实现

// queue.c 中的互斥锁初始化
static void prvInitialiseMutex( Queue_t * pxNewQueue )
{
    // 互斥锁特殊初始化
    pxNewQueue->u.xSemaphore.xMutexHolder = NULL;  // 无持有者
    pxNewQueue->u.xSemaphore.uxRecursiveCallCount = 0;  // 递归计数
    pxNewQueue->pcHead = NULL;  // 标记为互斥锁（不是队列）

    // 初始 Give：使互斥锁可用
    xQueueGenericSend( pxNewQueue, NULL, ( TickType_t ) 0U, queueSEND_TO_BACK );
}

// 优先级继承（queue.c 内部）
// 当高优先级任务等待互斥锁时：
// 1. 提升持有者优先级到等待者的优先级
// 2. 持有者释放锁后恢复基础优先级

// 递归互斥锁
// 同一线程可以多次获取（计数递增）
// 必须释放相同次数
BaseType_t xQueueTakeMutexRecursive( QueueHandle_t xMutex, TickType_t xTicksToWait )
{
    // 如果是当前线程持有，递增计数
    if( pxQueue->u.xSemaphore.xMutexHolder == xTaskGetCurrentTaskHandle() )
    {
        pxQueue->u.xSemaphore.uxRecursiveCallCount++;
        return pdTRUE;
    }
    // 否则正常 Take
    return xQueueGenericReceive( pxMutex, NULL, xTicksToWait, pdFALSE );
}

07 任务通知

// tasks.c 中的任务通知实现
// 每个任务自带通知值，无需创建队列

// 发送通知（tasks.c 第 4500+ 行）
BaseType_t xTaskNotify( TaskHandle_t xTaskToNotify,
                         uint32_t ulValue,
                         eNotifyAction eAction )
{
    tskTCB * pxTCB = xTaskToNotify;

    taskENTER_CRITICAL();
    {
        // 根据动作更新通知值
        switch( eAction )
        {
            case eSetBits:
                pxTCB->ulNotifiedValue[ 0 ] |= ulValue;
                break;
            case eIncrement:
                pxTCB->ulNotifiedValue[ 0 ]++;
                break;
            case eSetValueWithOverwrite:
                pxTCB->ulNotifiedValue[ 0 ] = ulValue;
                break;
            case eSetValueWithoutOverwrite:
                if( pxTCB->ucNotifyState[ 0 ] != taskNOTIFICATION_RECEIVED )
                {
                    pxTCB->ulNotifiedValue[ 0 ] = ulValue;
                }
                break;
        }

        // 标记通知已接收
        pxTCB->ucNotifyState[ 0 ] = taskNOTIFICATION_RECEIVED;

        // 如果任务在等待通知，唤醒它
        if( listIS_CONTAINED_WITHIN( &xSuspendedTaskList,
                                     &( pxTCB->xStateListItem ) ) != pdFALSE )
        {
            ( void ) uxListRemove( &( pxTCB->xStateListItem ) );
            prvAddNewTaskToReadyList( pxTCB );
        }
    }
    taskEXIT_CRITICAL();
    return pdTRUE;
}

递归互斥锁中无❌队列 + 递归计数

09 完整示例

// ===== FreeRTOS 队列完整示例 =====
// 演示：队列创建、发送、接收、ISR 中使用

#include "FreeRTOS.h"
#include "task.h"
#include "queue.h"
#include "semphr.h"
#include <stdio.h>

// 定义队列
#define QUEUE_LENGTH 10
#define ITEM_SIZE    sizeof( uint32_t )

QueueHandle_t xDataQueue;
SemaphoreHandle_t xBinarySem;
SemaphoreHandle_t xCountingSem;
SemaphoreHandle_t xMutex;

// 生产者任务
void vProducerTask( void * pvParameters )
{
    uint32_t ulValue = 0;

    for ( ;; )
    {
        ulValue++;
        // 发送到队列
        if( xQueueSend( xDataQueue, &ulValue, pdMS_TO_TICKS( 100 ) ) == pdTRUE )
        {
            printf( "Produced: %lu\\n", ulValue );
        }
        vTaskDelay( pdMS_TO_TICKS( 200 ) );
    }
}

// 消费者任务
void vConsumerTask( void * pvParameters )
{
    uint32_t ulReceived;

    for ( ;; )
    {
        // 从队列接收
        if( xQueueReceive( xDataQueue, &ulReceived, pdMS_TO_TICKS( 500 ) ) == pdTRUE )
        {
            printf( "Consumed: %lu\\n", ulReceived );
        }
    }
}

// ISR 示例
void vISR Handler( void )
{
    BaseType_t xHigherPriorityTaskWoken = pdFALSE;
    uint32_t ulISRValue = 999;

    // ISR 中发送到队列
    xQueueSendFromISR( xDataQueue, &ulISRValue, &xHigherPriorityTaskWoken );

    // ISR 中释放信号量
    xSemaphoreGiveFromISR( xBinarySem, &xHigherPriorityTaskWoken );

    // 必须调用此宏触发上下文切换
    portYIELD_FROM_ISR( xHigherPriorityTaskWoken );
}

// 互斥锁示例
void vCriticalSectionTask( void * pvParameters )
{
    for ( ;; )
    {
        // 获取互斥锁
        xSemaphoreTake( xMutex, portMAX_DELAY );

        // 临界区代码
        printf( "Task accessing shared resource\\n" );

        // 释放互斥锁
        xSemaphoreGive( xMutex );

        vTaskDelay( pdMS_TO_TICKS( 100 ) );
    }
}

int main( void )
{
    // 创建队列
    xDataQueue = xQueueCreate( QUEUE_LENGTH, ITEM_SIZE );

    // 创建二值信号量
    xBinarySem = xSemaphoreCreateBinary();

    // 创建计数信号量
    xCountingSem = xSemaphoreCreateCounting( 10, 0 );

    // 创建互斥锁
    xMutex = xSemaphoreCreateMutex();

    // 创建任务
    xTaskCreate( vProducerTask, "Producer", 128, NULL, 2, NULL );
    xTaskCreate( vConsumerTask, "Consumer", 128, NULL, 2, NULL );
    xTaskCreate( vCriticalSectionTask, "Critical", 128, NULL, 3, NULL );

    vTaskStartScheduler();

    for ( ;; ) {}
}
