01 概述

内核对象管理有一个经典问题：一个对象可能在多个子系统间共享（如 PCI 设备、网络套接字、驱动实例），需要知道最后一个使用者什么时候释放它。直接使用 refcount_t 虽然能实现计数，但语义不精确，容易误用——比如在引用计数归零后继续 get()，或者在释放时忘记检查是否归零。

kref 是 Linux 内核的"类智能指针"机制，用约 60 行代码将 refcount_t 包装成一套"难以误用"的 API。

struct kref {
    refcount_t refcount;  // 底层就是 refcount_t
};

kref 的核心设计：

  • 自动调用 release：refcount 到 0 时自动调用 release 回调
  • 防 double-free：release 只被调用一次
  • 防 UAF：kref_put() 返回后对象已无效
  • 安全竞态处理：kref_get_unless_zero() 防止 race with last put

02 核心 API

函数说明
kref_init(kref)将 refcount 初始化为 1
kref_read(kref)读取当前引用计数（仅用于 debug）
kref_get(kref)递增引用计数。调用者必须已持有一次引用。
kref_put(kref, release)递减引用计数。归零时调用 release，返回 true。
kref_get_unless_zero(kref)如果 refcount > 0 则递增，否则返回 false。竞态安全。
kref_put_mutex(kref, release, mutex)put + 在 release 前获得 mutex。防止 release 中访问共享数据。
kref_put_lock(kref, release, lock)put + 在 release 前获得 spinlock（irqsave 版本）。
kref_sub(kref, count, release)一次性减去 count（>=1），归零时调用 release。
// 典型用法：嵌入式 kref
struct my_obj {
    struct kref ref;
    int data;
};

static void my_release(struct kref *ref)
{
    struct my_obj *obj = container_of(ref, struct my_obj, ref);
    kfree(obj);
}

struct my_obj *obj = kzalloc(sizeof(*obj), GFP_KERNEL);
kref_init(&obj->ref);     // refcount = 1
kref_get(&obj->ref);      // refcount = 2
kref_put(&obj->ref, my_release);  // refcount = 1
kref_put(&obj->ref, my_release);  // refcount = 0 → 调 my_release → kfree(obj)

03 四条黄金法则

来自 Documentation/core-api/kref.rst 的官方规则：

规则 1：kref_get 必须对应已知活跃的对象

如果你不确定对象是否还活着，不能用 kref_get()。必须用 kref_get_unless_zero()。后者只有在 refcount > 0 时才会递增，防止在对象回收瞬间的竞态。

// 错误：可能 race with last put
struct my_obj *obj = lookup_table_find(key);
kref_get(&obj->ref);  // 如果 obj 已从 lookup 移除但尚未释放呢？
                      // 不对——lookup 本身必须保证对象活着

// 正确：如果 lookup 只持有引用但不保证存活
struct my_obj *obj = lookup_table_find(key);
if (obj && kref_get_unless_zero(&obj->ref)) {
    // 成功获取引用，obj 现在安全
} else {
    // obj 已经或即将被释放
}
规则 2：release 中不可持有 kref_get 路径上的锁

如果你在 release 回调中获取锁 A，而某个函数在持有锁 A 的时候调用了 kref_put()，就会死锁。

// 死锁场景
mutex_lock(&m);
kref_put(&obj->ref, my_release);  // → my_release 试图 mutex_lock(&m)
                                  // → 死锁！
规则 3：kref_put 返回后对象已释放

kref_put() 返回 true 表示 release 已被调用，对象已无效。不能再访问对象的任何字段。

if (kref_put(&obj->ref, my_release)) {
    // obj 已经被 kfree（或资源已释放）
    // 不能再访问 obj->anything
}
规则 4：kref_get_unless_zero 必须配对 kref_put

用 kref_get_unless_zero 获取的引用，最终还是用 kref_put 释放。

04 使用模式

模式一：对象查找 + 自动释放

// drivers/usb/common/ulpi.c — ULPI 总线
static struct ulpi *ulpi_alloc(struct device *parent)
{
    struct ulpi *ulpi;

    ulpi = kzalloc(sizeof(*ulpi), GFP_KERNEL);
    if (!ulpi)
        return NULL;
    kref_init(&ulpi->kref);  // refcount = 1
    // ...
    return ulpi;
}

void ulpi_put(struct ulpi *ulpi)
{
    // 递减引用计数，到 0 时调 ulpi_release
    kref_put(&ulpi->kref, ulpi_release);
}

static void ulpi_release(struct kref *kref)
{
    struct ulpi *ulpi = container_of(kref, struct ulpi, kref);
    // 最后一个引用已释放，清理资源
    kfree(ulpi);
}

模式二：kref_get_unless_zero 安全查找

// drivers/scsi/scsi_scan.c — SCSI target
static struct scsi_target *__scsi_find_target(
    struct device *parent, int channel, uint id)
{
    struct scsi_target *starget;
    // 遍历查找目标对象
    list_for_each_entry(starget, &shost->__targets, siblings) {
        if (starget->channel == channel && starget->id == id) {
            // 如果对象活着就递增引用，否则跳过
            if (kref_get_unless_zero(&starget->reap_ref))
                return starget;
        }
    }
    return NULL;  // 找不到或正在释放
}

// 之后始终配对 kref_put
scsi_target_reap(starget_put(starget));

模式三：kref_put_mutex — 带互斥的释放

// drivers/firmware/raspberrypi.c — RPi firmware
static void rpi_firmware_delete(struct kref *kref)
{
    struct rpi_firmware *fw = container_of(
        kref, struct rpi_firmware, consumers);

    // release 时需要访问共享结构
    // 用传入的 mutex 保护
    // ...
    device_unregister(fw->cl_dev);
    mbox_free_channel(fw->chan);
    kfree(fw);
}

void rpi_firmware_put(struct rpi_firmware *fw)
{
    // put + release 前自动获取 mutex
    // 确保 release 中对共享资源的访问是互斥的
    kref_put_mutex(&fw->consumers,
                   rpi_firmware_delete,
                   &fw->lock);
}

05 与原始 refcount_t 对比

操作refcount_tkref
初始化 refcount_set(&r, 1) kref_init(k) — 自动为 1
递增 refcount_inc(&r) kref_get(k)
安全递增 refcount_inc_not_zero(&r) kref_get_unless_zero(k)
递减 refcount_dec_and_test(&r) → 手动 if(true) release kref_put(k, release) — 自动释放
溢出保护 有（饱和到 0） 同 refcount_t
release 同步 手动实现 kref_put_mutex / kref_put_lock
核心区别:
refcount_t 是"原始计数工具"；kref 是"生命周期管理范式"。
kref 强迫你遵循正确的模式：release 通过回调注入，避免遗漏。

06 实现原理

kref 的整个实现都在 include/linux/kref.h 中，全是 static inline，没有单独的 .c 文件。

// include/linux/kref.h
struct kref {
    refcount_t refcount;
};

void kref_init(struct kref *kref)
{
    refcount_set(&kref->refcount, 1);
}

void kref_get(struct kref *kref)
{
    // refcount_inc 内部有溢出饱和保护
    refcount_inc(&kref->refcount);
}

int kref_put(struct kref *kref,
             void (*release)(struct kref *kref))
{
    // dec_and_test: 如果到 0 就调 release
    if (refcount_dec_and_test(&kref->refcount)) {
        release(kref);
        return 1;
    }
    return 0;
}

int kref_get_unless_zero(struct kref *kref)
{
    // 原子条件：如果当前 >0 就递增，否则不操作
    return refcount_inc_not_zero(&kref->refcount);
}

int kref_put_mutex(struct kref *kref,
    void (*release)(struct kref *kref),
    struct mutex *lock)
{
    // 先 dec，然后获取 mutex，再调 release
    // 这样 release 在 mutex 保护中执行
    if (refcount_dec_and_mutex_lock(&kref->refcount, lock)) {
        release(kref);
        mutex_unlock(lock);
        return 1;
    }
    return 0;
}

refcount_dec_and_mutex_lock() 是 refcount_t 和 mutex 配合的原子操作——先递减 refcount，成功后自动获取 mutex，期间不会丢失竞态。
