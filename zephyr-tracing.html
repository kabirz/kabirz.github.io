02 日志系统

Zephyr 日志系统支持多后端、多级别、编译时裁剪。

日志级别

级别宏说明
OFFLOG_LEVEL_OFF禁用日志
ERRLOG_ERR错误
WRNLOG_WRN警告
INFLOG_INF信息
DBGLOG_DBG调试
// 日志使用
#include <zephyr/logging/log.h>
LOG_MODULE_DECLARE(my_module, LOG_LEVEL_DBG);

void my_function(void)
{
    LOG_ERR("错误: %d", error_code);
    LOG_WRN("警告: 状态异常");
    LOG_INF("信息: 初始化完成");
    LOG_DBG("调试: 值 = %x", value);
}

日志后端

后端Kconfig说明
UARTCONFIG_LOG_BACKEND_UART=y串口输出
RAMCONFIG_LOG_BACKEND_RAM=y内存缓冲
RTTCONFIG_LOG_BACKEND_RTT=ySEGGER RTT
FSCONFIG_LOG_BACKEND_FS=y文件系统
NetCONFIG_LOG_BACKEND_NET=y网络输出

日志过滤

// 编译时过滤（零开销）
LOG_MODULE_REGISTER(my_module, LOG_LEVEL_WRN);

// 运行时过滤
log_filter_set(NULL, 0, my_module_id, LOG_LEVEL_DBG);

// 日志延迟处理
// CONFIG_LOG_DEFERRED=y
// 日志在后台线程中处理，减少 ISR 延迟

03 Shell 子系统

Zephyr Shell 提供交互式命令行接口，支持自定义命令。

// Shell 命令定义
#include <zephyr/shell/shell.h>

// 命令处理函数
static int cmd_hello(const struct shell *sh, size_t argc, char **argv)
{
    shell_print(sh, "Hello, %s!", argv[1] ? argv[1] : "World");
    return 0;
}

// 命令定义
SHELL_CMD_REGISTER(hello, NULL, "Say hello", cmd_hello);

// 命令组
SHELL_STATIC_SUBCMD_SET_CREATE(sub_sensor,
    SHELL_CMD(temperature, NULL, "Read temperature", cmd_temp),
    SHELL_CMD(humidity, NULL, "Read humidity", cmd_humidity),
    SHELL_SUBCMD_SET_END
);
SHELL_CMD_REGISTER(sensor, &sub_sensor, "Sensor commands", NULL);

// 使用
// uart:~$ hello Zephyr
// Hello, Zephyr!
// uart:~$ sensor temperature
// Temperature: 25.5°C

04 系统追踪

Zephyr 追踪子系统记录内核事件，用于性能分析和调试。

追踪事件

事件说明
线程切换上下文切换时间戳
中断中断进入/退出
信号量give/take 操作
互斥锁lock/unlock 操作
定时器定时器启动/停止/到期
内存分配/释放
// 追踪配置
// CONFIG_TRACING=y
// CONFIG_TRACING_NESTED=y

// 追踪后端
// CONFIG_TRACING_BACKEND_UART=y
// CONFIG_TRACING_BACKEND_RAM=y
// CONFIG_TRACING_BACKEND_POSIX=y

// 自定义追踪事件
#include <zephyr/tracing/tracing.h>

SYS_PORT_TRACING_FUNC(thread, create);
SYS_PORT_TRACING_FUNC(k_thread, sched_lock);

05 SystemView

SEGGER SystemView 是实时可视化追踪工具，显示线程调度和中断时序。

// SystemView 配置
// CONFIG_SEGGER_SYSTEMVIEW=y
// CONFIG_TRACING_BACKEND_RTT=y

// 功能：
// - 实时线程调度可视化
// - 中断时序显示
// - CPU 负载分析
// - 用户自定义事件

// 使用：
// 1. 固件启用 SystemView
// 2. J-Link 连接目标板
// 3. 打开 SEGGER SystemView 软件
// 4. 实时查看追踪数据

06 CTF 追踪

CTF（Common Trace Format）是标准追踪格式，可用 Babeltrace 分析。

// CTF 配置
// CONFIG_TRACING_CTF=y

// CTF 输出格式：
// - 线程事件
// - 中断事件
// - 内核对象事件
// - 自定义事件

// 分析：
// babeltrace trace.ctf | head -20
// 或导入 TraceCompass 可视化

07 硬件调试

GDB 调试

# GDB 调试
west debug

# 或手动启动 GDB
arm-zephyr-eabi-gdb build/zephyr/zephyr.elf
(gdb) target remote :3333
(gdb) break main
(gdb) continue
(gdb) bt  # 查看调用栈

# OpenOCD 调试
west debug --runner openocd

# J-Link 调试
west debug --runner jlink

断言与调试宏

// 断言
__ASSERT(condition, "message");
__ASSERT_NO_MSG(condition);

// 调试打印
#include <zephyr/sys/__assert.h>

// CONFIG_ASSERT=y 启用断言
# CONFIG_ASSERT_NO_FILE_INFO=y  ← 不打印文件信息
# CONFIG_ASSERT_NO_MESSAGE_INFO=y ← 不打印消息

08 调试 API

// 日志
LOG_MODULE_REGISTER(name, level);
LOG_ERR(fmt, ...);
LOG_WRN(fmt, ...);
LOG_INF(fmt, ...);
LOG_DBG(fmt, ...);

// Shell
SHELL_CMD_REGISTER(name, subcmds, help, handler);
shell_print(sh, fmt, ...);
shell_error(sh, fmt, ...);

// 断言
__ASSERT(expr, fmt, ...);
__ASSERT_NO_MSG(expr);

// 追踪
SYS_PORT_TRACING_FUNC(obj, func);
SYS_PORT_TRACING_OBJFunc(obj, func, ret);
