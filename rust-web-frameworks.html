02 Axum 详解

Axum 是 Tokio 团队官方维护的 Web 框架，强调类型安全和组合性。

// Cargo.toml:
// [dependencies]
// axum = "0.7"
// tokio = { version = "1", features = ["full"] }

use axum::{routing::get, Router, Json};
use serde::Serialize;

#[derive(Serialize)]
struct Hello { message: String }

#[tokio::main]
async fn main() {
    let app = Router::new()
        .route("/", get(root))
        .route("/users/:id", get(get_user).post(create_user))
        .route("/health", get(|| async { "ok" }));
    
    let listener = tokio::net::TcpListener::bind("0.0.0.0:3000").await.unwrap();
    axum::serve(listener, app).await.unwrap();
}

async fn root() -> Json<Hello> {
    Json(Hello { message: "hi".into() })
}

async fn get_user(Path(id): Path<u32>) -> Json<User> {
    Json(User { id, name: "test".into() })
}

async fn create_user(Json(payload): Json<CreateUser>) -> (StatusCode, Json<User>) {
    (StatusCode::CREATED, Json(User { ... }))
}

核心特性：

  • Extractor 模式：函数参数自动从请求提取
  • 类型安全路由：编译期检查 handler 签名
  • 无宏：纯 Rust 类型系统
  • Tree 路由：路径自动合并
  • 与 Tower 集成：Service trait 生态

Extractor 示例：

use axum::extract::{Path, Query, State, ConnectInfo};
use std::net::SocketAddr;

// 自定义 extractor
struct AuthUser { id: u32 }

#[async_trait]
impl FromRequestParts<AppState> for AuthUser {
    type Rejection = StatusCode;
    
    async fn from_request_parts(parts: &mut Parts, state: &AppState) -> Result<Self, Self::Rejection> {
        let token = parts.headers.get("Authorization")
            .and_then(|v| v.to_str().ok())
            .ok_or(StatusCode::UNAUTHORIZED)?;
        let user = verify_token(token).map_err(|_| StatusCode::UNAUTHORIZED)?;
        Ok(AuthUser { id: user })
    }
}

// 使用：
async fn handler(
    Path(id): Path<u32>,            // URL 参数
    Query(filter): Query<Filter>,    // query string
    State(db): State<DbPool>,        // 全局状态
    user: AuthUser,                  // 自定义 extractor
) -> Json<Item> {
    Json(db.get_item(id, &filter).await?)
}

03 Actix-web 详解

Actix-web 基于 Actor 模型，性能基准测试长期第一。

// [dependencies]
// actix-web = "4"

use actix_web::{web, App, HttpServer, Responder, HttpResponse};

async fn index() -> impl Responder {
    HttpResponse::Ok().body("Hello world!")
}

async fn get_user(path: web::Path<u32>) -> impl Responder {
    format!("User {}", path.into_inner())
}

#[actix_web::main]
async fn main() -> std::io::Result<()> {
    HttpServer::new(|| {
        App::new()
            .route("/", web::get().to(index))
            .route("/users/{id}", web::get().to(get_user))
    })
    .bind("127.0.0.1:8080")?
    .run()
    .await
}

核心特性：

  • Actor 模型：状态管理通过 actor message
  • App scope：模块化路由
  • Websocket 一等公民：内置支持
  • 高性能：经优化，常压基准第一
  • 成熟生态：5+ 年生产使用

State via Actor：

use actix::prelude::*;
use actix_web::web::Data;

struct DbActor { pool: Pool }

#[derive(Message)]
#[rtype(result = "Result<User, Error>")]
struct GetUser { id: u32 }

impl Handler<GetUser> for DbActor {
    type Result = Result<User, Error>;
    fn handle(&mut self, msg: GetUser, _: &mut Self::Context) -> Self::Result {
        self.pool.get_user(msg.id)
    }
}

async fn get_user(
    path: web::Path<u32>,
    db: Data<Addr<DbActor>>,
) -> impl Responder {
    let user = db.send(GetUser { id: path.into_inner() }).await.unwrap().unwrap();
    HttpResponse::Ok().json(user)
}

// App state
HttpServer::new(|| {
    let db_actor = DbActor { pool: create_pool() }.start();
    App::new()
        .app_data(Data::new(db_actor))
        .route("/users/{id}", web::get().to(get_user))
});

04 Rocket 详解

Rocket 是最早的 Rust Web 框架之一，强调开发者体验和宏。

// [dependencies]
// rocket = { version = "0.5", features = ["json"] }

#[macro_use] extern crate rocket;

#[get("/")]
fn index() -> &'static str {
    "Hello, world!"
}

#[get("/users/<id>")]
fn get_user(id: u32) -> Json<User> {
    Json(User { id, name: "test".into() })
}

#[launch]
fn rocket() -> _ {
    rocket::build()
        .mount("/", routes![index])
        .mount("/api", routes![get_user])
}

核心特性：

  • 过程宏路由：#[get("/path")]
  • 类型安全参数：FromParam、FromSegments
  • Request Guards：类似 extractor
  • 模板支持：Tera、Handlebars
  • 数据库集成： #[database]

Request Guard 示例：

use rocket::request::{FromRequest, Outcome, Request};

struct AdminUser(User);

#[rocket::async_trait]
impl<'r> FromRequest<'r> for AdminUser {
    type Error = ();
    
    async fn from_request(req: &'r Request<'_>) -> Outcome<Self, Self::Error> {
        let user = User::from_request(req).await?;
        if user.is_admin {
            Outcome::Success(AdminUser(user))
        } else {
            Outcome::Forward(Status::Forbidden)
        }
    }
}

#[get("/admin")]
fn admin_panel(admin: AdminUser) -> Json<AdminData> {
    // 自动调用 AdminUser::from_request
    Json(get_admin_data(admin.0.id))
}

06 底层：Hyper

Hyper 是 Rust 最成熟的 HTTP 实现，被所有主流框架使用。

// Hyper 提供底层 HTTP/1/2
// [dependencies]
// hyper = { version = "1", features = ["full"] }
// tokio = { version = "1", features = ["full"] }

use hyper::{Body, Request, Response, Server};
use hyper::service::{make_service_fn, service_fn};

async fn handle(_req: Request<Body>) -> Result<Response<Body>, hyper::Error> {
    Ok(Response::new(Body::from("Hello!")))
}

#[tokio::main]
async fn main() {
    let make_svc = make_service_fn(|_conn| async {
        Ok::<_, hyper::Error>(service_fn(handle))
    });
    
    let addr = ([0, 0, 0, 0], 3000).into();
    let server = Server::bind(&addr).serve(make_svc);
    
    if let Err(e) = server.await {
        eprintln!("server error: {}", e);
    }
}

// Hyper 关键 API：
//   - Body / Request / Response
//   - Server (incoming)
//   - Client (outgoing)
//   - Service trait

// HTTP/2 / HTTP/3：
//   h2 crate (HTTP/2)
//   h3 crate (HTTP/3，基于 quinn)

reqwest — HTTP 客户端（基于 Hyper）：

// [dependencies]
// reqwest = { version = "0.12", features = ["json"] }

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    let resp = reqwest::get("https://api.example.com/users").await?
        .json::<Vec<User>>().await?;
    
    for u in resp {
        println!("{}", u.name);
    }
    
    // POST
    let client = reqwest::Client::new();
    let res = client.post("https://api.example.com/users")
        .json(&CreateUser { name: "Alice".into() })
        .send()
        .await?;
    
    Ok(())
}

07 中间件机制

三种主流中间件模式：

// 1. Axum + Tower Layer
use axum::middleware::{self, Next};
use tower::ServiceBuilder;
use tower_http::{trace::TraceLayer, compression::CompressionLayer};

let app = Router::new()
    .route("/", get(handler))
    .layer(
        ServiceBuilder::new()
            .layer(TraceLayer::new_for_http())
            .layer(CompressionLayer::new())
            .layer(middleware::from_fn(auth_middleware))
    );

async fn auth_middleware(req: Request, next: Next) -> Response {
    if !is_authorized(&req) {
        return StatusCode::UNAUTHORIZED.into_response();
    }
    next.run(req).await
}

// 2. Actix middleware
use actix_web::middleware::Logger;
use actix_web::dev::{forward_ready, Service, ServiceRequest, ServiceResponse, Transform};

HttpServer::new(|| {
    App::new()
        .wrap(Logger::default())
        .wrap(Auth)
        .route("/", web::get().to(handler))
});

struct Auth;

impl<S, B> Transform<S, ServiceRequest> for Auth
where S: Service<ServiceRequest, Response = ServiceResponse<B>>
{
    // ...
}

// 3. Rocket Fairings
use rocket::fairing::{Fairing, Info, Kind};
use rocket::{Rocket, Orbit};

struct Counter;
#[rocket::async_trait]
impl Fairing for Counter {

08 Extractors 模式

Extractor 是从 Request 提取数据的统一抽象。所有主流框架都支持。

// Axum 内置 extractor：
async fn handler(
    Method(method): Method,                // GET/POST/...
    Uri(uri): Uri,                          // 完整 URI
    Path(id): Path<u32>,                    // /users/:id
    Query(filter): Query<Filter>,          // ?foo=bar
    Json(body): Json<CreateUser>,         // JSON body
    Form(form): Form<LoginForm>,          // form-urlencoded
    TypedHeader(auth): TypedHeader<Authorization<Bearer>>,
    State(db): State<Db>,                   // 共享状态
    Extension(user): Extension<AuthUser>,  // 中间件注入
    headers: HeaderMap,                     // 所有 headers
    ConnectInfo(addr): ConnectInfo<SocketAddr>,
    // ...
) -> Json<Item> { ... }

// 自定义 extractor：
struct ApiKey(String);

#[async_trait]
impl<S> FromRequestParts<S> for ApiKey
where S: Send + Sync
{
    type Rejection = (StatusCode, String);
    
    async fn from_request_parts(parts: &mut Parts, _state: &S) -> Result<Self, Self::Rejection> {
        let key = parts.headers.get("X-API-Key")
            .and_then(|v| v.to_str().ok())
            .ok_or((StatusCode::BAD_REQUEST, "missing api key".into()))?;
        Ok(ApiKey(key.to_string()))
    }
}

async fn protected(api_key: ApiKey) -> Json<Data> {
    // 自动从 header 提取
    Json(get_data(&api_key.0))
}

// 顺序很重要：
//   Path 和 Query 在 JSON 之前
//   extractor 消费 body（一次性的）
//   自定义 extractor 在内置之后

类型安全的路由：

// Axum 通过泛型约束检查
// 如果 handler 期望 u32，但路由参数不是数字 → 编译错误

#[derive(FromRequestParts)]   // 不存在，但概念
struct RequestContext {
    user: AuthUser,
    db: DbPool,
}

async fn handler(
    Path(id): Path<u32>,
    ctx: RequestContext,
) -> Json<Item> { ... }

// 错误的 handler 不会通过类型检查
async fn bad_handler(Path(id): Path<String>) -> ... 
// 如果路由写 /:id 而 id 不是 String → 编译报错

10 选型指南

决策树：

新项目？
├─ 是 → Axum
│       - Tokio 官方维护
│       - 类型安全
│       - 与 Tower 生态无缝
│       - 1.0 即将到来
│
├─ 已有 Actix 项目 → 继续 Actix-web
│       - 性能最好
│       - 成熟稳定
│
└─ 偏好宏风格 → Rocket
        - 学习曲线最低
        - 宏做大部分工作

特殊需求：
├─ WebSockets 重度 → Actix-web（内置）
├─ OpenAPI → Poem 或 Salvo
├─ 服务端渲染 → Axum + askama
├─ 微服务 → Axum + tonic（gRPC）
├─ 性能极致 → Actix-web
└─ 嵌入式 / no_std → 不用框架，直接 hyper
Web 框架总结:
1. Axum：新项目首选，Tokio 官方，类型安全
2. Actix-web：性能王者，Actor 模型
3. Rocket：宏友好，开发体验好
4. Hyper：所有框架的底层基础
5. Tower：中间件标准 trait
6. Extractor 模式：所有框架核心抽象
Rust Web 性能远超 Java/Python/Node.js，与 Go 相当甚至更快。
