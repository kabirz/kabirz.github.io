01 LLEXT 概览

LLEXT（Loadable ELF Extension）是 Zephyr 的运行时模块加载框架，允许在运行时动态加载 ELF 格式的代码模块。

核心价值

特性说明
运行时加载无需重新编译整个固件
ELF 格式标准 ELF 文件，可用 GCC 编译
符号解析自动解析内核 API 符号
内存管理支持 Flash/内存加载
Shell 集成通过 Shell 命令加载/卸载

02 架构设计

// LLEXT 核心数据结构
struct llext {
    sys_snode_t node;            // 全局链表节点
    const char *name;            // 扩展名称
    struct llext_loader *ldr;    // 加载器

    // ELF 段信息
    elf_shdr_t *sect_hdrs;       // 段头表
    size_t sect_cnt;             // 段数量

    // 段内存布局
    void *mem;                   // 分配的内存
    size_t mem_size;             // 内存大小

    // 符号表
    const char *sym_tab;         // 符号名表
    Elf_Sym *sym_tab_data;       // 符号数据
    size_t sym_cnt;              // 符号数量
};

03 ELF 加载器

LLEXT 支持多种加载源：内存缓冲区、文件系统。

// 从内存缓冲区加载
#include <zephyr/llext/llext.h>
#include <zephyr/llext/buf_loader.h>

// ELF 文件数据
extern char _binary_my_ext_start[];
extern char _binary_my_ext_end[];

struct llext_buf_loader buf_loader = {
    .buf = _binary_my_ext_start,
    .len = _binary_my_ext_end - _binary_my_ext_start,
};

struct llext *ext;
int ret = llext_load(&buf_loader.loader, "my_ext", &ext, NULL);

// 从文件系统加载
#include <zephyr/llext/fs_loader.h>

struct llext_fs_loader fs_loader;
llext_fs_loader_init(&fs_loader, "/littlefs/ext.bin");

ret = llext_load(&fs_loader.loader, "ext", &ext, NULL);
