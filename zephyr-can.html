CAN 帧格式

标准帧（CAN 2.0）：

SOF ID RTR Data (0-8 bytes) CRC ACK
1b 11b 1b 64b 16b 2b

CAN-FD 帧：

SOF ID RTR BRS Data (0-64 bytes) CRC ACK
1b 11b 1b 1b 512b 21b 2b

02 CAN 驱动 API

// CAN API（include/zephyr/drivers/can.h）

// 获取 CAN 设备
const struct device *can_dev = DEVICE_DT_GET(DT_NODELABEL(can0));

// 配置 CAN
struct can_timing timing;
can_calc_timing(can_dev, 500000, 100, &timing);
can_set_timing(can_dev, &timing);

// 设置模式
can_set_mode(can_dev, CAN_MODE_FD);

// 启动
can_start(can_dev);

// 发送帧
struct can_frame frame = {
    .id = 0x123,
    .dlc = 8,
    .data = {0x01, 0x02, 0x03, 0x04, 0x05, 0x06, 0x07, 0x08},
};

can_send(can_dev, &frame, K_MSEC(100), tx_callback, NULL);

// 接收回调
static void rx_callback(const struct device *dev, struct can_frame *frame, void *user_data)
{
    printk("收到帧: ID=0x%03x DLC=%d\n", frame->id, frame->dlc);
}

// 添加接收过滤器
struct can_filter filter = {
    .id = 0x456,
    .mask = CAN_STD_ID_MASK,
};
can_add_rx_filter(can_dev, rx_callback, NULL, &filter);

03 过滤器

// CAN 过滤器匹配标准/扩展帧

// 标准帧过滤（11位 ID）
struct can_filter std_filter = {
    .id = 0x100,
    .mask = CAN_STD_ID_MASK,  // 0x7FF
};

// 扩展帧过滤（29位 ID）
struct can_filter ext_filter = {
    .id = 0x12345678,
    .mask = CAN_EXT_ID_MASK,  // 0x1FFFFFFF
};

// 掩码过滤
// mask 中 bit=1 表示必须匹配
// mask 中 bit=0 表示不关心

// 示例：匹配 0x100-0x10F
struct can_filter range_filter = {
    .id = 0x100,
    .mask = 0x7F0,  // 高7位必须匹配
};

04 ISO-TP 传输层

05 CAN-FD

// CAN-FD 支持更高速率和更大数据

// 传统 CAN：最高 1Mbps，最大 8 字节
// CAN-FD：最高 8Mbps，最大 64 字节

// CAN-FD 配置
can_set_mode(can_dev, CAN_MODE_FD);

// CAN-FD 帧
struct canfd_frame fd_frame = {
    .id = 0x123,
    .dlc = 16,
    .flags = CANFD_FDF | CANFD_BRS,  // FD 格式 + 速率切换
    .data = {...},
};

can_send(can_dev, &fd_frame, K_MSEC(100), tx_cb, NULL);

08 CAN API

// 配置
int can_set_mode(const struct device *dev, enum can_mode mode);
int can_set_timing(const struct device *dev, const struct can_timing *timing);
int can_set_timing_data(const struct device *dev, const struct can_timing *timing_data);

// 控制
int can_start(const struct device *dev);
int can_stop(const struct device *dev);

// 收发
int can_send(const struct device *dev, const struct can_frame *frame,
             k_timeout_t timeout, can_tx_callback_t callback, void *user_data);

// 过滤器
int can_add_rx_filter(const struct device *dev, can_rx_callback_t callback,
                      void *user_data, const struct can_filter *filter);
void can_remove_rx_filter(const struct device *dev, int filter_id);

// 信息
int can_get_state(const struct device *dev, enum can_state *state, struct can_bus_err_cnt *err_cnt);
int can_get_capabilities(const struct device *dev, can_mode_t *cap);
