01 概述

Rust 宏是编译期代码生成，分两类：

类型定义用例
声明式宏macro_rules!vec![], println!(), format!()
过程宏 Derive#[derive(MyMacro)]Debug, Clone, serde::Serialize
过程宏 Attribute#[my_macro]#[tokio::main], #[get("/")]
过程宏 Function-likemy_macro!()sqlx::query!()

宏 vs 函数：

函数宏
时机运行时编译期
参数个数固定可变
参数类型固定任意 token
性能调用开销展开 + 编译
调试容易需要 cargo expand

典型场景：

  • 减少样板代码（#[derive(Debug)]）
  • DSL（领域特定语言，如 SQL、HTML 模板）
  • 编译期检查（如 sqlx::query!）
  • 配置生成（lazy_static!）
  • 自动实现 trait

02 声明式宏 macro_rules!

// 基本语法
macro_rules! my_macro {
    // 模式 => 展开
    ($x:expr) => { println!("{}", $x) };
}

my_macro!(5);   // println!("{}", 5)

// 多个模式（重载）
macro_rules! greet {
    () => { println!("Hello!") };
    ($name:expr) => { println!("Hello, {}!", $name) };
}

greet!();              // Hello!
greet!("World");       // Hello, World!

// vec! 宏的实现（简化）：
macro_rules! vec {
    ($($x:expr),* $(,)?) => {{
        let mut v = Vec::new();
        $(
            v.push($x);
        )*
        v
    }};
}

let v = vec![1, 2, 3];
let v = vec![1, 2, 3,];   // 末尾逗号 OK

// println! 的实现（简化）：
macro_rules! println {
    () => { $crate::print!("\n") };
    ($($arg:tt)*) => {{
        $crate::io::_print(format_args!($($arg)*));
        $crate::io::_print(format_args_nl!());
    }};
}

03 模式与匹配

// 元变量（fragment）类型：
//   $x:expr     表达式
//   $x:ident    标识符
//   $x:ty       类型
//   $x:path     路径（a::b::c）
//   $x:stmt     语句
//   $x:block    代码块
//   $x:pat      模式
//   $x:literal  字面量（数字、字符串）
//   $x:meta     属性
//   $x:lifetime 生命周期
//   $x:vis      可见性
//   $x:tt       单个 token tree

macro_rules! example {
    ($e:expr) => { println!("expr: {}", $e) };
    ($i:ident) => { println!("ident: {}", stringify!($i)) };
    ($t:ty) => { println!("type: {}", stringify!($t)) };
    ($($i:ident),*) => { println!("idents: $( $i ),*") };
}

example!(5);          // expr: 5
example!(foo);        // ident: foo
example!(Vec<i32>);   // type: Vec<i32>
example!(a, b, c);    // idents: $( $i ),*

// 重复：
//   $($x)...* 重复
//   分隔符：, ; 等
//   量词：* (0+), + (1+), ? (0 或 1)

macro_rules! sum {
    // 0 个
    () => { 0 };
    // 1+
    ($first:expr $(, $rest:expr)*) => {
        $first + sum!($($rest),*)
    };
}

sum!()           // 0
sum!(1)          // 1
sum!(1, 2, 3)    // 6

// 高级：嵌套重复
macro_rules! matrix {
    ($($($x:expr),*);*) => {{
        vec![
            $(
                vec![$($x),*],
            )*
        ]
    }};
}
let m = matrix![1, 2, 3; 4, 5, 6];
// vec![vec![1,2,3], vec![4,5,6]]

// 递归深度限制：默认 64，可通过 #![recursion_limit] 调整
#![recursion_limit = "256"]

04 宏卫生性

// 卫生宏：宏展开不污染调用方作用域

macro_rules! using_x {
    ($e:expr) => { x + $e };
}

let x = 5;
// using_x!(1) 不会引用这个 x（因为宏内部的 x 不存在）

// 这避免了名字捕获问题
// 例子：C 的宏 #define SQUARE(x) x*x 不卫生
// SQUARE(1+2) → 1+2*1+2 = 5（错误）
// Rust 宏不会这样

// 展开示例：
macro_rules! log {
    ($msg:expr) => {
        println!("[{}] {}", "LOG", $msg)
    };
}
// 展开为字面量 "LOG"，不会与调用方的 LOG 变量冲突

// 跨边界的 hygiene：
//   $crate — 指向定义宏的 crate
//   用于引用宏 crate 中的其他项

macro_rules! my_macro {
    () => {
        $crate::helper::do_something()  // 总是指向 my_macro 所在 crate
    };
}

// 局部 hygiene：
//   宏内 let 创建的变量，外部不可见
macro_rules! swap {
    ($a:expr, $b:expr) => {{
        let tmp = $a;
        $a = $b;
        $b = tmp;
    }};
}
// 调用方看不到 tmp

// 反例：需要"非卫生"的场景
//   使用 ident 元变量传入名字
macro_rules! make_var {
    ($name:ident, $val:expr) => {
        let $name = $val;
    };
}
make_var!(my_var, 5);
println!("{}", my_var);  // 5

05 过程宏

// 过程宏是 Rust 编译器调用的函数
// 接收 TokenStream 输入，返回 TokenStream 输出

// 三种类型：
// 1. derive     - 在 struct/enum 上添加 trait 实现
// 2. attribute  - 修改被标注的项
// 3. function-like - 像函数调用 my_macro!()

// Cargo.toml:
// [package]
// name = "my_macros"
// 
// [lib]
// proc-macro = true   # 必需

// 依赖：
// [dependencies]
// syn = "2.0"    # 解析
// quote = "1.0"  # 生成
// proc-macro2 = "1.0"

// 最小 derive 例子：
use proc_macro::TokenStream;
use quote::quote;
use syn::{parse_macro_input, DeriveInput};

#[proc_macro_derive(HelloMacro)]
pub fn hello_macro_derive(input: TokenStream) -> TokenStream {
    // 1. 解析输入
    let ast = parse_macro_input!(input as DeriveInput);
    
    // 2. 获取名字
    let name = &ast.ident;
    
    // 3. 生成代码
    let expanded = quote! {
        impl HelloMacro for #name {
            fn hello() {
                println!("Hello from {}!", stringify!(#name));
            }
        }
    };
    
    // 4. 返回
    expanded.into()
}

// 使用方：
// [dependencies]
// my_macros = { path = "../my_macros" }

#[derive(HelloMacro)]
struct Pancakes;

fn main() {
    Pancakes::hello();  // Hello from Pancakes!
}

06 Derive 宏

// 内置 derive：
//   Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord, Hash
//   Default

// 自定义 derive 用 #[derive(MyMacro)]

// 带 helper 属性：
#[proc_macro_derive(HelloMacro, attributes(hello))]
pub fn hello_macro_derive(input: TokenStream) -> TokenStream {
    // 可以读取 #[hello] 属性
}

// 使用：
#[derive(HelloMacro)]
#[hello(name = "My Type")]
struct MyType;

// 著名 derive 库：
//   serde::Serialize, Deserialize
//   thiserror::Error
//   clap::Parser
//   tokio::main（实际是 attribute，不是 derive）
//   diesel::Queryable

// serde::Serialize 实现（简化）：
#[proc_macro_derive(Serialize)]
pub fn derive_serialize(input: TokenStream) -> TokenStream {
    let ast = parse_macro_input!(input as DeriveInput);
    
    match &ast.data {
        syn::Data::Struct(s) => derive_struct(&ast, s),
        syn::Data::Enum(e) => derive_enum(&ast, e),
        syn::Data::Union(_) => panic!("unions not supported"),
    }
}

fn derive_struct(ast: &DeriveInput, s: &syn::DataStruct) -> TokenStream {
    let name = &ast.ident;
    let fields = s.fields.iter().map(|f| {
        let fname = f.ident.as_ref().unwrap();
        quote! {
            out.serialize_field(stringify!(#fname), &self.#fname)?;
        }
    });
    
    (quote! {
        impl Serialize for #name {
            fn serialize<S>(&self, ser: S) -> Result<S::Ok, S::Error>
                where S: Serializer
            {
                use serde::ser::SerializeStruct;
                let mut out = ser.serialize_struct(stringify!(#name), 3)?;
                #(#fields)*
                out.end()
            }
        }
    }).into()
}

07 Attribute 宏

// Attribute 宏：修改或替换被标注的代码

#[proc_macro_attribute]
pub fn log_calls(attr: TokenStream, item: TokenStream) -> TokenStream {
    // attr: 属性参数 #[log_calls(name = "MyFn")]
    // item: 被标注的函数
    
    let mut input = parse_macro_input!(item as syn::ItemFn);
    let name = &input.sig.ident;
    let block = &input.block;
    
    let expanded = quote! {
        fn #name() {
            println!("calling {}", stringify!(#name));
            #block
            println!("called {}", stringify!(#name));
        }
    };
    
    expanded.into()
}

// 使用：
#[log_calls]
fn my_function() {
    println!("hello");
}

// 著名 attribute 宏：
//   #[tokio::main]
//   #[get("/path")]          (rocket, axum)
//   #[pyo3::pyfunction]      (pyo3)
//   #[wasm_bindgen]          (wasm-bindgen)
//   #[test]
//   #[cfg(...)]

// tokio::main 例子：
#[tokio::main]
async fn main() {
    println!("hello");
}

// 展开为：
fn main() {
    tokio::runtime::Builder::new_multi_thread()
        .enable_all()
        .build()
        .unwrap()
        .block_on(async {
            println!("hello");
        });
}

08 Function-like 宏

// Function-like 宏：像函数调用，但参数是任意 token

#[proc_macro]
pub fn sql(input: TokenStream) -> TokenStream {
    // 解析 SQL，编译期检查
    let query = parse_sql(input);
    let expanded = generate_query_code(query);
    expanded.into()
}

// 使用：
let users = sql!("SELECT * FROM users WHERE id = $1");

// 著名 function-like 宏：
//   println!, format!, eprintln!, write!
//   vec!, vec_of_strings!
//   concat!, env!, include!, include_str!
//   sqlx::query!
//   tracing::instrument! (实际是 attribute)

// sqlx::query! 实现（简化）：
#[proc_macro]
pub fn query(input: TokenStream) -> TokenStream {
    // 1. 解析 SQL 字符串
    let sql = parse_to_string(input);
    
    // 2. 连接数据库（编译期！）
    let conn = connect_to_db();
    
    // 3. PREPARE 查询，获取 schema
    let prepared = conn.prepare(&sql);
    
    // 4. 生成强类型代码
    let expanded = quote! {
        sqlx::query_as!(#sql)
            // 强类型 row
    };
    
    expanded.into()
}
// 编译期就能发现 SQL 错误

// include_str! 实现：
// 编译期读文件内容，嵌入为 'static str
const SOURCE: &'static str = include_str!("input.txt");

// env!：读环境变量（编译期）
const VERSION: &'static str = env!("CARGO_PKG_VERSION");

编写过程宏的"圣三位一体"：

// 常用 syn 类型：
//   DeriveInput       - struct/enum 用于 derive
//   ItemFn            - 函数
//   ItemStruct        - 结构体
//   ItemEnum          - 枚举
//   Fields, Field     - 字段
//   Type, Path        - 类型
//   Attribute         - 属性
//   Lit               - 字面量

// quote! 宏：构造 TokenStream
let name = format_ident!("Foo");
let expanded = quote! {
    struct #name {
        x: i32,
    }
    
    impl #name {
        fn new() -> Self { Self { x: 0 } }
    }
};

// 常用 quote 操作：
//   #var            - 插入变量
//   #(#field)*      - 重复（类似 macro_rules）
//   #(#field,)*     - 带分隔符的重复
//   stringify!(...) - 转字符串

// 高级 quote：
let fields = vec!["a", "b", "c"];
let field_idents: Vec<_> = fields.iter().map(|f| format_ident!("{}", f)).collect();
let expanded = quote! {
    struct Foo {
        #(#field_idents: i32,)*
    }
};

10 最佳实践

// 1. 优先用函数，必要时才用宏
//   - 宏调试困难
//   - 增加编译时间
//   - 错误信息可能晦涩

// 2. macro_rules! 优先于 proc-macro
//   - 更快编译
//   - 不需要额外 crate
//   - 适用：简单文本替换

// 3. proc-macro 必须在独立 crate
//   - 项目结构：
//     my_lib/             # 用户用的 crate
//     my_lib_derive/      # proc-macro crate
//     my_lib_macros/      # 内部 helper

// 4. 错误信息
use proc_macro_error::{abort, proc_macro_error};

#[proc_macro_derive(MyMacro)]
#[proc_macro_error]
pub fn my_macro_derive(input: TokenStream) -> TokenStream {
    let ast = parse_macro_input!(input as DeriveInput);
    
    if !is_supported(&ast) {
        abort!(ast.ident, "not supported");
    }
    // ...
}

// 5. 提供 span 信息（错误定位）
use quote::ToTokens;
let name_str = ast.ident.to_token_stream();
let expanded = quote_spanned! {name_str.span() =>
    // 错误会指向原始位置
};

// 6. 文档化
/// My awesome macro.
/// 
/// # Example
/// 
/// ```
/// use my_macros::MyMacro;
/// 
/// #[derive(MyMacro)]
/// struct Foo;
/// ```
#[proc_macro_derive(MyMacro)]
pub fn ...

常见陷阱：

// 1. 宏展开顺序
//   - 不保证多次展开顺序
//   - 依赖副作用：危险

// 2. 递归深度
//   默认 64，超出错误
//   #![recursion_limit = "1024"] 调整

// 3. 跨 crate 的 hygiene
//   总是用 $crate 引用本 crate
//   避免名字冲突

// 4. 与 trait 方法的冲突
//   derive 可能添加 trait 方法
//   与用户实现冲突

// 5. 不应该用宏做的：
//   - 简单包装（用函数）
//   - 类型转换（用 From/Into）
//   - 配置（用 const fn）
宏系统总结:
1. 声明式宏：macro_rules!，pattern 匹配，简单场景
2. 过程宏：syn + quote，三种类型（derive/attribute/function-like）
3. 编译期代码生成：零运行时开销
4. 卫生性：避免名字捕获
5. 工具链：cargo expand、trybuild
Rust 宏比 C 的 #define 强百倍，是现代语言中最强的元编程系统之一。
