01 概述

io_uring 是 Jens Axboe 于 2019 年合入 Linux 5.1 的异步 I/O 框架。它解决了一个长期存在的痛点：传统系统调用（read/write/accept/connect/send/recv）都是同步的，AIO 又有诸多限制。

io_uring 的核心思想：内核和用户态共享一对环形队列（SQ + CQ），通过共享内存通信，避免系统调用和数据拷贝。

传统方式
// 每个 IO 都需要系统调用
// 上下文切换开销大
ssize_t r = read(fd, buf, len);
// 如果有 N 个 IO → N 次 syscall
// poll/epoll 在大量 IO 时也需多次 syscall
io_uring 方式
// 批量提交 N 个 IO
// 只需 1 次 syscall
struct io_uring_sqe sqes[N];
// 填充 sqes
io_uring_enter(ring_fd, N, 0, 0);
// 从 CQ 中回收结果

关键优势：

  • 零系统调用提交（SQPOLL 模式下完全无 syscall）
  • 批量处理：一次提交最多 4096 个请求
  • 操作丰富：read/write/send/recv/accept/connect/openat/statx/…
  • 自动热点路径优化：如果可以立刻完成，inline 完成，不入队

02 SQ / CQ 环形队列

核心数据结构是共享内存中的两个环形队列：

// include/uapi/linux/io_uring.h — 用户态可见结构

// 提交队列条目 (Submission Queue Entry)
struct io_uring_sqe {
    __u8    opcode;      // IORING_OP_READV / IORING_OP_WRITEV / …
    __u8    flags;
    __u16   ioprio;
    __s32   fd;          // 目标文件描述符
    __u64   off;         // 文件偏移
    __u64   addr;        // buffer 地址
    __u32   len;         // buffer 长度
    union {
        struct { __u32  buf_index;   // 注册 buffer 索引
                 __u32  buf_group; };
        struct { __u64  addr3; };
    };
    __u64   user_data;   // 用户自定义数据（用于匹配 CQE）
    union {
        struct { __u16  personality; };
        struct { __u8   splice_fd_in; };
    };
    // … 更多字段用于特定操作
};

// 完成队列条目 (Completion Queue Entry)
struct io_uring_cqe {
    __u64   user_data;   // 来自 SQE 的 user_data
    __s32   res;         // 结果（类似 read/write 的返回值）
    __u32   flags;       // 标记
};

共享内存的 Ring 结构：

// 内核维护的 ring 信息
struct io_rings {
    struct io_uring      cq;           // 完成队列
    struct io_uring      sq;           // 提交队列
    u32                  sq_ring_mask;
    u32                  cq_ring_mask;
    u32                  sq_ring_entries;
    u32                  cq_ring_entries;
};

// 用户态和内核通过 mmap 共享这片内存
// 用户态填充 SQE → 更新 SQ tail
// 内核消费 SQE → 用户态观察 SQ head
// 内核写入 CQE → 更新 CQ tail
// 用户态消费 CQE → 更新 CQ head

// 关键同步原语：
// smp_store_release() / smp_load_acquire()
// 保证顺序一致性，无需锁

03 提交路径

// 用户态调用 io_uring_enter()
// 系统调用入口
SYSCALL_DEFINE6(io_uring_enter, unsigned int, fd, u32, to_submit,
                u32, min_complete, u32, flags, const void __user *, argp,
                size_t, argsz)
{
    // 1. 通过 fd 找到 io_ring_ctx
    // 2. 通过 to_submit 解析 SQE
    // 3. 对每个 SQE：
    //    a. io_submit_sqe() → 初始化 io_kiocb
    //    b. __io_queue_sqe() → 尝试执行
    //    c. 如果不能立即完成 → 入队等待
    // 4. 如果 min_complete > 0 → 等待 CQE
    // 5. 返回完成的 CQE 数量
}

// 内部流程
io_submit_sqes(ctx, to_submit)
  └─ for each SQE:
       └─ io_init_req()          // 解析 SQE → 初始化 io_kiocb
       └─ io_submit_sqe()        // 提交请求
            └─ __io_queue_sqe()
                 ├─ io_issue_sqe()  // 尝试执行（hot path）
                 │    ├─ 如果可以立即完成 → 调用完成回调
                 │    └─ 如果需要等待 → io_queue_async_work()
                 └─ 如果成功 → 直接写入 CQE

关键数据结构 io_kiocb：

04 完成路径

// io_uring 中有多种完成方式：

// 1. 立即完成（hot path）
//    操作在同一个系统调用内完成（如 buffer 已在 page cache 中的 read）
//    → io_issue_sqe() 返回 IOU_OK
//    → io_put_req() → io_free_req()
//    → 写入 CQE

// 2. 任务工作（task_work）
//    操作需要等待（如 socket 数据未到）
//    → 操作注册等待回调
//    → 数据到达时通过 task_work 通知
//    → io_req_task_complete() 在原始进程上下文中完成
//    → 写入 CQE
//    这是最常见的方式，利用了内核的 task_work_add() 机制

// 3. IRQ 驱动的完成
//    块设备的 IO 完成时触发 IRQ
//    → IRQ handler → bio 完成 → kiocb_done()
//    → 调用 io_complete_rq()
//    → io_req_complete_post()
//    → 写入 CQE（可能 wakeup 等待的 io_uring_enter）

// CQE 写入
static void io_req_complete_post(struct io_kiocb *req)
{
    struct io_ring_ctx *ctx = req->ctx;

    // 1. 填充 CQE
    struct io_uring_cqe *cqe = io_get_cqe(ctx);
    cqe->user_data = req->user_data;
    cqe->res = req->result;
    cqe->flags = 0;

    // 2. smp_store_release 更新 CQ tail
    //    保证用户态读取时看到完整数据

    // 3. 如果有等待者 → wakeup
    if (ctx->submitter_task)
        wake_up_process(ctx->submitter_task);
}

完成路径的选择取决于操作类型和当前上下文：

完成方式延迟场景
Inline最小（~100ns）page cache 命中、socket buffer 有数据
task_work低（~1μs）大多数异步请求
IRQ中等块设备、网络设备的硬件中断
io-wq较高阻塞操作（O_DIRECT 未命中）

05 SQPOLL 模式

SQPOLL（Submission Queue Polling）是 io_uring 最突出的优化——内核中有一个专用内核线程持续轮询 SQ，用户态无需任何系统调用即可提交 IO。

// 设置 SQPOLL 模式
struct io_uring_params params;
params.flags |= IORING_SETUP_SQPOLL;
// 创建时自动启动一个内核线程
io_uring_queue_init(256, &ring, ¶ms);

// 用户态操作：
ring->sq.sqes[0].opcode = IORING_OP_READV;
ring->sq.sqes[0].fd = fd;
ring->sq.sqes[0].addr = (unsigned long)&iov;
ring->sq.sqes[0].len = 1;

// smp_store_release() 写入 SQ tail
// 不需要调用 io_uring_enter()！
// SQPOLL 线程自动发现新提交

// 从 CQ 回收结果
while (1) {
    if (ring->cq.head != ring->cq.tail) {
        // 有完成事件
        struct io_uring_cqe *cqe = &ring->cq.cqes[ring->cq.head & mask];
        // 处理结果
        ring->cq.head++;
    }
}

SQPOLL 线程的生命周期：

// io_uring/sqpoll.c
static int io_sq_thread(void *data)
{
    while (!kthread_should_stop()) {
        // 1. 检查 SQ tail 是否有新条目
        // 2. 如果有 → 消费并提交
        // 3. 如果没有 → 调用 schedule_timeout()
        // 4. 空闲超时后休眠
        // 5. 用户态写入 SQ tail → io_uring_enter(wake) 唤醒
        io_sq_thread_submit(ctx);
    }
}

三种 SQPOLL 模式：

标志行为最佳场景
默认SQPOLL 线程轮询，空闲休眠，唤醒有延迟通用
IORING_SETUP_SQPOLL + 保持忙碌持续轮询，CPU 100%极端低延迟
IORING_SETUP_SQ_AFFSQPOLL 线程绑定到指定 CPUCPU 隔离优化

06 IOPOLL 模式

IOPOLL（IO Polling）是块设备端的轮询模式：提交 IO 后，CPU 主动轮询硬件完成状态，而不是等待 IRQ。

// 设置
params.flags |= IORING_SETUP_IOPOLL;

// 对块设备生效（如 NVMe 支持 polling）
// NVMe 驱动中：
// 1. 提交 SQ 条目到硬件
// 2. 调用 blk_poll() → nvme_poll()
// 3. 直接读 NVMe 完成队列 MMIO 寄存器
// 4. 检查 CQ 是否有新完成

// 内核实现：
static int io_iopoll_check(struct io_ring_ctx *ctx,
                            long min_complete)
{
    while (min_complete) {
        // 轮询硬件完成队列
        for (poll = 0; poll < 4; poll++) {
            req = list_first_entry(&ctx->iopoll_list);
            if (req->iopoll_completed) {
                // 硬件已完成
                io_req_complete_post(req);
                min_complete--;
            }
        }
        // 没完成 → cpu_relax()
        cpu_relax();
    }
}

// IOPOLL + SQPOLL 可以组合使用
// 完全无中断的 IO 路径：从提交到完成都是轮询
IOPOLL 适用条件:
需要硬件支持 polling（NVMe 支持，大部分 SATA SSD 不支持）。
在 CPU 有空闲时能显著降低延迟（省去 IRQ 到调度到处理的延迟）。
CPU 繁忙时反而可能降低吞吐（浪费 CPU）。

07 注册资源

io_uring 允许预先注册文件描述符和 buffer，避免每次 IO 的重复开销。

// 注册 buffer
// io_uring/rsrc.c
struct io_ring_ctx {
    struct io_mapped_ubuf   **user_bufs; // 注册后的 buffer
    unsigned                 nr_user_bufs;
};

// 注册文件表
struct io_ring_ctx {
    struct file              **file_table; // 注册后的 file 指针
    unsigned                  nr_user_files;
};

// 注册流程
// IORING_REGISTER_BUFFERS:
//   1. 获取用户态 buffer 的物理页
//   2. 用 get_user_pages() 锁定页面（防止被 swap）
//   3. 记录 page 映射到 io_mapped_ubuf
//   4. 返回 buffer 索引
//
// IORING_REGISTER_FILES:
//   1. 获取用户态 fd 对应的 struct file
//   2. 递增引用计数
//   3. 存入 file_table 数组
//   4. 返回文件索引

// 在 SQE 中使用：
sqe->opcode = IORING_OP_READ_FIXED;  // 使用注册 buffer
sqe->fd = IORING_FILE_INDEX_FIXED;    // 使用注册 file
sqe->buf_index = buf_idx;              // buffer 索
注册类型IO 操作节省的开销
BUFFERSIORING_OP_READ_FIXED每次 IO 省去 get_user_pages() + page pin
FILES所有操作省去 fget() + fput()（原子操作 + 引用计数）

08 io-wq 异步线程池

当 IO 操作不能在内核快速路径中完成时（如 O_DIRECT 读未命中、lock 冲突），io_uring 将请求分流到 io-wq 线程池处理。

// io_uring/io-wq.c
struct io_wq {
    struct io_wq_hash        *hash;
    struct io_wq_node {
        struct io_wq_work    work;
        struct io_worker     *worker;
    };
    struct io_wq_worker_data {
        struct list_head     work_list;
        struct io_worker     *worker;
    };
};

// 工作分流流程
static int io_queue_async_work(struct io_kiocb *req)
{
    // 1. 将 req 包装为 io_wq_work
    // 2. 放入 io-wq 的工作队列
    // 3. io-wq 线程从队列取出执行
    // 4. 完成后调用完成回调（task_work）
    io_wq_enqueue(req->ctx->iou_wq, &req->work);
}

// io-wq 线程池：
// - 动态管理 worker 线程数量
// - worker 绑定到特定 NUMA 节点
// - 空闲 worker 超时后自动退出
// - 支持优先级队列
