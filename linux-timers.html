01 概述

Linux 时间子系统是内核最复杂的子系统之一，它由多个层次组成，每一层抽象下层硬件细节。

                 │

概念方向说明

clock_event_device触发事件可编程的定时器中断源

02 定时器轮 (Timer Wheel)

低精度定时器（timer_list）使用多级轮 (timer wheel)。它优化了大批量超时（如 TCP 重传）场景，不是精确触发。

// include/linux/timer_types.h
struct timer_list {
    struct hlist_node       entry;          // 链表节点
    unsigned long           expires;        // 超时时间（jiffies）
    void                  (*function)(struct timer_list *);
    u32                     flags;          // TIMER_* flags
#ifdef CONFIG_LOCKDEP
    struct lockdep_map      lockdep_map;
#endif
};

API 使用范式：

struct timer_list my_timer;

// 1. 初始化（旧 API setup_timer 已废弃）
timer_setup(&my_timer, my_callback, 0);

// 2. 设置超时和回调
static void my_callback(struct timer_list *t) {
    pr_info("Timer fired!\n");
}

// 3. 启动定时器
mod_timer(&my_timer, jiffies + msecs_to_jiffies(100));

// 4. 取消（可睡眠上下文中）
del_timer_sync(&my_timer);
// 或非阻塞版本
if (del_timer(&my_timer)) {
    // 已成功取消（callback 还没运行）
}

多级轮结构：

// kernel/time/timer.c

#define LVL_DEPTH   6                  // 6 级
#define LVL_BITS    6                  // 每级 6 位
#define LVL_SIZE    (1 << LVL_BITS)    // 64 个桶
#define LVL_CLKDIV  (1LL << (LVL_BITS * lvl))
#define WHEEL_SIZE  (LVL_SIZE * LVL_DEPTH)   // 384 桶

struct timer_base {
    raw_spinlock_t          lock;
    struct hlist_head       vectors[WHEEL_SIZE];
    unsigned long           clk;            // 当前 tick
    unsigned long           next_expiry;    // 下一个到期时间
    unsigned int            cpu;
    bool                    is_idle;
    bool                    must_forward_clk;
    // ...
};

// 6 级时间复杂度：
//   Lvl 0:   1 tick 内的定时器（≤ 1 jiffy）
//   Lvl 1:   2-64 jiffies
//   Lvl 2:   65-4096 jiffies (~40 ms - 4 s @250HZ)
//   Lvl 3:   4097-262144 jiffies
//   Lvl 4:   262145-16777216 jiffies
//   Lvl 5:   更长
// 级联 (cascade)：当 clk 进入下一级范围时，重新分桶
timer vs hrtimer 何时用:
timer_list：低精度（jiffies 级，~1-4ms），大批量超时，弱时序保证。
hrtimer：纳秒精度，红黑树组织，硬实时保证。
现代 Linux 已经把所有 timer 内部都用 hrtimer 模拟，但接口和精度不同。

03 高精度定时器 (hrtimer)

hrtimer (High Resolution Timer) 提供纳秒精度的定时器，使用红黑树组织所有挂起的定时器。

// include/linux/hrtimer_types.h
struct hrtimer {
    struct timerqueue_node      node;           // 红黑树节点
    ktime_t                     _softexpires;   // 软超时
    struct hrtimer_clock_base  *base;
    u8                          state;          // HRTIMER_STATE_*
    u8                          is_rel;         // 是否相对时间
    u8                          is_soft;        // 是否软中断上下文
    u8                          is_hard;        // 强制硬中断上下文
    enum hrtimer_restart      (*function)(struct hrtimer *);
};

使用范式：

struct hrtimer my_hr;

// 初始化
hrtimer_init(&my_hr, CLOCK_MONOTONIC, HRTIMER_MODE_REL);

// 设置回调
static enum hrtimer_restart my_callback(struct hrtimer *t) {
    // 返回 HRTIMER_RESTART 让定时器继续触发
    // 返回 HRTIMER_NORESTART 让定时器停止
    return HRTIMER_NORESTART;
}
my_hr.function = my_callback;

// 启动（1ms 后）
hrtimer_start(&my_hr, ms_to_ktime(1), HRTIMER_MODE_REL);

// 取消
hrtimer_cancel(&my_hr);

// 前向（重复触发的定时器）
hrtimer_forward(&my_hr, ktime_get(), ms_to_ktime(10));

hrtimer 基 (base) 的组织：

// 每个 CPU 有自己的 hrtimer_base 数组（按 clock type）
struct hrtimer_clock_base {
    struct hrtimer_cpu_base   *cpu_base;        // 所属 CPU base
    unsigned int               active;          // 是否有定时器
    ktime_t                    offset;          // 时间偏移
    enum hrtimer_base_type     index;
    struct timerqueue_head     head;            // 红黑树根
};

struct hrtimer_cpu_base {
    raw_spinlock_t             lock;
    unsigned int               cpu;
    unsigned int               active_bases;    // 哪些 base 有定时器
    unsigned int               clock_was_set_seq;
    unsigned int               hres_active:1,
                               hang_detected:1,
                               softirq_activated:1;
    ktime_t                    expires_next;    // 下一个到期时间
    struct hrtimer_clock_base  clock_base[HRTIMER_MAX_CLOCK_BASES];
    unsigned int               nr_retries;
    unsigned int               nr_hangs;
    // ...
};

// HRTIMER_MAX_CLOCK_BASES 通常是 4：HRTIMER_BASE_MONOTONIC、
// HRTIMER_BASE_REALTIME、HRTIMER_BASE_BOOTTIME、HRTIMER_BASE_TAI

触发流程：

// 1. 高精度 tick 设备到期，触发 IRQ
// 2. hrtimer_interrupt() 处理所有已到期定时器
//    - 红黑树最左节点为最近超时
//    - 遍历到期节点，调用 callback
//    - 重新编程硬件为下一个最近超时
// 3. 软中断版的 hrtimer：
//    - HRTIMER_MODE_SOFT 标记的 hrtimer 在软中断中执行
//    - 拆分硬/软中断，减少 IRQ 延迟

// hrtimer_run_queues() 在每个 tick 检查是否到期
// 在高精度模式下，硬件中断直接处理

04 时钟事件设备

clock_event_device 抽象可编程的中断源（如 LAPIC Timer、HPET、ARM Generic Timer）：

// include/linux/clockchips.h
struct clock_event_device {
    void                  (*event_handler)(struct clock_event_device *);
    int                   (*set_next_event)(unsigned long evt, struct clock_event_device *);
    int                   (*set_next_ktime)(ktime_t expires, struct clock_event_device *);
    enum clock_event_state   state;
    unsigned int             features;        // CLOCK_EVT_FEAT_*
    unsigned long            max_delta_ns;
    unsigned long            min_delta_ns;
    u32                      mult;
    u32                      shift;
    enum hrtimer_restart   (*mult_mult);     // unused
    ktime_t                  next_event;
    int                      retries;
    int                      set_state_periodic;
    int                      set_state_oneshot;
    int                      set_state_oneshot_stopped;
    int                      tick_resume;
    const char              *name;
    int                      rating;
    int                      irq;
    int                      bound_on;
    const struct cpumask    *cpumask;
    struct list_head         list;
    struct module           *owner;
} ____cacheline_aligned;

关键状态：

enum clock_event_state {
    CLOCK_EVT_STATE_DETACHED,   // 未使用
    CLOCK_EVT_STATE_SHUTDOWN,   // 已关闭
    CLOCK_EVT_STATE_PERIODIC,   // 周期模式（系统 tick）
    CLOCK_EVT_STATE_ONESHOT,    // 单发模式（hrtimer）
    CLOCK_EVT_STATE_ONESHOT_STOPPED,  // 单发但暂停
};

事件处理器的层次：

// tick_device 是对 clock_event_device 的封装
enum tick_device_mode {
    TICKDEV_MODE_PERIODIC,    // 周期 tick（低精度）
    TICKDEV_MODE_ONESHOT,     // 单发（高精度 / NOHZ）
};

struct tick_device {
    struct clock_event_device *evtdev;
    enum tick_device_mode      mode;
};

// 系统启动后：
// 1. 所有 tick_device 处于 PERIODIC 模式
// 2. 时间稳定后切换到 ONESHOT（高精度模式）
// 3. 在 ONESHOT 下每次重编程下一个事件

广播 (Broadcast) 机制：

// 问题：某些 CPU 进入深睡眠时停止本地定时器（如 C-states）
// 解决：用全局广播设备（如 HPET）代替，唤醒这些 CPU

// kernel/time/tick-broadcast.c
// tick_broadcast_setup_oneshot() 配置广播设备
// tick_handle_oneshot_broadcast() 在广播设备中断时
//   通知所有到期的 CPU

// 类型：
// - 周期性广播（旧）
// - 单发广播（NOHZ）
// - hrtimer 广播（高精度）

05 时间维护

Timekeeping 维护当前时间（wall time, monotonic, boottime 等），通过周期性读取 clocksource 累加 ns。

// kernel/time/timekeeping.c
static struct {
    seqcount_raw_spinlock_t   seq;
    struct timekeeper         tk;
    raw_spinlock_t            lock;
    bool                      ptp_reset_counter;
} tk_core ____cacheline_aligned;

// include/linux/timekeeper_internal.h
struct timekeeper {
    struct tk_read_base       tkr_mono;        // 单调时钟读取器
    struct tk_read_base       tkr_raw;         // 原始时钟读取器
    u64                       xtime_sec;       // 当前 wall time 秒
    unsigned long             xtime_nsec;      // 秒内 ns（按 shift 缩放）
    struct timespec64         wall_to_monotonic;  // wall → mono 偏移
    struct timespec64         offset_sec;
    ktime_t                   offs_real;       // mono → real 偏移
    ktime_t                   offs_boot;       // mono → boot 偏移
    ktime_t                   offs_tai;        // mono → TAI 偏移
    unsigned long             cycles_interval;  // NTP 周期
    unsigned long             xtime_interval;   // 每次 cycle 的 ns
    s64                       ntp_error;
    u32                       raw_interval;
    u32                       ntp_error_shift;
    u32                       cycle_last;
    unsigned long             flags;
    // ...
};

时间读取的快速路径：

// 用户态调用 clock_gettime(CLOCK_MONOTONIC, &ts)
// → ktime_get_ts64()
// → read_tkr(&tk->tkr_mono, ...) 
//   关键：使用 seqcount 防止读到中间状态
//   - seqcount_read_retry() 自旋等待写入完成
//   - readb() 读 cycle counter（如 rdtsc）
//   - 计算 delta = (cycle_now - cycle_last) & mask
//   - 转换为 ns：delta = cyc2ns(delta, mult, shift)
//   - 累加：now_ns = base + delta

// 关键的 ktime_get_* 函数家族：
ktime_get()                    // CLOCK_MONOTONIC（不可回退）
ktime_get_real()               // CLOCK_REALTIME（wall time）
ktime_get_boottime()           // CLOCK_BOOTTIME（含睡眠）
ktime_get_raw()                // CLOCK_MONOTONIC_RAW（无 NTP）
ktime_get_clocktai()           // CLOCK_TAI（国际原子时）
ktime_get_ns()                 // 仅 ns 值（高频使用）

NTP 调整：

// kernel/time/ntp.c
// 通过调整 mult（clocksource 到 ns 的转换系数）实现：
// - 长期偏快：降低 mult（让 ns 累加变慢）
// - 长期偏慢：升高 mult（让 ns 累加变快）
// - adjtimex() 系统调用让用户态（chronyd/ntpd）反馈

// 关键函数：timekeeping_advance() 在每个 tick 调用
//   - 累加 cycles → ns
//   - 更新 xtime_sec（满 1 秒进位）
//   - 应用 NTP 校正
//   - 检查是否需要插入/删除闰秒

06 Tick 管理 (NO_HZ)

Tick 是系统周期性中断（默认 HZ=250，每 4ms）。NO_HZ 优化减少不必要的 tick：

// kernel/time/tick-sched.h
struct tick_sched {
    struct hrtimer            sched_timer;       // hrtimer 模拟 tick
    u64                       last_tick;         // 上次 tick 时间
    u64                       next_tick;         // 下次 tick 时间
    unsigned long             check_clocks;      // ANNOYING标志
    unsigned int              inidle:1,          // 处于 idle
                              tick_stopped:1,    // tick 已停止
                              idle_active:1,     // 在 idle 中
                              do_timer_last:1,
                              nohz_enabled:1,
                              tick_programmed:1;
    ktime_t                   idle_entrytime;    // 进入 idle 时间
    ktime_t                   idle_waketime;     // 唤醒时间
    u64                       idle_sleeptime;    // 累计睡眠时间
    u64                       iowait_sleeptime;  // I/O 等待睡眠
    unsigned long             last_jiffies;
    unsigned long             next_jiffies;
    u32                       nohz_length;
    // ...
};

三种 NO_HZ 配置：

配置行为用途
无 NO_HZ每个 CPU 每 tick 都中断极旧系统
NO_HZ_IDLE空闲 CPU 停止 tick默认，省电
NO_HZ_FULL无任务 CPU 完全停止 tick实时 / HPC
// NO_HZ_IDLE 实现：
// 1. CPU 进入 idle
// 2. tick_nohz_idle_enter() 计算"下一次需要的 tick"
//    - 最近的 hrtimer 超时
//    - 最近的 RCU GP
// 3. 重编程 clock_event_device 为该时间
// 4. tick_stopped = 1
// 5. CPU 进入 idle（mwait/halt）
// 6. 唤醒后 tick_nohz_idle_exit() 恢复

// NO_HZ_FULL 实现：
// 1. 一个 CPU 上只有一个任务
// 2. tick_nohz_full_kick_cpu() 通知其他 CPU
// 3. 停止该 CPU 的所有 tick
// 4. 任务自行运行，无中断
// 5. 仍需处理：RCU 静默、定时器超时

// 系统级 timekeeping 用一个 housekeeping CPU：
//   - 该 CPU 始终保留 tick，维护 wall time
//   - 通过 isolcpus=domain nohz_full=2-N 隔离

07 时钟源

clocksource 抽象单调递增的计数器：

// include/linux/clocksource.h
struct clocksource {
    u64                   (*read)(struct clocksource *cs);
    u64                       mask;             // 计数器位宽 mask
    u32                       mult;             // cycle → ns 转换
    u32                       shift;            // 配合 mult 的位移
    u64                       max_idle_ns;      // 最长空闲
    u32                       maxadj;           // NTP 最大调整量
    u32                       uncert_fod;
    const char              *name;
    struct list_head         list;             // 全局链表
    int                      rating;           // 质量评分
    enum vclock_mode         vclock_mode;      // VDSO 模式
    enum clocksource_ids     id;               // 系统枚举
    // ... watchdog、suspend、resume ...
};

质量评分 (rating) 系统：

Rating 范围含义
1-99非常差，仅作 fallback
100-199基础（如 jiffies、PIT）
200-299可用（ACPI PM、HPET）
300-399快速且正确（如 HPET 64-bit）
400-499理想（TSC、ARM SysTimer）

常见时钟源对比：

// x86:
//   TSC (Time Stamp Counter) - rating 400, VDSO 可直接读
//   HPET (High Precision Event Timer) - rating 250
//   ACPI PM Timer - rating 200

// ARM:
//   arch_sys_counter (CNTVCT_EL0) - rating 400
//   各种 SoC 内部 timer

// Watchdog：
//   每秒检查一次，如果主时钟源偏移过大
//   自动降级并切换到次优时钟源
//   防止硬件故障导致时间错乱
