01 POSIX 概览

Zephyr 提供 POSIX 兼容层，允许使用 POSIX API 编写的应用代码迁移到 Zephyr。

POSIX 支持程度

API支持说明
pthread✅ 完整创建/退出/join/detach
pthread_mutex✅ 完整互斥锁（含属性）
pthread_cond✅ 完整条件变量
pthread_rwlock✅ 完整读写锁
semaphore✅ 完整POSIX 信号量
signal⚠️ 部分基础信号支持
mmap⚠️ 部分内存映射
timer✅ 完整POSIX 定时器
mqueue✅ 完整消息队列

02 pthread 线程

// POSIX pthread 在 Zephyr 上运行
#include <pthread.h>

// 创建线程
pthread_t thread;
pthread_attr_t attr;
pthread_attr_init(&attr);
pthread_attr_setstacksize(&attr, 2048);
pthread_attr_setschedpolicy(&attr, SCHED_RR);

struct sched_param param = { .sched_priority = 5 };
pthread_attr_setschedparam(&attr, &param);

pthread_create(&thread, &attr, thread_func, NULL);

// 等待线程结束
pthread_join(thread, NULL);

// 分离线程
pthread_detach(thread);

03 pthread_mutex

// POSIX mutex
pthread_mutex_t mutex;
pthread_mutex_init(&mutex, NULL);

pthread_mutex_lock(&mutex);
// 临界区
pthread_mutex_unlock(&mutex);

pthread_mutex_destroy(&mutex);

// 属性 mutex（递归锁等）
pthread_mutexattr_t attr;
pthread_mutexattr_init(&attr);
pthread_mutexattr_settype(&attr, PTHREAD_MUTEX_RECURSIVE);
pthread_mutex_init(&mutex, &attr);

04 pthread_cond

// POSIX 条件变量
pthread_cond_t cond;
pthread_cond_init(&cond, NULL);

// 等待条件
pthread_mutex_lock(&mutex);
while (!condition) {
    pthread_cond_wait(&cond, &mutex);
}
pthread_mutex_unlock(&mutex);

// 通知
pthread_cond_signal(&cond);    // 唤醒一个
pthread_cond_broadcast(&cond); // 唤醒所有

pthread_cond_destroy(&cond);

05 信号

// POSIX 信号（部分支持）
#include <signal.h>

// 注册信号处理
void handler(int sig) {
    printk("信号 %d 收到\n", sig);
}

signal(SIGUSR1, handler);

// 发送信号
raise(SIGUSR1);

// 注意：Zephyr 信号支持有限
// 不支持所有 POSIX 信号语义

06 POSIX 定时器

// POSIX 定时器
timer_t timerid;
struct sigevent sev;
struct itimerspec its;

sev.sigev_notify = SIGEV_THREAD;
sev.sigev_notify_function = timer_handler;

timer_create(CLOCK_REALTIME, &sev, &timerid);

its.it_value.tv_sec = 1;
its.it_interval.tv_sec = 1;
timer_settime(timerid, 0, &its, NULL);

// 回调
void timer_handler(union sigval val) {
    printk("定时器触发\n");
}

07 CMSIS-RTOS

CMSIS-RTOS 是 ARM 定义的 RTOS 抽象层，方便跨 RTOS 移植。

// CMSIS-RTOS v2（subsys/portability/cmsis_rtos_v2/）
// CONFIG_CMSIS_RTOS_V2=y

// CMSIS-RTOS API
osThreadId_t tid = osThreadNew(thread_func, NULL, &attr);
osDelay(1000);
osSignalSet(tid, 0x01);
osMutexAcquire(mutex, osWaitForever);
osSemaphoreAcquire(sem, osWaitForever);
