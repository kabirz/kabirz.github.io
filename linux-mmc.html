01 概述

MMC（MultiMediaCard）子系统统一管理 MMC/SD/eMMC/SDIO/UHS-II 卡片和存储设备。从嵌入式系统的 eMMC 存储到笔记本的 SD 卡，Linux MMC 框架覆盖了几乎所有闪存存储场景。

支持的存储类型
  • MMC：原始 MultiMediaCard
  • SD：Secure Digital（SDSC/SDHC/SDXC/SDUC）
  • eMMC：嵌入式 MMC（BGA 封装）
  • SDIO：SD I/O（WiFi/蓝牙/GPS 模块）
  • UHS-II：超高速 SD 接口
  • SD Express：NVMe over SD
速度等级
标准最大速率
MMC Legacy26 MHz
MMC HS52 MHz
MMC HS200200 MHz (SDR)
MMC HS400200 MHz (DDR)
SD UHS-I SDR104208 MB/s
SD UHS-II312 MB/s

07 编写驱动

SDHCI 主机驱动模板

#include <linux/mmc/host.h>
#include <linux/sdhci/pltfm.h>

static int my_sdhci_request(struct mmc_host *mmc,
                             struct mmc_request *mrq)
{
    struct sdhci_host *host = mmc_priv(mmc);

    // 检查命令类型
    if (mrq->cmd->flags & MMC_CMD_SPECIAL) {
        // 特殊命令处理
    }

    // 配置 DMA（如果支持）
    sdhci_prepare_data(host, mrq);

    // 发送命令
    sdhci_send_command(host, mrq->cmd);

    return 0;
}

static void my_sdhci_set_ios(struct mmc_host *mmc,
                               struct mmc_ios *ios)
{
    struct sdhci_host *host = mmc_priv(mmc);

    // 设置时钟频率
    sdhci_set_clock(host, ios->clock);

    // 设置总线宽度
    sdhci_set_bus_width(host, ios->bus_width);

    // 设置时序模式
    sdhci_set_timing(host, ios->timing);

    // 设置电源模式
    sdhci_set_power(host, ios->power_mode, ios->vdd);
}

static int my_sdhci_card_busy(struct mmc_host *mmc)
{
    struct sdhci_host *host = mmc_priv(mmc);
    u32 present_state = sdhci_readl(host, SDHCI_PRESENT_STATE);
    return !(present_state & SDHCI_DAT_0_SLAVE);
}

static const struct mmc_host_ops my_sdhci_ops = {
    .request     = my_sdhci_request,
    .set_ios     = my_sdhci_set_ios,
    .get_ro      = sdhci_get_ro,
    .get_cd      = sdhci_get_cd,
    .card_busy   = my_sdhci_card_busy,
    .start_signal_voltage_switch = sdhci_start_signal_voltage_switch,
    .execute_tuning = sdhci_execute_tuning,
};
