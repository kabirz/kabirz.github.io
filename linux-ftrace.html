01 概述

ftrace 是 Linux 内核最强大的跟踪框架，从 2008 年 (Linux 2.6.27) 开始合入，现已发展成覆盖函数跟踪、静态事件、动态探针的完整可观测性基础设施。

ftrace 能做什么
  • 跟踪任意内核函数的进入/返回
  • 通过静态 trace event 监控关键子系统
  • 动态插入 kprobe/uprobe 探针
  • 记录 IRQ/调度/内存等核心事件
  • 硬件延时分析（irqsoff、wakeup）
  • 函数调用图（function_graph）
关键特性
特性说明
零开销 (disabled)mcount 被 nop 填充，无性能影响
安全动态开启运行时即时修改函数入口
分层设计events / function / kprobe 三层可叠加
lockless ring bufferper-CPU 无锁缓冲，写入<50ns

用户接口：/sys/kernel/tracing/（tracefs），通过 echo 命令即可操作，不依赖任何用户态工具。

03 动态 ftrace

编译时，每个函数开头插入 5 字节的 mcount 调用（或 __fentry__）：

// gcc -pg 选项在每个函数开头插入：
__fentry__:
    ret    // 默认：直接返回（无操作）

// ftrace 启用后，改为：
__fentry__:
    jmp ftrace_caller  // 跳转到 ftrace 处理函数
    // 后续指令：保存寄存器、调用回调、恢复寄存器

运行时修改由 kernel/trace/ftrace.c 中的 DYNAMIC_FTRACE 机制完成：

// 关键数据结构
struct dyn_ftrace {
    unsigned long           ip;      // 函数入口地址
    unsigned long           flags;
    struct dyn_arch_ftrace  arch;
};

// 所有 mcount 调用点记录在 ftrace_pages 链表中
// 修改流程：
// 1. 找到目标函数对应的 dyn_ftrace record
// 2. stop_machine() 停止所有 CPU
// 3. 将 nop 替换为 jmp ftrace_caller
// 4. 恢复 CPU 运行
// 整个过程对运行中的系统完全透明

// arch/x86/kernel/ftrace.c
int ftrace_make_nop(struct module *mod,
                    struct dyn_ftrace *rec,
                    unsigned long addr)
{
    // 将 call mcount/fentry 替换为 5 字节 nop
    // (0xE8 call → 0x0F 0x1F nop)
}

int ftrace_make_call(struct dyn_ftrace *rec,
                     unsigned long addr)
{
    // 将 nop 替换为 call/jmp ftrace_caller
}
安全性:
ftrace 使用 stop_machine() 进行代码修改
确保修改时没有 CPU 正在执行被修改的指令
每次修改都是原子的（单条 jmp 指令）

04 ftrace_ops 与回调机制

// include/linux/ftrace.h
struct ftrace_ops {
    ftrace_func_t       func;            // 回调函数
    struct ftrace_ops   *next;           // 链表
    unsigned long       flags;           // TRACE_OPS_FL_*
    int                 priority;        // 优先级
    void                *private;

    // 过滤函数集合
    struct ftrace_hash  *notrace_hash;   // 黑名单
    struct ftrace_hash  *filter_hash;    // 白名单
};

// 回调函数签名
typedef void (*ftrace_func_t)(unsigned long ip,       // 函数地址
                              unsigned long parent_ip, // 调用者地址
                              struct ftrace_ops *op,   // ops 指针
                              struct ftrace_regs *fregs);

// 注册/注销 API
int register_ftrace_function(struct ftrace_ops *ops);
int unregister_ftrace_function(struct ftrace_ops *ops);

ftrace_ops 的链式调用机制：

// ftrace_caller (汇编入口)
// 1. 保存 regs
// 2. 进入 ftrace_call (C 函数)

void ftrace_call(unsigned long ip, unsigned long parent_ip)
{
    struct ftrace_ops *op;

    // 遍历 ops 链表，按 priority 从高到低
    // 每个 ops 检查自己的 filter_hash/notrace_hash
    // 匹配则调用 op->func()
    do_for_each_ftrace_op(op) {
        if (ftrace_ops_test(op, ip))
            op->func(ip, parent_ip, op, NULL);
    } while_for_each_ftrace_op(op);
}

// 多个 tracer 共存示例：
// function tracer: 每个函数都记录
// function_graph: 同时记录函数嵌套关系
// stack tracer: 记录调用栈
// 它们可以同时注册，按优先级依次调用
标志含义
TRACE_OPS_FL_GLOBAL全局追踪所有函数
TRACE_OPS_FL_DYNAMIC仅追踪 filter_hash 中的函数
TRACE_OPS_FL_RCU回调在 RCU 读锁内执行
TRACE_OPS_FL_PERMANENT不允许注销
TRACE_OPS_FL_IPMODIFY回调可能修改 IP（如 livepatch）

05 Trace Event 系统

Trace Event 是内核源码中静态插入的跟踪点，使用宏来声明和定义：

// 1. 创建事件类 (DECLARE_EVENT_CLASS)
//    定义事件的共同结构
DECLARE_EVENT_CLASS(sched_wakeup_template,
    TP_PROTO(struct task_struct *p),
    TP_ARGS(p),
    TP_STRUCT__entry(
        __field(pid_t, pid)
        __field(int, prio)
    ),
    TP_fast_assign(
        __entry->pid = p->pid;
        __entry->prio = p->prio;
    ),
    TP_printk("pid=%d prio=%d", __entry->pid, __entry->prio)
);

// 2. 定义具体事件 (DEFINE_EVENT)
DEFINE_EVENT(sched_wakeup_template,
             sched_wakeup_new,
             TP_PROTO(struct task_struct *p),
             TP_ARGS(p));

// 3. 在代码中触发
trace_sched_wakeup_new(p);
// 展开为：
// if (static_key_false(&tracepoint.__key))
//     tracepoint_sched_wakeup_new(p);

编译器优化：

  • 使用 static branch（static_key_false）：未启用时是单条 nop，零开销
  • 启用后改为 jmp，跳转到事件处理函数
  • 每个事件进入 ring buffer 前会检查 filter

主要事件簇位置：

分类文件内容
schedinclude/trace/events/sched.h调度：wakeup/migrate/switch
irqinclude/trace/events/irq.h中断：entry/exit/softirq
kmeminclude/trace/events/kmem.h内存：alloc/free/mm_page
blockinclude/trace/events/block.h块层：bio/rq/plug
netinclude/trace/events/net.h网络：napi/softirq
syscallsinclude/trace/events/syscalls.h系统调用 enter/exit

06 Ring Buffer

ftrace 使用自己实现的 lockless ring buffer（kernel/trace/ring_buffer.c），核心设计：

// 核心数据结构
struct ring_buffer {
    struct ring_buffer_per_cpu **buffers;  // 每个 CPU 一个
    unsigned             pages;
    unsigned             flags;
};

struct ring_buffer_per_cpu {
    struct buffer_page    *reader_page;      // 读取页
    struct buffer_page    *head_page;        // 最旧数据页
    struct buffer_page    *tail_page;        // 最新数据页
    struct buffer_page    *commit_page;      // 已提交页
    unsigned long         nr_pages;
    local_t               commit_overrun;    // 覆盖计数
};

struct buffer_page {
    struct list_head      list;              // 环形链表
    local_t               write;             // 写入偏移
    unsigned long         real_end;          // 实际结束位置
    struct page           *page;             // 真实物理页
    local_t               entries;           // 条目数
};

写入流程：

// 写入者（在任意上下文均可）
// 1. 从当前 CPU 的 tail_page 中预留空间
// 2. 拷贝事件数据到预留位置
// 3. 更新 commit_page（通过 smp_store_release）
// 4. 如果 tail_page 写满，切换到下一个 buffer_page

// 读取者
// 1. 交换 reader_page 和 head_page
// 2. 在 reader_page 上读取数据
// 3. 将 reader_page 放回链表尾部

// 关键特性：
// - 写入者和读取者无需锁
// - 写入者之间通过 local_t（本地原子操作）互斥
// - 通过 smp_store_release + smp_load_acquire 实现无锁读取
// - 写入者可以被抢占，但不影响一致性
性能数据:
单次写入 ring buffer：~30-50ns（无事件时）
加上 trace event 的 TP_fast_assign：~80-150ns
远低于 printk 的 ~1-10μs

07 tracefs 接口

所有用户态操作通过 /sys/kernel/tracing/ 下的文件完成：

文件用途
available_tracers列出可用 tracer：function、function_graph、irqsoff、wakeup…
current_tracer选择当前 tracer：echo function > current_tracer
available_events列出所有可用事件（数千个）
set_event启用事件：echo sched:* > set_event
trace读取跟踪结果
trace_pipe流式读取（blocking）
buffer_size_kb设置 per-CPU 缓冲大小
ftrace_filter函数过滤器（白名单）
ftrace_notrace函数过滤器（黑名单）
kprobe_events动态 kprobe 定义
uprobe_events动态 uprobe 定义
events/事件分类目录（sched/irq/kmem/…）
// 典型使用流程
# 挂载 tracefs
mount -t tracefs none /sys/kernel/tracing

# 查看可用 tracer
cat available_tracers
# function function_graph irqsoff wakeup ...

# 启用函数跟踪
echo function > current_tracer

# 只跟踪特定函数
echo 'schedule' > set_ftrace_filter
echo '__do_fault' >> set_ftrace_filter

# 查看结果
cat trace
# <...>-1139  [001] ...1.  123.456: schedule <= do_nanosleep

08 kprobe/uprobe 集成

ftrace 与 kprobe 深度集成，使得 kprobe 事件也可以通过 trace event 接口管理：

// kprobe 事件定义（通过 tracefs 动态添加）
# 在函数入口处插入 kprobe
echo 'p:my_probe do_sys_open file=%di:u32 flags=%si:u32' \
     > kprobe_events

# 在函数返回处插入 kretprobe
echo 'r:my_ret do_sys_open ret=$retval:s32' \
     > kprobe_events

# 启用
echo 1 > events/kprobes/my_probe/enable

# 底层实现：
// kernel/trace/trace_kprobe.c
// kprobe 注册到 ftrace（如果目标函数已被 ftrace 跟踪）
// → 利用 ftrace 的 mcount 框架，避免单独插入 breakpoint
// → 性能远优于传统 kprobe（~500ns vs ~1μs）

// uprobe 事件定义
echo 'p:my_uprobe /bin/bash:0x12345' > uprobe_events
// 在 bash 的 0x12345 偏移处插入用户态探针
// 通过 UPROBE_EVENT 送往 trace event 系统

三种方式可以同时使用，统一输出到 ring buffer：

# 同时启用三种跟踪
echo function > current_tracer
echo 1 > events/sched/sched_switch/enable
echo 'p:my_kprobe schedule' > kprobe_events

# 结果混合在 trace 中
cat trace
# 函数跟踪  ...
# sched_switch 事件 ...
# kprobe 事件 ...
