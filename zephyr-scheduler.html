01 调度器概览

Zephyr 调度器是内核的核心组件，负责决定哪个线程获得 CPU 执行权。它支持优先级抢占式调度、时间片轮转和截止期限（EDF）调度三种模式，可配置为单处理器或多处理器（SMP）运行。

02 就绪队列

Zephyr 使用两级优先级队列管理就绪线程：外层按优先级索引，内层按具体优先级排序。

数据结构

// 内核全局调度数据（kernel/include/ksched.h）
struct z_kernel {
    // 单核模式：全局就绪队列
    struct {
        _priq_run runq;      // 就绪线程优先级队列
    } ready_q;

#ifdef CONFIG_SMP
    // SMP 模式：每 CPU 一个就绪队列
    struct {
        _priq_run runq;
    } cpus[CONFIG_MP_MAX_NUM_CPUS];
#endif

    struct k_thread *current;  // 当前线程
    struct k_thread *idle;     // 空闲线程
};

// 优先级队列节点
struct _priq_rb {
    struct rbtree tree;       // 红黑树（按优先级排序）
    int min_pri;              // 最高优先级缓存
    int max_pri;              // 最低优先级缓存
};

就绪队列工作流程

// 就绪队列操作（kernel/priority_queues.c）

// 添加线程到就绪队列
void z_priq_run_add(_priq_run *pq, struct k_thread *thread)
{
    RB_TREE_INSERT(&pq->tree, &thread->base.qnode_rb);
    // 更新最小优先级缓存
    if (thread->base.prio < pq->min_pri) {
        pq->min_pri = thread->base.prio;
    }
}

// 从就绪队列移除线程
void z_priq_run_remove(_priq_run *pq, struct k_thread *thread)
{
    RB_TREE_REMOVE(&pq->tree, &thread->base.qnode_rb);
    // 重新计算最小优先级
    struct rbnode *n = rbtree_get_min(&pq->tree);
    pq->min_pri = n ? CONTAINER_OF(n, struct k_thread, base.qnode_rb)->base.prio
                    : CONFIG_NUM_PREEMPT_PRIORITIES;
}

// 获取最高优先级就绪线程
struct k_thread *z_priq_run_best(_priq_run *pq)
{
    struct rbnode *n = rbtree_get_min(&pq->tree);
    return n ? CONTAINER_OF(n, struct k_thread, base.qnode_rb) : NULL;
}

红黑树 + 位图加速

Zephyr 使用红黑树按优先级排序就绪线程，同时维护一个位图来快速找到最高优先级非空队列：

// 位图加速（用于快速查找最高优先级）
// 每个 bit 对应一个优先级，bit=1 表示该优先级有就绪线程
// 使用 __builtin_clz() 一条指令找到最高优先级

static inline int runq_best_prio(_priq_run *pq)
{
    // 红黑树 O(log n) 查找
    struct rbnode *n = rbtree_get_min(&pq->tree);
    return n ? CONTAINER_OF(n, struct k_thread, base.qnode_rb)->base.prio : -1;
}

03 优先级调度

Zephyr 的核心调度策略是严格优先级抢占：更高优先级的线程总是优先执行。

调度决策流程

// 调度决策核心函数（kernel/sched.c）
static ALWAYS_INLINE struct k_thread *next_up(void)
{
    // 1. 处理当前线程挂起/中止（SMP）
#ifdef CONFIG_SMP
    if (z_is_thread_halting(_current)) {
        halt_thread(_current, ...);
    }
#endif

    // 2. 从就绪队列选择最高优先级线程
    struct k_thread *thread = runq_best();

    // 3. 元 IRQ 抢占恢复
    thread = metairq_preempt_recover(thread);

#ifndef CONFIG_SMP
    // 4. 单核：如果当前线程仍在队列中，不切换
    if (thread == _current) {
        return thread;
    }
#endif

    return thread;
}

抢占时机

事件抢占行为
高优先级线程就绪立即抢占当前线程
低优先级线程释放锁唤醒等待的高优先级线程并抢占
ISR 返回检查是否有更高优先级线程就绪
时间片到期同优先级线程轮转（非抢占）
调度锁释放重新评估调度决策

04 时间片轮转

当多个相同优先级的抢占线程同时就绪时，Zephyr 使用时间片轮转（Round-Robin）在它们之间切换。

时间片实现

// 时间片管理（kernel/timeslicing.c）

// 时间片到期回调
static void timeslice_expiration(void *handle)
{
    struct k_thread *thread = handle;

    // 1. 取消当前时间片定时器
    z_abort_timeout(&thread->base.timeout);

    // 2. 标记时间片到期
    thread->base.slice_expired = true;

    // 3. 触发重新调度
    z_reschedule(&lock, key);
}

// 设置时间片（CONFIG_TIMESLICING）
void z_thread_timeout(struct k_thread *thread)
{
    // 线程被唤醒时启动时间片定时器
    if (CONFIG_TIMESLICING > 0) {
        z_reset_timeout(&thread->base.timeout,
                        timeslice_expiration,
                        thread,
                        K_TICKS(CONFIG_TIMESLICING));
    }
}

时间片参数

Kconfig 选项默认值说明
CONFIG_TIMESLICINGy启用时间片轮转
CONFIG_TIMESLICE_SIZE0时间片大小（tick 数），0 表示等于系统 tick 周期
CONFIG_TIMESLICE_PRIORITY0启用时间片的最高优先级

时间片工作流程

// 时间片轮转决策
static ALWAYS_INLINE void update_cache(int preempt_ok)
{
    struct k_thread *thread = _current;

    // 如果时间片到期
    if (thread->base.slice_expired) {
        thread->base.slice_expired = false;

        // 重置时间片定时器
        z_reset_timeout(&thread->base.timeout,
                        timeslice_expiration,
                        thread,
                        K_TICKS(CONFIG_TIMESLICING));

        // 移到同优先级队列尾部
        z_priq_run_remove(&_kernel.ready_q.runq, thread);
        z_priq_run_add(&_kernel.ready_q.runq, thread);

        // 标记需要重新调度
        z_set_reschedule_pending();
    }
}

05 EDF 截止期限调度

Zephyr 支持最早截止期限优先（Earliest Deadline First, EDF）调度，适用于实时控制系统。

EDF 原理

EDF 调度器为每个周期性任务维护三个参数：周期（period）、截止期限（deadline）和执行预算（budget）。调度器总是选择截止期限最近的任务执行。

// EDF 任务参数
struct k_threadDeadline {
    uint32_t deadline;    // 绝对截止期限
    uint32_t period;      // 周期
    uint32_t budget;      // 执行预算
    uint32_t consumed;    // 已消耗预算
};

// EDF 调度决策
// 在 deadline.c 中实现
// 当启用 CONFIG_DEADLINE_SCHEDULING 时：
// 1. 从 EDF 队列选择 deadline 最小的线程
// 2. 与优先级队列结果比较
// 3. 选择更紧急的线程执行

EDF vs 优先级调度

维度优先级调度EDF 调度
策略静态优先级动态截止期限

06 调度锁

调度锁（k_sched_lock()）禁止当前线程被同优先级或低优先级线程抢占，但高优先级线程仍可抢占。

// 调度锁实现（kernel/scheduler.c）
void k_sched_lock(void)
{
    K_SPINLOCK(&_sched_spinlock) {
        __ASSERT(!arch_is_in_isr(), "不能在 ISR 中加锁");
        __ASSERT(_current->base.sched_locked != 1U,
                 "已经锁定");

        // sched_locked 计数器：0=解锁，负数=锁定
        --_current->base.sched_locked;

        compiler_barrier();
    }
}

void k_sched_unlock(void)
{
    k_spinlock_key_t key = k_spin_lock(&_sched_spinlock);

    __ASSERT(_current->base.sched_locked != 0U, "未锁定");
    __ASSERT(!arch_is_in_isr(), "不能在 ISR 中解锁");

    ++_current->base.sched_locked;

    // 解锁后检查是否需要重新调度
    z_sched_lock_reschedule(key);
}

调度锁 vs 中断锁

特性k_sched_lock()irq_lock()
中断响应仍响应中断屏蔽所有中断
抢占禁止同/低优先级抢占禁止所有抢占
延迟极低（纳秒级）极低（纳秒级）
适用场景保护临界区（短）保护硬件临界区

07 SMP 调度

Zephyr SMP 支持在多核处理器上运行，每个 CPU 维护独立的就绪队列。

SMP 调度架构

// SMP 模式下的调度（kernel/smp/）

// 1. 每 CPU 独立就绪队列
_kernel.cpus[cpu].ready_q.runq;

// 2. 线程亲和性
thread->base.cpu_mask = BIT(cpu);  // 绑定到特定 CPU

// 3. IPI（处理器间中断）
// 当线程状态改变时，发送 IPI 通知其他 CPU
void z_sched_ipi(void)
{
    // 检查本 CPU 是否需要重新调度
    z_reschedule_unlocked();
}

// 4. 负载均衡
// 当 CPU 空闲时，从其他 CPU 偷取线程
static struct k_thread *thread_expired(void)
{
    // 从其他 CPU 的就绪队列偷取最高优先级线程
}

SMP 配置选项

Kconfig说明
CONFIG_SMP启用 SMP 支持
CONFIG_MP_MAX_NUM_CPUS最大 CPU 数量
CONFIG_SCHED_CPU_MASK_PIN_ONLY仅允许线程绑定到特定 CPU
