01 同步原语概览

Zephyr 提供丰富的同步原语，从轻量级自旋锁到高级互斥锁，满足不同并发场景的需求。

原语核心文件等待方式适用场景
k_mutexmutex.c阻塞保护共享资源（支持优先级继承）
k_semsem.c阻塞资源计数、任务通知
k_eventevents.c阻塞位掩码事件等待
k_condvarcondvar.c阻塞条件等待（与 mutex 配合）
k_futexmutex.c阻塞用户态原子操作等待
k_spinlockspinlock.c自旋极短临界区（中断上下文）

02 互斥锁 (k_mutex)

Zephyr 互斥锁实现了优先级继承算法，防止优先级反转问题。

数据结构

// include/zephyr/kernel.h
struct k_mutex {
    struct k_wait_q wait_q;    // 等待队列
    struct k_thread *owner;    // 当前持有者线程
    int lock_count;            // 锁计数（支持递归）
    int owner_orig_prio;       // 持有者原始优先级（优先级继承）
};

优先级继承

// kernel/mutex.c

// 加锁（核心逻辑）
int z_impl_k_mutex_lock(struct k_mutex *mutex, k_timeout_t timeout)
{
    k_spinlock_key_t key = k_spin_lock(&lock);

    // 1. 无持有者，直接获取
    if (mutex->owner == NULL) {
        mutex->owner = _current;
        mutex->lock_count = 1;
        mutex->owner_orig_prio = _current->base.prio;
        k_spin_unlock(&lock, key);
        return 0;
    }

    // 2. 同一线程递归加锁
    if (mutex->owner == _current) {
        mutex->lock_count++;
        k_spin_unlock(&lock, key);
        return 0;
    }

    // 3. 优先级继承：提升持有者优先级
    if (_current->base.prio < mutex->owner->base.prio) {
        // 记录原始优先级
        mutex->owner_orig_prio = mutex->owner->base.prio;
        // 提升到等待者的优先级
        z_thread_priority_inherit(mutex->owner, _current->base.prio);
    }

    // 4. 当前线程加入等待队列
    int ret = z_pend_curr(&lock, key, &mutex->wait_q, timeout);
    return ret;
}

// 解锁
int z_impl_k_mutex_unlock(struct k_mutex *mutex)
{
    k_spinlock_key_t key = k_spin_lock(&lock);

    // 1. 递归锁计数减一
    mutex->lock_count--;
    if (mutex->lock_count > 0) {
        k_spin_unlock(&lock, key);
        return 0;
    }

    // 2. 优先级恢复
    z_thread_priority_uninherit(mutex->owner, mutex->owner_orig_prio);

    // 3. 唤醒最高优先级等待者
    struct k_thread *pending = z_waitq_head(&mutex->wait_q);
    if (pending != NULL) {
        mutex->owner = pending;
        mutex->lock_count = 1;
        mutex->owner_orig_prio = pending->base.prio;
        z_thread_return_value(pending, 0);
        z_unready_thread(pending);
    } else {
        mutex->owner = NULL;
    }

    k_spin_unlock(&lock, key);
    return 0;
}

03 信号量 (k_sem)

信号量是计数同步原语，用于资源管理和任务通知。

数据结构

// include/zephyr/kernel.h
struct k_sem {
    struct k_wait_q wait_q;    // 等待队列
    atomic_t count;            // 计数器
    atomic_t limit;            // 最大计数值
};

实现

// kernel/sem.c

// 初始化
void z_impl_k_sem_init(struct k_sem *sem, unsigned int initial_count,
                       unsigned int limit)
{
    __ASSERT_NO_MSG(initial_count <= limit);
    sem->count = ATOMIC_INIT(initial_count);
    sem->limit = ATOMIC_INIT(limit);
    z_waitq_init(&sem->wait_q);
}

// 释放（give）
int z_impl_k_sem_give(struct k_sem *sem)
{
    k_spinlock_key_t key = k_spin_lock(&lock);

    // 检查是否有等待者
    struct k_thread *pending = z_waitq_head(&sem->wait_q);
    if (pending != NULL) {
        // 有等待者，直接唤醒（不增加计数）
        z_thread_return_value(pending, 0);
        z_unready_thread(pending);
    } else {
        // 无等待者，增加计数
        int count = atomic_get(&sem->count);
        if (count < atomic_get(&sem->limit)) {
            atomic_set(&sem->count, count + 1);
        }
    }

    k_spin_unlock(&lock, key);
    return 0;
}

// 获取（take）
int z_impl_k_sem_take(struct k_sem *sem, k_timeout_t timeout)
{
    k_spinlock_key_t key = k_spin_lock(&lock);

    // 尝试获取
    int count = atomic_get(&sem->count);
    if (count > 0) {
        atomic_set(&sem->count, count - 1);
        k_spin_unlock(&lock, key);
        return 0;
    }

    // 无法获取，等待
    int ret = z_pend_curr(&lock, key, &sem->wait_q, timeout);
    return ret;
}

典型用法

// 资源计数（例如：3 个可用缓冲区）
K_SEM_DEFINE(buf_sem, 3, 3);  // 初始3，最大3

// 生产者
void producer(void)
{
    k_sem_give(&buf_sem);  // 释放资源
}

// 消费者

{
    k_sem_take(&buf_sem, K_FOREVER);  // 等待资源
    // 使用缓冲区...

}

04 事件 (k_event)

事件对象用于线程间基于位掩码的事件通知，支持等待任意或所有事件。

// 事件定义
K_EVENT_DEFINE(my_event);

// 发布事件（设置位）
k_event_post(&my_event, BIT(0) | BIT(3));

// 等待任意事件
struct k_event_wait ev = {
    .events = BIT(0) | BIT(1),
    .options = K_EVENT_WAIT_ANY,
};
int ret = k_event_wait(&my_event, &ev, 1, K_FOREVER);
// ret = 匹配的事件位

// 等待所有事件
struct k_event_wait ev = {
    .events = BIT(0) | BIT(1),
    .options = K_EVENT_WAIT_ALL,
};
int ret = k_event_wait(&my_event, &ev, 1, K_FOREVER);

// 带选项等待
// K_EVENT_WAIT_RESET  - 等待前重置事件
// K_EVENT_WAIT_CLEAR  - 匹配后清除事件位

内部实现

// kernel/events.c

// 等待条件检查
// K_EVENT_WAIT_ANY: 匹配任一事件位
// K_EVENT_WAIT_ALL: 匹配所有事件位

static int check_events(struct k_event *event, struct k_event_wait *wait)
{
    uint32_t matched = event->events & wait->events;

    if (wait->options & K_EVENT_WAIT_ALL) {
        // 必须匹配所有位
        return (matched == wait->events) ? matched : 0;
    } else {
        // 匹配任一位即可
        return matched;
    }
}

05 条件变量 (k_condvar)

条件变量用于线程等待特定条件成立，必须与互斥锁配合使用。

// 条件变量定义
K_CONDVAR_DEFINE(my_condvar);

// 等待条件
K_MUTEX_DEFINE(my_mutex);

void consumer(void)
{
    k_mutex_lock(&my_mutex, K_FOREVER);
    while (!condition_met) {
        // 原子地释放 mutex 并等待条件变量
        k_condvar_wait(&my_condvar, &my_mutex, K_FOREVER);
    }
    // 条件满足，继续执行（mutex 已重新获取）
    k_mutex_unlock(&my_mutex);
}

// 通知条件
void producer(void)
{
    k_mutex_lock(&my_mutex, K_FOREVER);
    condition_met = true;
    k_mutex_broadcast(&my_condvar);  // 唤醒所有等待者
    // 或 k_condvar_signal(&my_condvar);  唤醒一个
    k_mutex_unlock(&my_mutex);
}

06 Futex

Futex（Fast Userspace Mutex）允许用户态线程在无竞争时避免系统调用。

// Futex 定义
K_FUTEX_DEFINE(my_futex);

// 等待（原子操作 + 系统调用）
// 如果 futex 值为 0，线程阻塞
// 如果 futex 值非 0，立即返回
int ret = k_futex_wait(&my_futex, 0, K_FOREVER);

// 释放（原子操作 + 系统调用）
// 设置 futex 值为 1，唤醒等待线程
k_futex_wake(&my_futex, 1);

无竞争快速路径

// Futex 的核心优势：无竞争时无需内核介入
// 1. 用户态检查 atomic 值
// 2. 如果值非 0，直接获取（无系统调用）
// 3. 如果值为 0，调用系统调用阻塞

// 适用于：
// - 用户态 pthread_mutex 实现
// - 高性能锁（大部分情况无竞争）

07 自旋锁 (k_spinlock)

自旋锁是最轻量级的同步原语，在中断上下文和极短临界区中使用。

// 自旋锁定义
static struct k_spinlock my_lock;

// 使用（中断上下文）
k_spinlock_key_t key = k_spin_lock(&my_lock);

// 临界区（极短操作）
shared_var++;

k_spin_unlock(&my_lock, key);

// 注意：自旋锁会禁止抢占（和中断，取决于架构）
// 临界区必须非常短，否则会影响系统响应性

自旋锁 vs 互斥锁

08 选型对比

场景推荐原语原因
保护共享资源（可睡眠）k_mutex支持优先级继承，避免优先级反转
资源计数k_sem轻量级，支持计数
事件通知（位掩码）k_event支持等待任意/所有事件
条件等待k_condvar + k_mutex标准条件变量模式
用户态高性能锁k_futex无竞争时无系统调用
中断上下文k_spinlock唯一可在 ISR 中使用的锁
ISR 通知线程k_sem_give()ISR 中可调用
ISR 通知线程（事件）k_event_post()ISR 中可调用
