01 概述

Linux 时钟子系统（Common Clock Framework, CCF）提供了统一的时钟管理框架，负责 SoC 内部时钟树的创建、速率设置、门控开关、父子关系管理。几乎所有 SoC 外设都依赖时钟子系统来获取工作时钟。

核心职责
  • 时钟树管理：PLL → 分频器 → 门控 → 外设
  • 速率设置：设置/查询/舍入时钟频率
  • 使能/禁用：引用计数式门控
  • 父子关系：速率传播与父时钟切换
  • 通知机制：速率变更通知
  • 调试接口：debugfs 查看时钟树
时钟类型
类型说明
固定速率不可更改的参考时钟
门控可开关的时钟
分频器父时钟 / 固定因子
倍频器父时钟 × 固定因子
多路选择从多个父时钟中选择
复合组合多种功能
PLL锁相环（可变倍频）

时钟树层次

OSC（24 MHz 晶振）
  ├──→ PLL（倍频到 1.2 GHz）
  │      ├──→ CPU_CLK（CPU 核心时钟）
  │      │      ├──→ cpu_div（÷2 → 600 MHz）
  │      │      └──→ cpu_gate（门控）
  │      ├──→ BUS_CLK（总线时钟）
  │      └──→ MEM_CLK（内存控制器时钟）
  ├──→ PLL2（倍频到 500 MHz）
  │      └──→ PERIPH_CLK
  │             ├──→ uart_clk
  │             ├──→ spi_clk
  │             └──→ i2c_clk
  └──→ RTC_CLK（32.768 KHz）

三层模型

层次核心结构说明
消费者clk_get() / clk_set_rate()设备驱动使用时钟
核心层clk_core + clk.c时钟树管理、引用计数、通知
提供者clk_ops + clk_hw硬件操作回调

prepare/enable 分离

两阶段使能：Linux 将时钟使能分为 clk_prepare()（可睡眠，如使能稳压器）和 clk_enable()（原子上下文，如写寄存器）。这使得中断上下文中的时钟使能可以快速完成。使用 clk_prepare_enable() 组合两者。

struct clk_ops {
    // 使能控制
    int (*prepare)(struct clk_hw *hw);      // 可睡眠
    void (*unprepare)(struct clk_hw *hw);
    int (*enable)(struct clk_hw *hw);      // 原子上下文
    void (*disable)(struct clk_hw *hw);

    // 频率管理
    unsigned long (*recalc_rate)(struct clk_hw *hw, unsigned long parent_rate);
    long (*round_rate)(struct clk_hw *hw, unsigned long rate, unsigned long *parent_rate);
    int (*determine_rate)(struct clk_hw *hw, struct clk_rate_request *req);
    int (*set_rate)(struct clk_hw *hw, unsigned long rate, unsigned long parent_rate);

    // 父时钟管理
    int (*set_parent)(struct clk_hw *hw, u8 index);
    u8 (*get_parent)(struct clk_hw *hw);

    // 相位/占空比
    int (*get_phase)(struct clk_hw *hw);
    int (*set_phase)(struct clk_hw *hw, int degrees);
    int (*get_duty_cycle)(struct clk_hw *hw, struct clk_duty *duty);
    int (*set_duty_cycle)(struct clk_hw *hw, unsigned int num, unsigned int den);
};

04 clk_ops

关键回调详解

回调上下文说明
prepare()可睡眠使能前准备（使能稳压器等）
enable()原子写寄存器使能时钟
recalc_rate()可睡眠查询硬件当前频率
determine_rate()可睡眠计算最优频率和父时钟
set_rate()可睡眠设置硬件频率
set_parent()可睡眠切换父时钟
get_parent()可睡眠查询当前父时钟
set_phase()可睡眠设置输出相位
save_context()可睡眠保存时钟状态（休眠前）
restore_context()可睡眠恢复时钟状态（唤醒后）

引用计数

引用计数：每个时钟维护 enable_count 和 prepare_count。只有当计数从 0→1 时才真正使能硬件，从 1→0 时才禁用。这允许多个驱动共享同一时钟。

设备树中的时钟

// 常见时钟节点
clk@0x00 {
    // 晶振（固定频率）
    osc: oscillator@0 {
        compatible = "fixed-clock";
        #clock-cells = <0>;
        clock-frequency = <24000000>;  // 24 MHz
        clock-output-names = "osc";
    };

    // PLL（锁相环）
    pll: pll@0x100 {
        compatible = "vendor,pll";
        clocks = &osc;
        clock-output-names = "pll";
        #clock-cells = <0>;
    };

    // 门控时钟
    uart_gate: gate@0x200 {
        compatible = "clk-gate";
        clocks = &pll;
        #clock-cells = <0>;
        reg = <0x200 0x4>;
        bit-shift = <0>;
    };

    // 分频器
    cpu_div: div@0x300 {
        compatible = "divider-clock";
        clocks = &pll;
        #clock-cells = <0>;
        clock-divisor = <2>;  // ÷2
    };
};

时钟引用语法

格式说明
clocks = &pll;引用单一时钟（#clock-cells=0）
clocks = &pll 0;引用时钟 + 索引（#clock-cells>0）
clock-names = "core";为时钟引用命名
assigned-clocks启动时需要配置的时钟
assigned-clock-rates分配的频率值
assigned-clock-parents分配的父时钟

07 编写驱动

时钟控制器驱动模板

#include <linux/clk-provider.h>
#include <linux/platform_device.h>

struct my_clk_priv {
    void __iomem *base;
    struct clk_hw hw;
    u32 reg;
};

#define to_my_clk(_hw) container_of(_hw, struct my_clk_priv, hw)

static int my_clk_enable(struct clk_hw *hw)
{
    struct my_clk_priv *priv = to_my_clk(hw);
    u32 val = readl(priv->base + priv->reg);
    val |= 1;  // 使能位
    writel(val, priv->base + priv->reg);
    return 0;
}

static void my_clk_disable(struct clk_hw *hw)
{
    struct my_clk_priv *priv = to_my_clk(hw);
    u32 val = readl(priv->base + priv->reg);
    val &= ~1;
    writel(val, priv->base + priv->reg);
}

static const struct clk_ops my_clk_ops = {
    .enable  = my_clk_enable,
    .disable = my_clk_disable,
};

static int my_clk_probe(struct platform_device *pdev)
{
    struct my_clk_priv *priv;
    struct clk_init_data init = {};

    priv = devm_kzalloc(&pdev->dev, sizeof(*priv), GFP_KERNEL);
    priv->base = devm_platform_ioremap_resource(pdev, 0);

    init.name = of_get_property(pdev->dev.of_node, "clock-output-names", NULL);
    init.ops = &my_clk_ops;
    init.parent_names = &parent_name;
    init.num_parents = 1;

    priv->hw.init = &init;
    devm_clk_hw_register(&pdev->dev, &priv->hw);

    return of_clk_add_hw_provider(pdev->dev.of_node, of_clk_hw_simple_get, &priv->hw);
}
