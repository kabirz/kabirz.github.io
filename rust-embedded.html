01 概述

嵌入式 Rust 是 Rust 在资源受限的微控制器（MCU）上的应用。提供 C/C++ 的性能 + Rust 的安全。

嵌入式 vs 桌面：

桌面 Rust嵌入式 Rust
内存GBKB（2KB-1MB）
存储GBKB-MB（8KB-4MB）
OSLinux/Windows裸机 / RTOS
标准库完整no_std
分配器malloc/free无（或 alloc）
典型平台x86_64ARM Cortex-M, RISC-V
线程OS 线程中断 + 协程

嵌入式 Rust 分层：

┌─────────────────────────────────────────┐
│   Application Code                      │
├─────────────────────────────────────────┤
│   Framework（RTIC / Embassy / 自管）     │
├─────────────────────────────────────────┤
│   BSP（Board Support Package）           │
│   (板级支持，如 rp-pico)                 │
├─────────────────────────────────────────┤
│   HAL（Hardware Abstraction Layer）      │
│   (芯片级，如 rp2040-hal)                │
├─────────────────────────────────────────┤
│   PAC（Peripheral Access Crate）         │
│   (寄存器级，svd2rust 生成)              │
├─────────────────────────────────────────┤
│   MCU（微控制器硬件）                    │
└─────────────────────────────────────────┘

02 no_std 环境

// #![no_std] 移除 std 依赖
// 只能用 core + alloc（可选）

#![no_std]
#![no_main]   // 不用标准 main

use core::panic::PanicInfo;

#[panic_handler]
fn panic(_info: &PanicInfo) -> ! {
    // 自定义 panic 处理
    loop {}
}

// 自定义入口（Cortex-M）：
#[cortex_m_rt::entry]
fn main() -> ! {
    // 应用代码
    loop {}
}

// core crate：
//   - 类型：i32, f32, Option, Result, ...
//   - 操作：Add, Eq, Iterator, ...
//   - mem, convert, slice, ...

// alloc crate（需要 alloc）：
//   - Vec, String, Box, Arc, ...
//   - 需要实现 #[global_allocator]
//   - 嵌入式很少用（耗内存）

// common no_std deps：
//   heapless  - 无堆数据结构（Vec, String, Map）
//   nb        - 非阻塞 IO
//   embedded-hal - 硬件抽象
//   void      - Void 类型
//   bitflags  - 位标志

无堆数据结构：

// heapless：基于数组的容器
use heapless::spsc::Queue;
use heapless::Vec as HVec;
use heapless::String as HString;

let mut queue: Queue<u8, 16> = Queue::new();
queue.enqueue(1).unwrap();
let v = queue.dequeue().unwrap();

let mut vec: HVec<u8, 8> = HVec::new();
vec.push(1).unwrap();
vec.push(2).unwrap();
// vec.push(3).unwrap();   // 第 9 个会失败

let mut s: HString<16> = HString::new();
s.push_str("hello").unwrap();

03 embedded-hal 抽象

// embedded-hal 是嵌入式 Rust 的核心 trait 库
// 类似 std::io::Read/Write，但针对硬件

// 主要 trait：
// digital::
//   InputPin   - 数字输入
//   OutputPin  - 数字输出
//   StatefulOutputPin  - 状态输出
//   ToggleableOutputPin
// 
// i2c::I2c    - I2C 主机
// spi::SpiBus / SpiDevice
// serial::Read / Write  - 串口
// 
// blocking vs nb vs async：
//   blocking::Read / Write - 阻塞 API
//   nb::Read / Write - non-blocking
//   async（embedded-hal-async）

// 使用 trait：
use embedded_hal::digital::OutputPin;

fn blink_led<LED: OutputPin>(led: &mut LED, times: u8) {
    for _ in 0..times {
        led.set_high().ok();
        delay(100);
        led.set_low().ok();
        delay(100);
    }
}

// 任何实现了 OutputPin 的都可以
struct MyLed;
impl OutputPin for MyLed {
    type Error = ();
    fn set_low(&mut self) -> Result<(), Self::Error> { Ok(()) }
    fn set_high(&mut self) -> Result<(), Self::Error> { Ok(()) }
}

blink_led(&mut MyLed, 5);

I2C 示例（温度传感器）：

use embedded_hal::i2c::I2c;

const TMP102_ADDR: u8 = 0x48;

fn read_temperature<I2C: I2c>(i2c: &mut I2C) -> Result<f32, I2C::Error> {
    let mut buf = [0u8; 2];
    i2c.write_read(TMP102_ADDR, &[0x00], &mut buf)?;
    let raw = ((buf[0] as u16) << 4) | (buf[1] as u16 >> 4);
    Ok(raw as f32 * 0.0625)
}

04 PAC / HAL / BSP

// PAC (Peripheral Access Crate)
//   - 由 SVD 文件自动生成
//   - 直接访问寄存器
//   - svd2rust 工具
//   - 例：rp2040-pac, stm32f4-pac

// PAC 使用：
use rp2040_pac;

let p = rp2040_pac::Peripherals::take().unwrap();
p.IO_BANK0.gpio[25].gpio_ctrl.write(|w| w.funcsel().sio_25());
p.SIO.gpio_oe_set.write(|w| w.bits(1 << 25));
p.SIO.gpio_out_set.write(|w| w.bits(1 << 25));

// 优点：最细粒度控制
// 缺点：繁琐、易错、与硬件绑定

// HAL (Hardware Abstraction Layer)
//   - 包装 PAC，提供高层 API
//   - 例：rp2040-hal, stm32f4xx-hal
//   - 实现 embedded-hal trait

use rp2040_hal::{
    clocks::{Clock, ClockSource},
    gpio::{Pins, Output},
    pac,
    sio::Sio,
    watchdog::Watchdog,
};

let mut pac = pac::Peripherals::take().unwrap();
let mut watchdog = Watchdog::new(pac.WATCHDOG);

// 配置时钟
let clocks = clocks::init_clocks_and_plls(
    12_000_000,   // 外部晶振
    pac.XOSC,
    pac.CLOCKS,
    pac.PLL_SYS,
    pac.PLL_USB,
    &mut pac.RESETS,
    &mut watchdog,
).ok().unwrap();

// 配置 GPIO
let sio = Sio::new(pac.SIO);
let pins = Pins::new(pac.IO_BANK0, pac.PADS_BANK0, sio.gpio_bank0, &mut pac.RESETS);
let mut led = pins.gpio25.into_push_pull_output();
led.set_high().unwrap();

// BSP (Board Support Package)
//   - 特定板子的封装
//   - 例：rp-pico (Raspberry Pi Pico)
//   - 预配置时钟、关键外设

use rp_pico::hal::prelude::*;
use rp_pico::{Pins, XOSC_CRYSTAL_FREQ};

let mut pac = hal::pac::Peripherals::take().unwrap();
let mut watchdog = hal::Watchdog::new(pac.WATCHDOG);
let clocks = hal::clocks::init_clocks_and_plls(
    XOSC_CRYSTAL_FREQ,
    pac.XOSC,
    pac.CLOCKS,
    pac.PLL_SYS,
    pac.PLL_USB,
    &mut pac.RESETS,
    &mut watchdog,
).ok().unwrap();

let pins = Pins::new(
    pac.IO_BANK0, pac.PADS_BANK0,
    sio.gpio_bank0, &mut pac.RESETS,
);
let mut led = pins.led.into_push_pull_output();

05 内存布局

// 嵌入式可执行文件 = ELF + 自定义段

// 内存映射（典型 Cortex-M）：
// 0x00000000 +----------+
//            | .vector  |  (中断向量表)
//            +----------+
// 0x08000000 | .text    |  (代码 in flash)
//  | .rodata |  (只读数据)
//            +----------+
// 0x20000000 | .data    |  (初始化的全局)
//            | .bss     |  (零初始化的全局)
//            | heap     |  (可选)
//            | stack    |
//            +----------+
// 0x2000FFFF (RAM 顶)

// linker script：memory.x
MEMORY {
    FLASH : ORIGIN = 0x10000000, LENGTH = 2048K
    RAM   : ORIGIN = 0x20000000, LENGTH = 256K
}

// link.x：
//   - 自动生成（cortex-m-rt）
//   - 定义段位置

// 中断向量表：
//   cortex-m-rt::entry / exception / interrupt 宏注册

// 静态可变：
static mut COUNTER: u32 = 0;

#[entry]
fn main() -> ! {
    unsafe { COUNTER += 1; }   // unsafe，单线程但编译器保守
    loop {}
}

alloc 在嵌入式：

// 需要实现 #[global_allocator]
// 用 linked_list_allocator 或 alloc-cortex-m

use alloc_cortex_m::CortexMHeap;

#[global_allocator]
static ALLOCATOR: CortexMHeap = CortexMHeap::empty();

fn init_heap() {
    let start = ...;
    let size = 1024 * 16;   // 16KB
    unsafe {
        ALLOCATOR.init(start, size);
    }
}

// 之后可以用 Vec, Box, ...
extern crate alloc;
use alloc::vec::Vec;

let v: Vec<u8> = Vec::new();
v.push(1);

06 RTIC 框架

RTIC (Real-Time Interrupt-driven Concurrency) 是 Rust 嵌入式主要并发框架。

// [dependencies]
// cortex-m-rtic = "1.0"
// 
// 关键概念：
// - resources: 共享状态
// - tasks: 中断处理
// - spawn: 软件触发任务
// - schedule: 定时触发

#![no_std]
#![no_main]

use rtic::app;

#[app(device = rp_pico::hal::pac)]
mod app {
    use super::*;
    
    #[shared]
    struct Shared {
        counter: u32,
    }
    
    #[local]
    struct Local {}
    
    #[init]
    fn init(_: init::Context) -> (Shared, Local) {
        // 配置硬件、注册中断
        (Shared { counter: 0 }, Local {})
    }
    
    // 中断任务
    #[task(binds = TIMER_IRQ, shared = [counter])]
    fn timer_irq(ctx: timer_irq::Context) {
        (ctx.shared.counter).lock(|c| *c += 1);
    }
    
    // 软件任务（spawn 触发）
    #[task(shared = [counter])]
    fn process(ctx: process::Context) {
        let c = (ctx.shared.counter).lock(|c| *c);
        // ...
    }
    
    #[idle]
    fn idle(_: idle::Context) -> ! {
        loop {
            cortex_m::asm::wfi();
        }
    }
}

RTIC 优势：

  • 编译期分析：无数据竞争（类似 ownership）
  • 无锁优先：高优先级抢占时不阻塞
  • 无 OS：直接跑在裸机
  • 静态分析：worst-case execution time 可分析
  • 资源使用：无堆、无 OS

优先级抢占：

// RTIC 通过 priority 实现抢占：
#[task(binds = IRQ_LOW, priority = 1)]
fn low(ctx: low::Context) { ... }

#[task(binds = IRQ_HIGH, priority = 3)]
fn high(ctx: high::Context) { ... }

// high 抢占 low
// 中断嵌套由 RTIC 自动处理

// 共享资源用 lock：
(ctx.shared.counter).lock(|c| *c += 1);
// lock 自动屏蔽低于当前优先级的中断
// 高优先级仍可抢占

07 Embassy 异步

Embassy 是基于 Rust async/await 的异步嵌入式框架。

// [dependencies]
// embassy-executor = { version = "0.5", features = ["arch-cortex-m", "executor-thread"] }
// embassy-time = "0.3"
// embassy-rp = { version = "0.1", features = ["time-driver"] }
// 
// Embassy 核心：
// - async/await 替代中断
// - 多任务并发
// - 类型安全的 HAL
// - 内置 USB、网络、蓝牙协议栈

#![no_std]
#![no_main]

use embassy_executor::Spawner;
use embassy_rp::gpio::{Level, Output};
use embassy_time::Timer;

#[embassy_executor::main]
async fn main(_spawner: Spawner) {
    let p = embassy_rp::init(Default::default());
    let mut led = Output::new(p.PIN_25, Level::Low);
    
    loop {
        led.set_high();
        Timer::after_millis(500).await;
        led.set_low();
        Timer::after_millis(500).await;
    }
}

// 多任务：
#[embassy_executor::task]
async fn blinker(mut led: Output) {
    loop {
        led.set_high();
        Timer::after_millis(500).await;
        led.set_low();
        Timer::after_millis(500).await;
    }
}

#[embassy_executor::task]
async fn logger() {
    loop {
        log_info!("tick");
        Timer::after_secs(1).await;
    }
}

#[embassy_executor::main]
async fn main(spawner: Spawner) {
    let p = embassy_rp::init(Default::default());
    let led = Output::new(p.PIN_25, Level::Low);
    spawner.spawn(blinker(led)).unwrap();
    spawner.spawn(logger()).unwrap();
}

Embassy 优势：

  • async 替代中断：代码更线性
  • 无 dynamic allocation：栈固定大小
  • 多个任务并发：单核也能高效
  • 丰富生态：USB、网络、蓝牙
  • 编译期分析：内存使用可预测

协议栈：

// Embassy 提供完整网络栈
use embassy_net::{
    Stack, StackResources, Config,
    Ipv4Address, StaticConfigurator,
};
use embassy_rp::peripherals::{PIO0};
use embassy_rp::pio::Pio;

// cyw43 (Pico W WiFi)
let (mut control, stack) = embassy_rp::init_wifi(p.PIO0, p.DMA_CH0, p.CYW43_PWR, ...);

spawner.spawn(net_task(stack)).unwrap();

control.init().await;
control.join("SSID", "password").await;

// TCP/UDP
use embassy_net::tcp::TcpSocket;
let mut rx = [0; 4096];
let mut tx = [0; 4096];
let mut socket = TcpSocket::new(stack, &mut rx, &mut tx);
socket.connect(8080, Ipv4Address::new(192, 168, 1, 1)).await.unwrap();
socket.write(b"hello").await.unwrap();

嵌入式 Rust 总结:
1. no_std：无 OS、无标准库环境
2. embedded-hal：硬件抽象 trait（生态核心）
3. PAC → HAL → BSP：分层抽象
4. RTIC：实时并发框架（同步）
5. Embassy：异步嵌入式框架
6. probe-rs + defmt：调试标准工具链
Rust 嵌入式生态已相当成熟，2024 起 NASA、SpaceX、特斯拉都在生产用 Rust 嵌入式。
