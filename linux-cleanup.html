01 概述

Linux 内核中资源管理的传统做法是 goto error 模式 — 每个错误出口都要手动释放已获取的资源。

传统 goto error 模式
p = kzalloc(...);
if (!p) goto out;
r = devm_request_irq(..., handler, ...);
if (r < 0) goto free_p;
m = kzalloc(...);
if (!m) goto free_irq;
// ... 正常路径 ...
return 0;

free_irq: free_irq(...);
free_p: kfree(p);
out: return -ENOMEM;
cleanup 模式
struct my *p __free(kfree) = kzalloc(...);
if (!p) return -ENOMEM;
r = devm_request_irq(..., handler, ...);
if (r < 0) return r;
struct my2 *m __free(kfree) = kzalloc(...);
if (!m) return -ENOMEM;
// ... 正常路径 ...
return 0;
// 无需任何 cleanup 代码

cleanup.h 利用 C 编译器的 __attribute__((__cleanup__))，在变量出作用域时自动调用释放函数。这可以看作 C 语言领域的 RAII。

该机制提供四个宏家族：

宏族用途
__free()自动释放内存/资源
guard()作用域锁（整个块）
scoped_guard()块级作用域锁
CLASS()带状态的资源包装

02 核心宏 API

DEFINE_FREE / __free

// 定义释放函数：free_expr 在变量出作用域时执行
DEFINE_FREE(name, type, free_expr)

// 声明变量：value 出作用域时自动调用 DEFINE_FREE 指定的表达式
type var __free(name) = value;

no_free_ptr / return_ptr

// 释放所有权，不释放资源（从 cleanup 托管中取出）
type *no_free_ptr(var)

// 释放所有权并返回指针
return_ptr(var)

DEFINE_GUARD / guard

// 定义锁的类型：
// lock_expr 获取锁，unlock_expr 释放锁
DEFINE_GUARD(name, type, lock_expr, unlock_expr)

// 在作用域内持有锁，作用域结束时自动释放
guard(name)(lock_var)

DEFINE_GUARD_COND / ACQUIRE

// 条件锁（如 trylock）
DEFINE_GUARD_COND(name, ext, condlock_expr)

// 获取条件锁，失败时跳转到 scope_err
ACQUIRE(name, lock_var)

// 获取条件锁，返回错误码
ACQUIRE_ERR(name, lock_var)

scoped_guard

// 块级作用域锁，锁仅在该块内持有
scoped_guard(name, lock_var) {
    // 在此块中持有锁
    // 块结束时自动释放
}

CLASS

// 声明一个带有 init/exit 作用域的类实例
CLASS(name, var) { ... }

04 使用模式

模式一：自动释放 kmalloc

struct foo *p __free(kfree) = kzalloc(sizeof(*p), GFP_KERNEL);
if (!p)
    return -ENOMEM;
// 使用 p... 无需手动 kfree
if (err)
    return -EINVAL;  // p 自动释放
return 0;            // p 自动释放

模式二：作用域锁

// 在整个函数作用域持有锁
int my_func(struct my *m)
{
    guard(mutex)(&m->lock);
    // 互斥访问
    // 函数返回时自动释放
    return 0;
}

模式三：块级锁

// 锁仅在 if 块内持有
if (need_lock) {
    scoped_guard(mutex, &m->lock) {
        // 锁在此块中
        update_locked_data();
    } // 离开块自动释放
}
// 锁已释放

模式四：嵌套锁

void nested(void)
{
    guard(mutex)(&outer);
    guard(spinlock_irqsave)(&inner);
    // 按 LIFO 顺序释放：先 inner 再 outer
}

05 高级用法

return_ptr — 转移所有权

struct obj *create_obj(void)
{
    struct obj *o __free(kfree) = kzalloc(sizeof(*o), GFP_KERNEL);
    if (!o)
        return NULL;

    if (init_failed) {
        return NULL;  // 自动 kfree
    }

    // 转移所有权给调用者
    return_ptr(o);
}

ACQUIRE — trylock 模式

// 尝试获取锁，失败则退出
ACQUIRE(mutex_try, &m->lock);
// 成功获取锁
// 作用域结束时自动释放

scoped_guard 的 trylock 变体

// 尝试获取锁，锁可用才进入块
scoped_guard(mutex_try, &m->lock) {
    // 成功获取锁
} else {
    // 锁不可用时的处理
}

06 与 goto error 对比

传统驱动的 probe 函数 — 这是最容易产生资源泄漏的模式。

传统写法（冗长）
static int my_probe(struct platform_device *pdev)
{
    struct my *m;
    int r;

    m = kzalloc(sizeof(*m), GFP_KERNEL);
    if (!m) { r = -ENOMEM; goto out; }

    m->res = platform_get_resource(pdev,...);
    if (!m->res) { r = -ENXIO; goto free_m; }

    m->base = devm_ioremap(dev, m->res);
    if (!m->base) { r = -ENOMEM; goto free_m; }

    r = request_irq(irq, handler, 0, "my", m);
    if (r) goto free_m;

    r = register_something(m);
    if (r) goto free_irq;

    return 0;

free_irq: free_irq(irq, m);
free_m: kfree(m);
out: return r;
}
cleanup 写法（简洁）
static int my_probe(struct platform_device *pdev)
{
    struct my *m __free(kfree) = kzalloc(...);
    if (!m) return -ENOMEM;

    m->res = platform_get_resource(pdev,...);
    if (!m->res) return -ENXIO;

    m->base = devm_ioremap(dev, m->res);
    if (!m->base) return -ENOMEM;

    // 注意：这里 request_irq 注册了回调引用了 m
    // 但 __free(kfree) 在成功路径上也被调用了
    // 需要 return_ptr(m) 转移所有权
    // 或使用 devm_request_irq

    int r = devm_request_irq(dev, irq, handler,...);
    if (r) return r;

    r = register_something(m);
    if (r) return r;

    // 所有权转移给 devm
    return_ptr(m);
}

07 实现原理

编译器基础：__attribute__((__cleanup__))

// GCC/Clang 内置特性
void cleanup_func(void *ptr) { kfree(*(void **)ptr); }

int test(void)
{
    // 当 'p' 出作用域时，自动调用 cleanup_func(&p)
    void *p __attribute__((__cleanup__(cleanup_func))) = kmalloc(100);
    return p ? 0 : -ENOMEM;
}

__free 的实现

// cleanup.h
#define __free(name) \
    __attribute__((__cleanup__(__free_##name)))

// DEFINE_FREE 展开为：
// static inline void __free_kfree(void *p) { kfree(*(void **)p); }

guard 的实现

// guard(mutex)(&m) 展开为：
// 1. 在栈上创建一个匿名变量，类型为 guard_mutex
// 2. 初始化时调用 mutex_lock(&m)
// 3. 析构时调用 mutex_unlock(&m)

#define guard(name) \
    DEFINE_GUARD(name, ...)

#define DECLARE_GUARD(name, type, lock, unlock) \
    static inline void __guard_##name(type *p) { unlock(*p); } \
    typedef struct { type lock; } guard_##name; \
    static inline guard_##name *__guard_##name##_init(type *l) \
    { lock(*l); return (guard_##name *)l; }

scoped_guard 的 for 循环技巧

// scoped_guard(mutex, &m)
// 展开为利用 for 循环创建块级作用域：
for (guard(mutex)(&m), *done = 0; done == 0; done = 1)
    // 块只执行一次

// 因为 for 循环的初始化语句创建了一个作用域
// 锁在初始化时获取，在循环迭代间释放
