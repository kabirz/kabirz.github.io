01 内存管理概览

Zephyr 提供多种内存分配机制，适用于不同的使用场景和资源约束。

02 内核堆 (k_heap)

k_heap 是 Zephyr 的通用动态内存分配器，基于 sys_heap 实现，支持任意大小分配。

数据结构

// 内核堆结构体
struct k_heap {
    struct sys_heap heap;     // 底层堆管理
    struct k_spinlock lock;   // 自旋锁保护
    struct k_wait_q wait_q;   // 等待队列（阻塞分配）
};

堆分配实现

// kernel/kheap.c

// 堆初始化
void k_heap_init(struct k_heap *heap, void *mem, size_t bytes)
{
    z_waitq_init(&heap->wait_q);
    heap->lock = (struct k_spinlock) {};
    sys_heap_init(&heap->heap, mem, bytes);
}

// 堆分配
static void *z_heap_alloc_helper(struct k_heap *heap, size_t align,
                                 size_t bytes, k_timeout_t timeout,
                                 sys_heap_allocator_t *alloc_fn)
{
    k_timepoint_t end = sys_timepoint_calc(timeout);
    void *ret = NULL;

    k_spinlock_key_t key = k_spin_lock(&heap->lock);

    // 循环尝试分配
    while (ret == NULL) {
        // 1. 尝试从 sys_heap 分配
        ret = alloc_fn(&heap->heap, align, bytes);

        // 2. 分配成功则退出
        if (ret != NULL) {
            break;
        }

        // 3. 如果是 K_NO_WAIT，立即返回
        if (K_TIMEOUT_EQ(timeout, K_NO_WAIT)) {
            break;
        }

        // 4. 等待内存释放
        (void)z_pend_curr_unlocked(&heap->wait_q, NULL, end);
    }

    k_spin_unlock(&heap->lock, key);
    return ret;
}

// k_malloc 底层实现
void *k_malloc(size_t size)
{
    // 从默认堆分配
    return k_heap_alloc(&_k_thread_heap, size, K_NO_WAIT);
}

// k_free 底层实现
void k_free(void *ptr)
{
    // 释放到默认堆
    k_heap_free(&_k_thread_heap, ptr);
}

静态堆定义

// 静态定义堆
K_HEAP_DEFINE(my_heap, 4096);  // 定义 4KB 堆

// 或者动态初始化
static char heap_mem[4096];
struct k_heap my_heap;

k_heap_init(&my_heap, heap_mem, sizeof(heap_mem));

// 从堆分配
void *ptr = k_heap_alloc(&my_heap, 128, K_NO_WAIT);

// 释放
k_heap_free(&my_heap, ptr);

03 内存 Slab

k_mem_slab 是固定大小对象的分配器，O(1) 分配/释放，零碎片，适合频繁分配相同大小对象的场景。

数据结构

// include/zephyr/kernel.h
struct k_mem_slab {
    struct k_wait_q wait_q;   // 等待队列
    struct k_spinlock lock;   // 自旋锁
    char *buffer;             // 内存块起始地址
    size_t block_size;        // 每个块的大小
    uint32_t num_blocks;      // 总块数
    uint32_t num_used;        // 已使用块数
    uint32_t max_used;        // 历史最大使用块数
    struct sys_memory_stats info;
    // 空闲块链表（每块前 4 字节指向下一个空闲块）
};

Slab 分配实现

// kernel/mem_slab.c

// 分配一个内存块
int k_mem_slab_alloc(struct k_mem_slab *slab, void **mem, k_timeout_t timeout)
{
    k_timepoint_t end = sys_timepoint_calc(timeout);

    k_spinlock_key_t key = k_spin_lock(&slab->lock);

    // 1. 检查是否有空闲块
    if (slab->num_used < slab->num_blocks) {
        // 从空闲链表取第一个块
        *mem = slab->free_list;
        slab->free_list = *(void **)*mem;
        slab->num_used++;

        // 更新最大使用统计
        if (slab->num_used > slab->max_used) {
            slab->max_used = slab->num_used;
        }

        k_spin_unlock(&slab->lock, key);
        return 0;
    }

    // 2. 无空闲块，等待或立即返回
    if (K_TIMEOUT_EQ(timeout, K_NO_WAIT)) {
        k_spin_unlock(&slab->lock, key);
        *mem = NULL;
        return -ENOMEM;
    }

    // 3. 加入等待队列
    int ret = z_pend_curr(&slab->lock, key, &slab->wait_q, end);
    if (ret == 0) {
        *mem = k_current_get()->base.swap_data;
    } else {
        *mem = NULL;
    }
    return ret;
}

// 释放内存块
void k_mem_slab_free(struct k_mem_slab *slab, void *mem)
{
    k_spinlock_key_t key = k_spin_lock(&slab->lock);

    // 将块加入空闲链表头部
    *(void **)mem = slab->free_list;
    slab->free_list = mem;
    slab->num_used--;

    // 唤醒等待的线程
    struct k_thread *pending = z_waitq_head(&slab->wait_q);
    if (pending != NULL) {
        z_thread_return_value(pending, 0);
        z_unready_thread(pending);
    }

    k_spin_unlock(&slab->lock, key);
}

静态 Slab 定义

// 静态定义 Slab
K_MEM_SLAB_DEFINE(my_slab, 128, 16, 4);
// 参数：name, block_size, num_blocks, alignment
// 即：16 个 128 字节的块，4 字节对齐

// 使用
void *buf;
int ret = k_mem_slab_alloc(&my_slab, &buf, K_NO_WAIT);
if (ret == 0) {
    // 使用 buf...
    k_mem_slab_free(&my_slab, buf);
}

04 内存池

k_mem_pool 支持 2^n 大小的块分配，提供介于 Slab 和 Heap 之间的灵活性。

// 内存池结构（简化版）
// 内部使用多级空闲列表管理不同大小的块

// 静态定义内存池
K_MEM_POOL_DEFINE(my_pool, 64, 1024, 4, 4);
// 参数：name, min_block_size, max_block_size, num_blocks, align
// 即：支持 64/128/256/512/1024 字节的块分配

// 分配
struct k_mem_block block;
int ret = k_mem_pool_alloc(&my_pool, &block, 256, K_NO_WAIT);

// 释放
k_mem_pool_free(&block);

05 内存域

内存域（Memory Domain）用于用户态线程的内存隔离，通过 MPU/MMU 保护不同线程的内存空间。

// 内存域定义
K_MEM_DOMAIN_DEFINE(my_domain);

// 添加内存分区到域
K_MEM_PARTITION_DEFINE(part1, part1_mem, PART1_SIZE,
                       K_MEM_PARTITION_P_RW_U_RW);

k_mem_domain_add_partition(&my_domain, &part1);

// 将线程添加到域
k_mem_domain_add_thread(&my_domain, user_tid);

// 内存分区权限
// K_MEM_PARTITION_P_RW_U_RW  - 内核/用户 可读写
// K_MEM_PARTITION_P_RO_U_RO  - 内核/用户 只读
// K_MEM_PARTITION_P_RX_U_RX  - 内核/用户 可执行
// K_MEM_PARTITION_P_RW_U_RO  - 内核读写，用户只读
// K_MEM_PARTITION_P_RWX_U_RWX - 全权限

06 用户态内存

用户态线程不能直接访问内核内存，需要通过系统调用分配内存。

// 用户态线程内存分配
// 1. 从内核堆分配（通过系统调用）
void *ptr = k_malloc(size);
k_free(ptr);

// 2. 从内存域分配（MPU 保护）
// 线程只能访问自己内存域中的内存

// 3. 线程栈独立保护
// 每个用户态线程的栈有独立的 MPU 保护
// 栈溢出触发 MemManage 异常
