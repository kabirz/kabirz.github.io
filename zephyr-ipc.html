02 FIFO / LIFO 队列

FIFO 和 LIFO 是最简单的 IPC 机制，传递数据指针而非数据本身。

数据结构

// FIFO 本质是双向链表
// include/zephyr/kernel.h
struct k_fifo {
    struct k_wait_q wait_q;    // 等待队列（消费者等待）
    struct _dlist list;        // 数据链表
};

使用示例

// 定义 FIFO
K_FIFO_DEFINE(my_fifo);

// 数据结构（必须包含 sys_dnode_t 节点）
struct data_item {
    sys_dnode_t node;          // 链表节点
    uint32_t value;
    char payload[32];
};

// 生产者（线程或 ISR）
void producer(void)
{
    struct data_item *item = k_malloc(sizeof(*item));
    item->value = 42;

    k_fifo_put(&my_fifo, item);  // 放入 FIFO
    // ISR 中也可调用 k_fifo_put()
}

// 消费者
void consumer(void)
{
    struct data_item *item = k_fifo_get(&my_fifo, K_FOREVER);
    // 处理 item...
    k_free(item);
}

ISR 安全性

// FIFO/LIFO 是 ISR 安全的：
// k_fifo_put() / k_lifo_put() 可在 ISR 中调用
// 如果有线程在等待，ISR 会唤醒它

// ISR 中使用
void my_isr(void *arg, void *arg2)
{
    struct data_item *item = ...;
    k_fifo_put(&my_fifo, item);
    // ISR 返回后，被唤醒的线程继续执行
}

03 消息队列 (k_msgq)

消息队列传递固定大小的消息，使用拷贝语义，无需动态内存分配。

数据结构

// include/zephyr/kernel.h
struct k_msgq {
    struct k_wait_q wait_q;    // 等待队列
    char *buffer;              // 消息缓冲区
    size_t msg_size;           // 每条消息大小
    uint32_t max_msgs;         // 最大消息数
    uint32_t head;             // 写入位置
    uint32_t tail;             // 读取位置
    uint32_t used_msgs;        // 当前消息数
};

使用示例

// 定义消息队列（16 条消息，每条 64 字节）
K_MSGQ_DEFINE(my_msgq, 64, 16, 4);

// 发送消息
struct sensor_data {
    uint32_t timestamp;
    int32_t temperature;
};

struct sensor_data data = { .timestamp = 1000, .temperature = 25 };
k_msgq_put(&my_msgq, &data, K_FOREVER);

// 接收消息
struct sensor_data rx_data;
k_msgq_get(&my_msgq, &rx_data, K_FOREVER);

// 非阻塞收发
int ret = k_msgq_put(&my_msgq, &data, K_NO_WAIT);
if (ret == -EAGAIN) {
    // 队列满
}

ret = k_msgq_get(&my_msgq, &rx_data, K_NO_WAIT);
if (ret == -EAGAIN) {
    // 队列空
}

实现原理

// kernel/msg_q.c

int z_impl_k_msgq_put(struct k_msgq *msgq, const void *data, k_timeout_t timeout)
{
    k_spinlock_key_t key = k_spin_lock(&lock);

    // 1. 队列未满，直接写入
    if (msgq->used_msgs < msgq->max_msgs) {
        memcpy(msgq->buffer + (msgq->head * msgq->size), data, msgq->size);
        msgq->head = (msgq->head + 1) % msgq->max_msgs;
        msgq->used_msgs++;

        // 如果有消费者在等待，唤醒它
        struct k_thread *pending = z_waitq_head(&msgq->wait_q);
        if (pending != NULL) {
            // 直接将消息拷贝给消费者
            memcpy(pending->base.swap_data, data, msgq->size);
            z_thread_return_value(pending, 0);
            z_unready_thread(pending);
            msgq->used_msgs--;  // 消费者取走了一条
        }

        k_spin_unlock(&lock, key);
        return 0;
    }

    // 2. 队列满，等待
    int ret = z_pend_curr(&lock, key, &msgq->wait_q, timeout);
    return ret;
}

04 邮箱 (k_mbox)

邮箱支持任意大小的异步消息传递，使用描述符管理数据传输。

// 定义邮箱
K_MBOX_DEFINE(my_mbox);

// 发送消息（异步）
struct k_mbox_msg send_msg;
send_msg.info = 0;
send_msg.size = sizeof(data);
send_msg.rx_source_thread = k_current_get();
send_msg.tx_target_thread = target_tid;
send_msg.tx_block.data = &data;
send_msg.tx_block.size = sizeof(data);

k_mbox_async_put(&my_mbox, &send_msg, &sem);

// 接收消息
struct k_mbox_msg recv_msg;
recv_msg.rx_block.data = buffer;
recv_msg.rx_block.size = sizeof(buffer);
k_mbox_get(&my_mbox, &recv_msg, K_FOREVER);

05 管道 (k_pipe)

管道提供字节流通信，类似 Unix 管道，支持部分读写。

// 定义管道（环形缓冲区）
K_PIPE_DEFINE(my_pipe, 256);  // 256 字节缓冲区

// 写入
size_t bytes_written;
k_pipe_put(&my_pipe, data, sizeof(data), &bytes_written, sizeof(data), K_FOREVER);

// 读取
size_t bytes_read;
k_pipe_get(&my_pipe, buffer, sizeof(buffer), &bytes_read, 1, K_FOREVER);
// 至少读取 1 字节，最多 sizeof(buffer) 字节

// 管道特点：
// 1. 字节流，无消息边界
// 2. 支持部分读写
// 3. 内部使用环形缓冲区
// 4. 支持等待最小字节数

06 Poll 事件多路复用

k_poll 允许线程同时等待多个同步对象，类似 Linux 的 poll/select。

// 定义事件数组
struct k_poll_event events[3] = {
    K_POLL_EVENT_INITIALIZER(K_POLL_TYPE_SEM_AVAILABLE,
                              K_POLL_MODE_NOTIFY_ONLY, &sem1),
    K_POLL_EVENT_INITIALIZER(K_POLL_TYPE_FIFO_DATA_AVAILABLE,
                              K_POLL_MODE_NOTIFY_ONLY, &fifo),
    K_POLL_EVENT_INITIALIZER(K_POLL_TYPE_STATE,
                              K_POLL_MODE_NOTIFY_ONLY, &state),
};

// 等待任一事件
int ret = k_poll(events, 3, K_FOREVER);

// 检查哪个事件就绪
for (int i = 0; i < 3; i++) {
    if (events[i].state == K_POLL_STATE_READY) {
        // 处理事件 i
        events[i].state = K_POLL_STATE_NOT_READY;
    }
}

// 支持的事件类型：
// K_POLL_TYPE_SEM_AVAILABLE       - 信号量可用
// K_POLL_TYPE_FIFO_DATA_AVAILABLE - FIFO 有数据
// K_POLL_TYPE_STATE              - 状态变化
// K_POLL_TYPE_SIGNAL             - 信号触发

07 zbus 消息总线

zbus 是 Zephyr 的发布-订阅消息总线，实现松耦合的线程间通信。

// 定义通道
ZBUS_CHAN_DEFINE accel_chan,
    ZBUS_CHAN_NAME("accel"),
    ZBUS_MSG_INIT(0),
    &accel_data,
    NULL,  // 无用户态访问
    ZBUS_OBSERVERS_EMPTY,
    ZBUS_CHAN_PREPARE_INIT(0)
);

// 定义观察者
static void accel_handler(const struct zbus_channel *chan)
{
    // 处理加速度数据
}

ZBUS_OBS_DEFINE(accel_handler);
ZBUS_CHAN_ADD_OBS(accel_chan, accel_handler, 1);

// 发布消息
struct accel_data data = { .x = 10, .y = 20, .z = 30 };
zbus_chan_pub(&accel_chan, &data, K_FOREVER);

// zbus 优势：
// 1. 发布-订阅模式，松耦合
// 2. 编译时构建，零运行时开销
// 3. 支持观察者优先级
// 4. 支持远程代理（跨设备通信）
