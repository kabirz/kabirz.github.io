02 GPIO 驱动

GPIO 是最基础的外设接口，用于数字输入输出和中断。

GPIO API

// GPIO API（include/zephyr/drivers/gpio.h）

// 获取 GPIO 设备
const struct device *gpio_dev = DEVICE_DT_GET(DT_NODELABEL(gpio0));

// 配置引脚
gpio_pin_configure(gpio_dev, 5, GPIO_OUTPUT | GPIO_ACTIVE_HIGH);

// 设置输出值
gpio_pin_set(gpio_dev, 5, 1);  // 高电平
gpio_pin_set(gpio_dev, 5, 0);  // 低电平

// 读取输入值
int value = gpio_pin_get(gpio_dev, 5);

// 中断配置
gpio_pin_interrupt_configure(gpio_dev, 5, GPIO_INT_EDGE_TOACTIVE);
gpio_init_callback(&button_cb, button_handler, BIT(5));
gpio_add_callback(gpio_dev, &button_cb);

GPIO 中断

// GPIO 中断处理
static void button_handler(const struct device *dev, struct gpio_callback *cb,
                           uint32_t pins)
{
    printk("Button pressed!\n");
}

K_GPIO(button_pin, DT_GPIO_PIN(DT_NODELABEL(button), gpios),
       DT_GPIO_FLAGS(DT_NODELABEL(button), gpios));

void button_isr(void)
{
    int value = gpio_pin_get_dt(&button_pin);
    printk("Button state: %d\n", value);
}

03 SPI 驱动

SPI（Serial Peripheral Interface）是高速同步串行接口。

// SPI API（include/zephyr/drivers/spi.h）

// SPI 配置
const struct device *spi_dev = DEVICE_DT_GET(DT_NODELABEL(spi1));

struct spi_config spi_cfg = {
    .frequency = 1000000,  // 1MHz
    .operation = SPI_WORD_SET(8) | SPI_OP_MODE_MASTER,
    .slave = 0,
    .cs = SPI_CS_CONTROL_PTR_DT(DT_NODELABEL(spi1), 0),
};

// SPI 传输
uint8_t tx_buf[] = {0x01, 0x02, 0x03};
uint8_t rx_buf[3];

struct spi_buf tx_spi_buf = {
    .buf = tx_buf,
    .len = sizeof(tx_buf),
};

struct spi_buf_set tx_bufs = {
    .buffers = &tx_spi_buf,
    .count = 1,
};

struct spi_buf rx_spi_buf = {
    .buf = rx_buf,
    .len = sizeof(rx_buf),
};

struct spi_buf_set rx_bufs = {
    .buffers = &rx_spi_buf,
    .count = 1,
};

spi_transceive(spi_dev, &spi_cfg, &tx_bufs, &rx_bufs);

04 I2C 驱动

// I2C API（include/zephyr/drivers/i2c.h）

const struct device *i2c_dev = DEVICE_DT_GET(DT_NODELABEL(i2c0));

// I2C 读取
uint8_t reg = 0x00;
uint8_t data[6];
i2c_burst_read(i2c_dev, 0x76, reg, data, sizeof(data));

// I2C 写入
uint8_t config[] = {0x24, 0x00};
i2c_burst_write(i2c_dev, 0x76, config[0], &config[1], 1);

// I2C 消息传输
struct i2c_msg msgs[2];
msgs[0].buf = &reg;
msgs[0].len = 1;
msgs[0].flags = I2C_MSG_WRITE;

msgs[1].buf = data;
msgs[1].len = sizeof(data);
msgs[1].flags = I2C_MSG_READ | I2C_MSG_RESTART;

i2c_transfer(i2c_dev, msgs, 2, 0x76);

05 UART 驱动

// UART API（include/zephyr/drivers/serial.h）

const struct device *uart_dev = DEVICE_DT_GET(DT_NODELABEL(uart0));

// UART 配置
struct uart_config uart_cfg = {
    .baudrate = 115200,
    .parity = UART_CFG_PARITY_NONE,
    .stop_bits = UART_CFG_STOP_BITS_1,
    .data_bits = UART_CFG_DATA_BITS_8,
    .flow_ctrl = UART_CFG_FLOW_CTRL_NONE,
};
uart_configure(uart_dev, &uart_cfg);

// 发送数据
char tx_buf[] = "Hello, UART!\n";
uart_tx(uart_dev, tx_buf, sizeof(tx_buf), K_FOREVER);

// 接收数据
uint8_t rx_buf[32];
uart_rx_enable(uart_dev, rx_buf, sizeof(rx_buf), K_FOREVER);

// UART 中断回调
static void uart_isr(const struct device *dev, void *user_data)
{
    uint8_t buf[32];
    int len = uart_fifo_read(dev, buf, sizeof(buf));
    uart_tx(dev, buf, len, K_FOREVER);  // 回显
}

06 传感器驱动

Zephyr 传感器 API 提供统一的传感器接口。

// 传感器 API（include/zephyr/drivers/sensor.h）

const struct device *sensor_dev = DEVICE_DT_GET(DT_NODELABEL(bme280));

// 读取传感器数据
struct sensor_value temp, humidity, pressure;

sensor_sample_fetch(sensor_dev);
sensor_channel_get(sensor_dev, SENSOR_CHAN_AMBIENT_TEMP, &temp);
sensor_channel_get(sensor_dev, SENSOR_CHAN_HUMIDITY, &humidity);
sensor_channel_get(sensor_dev, SENSOR_CHAN_PRESS, &pressure);

printk("温度: %.2f°C, 湿度: %.2f%%, 气压: %.2f hPa\n",
       sensor_value_to_double(&temp),
       sensor_value_to_double(&humidity),
       sensor_value_to_double(&pressure) / 100.0);

// 传感器触发（中断模式）
sensor_trigger_set(sensor_dev,
                   SENSOR_TRIG_THRESHOLD,
                   &trigger_handler);

07 DMA 驱动

// DMA API（include/zephyr/drivers/dma.h）

const struct device *dma_dev = DEVICE_DT_GET(DT_NODELABEL(dma0));

// DMA 传输配置
struct dma_config dma_cfg = {
    .channel_direction = PERIPHERAL_TO_MEMORY,
    .dma_callback = dma_callback,
    .block_count = 1,
    .head_block = &dma_block,
};

// 启动 DMA 传输
dma_config(dma_dev, channel, &dma_cfg);
dma_start(dma_dev, channel);

// DMA 回调
static void dma_callback(const struct device *dev, void *user_data,
                         uint32_t channel, int status)
{
    printk("DMA transfer complete: %d\n", status);
}
