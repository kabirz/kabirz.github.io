01 RTIO 概览

RTIO（Real-Time I/O）是 Zephyr 的异步 I/O 框架，支持零拷贝、批量操作和传感器采集。

02 架构设计

// RTIO 核心数据结构
struct rtio {
    struct rtio_iodev **iovs;    // I/O 设备列表
    struct rtio_sqe *sqes;       // 提交队列
    struct rtio_cqe *cqes;       // 完成队列
    uint32_t sqe_count;          // SQ 大小
    uint32_t cqe_count;          // CQ 大小
};

// SQE（Submission Queue Entry）
struct rtio_sqe {
    uint8_t op;                  // 操作类型
    uint8_t prio;                // 优先级
    uint16_t flags;              // 标志
    struct rtio_iodev *iodev;    // 目标设备
    void *userdata;              // 用户数据
    // 操作特定字段...
};

// CQE（Completion Queue Entry）
struct rtio_cqe {
    int32_t result;              // 操作结果
    void *userdata;              // 用户数据
    struct rtio_sqe *sqe;        // 对应的 SQE
};

08 RTIO API

// 初始化
int rtio_init(struct rtio *r, const struct rtio_config *cfg);

// SQE 操作
struct rtio_sqe *rtio_sqe_acquire(struct rtio *r);
void rtio_sqe_prep_read(struct rtio_sqe *sqe, struct rtio_iodev *iodev, void *buf, size_t len);
void rtio_sqe_prep_write(struct rtio_sqe *sqe, struct rtio_iodev *iodev, const void *buf, size_t len);

// 提交
int rtio_submit(struct rtio *r, uint32_t head);

// CQ 操作
struct rtio_cqe *rtio_cqe_acquire(struct rtio *r);
void rtio_cqe_release(struct rtio *r, struct rtio_cqe *cqe);

// 批量操作
int rtio_sqe_batch(struct rtio *r, struct rtio_sqe *sqes, uint32_t count);
