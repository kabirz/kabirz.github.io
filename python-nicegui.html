01 NiceGUI 概述

NiceGUI 是一个基于 Vue 3 + Quasar Framework 的纯 Python GUI 框架。你无需编写 HTML/CSS/JavaScript，只需写 Python 代码就能构建交互式 Web 界面或桌面应用。

核心架构

# NiceGUI 的技术栈
"""
┌──────────────────────────────────────────────┐
│              你的 Python 代码                  │
│  ui.label / ui.button / ui.input / ...        │
├──────────────────────────────────────────────┤
│           NiceGUI 后端引擎                    │
│  - 自动生成 Vue 3 组件                        │
│  - WebSocket 双向实时通信                     │
│  - 响应式数据绑定                             │
│  - 事件处理系统                               │
├──────────────────────────────────────────────┤
│         前端 (自动管理)                       │
│  - Vue 3 + Quasar Framework                  │
│  - Tailwind CSS 样式                          │
│  - 自动热更新（开发模式）                      │
├──────────────────────────────────────────────┤
│               运行模式                        │
│  - Web 模式: 浏览器访问                      │
│  - 桌面模式: 原生窗口 (配合 pywebview)        │
│  - 混合模式: 同时 Web + 桌面                 │
└──────────────────────────────────────────────┘
"""

# 最小示例
from nicegui import ui

# 创建 UI
ui.label("Hello NiceGUI!")
ui.button("Click me", on_click=lambda: ui.notify("Clicked!"))

# 启动应用
# ui.run()

# 可选参数
ui.run(
    title="My App",
    port=8080,
    host="0.0.0.0",
    dark=False,
    reload=True,       # 开发模式热重载
    show=True,         # 自动打开浏览器
    native=False,      # 是否使用原生窗口
    window_size=(800, 600),
    favicon="🚀",
)

设计理念

# 1. 纯 Python — 无需前端知识
# 2. 声明式 UI — 代码结构与 UI 结构对应
# 3. 响应式更新 — 数据变化自动反映到 UI
# 4. 双向绑定 — 前端交互反向更新 Python 数据

# 与其他 Python GUI 框架对比
"""
框架          类型          学习曲线   安装大小   平台
───────────   ──────────   ────────   ────────   ────────
NiceGUI       Web+桌面      低         小         跨平台
Tkinter       原生桌面       中         内置       Windows优先
PyQt/PySide   原生桌面       高         大         跨平台
Streamlit     Web 仪表盘    极低        小         仅 Web
Gradio        ML 演示       极低        小         仅 Web
Flet          Flutter桌面   中         中         跨平台
Remi          Web 界面      极低        极小       仅 Web
"""

提示:
NiceGUI 特别适合快速原型、内部工具、IoT 仪表盘和简单的桌面应用。它的优势在于响应式数据绑定和自动生成前端代码，但复杂的交互逻辑还是需要深入理解其事件循环模型。生产环境中建议使用 native=False 的 Web 模式，通过反向代理部署。

04 数据绑定

NiceGUI 提供强大的响应式数据绑定系统，数据变化时 UI 自动更新，UI 交互时数据自动同步。

基础绑定

from nicegui import ui

# 基础响应式变量
from nicegui import binding

# 使用 reactive 创建响应式数据
count = ui.reactive(0)

# 自动绑定：数据变化触发 UI 更新
ui.label().bind_text_from(count, "value")

# 手动更新
def increment():
    count.value += 1  # 自动更新所有绑定的 UI
    ui.notify(f"计数: {count.value}")

ui.button("增加", on_click=increment)

# 双向绑定示例
name = ui.reactive("张三")
ui.input("姓名").bind_value(name)  # 双向绑定

# 显示更新
ui.label().bind_text_from(name, "value")

# 复杂对象
class User:
    def __init__(self):
        self.name = "张三"
        self.age = 25

user = User()
ui.label().bind_text_from(user, "name")
ui.input().bind_value(user, "name")
ui.number().bind_value(user, "age")

bind_* 方法

from nicegui import ui

# 绑定文本
label = ui.label()
label.bind_text_from(source_obj, "field_name")

# 绑定值
input_field = ui.input()
input_field.bind_value(source_obj, "field_name")

# 绑定可见性
button = ui.button("动态按钮")
button.bind_visible_from(source_obj, "is_visible")

# 绑定禁用状态
button.bind_enabled_from(source_obj, "is_enabled")

# 绑定样式类
label.bind_classes_from(source_obj, "class_name", {
    True: "text-red",
    False: "text-green",
})

# 绑定样式
label.bind_style_from(source_obj, "color_value", backward=lambda v: f"color: {v}")

# 绑定属性
button.props("color=primary")
button.bind_props_from(source_obj, "button_color", "color")

# 绑定图标
icon = ui.icon("home")
icon.bind_name_from(source_obj, "icon_name")

# 完整示例
class ViewModel:
    def __init__(self):
        self.count = 0
        self.is_even = True
        self.color = "primary"

model = ViewModel()

# 响应式更新
def update_count():
    model.count += 1
    model.is_even = model.count % 2 == 0
    model.color = "positive" if model.is_even else "negative"

# UI 绑定
ui.label().bind_text_from(model, "count")
ui.button("增加", on_click=update_count)
ui.label("偶数").bind_visible_from(model, "is_even")
ui.label("奇数").bind_visible_from(model, "is_even", backward=lambda v: not v)

class Temperature:
    def __init__(self):
        self.celsius = 25

# 显示华氏温度（自动转换）
def to_fahrenheit(c):
    return f"{c * 9/5 + 32:.1f}°F"

def from_fahrenheit(f_str):
    # 从字符串提取数值
    try:
        return (float(f_str.replace("°F", "")) - 32) * 5/9
    except:
        return temp.celsius

自定义事件与定时器

from nicegui import ui
import asyncio

# 定时器
from nicegui import app

# 周期性任务
@ui.page("/timer")
def timer_page():
    counter = ui.reactive(0)

    # 每秒更新
    timer = ui.timer(1.0, lambda: counter.increment())

    ui.label().bind_text_from(counter, "value")

    ui.button("暂停", on_click=timer.deactivate)
    ui.button("恢复", on_click=timer.activate)

# 一次性延迟任务
async def delayed_action():
    await asyncio.sleep(2)
    ui.notify("2 秒后执行")

ui.button("延迟通知", on_click=delayed_action)

# 复杂自定义事件
class Counter:
    def __init__(self):
        self.count = 0
        self.history = []

    def increment(self):
        self.count += 1
        self.history.append(("+", self.count))
        self.emit_event("change")

    def decrement(self):
        self.count -= 1
        self.history.append(("-", self.count))
        self.emit_event("change")

# 自定义 UI 事件
@ui.page("/custom-events")
def custom_events():
    c = Counter()

    # 使用 ui.timer 实现事件驱动
    ui.label("计数器").classes("text-h4")
    count_label = ui.label().bind_text_from(c, "count")

    with ui.row():
        ui.button("-", on_click=c.decrement)
        ui.button("+", on_click=c.increment)

    # 监听自定义变化
    ui.label("历史记录:")
    history_log = ui.column()

    def update_history():
        history_log.clear()
        for op, val in c.history[-5:]:
            with history_log:
                ui.label(f"{op} {val}")

    # 每次更新后刷新历史
    ui.timer(0.5, update_history, once=False)

def custom_dark():
    dark = ui.dark_mode()
    is_dark = ui.reactive(False)

    def toggle_dark():
        is_dark.value = not is_dark.value
        if is_dark.value:
            dark.enable()
        else:
            dark.disable()

    ui.button("切换主题", on_click=toggle_dark)

        data.append(randint(10, 100))

            "data": [list(enumerate(data))],
        })
