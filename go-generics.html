01 泛型演进

Go 从诞生之初就有"泛型呼声"，但直到 Go 1.18（2022 年 3 月）才正式引入泛型。从设计草案到正式发布经历了 10 多年的讨论和反复迭代。

Go 版本里程碑

Go 1.21 新增内置泛型函数

// Go 1.21 新增的三个泛型函数
package main

import (
    "cmp"
    "maps"
    "slices"
)

func main() {
    // min/max：比较任意可排序类型
    x := min(3.14, 2.71)       // float64 - 2.71
    y := max("abc", "xyz")     // string - "xyz"
    z := min(100, 200, 300)    // int - 100

    // clear：清空 map 或切片元素为零值
    m := map[string]int{"a": 1}
    clear(m)

    s := []int{1, 2, 3}
    clear(s)                   // s 变为 [0, 0, 0]

    // cmp 包：可比较类型的排序工具
    cmp.Compare(1, 2)          // -1
    cmp.Less("a", "b")         // true
    cmp.Or(0, 1, 2)            // 返回第一个非零值 - 1

    // slices 包：泛型切片操作
    slices.Sort([]int{3, 1, 2})
    slices.Compact([]int{1, 1, 2, 2, 3})   // [1, 2, 3]

    // maps 包：泛型 map 操作
    maps.Clone(map[string]int{"a": 1})
}
设计哲学: Go 的泛型在不引入类型擦除的前提下实现。编译器为每个类型参数的具体实例化生成专门的代码（类似 C++ 模板实例化），而非运行时装箱拆箱。

02 类型参数

类型参数（Type Parameters）是泛型的核心。Go 使用方括号 [] 声明类型参数，而不是像 C++ 的尖括号 <>。

函数泛型

package main

import "fmt"

// 单个类型参数
func Print[T any](v T) {
    fmt.Println(v)
}

// 多个类型参数
func Pair[A, B any](a A, b B) (A, B) {
    return a, b
}

// 约束类型参数
func Sum[T ~int | ~float64](values []T) T {
    var total T
    for _, v := range values {
        total += v
    }
    return total
}

// 调用方式
func main() {
    Print[string]("hello")   // 显式指定类型参数
    Print(42)                // 类型推导（隐式）

    result := Pair(1, "go")
    fmt.Println(result)

    ints := []int{1, 2, 3}
    fmt.Println(Sum(ints))   // 6
}

类型泛型

// 泛型结构体
type Stack[T any] struct {
    items []T
}

func (s *Stack[T]) Push(item T) {
    s.items = append(s.items, item)
}

func (s *Stack[T]) Pop() (T, bool) {
    if len(s.items) == 0 {
        var zero T
        return zero, false
    }
    item := s.items[len(s.items)-1]
    s.items = s.items[:len(s.items)-1]
    return item, true
}

func main() {
    intStack := Stack[int]{}
    intStack.Push(1)
    intStack.Push(2)

    strStack := Stack[string]{}
    strStack.Push("a")
    strStack.Push("b")

    v, _ := intStack.Pop()
    fmt.Println(v)  // 2
}

方法泛型限制

// 方法不能引入新的类型参数（Go 1.18+ 限制）
type Container[T any] struct {
    value T
}

func (c Container[T]) Get() T {
    return c.value
}

// 编译错误：方法不能有额外的类型参数
// func (c Container[T]) Map[U any](fn func(T) U) Container[U]

// 解决：使用顶层函数
func MapContainer[T, U any](c Container[T], fn func(T) U) Container[U] {
    return Container[U]{value: fn(c.value)}
}

// 泛型切片映射函数
func Map[T, U any](s []T, fn func(T) U) []U {
    result := make([]U, len(s))
    for i, v := range s {
        result[i] = fn(v)
    }
    return result
}

func main() {
    nums := []int{1, 2, 3}
    strs := Map(nums, func(i int) string {
        return fmt.Sprintf("num-%d", i)
    })
    fmt.Println(strs)
}
已知限制: Go 1.18~1.23 均不支持在方法上声明额外的类型参数。这是 Go 泛型设计的主要限制之一，但可以通过泛型函数参数来绕过。

03 约束

约束（Constraints）定义了类型参数允许的类型集合。Go 通过接口类型来定义约束。

内置约束

import "cmp"

// any：所有类型的约束（interface{} 的别名）
func Identity[T any](v T) T { return v }

// comparable：可比较类型（支持 == 和 !=）
func Contains[T comparable](slice []T, target T) bool {
    for _, v := range slice {
        if v == target {
            return true
        }
    }
    return false
}

// cmp.Ordered：可排序类型（Go 1.21）
func Sort[T cmp.Ordered](slice []T) {
    slices.Sort(slice)
}

自定义约束

// 使用接口定义约束
type Numeric interface {
    ~int | ~int8 | ~int16 | ~int32 | ~int64 |
    ~uint | ~uint8 | ~uint16 | ~uint32 | ~uint64 |
    ~float32 | ~float64
}

type Integer interface {
    ~int | ~int8 | ~int16 | ~int32 | ~int64 |
    ~uint | ~uint8 | ~uint16 | ~uint32 | ~uint64
}

type Float interface {
    ~float32 | ~float64
}

// 组合约束
type Number interface {
    Integer | Float
}

// 使用约束
func Add[T Number](a, b T) T {
    return a + b
}

// 带方法的约束
type Stringer interface {
    ~string
    String() string
}

~ 底层类型匹配

type MyInt int
type MyString string

// 严格匹配：不接受 MyInt
func AddStrict[T int](a, b T) T {
    return a + b
}
// AddStrict(MyInt(1), MyInt(2))  // 编译错误

// 宽松匹配：接受底层类型为 int 的所有类型
func AddLoose[T ~int](a, b T) T {
    return a + b
}
func main() {
    AddLoose(MyInt(1), MyInt(2))  // 正确编译
}

// 类型集（Type Set）概念
// ~int 的类型集：int, MyInt, MyCount, ...
// int 的类型集：只有 int 本身

约束字面量

// 内联约束：无需定义具名约束
func Min[T ~int | ~float64](a, b T) T {
    if a < b {
        return a
    }
    return b
}

// 复杂约束组合
func Process[T interface {
    ~[]byte
    Clone() T
}](data T) T {
    return data.Clone()
}

// 近似元素的限制
// ~ 只能用于具体类型，不能用于接口
// ~ 不能用于类型参数

04 类型推导

类型推导（Type Inference）让调用泛型函数时可以省略大部分类型参数，编译器根据函数参数和返回值的上下文自动推导。

函数参数推导

func Map[T, U any](s []T, fn func(T) U) []U {
    result := make([]U, len(s))
    for i, v := range s {
        result[i] = fn(v)
    }
    return result
}

func main() {
    // 完全显式
    Map[int, string]([]int{1, 2, 3}, func(i int) string {
        return fmt.Sprintf("%d", i)
    })

    // 部分推导：编译器从参数推断 int，从返回值推断 string
    Map([]int{1, 2, 3}, func(i int) string {
        return fmt.Sprintf("%d", i)
    })

    // 完全推导（Go 1.21+）
    result := Map([]int{1, 2, 3}, func(i int) string {
        return fmt.Sprintf("%d", i)
    })
    _ = result
}

05 泛型数据容器

在泛型之前，Go 的标准容器使用 interface{}，需要运行时类型断言。泛型让容器变得类型安全且更高效。

泛型 List

package main

import "fmt"

type List[T any] struct {
    head *node[T]
    tail *node[T]
    size int
}

type node[T any] struct {
    value T
    next  *node[T]
}

func (l *List[T]) PushBack(v T) {
    n := &node[T]{value: v}
    if l.tail == nil {
        l.head = n
        l.tail = n
    } else {
        l.tail.next = n
        l.tail = n
    }
    l.size++
}

func (l *List[T]) PushFront(v T) {
    n := &node[T]{value: v, next: l.head}
    l.head = n
    if l.tail == nil {
        l.tail = n
    }
    l.size++
}

func (l *List[T]) ForEach(fn func(T)) {
    for cur := l.head; cur != nil; cur = cur.next {
        fn(cur.value)
    }
}

泛型 Set

type Set[T comparable] struct {
    items map[T]struct{}
}

func NewSet[T comparable]() *Set[T] {
    return &Set[T]{items: make(map[T]struct{})}
}

func (s *Set[T]) Add(v T) {
    s.items[v] = struct{}{}
}

func (s *Set[T]) Remove(v T) {
    delete(s.items, v)
}

func (s *Set[T]) Contains(v T) bool {
    _, ok := s.items[v]
    return ok
}

func (s *Set[T]) Len() int {
    return len(s.items)
}

func (s *Set[T]) Union(other *Set[T]) *Set[T] {
    result := NewSet[T]()
    for v := range s.items {
        result.Add(v)
    }
    for v := range other.items {
        result.Add(v)
    }
    return result
}

func (s *Set[T]) Intersect(other *Set[T]) *Set[T] {
    result := NewSet[T]()
    for v := range s.items {
        if other.Contains(v) {
            result.Add(v)
        }
    }
    return result
}

泛型 OrderedMap

type OrderedMap[K comparable, V any] struct {
    keys   []K
    values map[K]V
}

func NewOrderedMap[K comparable, V any]() *OrderedMap[K, V] {
    return &OrderedMap[K, V]{
        keys:   make([]K, 0),
        values: make(map[K]V),
    }
}

func (m *OrderedMap[K, V]) Set(key K, value V) {
    if _, exists := m.values[key]; !exists {
        m.keys = append(m.keys, key)
    }
    m.values[key] = value
}

func (m *OrderedMap[K, V]) Get(key K) (V, bool) {
    v, ok := m.values[key]
    return v, ok
}

func (m *OrderedMap[K, V]) Delete(key K) {
    if _, ok := m.values[key]; ok {
        delete(m.values, key)
        for i, k := range m.keys {
            if k == key {
                m.keys = append(m.keys[:i], m.keys[i+1:]...)
                break
            }
        }
    }
}

func (m *OrderedMap[K, V]) Iterate(fn func(K, V) bool) {
    for _, k := range m.keys {
        if !fn(k, m.values[k]) {
            break
        }
    }
}

泛型 Heap

type Heap[T any] struct {
    items []T
    less  func(a, b T) bool
}

func NewHeap[T any](less func(a, b T) bool) *Heap[T] {
    return &Heap[T]{
        items: make([]T, 0),
        less:  less,
    }
}

func (h *Heap[T]) Push(v T) {
    h.items = append(h.items, v)
    h.siftUp(len(h.items) - 1)
}

func (h *Heap[T]) Pop() (T, bool) {
    if len(h.items) == 0 {
        var zero T
        return zero, false
    }
    v := h.items[0]
    h.items[0] = h.items[len(h.items)-1]
    h.items = h.items[:len(h.items)-1]
    h.siftDown(0)
    return v, true
}

func (h *Heap[T]) siftUp(i int) {
    for i > 0 {
        parent := (i - 1) / 2
        if h.less(h.items[i], h.items[parent]) {
            h.items[i], h.items[parent] = h.items[parent], h.items[i]
            i = parent
        } else {
            break
        }
    }
}

func (h *Heap[T]) siftDown(i int) {
    n := len(h.items)
    for {
        smallest := i
        left := 2*i + 1
        right := 2*i + 2
        if left < n && h.less(h.items[left], h.items[smallest]) {
            smallest = left
        }
        if right < n && h.less(h.items[right], h.items[smallest]) {
            smallest = right
        }
        if smallest != i {
            h.items[i], h.items[smallest] = h.items[smallest], h.items[i]
            i = smallest
        } else {
            break
        }
    }
}

func (h *Heap[T]) Len() int {
    return len(h.items)
}

// 使用示例
func main() {
    minHeap := NewHeap(func(a, b int) bool { return a < b })
    minHeap.Push(3)
    minHeap.Push(1)
    minHeap.Push(2)
    v, _ := minHeap.Pop()
    fmt.Println(v)  // 1
}

06 泛型算法

泛型使得编写类型安全的算法库成为可能。以下演示常见的泛型算法模式。

排序与搜索

import (
    "cmp"
    "slices"
)

// 通用排序（Go 1.21 slices.Sort 已是泛型）
func Sort[T cmp.Ordered](slice []T) {
    slices.Sort(slice)
}

// 倒序
func SortDesc[T cmp.Ordered](slice []T) {
    slices.SortFunc(slice, func(a, b T) int {
        return cmp.Compare(b, a)
    })
}

// 二分查找
func BinarySearch[T cmp.Ordered](slice []T, target T) (int, bool) {
    left, right := 0, len(slice)-1
    for left <= right {
        mid := left + (right-left)/2
        if slice[mid] == target {
            return mid, true
        }
        if slice[mid] < target {
            left = mid + 1
        } else {
            right = mid - 1
        }
    }
    return -1, false
}

过滤与映射

// 过滤
func Filter[T any](slice []T, predicate func(T) bool) []T {
    result := make([]T, 0, len(slice))
    for _, v := range slice {
        if predicate(v) {
            result = append(result, v)
        }
    }
    return result
}

// 映射
func Map[T, U any](slice []T, transform func(T) U) []U {
    result := make([]U, len(slice))
    for i, v := range slice {
        result[i] = transform(v)
    }
    return result
}

// 归约
func Reduce[T, U any](slice []T, initial U, f func(U, T) U) U {
    result := initial
    for _, v := range slice {
        result = f(result, v)
    }
    return result
}

// 测试
func main() {
    nums := []int{1, 2, 3, 4, 5, 6}

    // 过滤偶数
    evens := Filter(nums, func(n int) bool {
        return n%2 == 0
    })
    fmt.Println(evens)  // [2 4 6]

    // 映射为字符串
    strs := Map(nums, func(n int) string {
        return fmt.Sprintf("n-%d", n)
    })

    // 归约求和
    sum := Reduce(nums, 0, func(acc, n int) int {
        return acc + n
    })
    fmt.Println(sum)  // 21
}

去重与分组

// 去重（stable）
func Unique[T comparable](slice []T) []T {
    seen := make(map[T]struct{})
    result := make([]T, 0, len(slice))
    for _, v := range slice {
        if _, ok := seen[v]; !ok {
            seen[v] = struct{}{}
            result = append(result, v)
        }
    }
    return result
}

// 分组
func GroupBy[T any, K comparable](slice []T, keyFn func(T) K) map[K][]T {
    result := make(map[K][]T)
    for _, v := range slice {
        key := keyFn(v)
        result[key] = append(result[key], v)
    }
    return result
}

// 分区
func Partition[T any](slice []T, predicate func(T) bool) ([]T, []T) {
    matched := make([]T, 0, len(slice)/2)
    unmatched := make([]T, 0, len(slice)/2)
    for _, v := range slice {
        if predicate(v) {
            matched = append(matched, v)
        } else {
            unmatched = append(unmatched, v)
        }
    }
    return matched, unmatched
}

// 扁平化
func Flatten[T any](slices [][]T) []T {
    total := 0
    for _, s := range slices {
        total += len(s)
    }
    result := make([]T, 0, total)
    for _, s := range slices {
        result = append(result, s...)
    }
    return result
}
实时性能: Go 1.21 的 slices 和 maps 包提供了官方泛型算法实现。在大多数情况下优先使用标准库而非自己实现。

08 性能对比

泛型在性能上通常优于接口（interface{}），但可能略低于手写特殊化代码。以下通过基准测试对比三种方案的差异。

性能基准测试

package main

import (
    "testing"
)

// Interface 方案
type IntStackInterface struct {
    items []interface{}
}

func (s *IntStackInterface) Push(v interface{}) {
    s.items = append(s.items, v)
}

func (s *IntStackInterface) Pop() interface{} {
    if len(s.items) == 0 {
        return nil
    }
    v := s.items[len(s.items)-1]
    s.items = s.items[:len(s.items)-1]
    return v
}

// 泛型方案
type IntStackGeneric[T any] struct {
    items []T
}

func (s *IntStackGeneric[T]) Push(v T) {
    s.items = append(s.items, v)
}

func (s *IntStackGeneric[T]) Pop() T {
    v := s.items[len(s.items)-1]
    s.items = s.items[:len(s.items)-1]
    return v
}

// 手写特殊化方案
type IntStack struct {
    items []int
}

func (s *IntStack) Push(v int) {
    s.items = append(s.items, v)
}

func (s *IntStack) Pop() int {
    v := s.items[len(s.items)-1]
    s.items = s.items[:len(s.items)-1]
    return v
}

09 最佳实践与陷阱

最佳实践

// 1. 优先使用标准库泛型
// Go 1.21 的 slices 和 maps 包已经覆盖大部分场景
import (
    "slices"
    "maps"
)

// 2. 使用 cmp.Ordered 而非自定义排序约束
func SortAndSearch[T cmp.Ordered](slice []T, target T) bool {
    slices.Sort(slice)
    _, found := slices.BinarySearch(slice, target)
    return found
}

// 3. 避免不必要的类型参数
// ❌ 不好的设计
type Wrapper[T any] struct {
    Value T
}
func ProcessInt(w Wrapper[int]) {
    fmt.Println(w.Value)
}

// ✅ 直接使用具体类型
type IntWrapper struct {
    Value int
}

// 4. 用类型推导简化调用
// 调用者无需显式指定类型参数
func NewStack[T any]() *Stack[T] {
    return &Stack[T]{}
}
// s := NewStack[int]()  // 简洁

// 5. 在公共 API 中优先使用泛型而非 interface{}
// ❌ 旧 API
func Sort(data sort.Interface)
// ✅ 新 API（泛型）
func Sort[T cmp.Ordered](data []T)

常见陷阱

// 陷阱 1：误用 any 导致运行时错误
func Unmarshal[T any](data []byte) (T, error) {
    var v T
    // 不是所有 T 都实现了 json.Unmarshaler
    err := json.Unmarshal(data, &v)
    return v, err
}

// 正确做法：约束具体的反序列化方式
type JSONable interface {
    ~string | ~[]byte
}

// 陷阱 2：零值陷阱
func First[T any](items []T) T {
    if len(items) == 0 {
        var zero T  // 创建零值（不是 nil！）
        return zero
    }
    return items[0]
}

// Go 1.21 可以使用 clear 清零
func Reset[T any](items []T) {
    clear(items)  // 所有元素归零
}

// 陷阱 3：无法约束方法集合
// 不能这样定义约束：
// type HasString interface {
//     String() string  // OK
//     ~string           // 编译错误：不能混合方法和类型集
// }

// 陷阱 4：约束可以很复杂
type ComplexConstraint[T any] interface {
    ~[]T
    ~string
}
// 这种约束只能匹配同时是 []T 和 string 的类型（不现实）

// 陷阱 5：性能陷阱 - 频繁实例化
// 以下导致大量代码膨胀
type ManyInstances struct {
    a Stack[float64]
    b Stack[float32]
    c Stack[int]
    d Stack[int8]
    e Stack[int16]
    f Stack[int32]
    g Stack[int64]
    h Stack[uint]
    i Stack[string]
}

// 改进：使用接口作为内部实现
type StackCore struct {
    items []interface{}
}
// 但失去类型安全性

何时使用泛型

适合使用泛型不适合使用泛型
通用容器（List、Set、Map、Stack、Queue）业务逻辑中单一类型的函数
通用算法（排序、搜索、映射、过滤、归约）需要类型特化或性能极致优化
数据管道和转换函数枚举类型方法生成
需要类型安全的包装器需要反射和运行时类型检查
减少相同逻辑的重复代码API 设计不需要泛型灵活性
决策指南: 如果一个函数或类型需要为三种以上不同类型重复实现，且这些类型满足相同的约束，就应该用泛型。如果只有一两种类型，手写特殊化代码更简洁高效。
