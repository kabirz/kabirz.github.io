01 k_thread 结构体

struct k_thread 是 Zephyr 内核中最重要的数据结构之一，代表一个内核线程的所有状态信息。

核心字段

// include/zephyr/kernel_structs.h
struct k_thread {
    // 架构相关上下文（必须在偏移量 0）
    struct _thread_arch arch;

    // 内核管理字段
    struct {
        // 线程状态位图
        uint8_t thread_state;

        // 优先级（数值越小优先级越高）
        int8_t prio;

        // 调度锁计数（0=解锁，负数=锁定）
        int8_t sched_locked;

        // 线程选项（_THREAD_* 标志位）
        uint8_t options;

#ifdef CONFIG_THREAD_NAME
        // 线程名（调试用）
        char name[CONFIG_THREAD_MAX_NAME_LEN];
#endif

        // 线程栈指针
        void *stack_ptr;

        // 线程栈大小
        uint32_t stack_size;

        // 用户态线程的栈边界
        k_thread_stack_t *stack_obj;

        // 超时节点（用于 sleep/timeout）
        struct _timeout timeout;

        // 就绪队列节点
        struct _priq_rb *priq;

        // 红黑树节点
        struct rbnode qnode_rb;

        // 时间片计数
        uint32_t slice_count;

        // 时间片到期标志
        bool slice_expired;

#ifdef CONFIG_SMP
        // CPU 亲和性掩码
        uint32_t cpu_mask;
#endif

#ifdef CONFIG_USERSPACE
        // 线程权限
        uint32_t permissions;
#endif
    } base;

    // 等待队列链表（用于 poll 事件）
    sys_dlist_t poll_events;

    // 对象核心（可选，用于 ObjCore 统计）
#ifdef CONFIG_OBJ_CORE_THREAD
    struct k_obj_core obj_core;
#endif
};

线程状态位图

位宏含义
bit 0_THREAD_PENDING线程在等待队列中
bit 1_THREAD_PRESTART线程已创建但未启动
bit 2_THREAD_SUSPENDED线程被挂起
bit 3_THREAD_ABORTING线程正在被中止
bit 4_THREAD_DEAD线程已终止
bit 5_THREAD_QUEUED线程在就绪队列中

02 线程创建

Zephyr 支持动态创建和静态定义两种线程创建方式。

动态创建

// 动态创建线程
K_THREAD_DEFINE(thread_id,        // 线程 ID（宏生成的变量名）
                STACK_SIZE,       // 栈大小
                thread_entry,     // 入口函数
                p1, p2, p3,       // 入口参数
                PRIORITY,         // 优先级
                OPTIONS,          // 选项（K_USER | K_ESSENTIAL）
                START_DELAY);     // 启动延迟

// 或者使用 API 动态创建
struct k_thread my_thread;
K_THREAD_STACK_DEFINE(my_stack, STACK_SIZE);

k_tid_t tid = k_thread_create(
    &my_thread,              // 线程结构体
    my_stack,                // 栈内存
    STACK_SIZE,              // 栈大小
    thread_entry,            // 入口函数
    p1, p2, p3,              // 参数
    PRIORITY,                // 优先级
    OPTIONS,                 // 选项
    K_NO_WAIT                // 超时（立即创建）
);

线程创建内部流程

// kernel/thread.c - k_thread_create 核心实现
k_tid_t z_impl_k_thread_create(struct k_thread *new_thread,
                               k_thread_stack_t *stack,
                               size_t stack_size,
                               k_thread_entry_t entry,
                               void *p1, void *p2, void *p3,
                               int prio, uint32_t options,
                               k_timeout_t start_delay)
{
    // 1. 参数验证
    CHECKIF(new_thread == NULL || stack == NULL || entry == NULL) {
        return NULL;
    }

    // 2. 初始化线程结构体
    (void)memset(new_thread, 0, sizeof(*new_thread));

    // 3. 设置栈
    new_thread->base.stack_ptr = stack;
    new_thread->base.stack_size = stack_size;
    new_thread->base.stack_obj = stack;

    // 4. 设置优先级和选项
    new_thread->base.prio = prio;
    new_thread->base.options = options;

    // 5. 初始化架构上下文
    z_new_thread(new_thread, stack, stack_size, entry, p1, p2, p3, prio, options);

    // 6. 初始化等待队列
    z_waitq_init(&new_thread->base.wait_q);

    // 7. 设置超时（延迟启动）
    z_init_timeout(&new_thread->base.timeout);
    if (!K_TIMEOUT_EQ(start_delay, K_NO_WAIT)) {
        z_add_timeout(&new_thread->base.timeout, z_thread_timeout, start_delay);
    } else {
        // 立即加入就绪队列
        z_ready_thread(new_thread);
    }

    return new_thread;
}

03 协作式 vs 抢占式

Zephyr 支持两种线程调度模式，通过优先级和选项控制。

协作线程

// 协作线程：优先级为负数，不会被同/低优先级线程抢占
K_THREAD_DEFINE(coop_thread, 1024,
                coop_entry, NULL, NULL, NULL,
                -1,             // 负优先级 → 协作式
                0, K_NO_WAIT);

void coop_entry(void *p1, void *p2, void *p3)
{
    while (1) {
        // 处理任务...

        // 主动让出 CPU
        k_yield();

        // 或者休眠一段时间
        k_msleep(10);
    }
}

抢占线程

// 抢占线程：优先级为非负数，可被更高优先级线程抢占
K_THREAD_DEFINE(preempt_thread, 1024,
                preempt_entry, NULL, NULL, NULL,
                5,              // 非负优先级 → 抢占式
                0, K_NO_WAIT);

void preempt_entry(void *p1, void *p2, void *p3)
{
    // 高优先级线程可以随时抢占此线程
    while (1) {
        // 处理任务（不主动让出）
        process_data();
    }
}

对比

特性协作线程抢占线程
优先级范围-N ~ -10 ~ N-1
可被抢占仅被更高优先级抢占可被更高优先级抢占
CPU 使用直到主动让出被时间片或抢占中断
适用场景关键实时任务一般应用逻辑
响应延迟取决于协作点取决于优先级

04 线程栈

每个线程需要独立的栈空间，用于函数调用、局部变量和中断上下文。

栈定义方式

// 方式 1：静态定义栈（编译时分配）
K_THREAD_STACK_DEFINE(my_stack, 1024);

// 方式 2：栈数组（用于动态线程）
K_THREAD_STACK_ARRAY_DEFINE(stack_array, NUM_THREADS, STACK_SIZE);

// 方式 3：动态分配（运行时分配）
// 注意：需要 ARCH_STACK_PTR_ALIGN 对齐
void *stack = k_thread_stack_alloc(STACK_SIZE, K_NO_WAIT);

05 生命周期管理

线程状态转换

// 线程启动
int k_thread_start(k_tid_t thread)
{
    // 将线程从 PRESTART 状态转为 READY
    z_ready_thread(thread);
    z_reschedule_unlocked();
    return 0;
}

// 线程中止
int k_thread_abort(k_tid_t thread)
{
    k_spinlock_key_t key = k_spin_lock(&_sched_spinlock);

    // 标记线程为 ABORTING
    thread->base.thread_state |= _THREAD_ABORTING;

    // 如果线程在等待队列中，移除
    if (z_is_thread_pending(thread)) {
        z_unready_thread(thread);
    }

    // 触发线程清理
    z_thread_abort(thread);

    k_spin_unlock(&_sched_spinlock, key);
    return 0;
}

// 线程挂起
int k_thread_suspend(k_tid_t thread)
{
    // 标记为 SUSPENDED，从就绪队列移除
    z_suspend_thread(thread);
    z_reschedule_unlocked();
    return 0;
}

// 线程恢复
int k_thread_resume(k_tid_t thread)
{
    // 清除 SUSPENDED 标志，重新加入就绪队列
    z_ready_thread(thread);
    z_reschedule_unlocked();
    return 0;
}

06 用户态线程

Zephyr 支持用户态/内核态隔离，用户态线程通过系统调用访问内核服务。

// 启用用户态（CONFIG_USERSPACE + CONFIG_STACKProtection）
// 用户态线程使用 K_USER 选项创建

// 内核侧定义
K_THREAD_DEFINE(user_thread, STACK_SIZE,
                user_entry, NULL, NULL, NULL,
                K_PRIO_PREEMPT(5),
                K_USER,         // 用户态标志
                K_NO_WAIT);

// 用户态线程限制：
// 1. 不能直接访问内核数据结构
// 2. 通过系统调用访问内核 API
// 3. 受 MPU/MMU 保护
// 4. 栈空间独立且受保护

07 静态线程

静态线程在编译时定义，不需要运行时创建开销，适合嵌入式系统。

// 静态线程定义（最常用方式）
// K_THREAD_DEFINE 宏同时定义线程结构体和栈

// 定义线程
#define STACK_SIZE 1024
#define THREAD_PRIO K_PRIO_PREEMPT(5)

K_THREAD_DEFINE(sensor_tid, STACK_SIZE,
                sensor_thread,      // 入口函数
                NULL, NULL, NULL,   // 参数
                THREAD_PRIO,        // 优先级
                0,                  // 选项
                K_NO_WAIT);         // 延迟
