02 核心配置

// FreeRTOSConfig.h 核心配置

// ===== 调度器 =====
#define configUSE_PREEMPTION            1   // 1=抢占式, 0=协作式
#define configUSE_TIME_SLICING          1   // 1=时间片轮转
#define configUSE_PORT_OPTIMISED_TASK_SELECTION  0  // 1=硬件加速选择

// ===== 时钟 =====
#define configCPU_CLOCK_HZ             72000000  // CPU 时钟频率
#define configTICK_RATE_HZ             1000      // tick 频率（1ms）

// ===== 任务 =====
#define configMAX_PRIORITIES           56        // 最大优先级数
#define configMINIMAL_STACK_SIZE       128       // 最小栈大小（字）
#define configMAX_TASK_NAME_LEN        16        // 最大任务名长度
#define configTOTAL_HEAP_SIZE          32768     // 堆总大小（字节）

// ===== 钩子 =====
#define configUSE_IDLE_HOOK            0
#define configUSE_TICK_HOOK            0
#define configUSE_MALLOC_FAILED_HOOK   0
#define configCHECK_FOR_STACK_OVERFLOW 0
