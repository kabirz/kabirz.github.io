01 输入子系统概览

Zephyr 输入子系统提供统一的输入事件处理框架，支持按键、触摸屏、编码器等。

输入设备类型

触摸 (Touch)电容触摸屏、触摸板

02 输入事件

// 输入事件结构
struct input_event {
    const struct device *dev;   // 来源设备
    uint32_t sync;              // 同步标志
    uint16_t type;              // 事件类型
    uint16_t code;              // 事件代码
    int32_t value;              // 事件值
};

// 事件类型
// INPUT_EV_KEY    - 按键事件
// INPUT_EV_REL    - 相对位移（鼠标）
// INPUT_EV_ABS    - 绝对坐标（触摸屏）
// INPUT_EV_MSC    - 杂项事件

// 按键代码
// INPUT_KEY_0 ~ INPUT_KEY_9
// INPUT_KEY_A ~ INPUT_KEY_Z
// INPUT_KEY_UP / DOWN / LEFT / RIGHT
// INPUT_KEY_ENTER / ESCAPE / SPACE

03 输入 API

// 输入 API
#include <zephyr/input/input.h>

// 注册输入回调
static void input_callback(struct input_event *event, void *user_data)
{
    printk("设备: %s, 类型: %d, 代码: %d, 值: %d\n",
           event->dev->name, event->type, event->code, event->value);
}

INPUT_LISTENER_CB_DEFINE(my_listener, input_callback, NULL);

// 绑定到特定设备
const struct device *input_dev = DEVICE_DT_GET(DT_NODELABEL(buttons));
INPUT_LISTENER_DATAGEN(my_listener, input_callback, NULL);
input_register_listener(&my_listener, input_dev);

08 使用示例

// 按键处理示例
#include <zephyr/input/input.h>

// DTS 中定义的按键
// buttons {
//     compatible = "gpio-keys";
//     button_0: button_0 {
//         gpios = <&gpio0 5 (GPIO_PULL_UP | GPIO_ACTIVE_LOW)>;
//         label = "SW0";
//     };
// };

static void button_handler(struct input_event *event, void *user_data)
{
    if (event->type == INPUT_EV_KEY && event->code == INPUT_KEY_0) {
        if (event->value) {
            printk("按钮按下\n");
        } else {
            printk("按钮释放\n");
        }
    }
}

INPUT_LISTENER_DATAGEN(button_listener, button_handler, NULL);

// 注册监听
const struct device *gpio_keys = DEVICE_DT_GET(DT_NODELABEL(buttons));
input_register_listener(&button_listener, gpio_keys);
