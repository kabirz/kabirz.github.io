01 任务控制块 (TCB)

TCB（Task Control Block）是 FreeRTOS 任务管理的核心数据结构，存储在 tasks.c 内部。

TCB 完整结构

// tasks.c 内部定义（简化）
// 实际定义在 tasks.c 第 250-350 行左右

typedef struct tskTaskControlBlock
{
    // ===== 栈相关 =====
    volatile StackType_t * pxTopOfStack;   // 当前栈顶（上下文切换时保存/恢复）
    StackType_t * pxStackBase;             // 栈基地址（用于栈溢出检测）
    configSTACK_DEPTH_TYPE usStackDepth;   // 栈深度（单位：字）

    // ===== 链表节点 =====
    ListItem_t xStateListItem;    // 状态链表节点（就绪/阻塞/挂起）
    ListItem_t xEventListItem;    // 事件链表节点（等待事件时使用）

    // ===== 优先级 =====
    UBaseType_t uxPriority;       // 当前优先级（可能因优先级继承而改变）
    UBaseType_t uxBasePriority;   // 基础优先级（互斥锁释放后恢复）

    // ===== 互斥锁 =====
#if ( configUSE_MUTEXES == 1 )
    UBaseType_t uxMutexesHeld;   // 持有的互斥锁数量
#endif

    // ===== 任务通知 =====
#if ( configUSE_TASK_NOTIFICATIONS == 1 )
    volatile uint32_t ulNotifiedValue[ configTASK_NOTIFICATION_ARRAY_ENTRIES ];
    volatile uint8_t ucNotifyState[ configTASK_NOTIFICATION_ARRAY_ENTRIES ];
#endif

    // ===== 调试信息 =====
    char pcTaskName[ configMAX_TASK_NAME_LEN ];  // 任务名（调试用）

    // ===== 内存分配标记 =====
    uint8_t ucStaticallyAllocated;  // 0=动态, 1=静态栈, 2=静态栈+TCB

    // ===== 栈溢出检测 =====
#if ( configCHECK_FOR_STACK_OVERFLOW > 0 )
    // 栈末尾的魔数（用于检测溢出）
#endif

    // ===== SMP 相关 =====
#if ( configNUMBER_OF_CORES > 1 )
    volatile BaseType_t xTaskRunState;      // 运行状态
    UBaseType_t uxTaskAccessedState;        // 访问状态
    volatile UBaseType_t uxSuspendedSmpSpinlockCount; // SMP 挂起计数
#endif

    // ===== 时间统计 =====
#if ( configGENERATE_RUN_TIME_STATS == 1 )
    uint32_t ulRunTimeCounter;   // CPU 运行时间
#endif
} tskTCB;

TCB 初始化流程

// tasks.c 中的 prvInitialiseTCB() 函数
// 在 xTaskCreate() 或 xTaskCreateStatic() 中调用

static void prvInitialiseTCB( tskTCB * const pxNewTCB,
                               const StackType_t * const puxStackBuffer,
                               configSTACK_DEPTH_TYPE usStackDepth,
                               UBaseType_t uxStackType )
{
    // 1. 初始化栈
    pxNewTCB->pxStackBase = ( StackType_t * ) puxStackBuffer;
    pxNewTCB->usStackDepth = usStackDepth;

#if ( portSTACK_GROWTH < 0 )
    // 向下增长的栈（ARM Cortex-M）
    pxNewTCB->pxTopOfStack = &( puxStackBuffer[ usStackDepth - 1 ] );
    pxNewTCB->pxEndOfStack = ( StackType_t * ) puxStackBuffer;
#else
    // 向上增长的栈
    pxNewTCB->pxTopOfStack = ( StackType_t * ) puxStackBuffer;
    pxNewTCB->pxEndOfStack = &( puxStackBuffer[ usStackDepth - 1 ] );
#endif

    // 2. 初始化链表节点
    listINIT_LIST_ITEM( &( pxNewTCB->xStateListItem ) );
    listINIT_LIST_ITEM( &( pxNewTCB->xEventListItem ) );
    listSET_LIST_ITEM_OWNER( &( pxNewTCB->xStateListItem ), pxNewTCB );
    listSET_LIST_ITEM_OWNER( &( pxNewTCB->xEventListItem ), pxNewTCB );

    // 3. 设置事件列表项优先级（用于按优先级排序）
    listSET_LIST_ITEM_VALUE( &( pxNewTCB->xEventListItem ),
                             configMAX_PRIORITIES - ( TickType_t ) configMAX_PRIORITIES );
}

02 任务创建

FreeRTOS 提供两种创建方式：动态分配和静态分配。

动态创建 (xTaskCreate)

// tasks.c 中的 xTaskCreate() 实现
BaseType_t xTaskCreate( TaskFunction_t pxTaskCode,
                        const char * const pcName,
                        configSTACK_DEPTH_TYPE usStackDepth,
                        void * const pvParameters,
                        UBaseType_t uxPriority,
                        TaskHandle_t * const pxCreatedTask )
{
    tskTCB * pxNewTCB;
    BaseType_t xReturn;

    // 1. 分配栈内存（从堆中）
    StackType_t * pxStackBase = ( StackType_t * ) pvPortMalloc(
        ( ( size_t ) usStackDepth ) * sizeof( StackType_t ) );

    // 2. 分配 TCB（从堆中）
    pxNewTCB = ( tskTCB * ) pvPortMalloc( sizeof( tskTCB ) );

    // 3. 初始化 TCB
    prvInitialiseTCB( pxNewTCB, pxStackBase, usStackDepth, portSTACK_TYPE );

    // 4. 初始化栈（填充魔数 0xa5 用于高水位检测）
#if ( tskSET_NEW_STACKS_TO_KNOWN_VALUE == 1 )
    {
        ( void ) memset( pxStackBase, ( int ) tskSTACK_FILL_BYTE,
                         ( size_t ) usStackDepth * sizeof( StackType_t ) );
    }
#endif

    // 5. 初始化架构上下文（设置初始寄存器值）
    pxNewTCB->pxTopOfStack = pxPortInitialiseStack(
        pxNewTCB->pxTopOfStack, pxStackBase, pxTaskCode, pvParameters );

    // 6. 添加到就绪列表
    prvAddNewTaskToReadyList( pxNewTCB );

    // 7. 返回任务句柄
    * pxCreatedTask = ( TaskHandle_t ) pxNewTCB;
    return pdPASS;
}

静态创建 (xTaskCreateStatic)

// 静态创建 — 不使用堆，所有内存由调用者提供
TaskHandle_t xTaskCreateStatic( TaskFunction_t pxTaskCode,
                                 const char * const pcName,
                                 configSTACK_DEPTH_TYPE usStackDepth,
                                 void * const pvParameters,
                                 UBaseType_t uxPriority,
                                 StackType_t * const puxStackBuffer,
                                 StaticTask_t * const pxTaskBuffer )
{
    tskTCB * pxNewTCB;

    // 使用调用者提供的 TCB 缓冲区
    pxNewTCB = ( tskTCB * ) pxTaskBuffer;

    // 初始化 TCB（使用调用者提供的栈）
    prvInitialiseTCB( pxNewTCB, puxStackBuffer, usStackDepth, portSTACK_TYPE );

    // 标记为静态分配
    pxNewTCB->ucStaticallyAllocated = tskSTATICALLY_ALLOCATED_STACK_AND_TCB;

    // 初始化架构上下文
    pxNewTCB->pxTopOfStack = pxPortInitialiseStack(
        pxNewTCB->pxTopOfStack, puxStackBuffer, pxTaskCode, pvParameters );

    // 添加到就绪列表
    prvAddNewTaskToReadyList( pxNewTCB );

    return ( TaskHandle_t ) pxNewTCB;
}

// 示例：创建两个任务，演示不同优先级
#include "FreeRTOS.h"
#include "task.h"
#include <stdio.h>

// 任务 1：高优先级（LED 闪烁）
void vHighPriorityTask( void * pvParameters )
{
    for ( ;; )
    {
        printf( "High priority task running\\n" );
        vTaskDelay( pdMS_TO_TICKS( 500 ) );
    }
}

// 任务 2：低优先级（传感器读取）
void vLowPriorityTask( void * pvParameters )
{
    for ( ;; )
    {
        printf( "Low priority task reading sensor\\n" );
        vTaskDelay( pdMS_TO_TICKS( 2000 ) );
    }
}

int main( void )
{
    // 创建高优先级任务
    xTaskCreate( vHighPriorityTask,
                 "HighPrio",
                 configMINIMAL_STACK_SIZE * 2,
                 NULL,
                 configMAX_PRIORITIES - 2,  // 高优先级
                 NULL );

    // 创建低优先级任务
    xTaskCreate( vLowPriorityTask,
                 "LowPrio",
                 configMINIMAL_STACK_SIZE * 2,
                 NULL,
                 2,  // 低优先级
                 NULL );

    // 启动调度器（不返回）
    vTaskStartScheduler();

    // 不应到达这里
    for ( ;; ) {}
}

03 任务删除

// tasks.c 中的 vTaskDelete() 实现
void vTaskDelete( TaskHandle_t xTaskToDelete )
{
    tskTCB * pxTCB;

    taskENTER_CRITICAL();
    {
        // 1. 获取要删除的任务 TCB
        pxTCB = prvGetTCBFromHandle( xTaskToDelete );

        // 2. 从就绪/阻塞/挂起列表中移除
        if( uxListRemove( &( pxTCB->xStateListItem ) ) == ( UBaseType_t ) 0 )
        {
            taskRESET_READY_PRIORITY( pxTCB->uxPriority );
        }

        // 3. 如果删除自身，切换到空闲任务
        if( pxTCB == pxCurrentTCB )
        {
            listINSERT_END( &xSuspendedTaskList, &( pxTCB->xStateListItem ) );
            ++uxCurrentNumberOfTasks;
            --uxCurrentNumberOfTasks;
            portPRE_TASK_DELETE_HOOK();
        }
        else
        {
            // 4. 标记为已删除，等待空闲任务回收
            --uxCurrentNumberOfTasks;
            prvDeleteTCB( pxTCB );
        }
    }
    taskEXIT_CRITICAL();
}

05 调度器实现

// tasks.c 中的调度器核心逻辑

// 选择最高优先级任务
// 这是调度决策的核心宏
#define taskSELECT_HIGHEST_PRIORITY_TASK()                                 \
    do {                                                                   \
        UBaseType_t uxTopPriority = uxTopReadyPriority;                    \
                                                                           \
        // 从最高优先级向下搜索非空链表                                  \
        while( listLIST_IS_EMPTY( &( pxReadyTasksLists[ uxTopPriority ] ) ) != pdFALSE ) \
        {                                                                  \
            configASSERT( uxTopPriority );                                 \
            --uxTopPriority;                                               \
        }                                                                  \
                                                                           \
        // 使用循环轮转实现时间片                                       \
        // listGET_OWNER_OF_NEXT_ENTRY 会让同优先级任务轮流执行        \
        listGET_OWNER_OF_NEXT_ENTRY( pxCurrentTCB,                        \
                                     &( pxReadyTasksLists[ uxTopPriority ] ) ); \
        uxTopReadyPriority = uxTopPriority;                                \
    } while( 0 )

// 时间片轮转原理：
// 每个优先级的就绪链表是一个循环链表
// listGET_OWNER_OF_NEXT_ENTRY 会移动到下一个节点
// 这样同优先级任务获得相等的 CPU 时间

    xIdleTaskHandles[ 0 ] = xTaskCreateStatic( prvTaskDeleteTask,
                                                configIDLE_TASK_NAME,

                                          pxTimerTaskStack,
                                          &xTimerTaskTCB );
#endif

    // 3. 启动第一个任务（不返回）
    xPortStartScheduler();

06 优先级管理

FreeRTOS 优先级：数值越大，优先级越高（与 Zephyr 相反）。

// tasks.c 中的优先级相关操作

// 获取任务优先级
UBaseType_t uxTaskPriorityGet( const TaskHandle_t xTask )
{
    tskTCB * pxTCB;

    taskENTER_CRITICAL();
    {
        pxTCB = prvGetTCBFromHandle( xTask );
        return pxTCB->uxPriority;
    }
    taskEXIT_CRITICAL();
}

// 设置任务优先级
void vTaskPrioritySet( TaskHandle_t xTask, UBaseType_t uxNewPriority )
{
    tskTCB * pxTCB;

    taskENTER_CRITICAL();
    {
        pxTCB = prvGetTCBFromHandle( xTask );

        // 如果当前任务优先级被提升（互斥锁），恢复基础优先级
        if( ( pxTCB->uxBasePriority == pxTCB->uxPriority ) == pdTRUE )
        {
            // 未被提升，直接设置新优先级
            pxTCB->uxPriority = uxNewPriority;
            pxTCB->uxBasePriority = uxNewPriority;
        }
        else
        {
            // 被提升了，只更新基础优先级
            pxTCB->uxBasePriority = uxNewPriority;
        }

        // 重新排列链表
        listSET_LIST_ITEM_VALUE( &( pxTCB->xEventListItem ),
                                 configMAX_PRIORITIES - ( TickType_t ) uxNewPriority );
    }
    taskEXIT_CRITICAL();
}

07 栈管理

// 栈溢出检测（tasks.c 内部）

// 方法 1：检查栈指针是否超出范围（configCHECK_FOR_STACK_OVERFLOW = 1）
// 方法 2：检查栈末尾魔数 0xa5 是否被覆盖（configCHECK_FOR_STACK_OVERFLOW = 2）

// 栈高水位查询
UBaseType_t uxTaskGetStackHighWaterMark( TaskHandle_t xTask )
{
    tskTCB * pxTCB = prvGetTCBFromHandle( xTask );
    UBaseType_t uxReturn;

    // 从栈底向上搜索第一个非 0xa5 的位置
    // 这就是栈使用的最高水位
    pxEndOfStack = pxTCB->pxStackBase;

    while( *pxEndOfStack == tskSTACK_FILL_BYTE )
    {
        pxEndOfStack++;
    }

    // 返回栈剩余字数
    uxReturn = ( UBaseType_t ) ( ( ( uint32_t ) pxEndOfStack - ( uint32_t ) pxTCB->pxStackBase ) / sizeof( StackType_t ) );
    return uxReturn;
}

08 任务通知

// 任务通知是最轻量的 IPC 机制
// 每个任务自带一个 32 位通知值（或数组）
// 不需要创建队列，开销最小

// 发送通知
BaseType_t xTaskNotify( TaskHandle_t xTaskToNotify,
                         uint32_t ulValue,
                         eNotifyAction eAction )
{
    tskTCB * pxTCB = xTaskToNotify;

    taskENTER_CRITICAL();
    {
        // 根据动作更新通知值
        switch( eAction )
        {
            case eSetBits:
                pxTCB->ulNotifiedValue[ 0 ] |= ulValue;
                break;
            case eIncrement:
                pxTCB->ulNotifiedValue[ 0 ]++;
                break;
            case eSetValueWithOverwrite:
                pxTCB->ulNotifiedValue[ 0 ] = ulValue;
                break;
            case eSetValueWithoutOverwrite:
                if( pxTCB->ucNotifyState[ 0 ] != taskNOTIFICATION_RECEIVED )
                {
                    pxTCB->ulNotifiedValue[ 0 ] = ulValue;
                }
                break;
        }

        // 标记通知已接收
        pxTCB->ucNotifyState[ 0 ] = taskNOTIFICATION_RECEIVED;

        // 如果任务在等待通知，唤醒它
        if( listIS_CONTAINED_WITHIN( &xSuspendedTaskList,
                                     &( pxTCB->xStateListItem ) ) != pdFALSE )
        {
            ( void ) uxListRemove( &( pxTCB->xStateListItem ) );
            prvAddNewTaskToReadyList( pxTCB );
        }
    }
    taskEXIT_CRITICAL();

    return pdTRUE;
}

10 完整示例

// ===== FreeRTOS 任务管理完整示例 =====
// 演示：创建、优先级、延时、删除、通知

#include "FreeRTOS.h"
#include "task.h"
#include <stdio.h>

// 任务句柄
TaskHandle_t xSensorTaskHandle = NULL;

// 传感器任务（低优先级）
void vSensorTask( void * pvParameters )
{
    uint32_t ulNotificationValue;

    for ( ;; )
    {
        // 等待通知（阻塞，最多等 1000ms）
        if( xTaskNotifyWait( 0, 0, &ulNotificationValue, pdMS_TO_TICKS( 1000 ) ) == pdTRUE )
        {
            printf( "Sensor: received notification 0x%08lX\\n", ulNotificationValue );
        }
        else
        {
            printf( "Sensor: timeout, reading data\\n" );
        }
    }
}

// 控制任务（高优先级）
void vControlTask( void * pvParameters )
{
    uint32_t ulNotificationCount = 0;

    for ( ;; )
    {
        // 每 500ms 发送通知给传感器任务
        vTaskDelay( pdMS_TO_TICKS( 500 ) );
        ulNotificationCount++;
        xTaskNotify( xSensorTaskHandle, ulNotificationCount, eIncrement );
    }
}

// 空闲钩子
void vApplicationIdleHook( void )
{
    // 可以在这里执行低功耗操作
}

// 栈溢出钩子
void vApplicationStackOverflowHook( TaskHandle_t xTask, char * pcTaskName )
{
    printf( "Stack overflow in task: %s\\n", pcTaskName );
    for ( ;; ) {}
}

int main( void )
{
    // 创建传感器任务（低优先级）
    xTaskCreate( vSensorTask,
                 "Sensor",
                 configMINIMAL_STACK_SIZE * 2,
                 NULL,
                 2,
                 &xSensorTaskHandle );

    // 创建控制任务（高优先级）
    xTaskCreate( vControlTask,
                 "Control",
                 configMINIMAL_STACK_SIZE * 2,
                 NULL,
                 4,
                 NULL );

    // 启动调度器
    vTaskStartScheduler();

    for ( ;; ) {}
