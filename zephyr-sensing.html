02 Sensing 框架

Sensing 是 Zephyr 的高层传感管理框架，支持虚拟传感器和数据管道。

// Sensing 框架（subsys/sensing/）
// CONFIG_SENSING=y

// 核心概念：
// 1. 传感器（sensor）- 物理或虚拟传感器
// 2. 管道（pipeline）- 数据处理管道
// 3. 处理器（processor）- 数据变换
// 4. 接收器（receiver）- 数据消费

// 虚拟传感器
// 从物理传感器数据派生新数据
// 例如：融合加速度+陀螺仪 → 旋转矩阵

03 传感器 API

// 传感器 API（include/zephyr/drivers/sensor.h）

// 获取设备
const struct device *dev = DEVICE_DT_GET(DT_NODELABEL(bme280));

// 采样数据
sensor_sample_fetch(dev);

// 获取通道数据
struct sensor_value temp, hum, press;
sensor_channel_get(dev, SENSOR_CHAN_AMBIENT_TEMP, &temp);
sensor_channel_get(dev, SENSOR_CHAN_HUMIDITY, &hum);
sensor_channel_get(dev, SENSOR_CHAN_PRESS, &press);

// 打印数据
printk("温度: %.2f°C\n", sensor_value_to_double(&temp));
printk("湿度: %.2f%%\n", sensor_value_to_double(&hum));
printk("气压: %.2f hPa\n", sensor_value_to_double(&press) / 100.0);

// 非阻塞采样
sensor_sample_fetch-chan(dev, SENSOR_CHAN_AMBIENT_TEMP);

// 批量读取
struct sensor_data采集 {
    struct sensor_value temp;
    struct sensor_value hum;
};
sensor_channel_get(dev, SENSOR_CHAN_AMBIENT_TEMP, &采集.temp);
sensor_channel_get(dev, SENSOR_CHAN_HUMIDITY, &采集.hum);

04 RTIO 异步 I/O

RTIO 提供异步 I/O 框架，用于传感器批量采集和零拷贝数据处理。

// RTIO（subsys/rtio/）
// CONFIG_RTIO=y

// RTIO 核心概念：
// 1. Submission Queue (SQ) - 提交 I/O 请求
// 2. Completion Queue (CQ) - 完成事件
// 3. Executor - 执行 I/O 操作
// 4. Scheduler - I/O 调度

// 传感器 RTIO 集成
// - 异步采样
// - 批量采集
// - 零拷贝传输
// - 多传感器并发

06 触发模式

// 传感器触发
// 替代轮询，使用中断通知

// 触发类型
// SENSOR_TRIG_DATA_READY   - 数据就绪
// SENSOR_TRIG_THRESHOLD    - 阈值触发
// SENSOR_TRIG_TAP          - 敲击检测
// SENSOR_TRIG_FREEFALL     - 自由落体
// SENSOR_TRIG_MOTION       - 运动检测
// SENSOR_TRIG_STATIONARY   - 静止检测

// 设置触发
struct sensor_trigger trig = {
    .type = SENSOR_TRIG_DATA_READY,
    .chan = SENSOR_CHAN_ALL,
};
sensor_trigger_set(dev, &trig, data_ready_handler);

// 触发回调
void data_ready_handler(const struct device *dev,
                        const struct sensor_trigger *trigger)
{
    // ISR 上下文，不能阻塞
    k_sem_give(&data_sem);
}

08 传感器 API

// 采样
int sensor_sample_fetch(const struct device *dev);
int sensor_sample_fetch_chan(const struct device *dev, enum sensor_channel chan);
int sensor_sample_fetch_chan_list(const struct device *dev,
                                  const enum sensor_channel *chan_list, size_t count);

// 数据获取
int sensor_channel_get(const struct device *dev, enum sensor_channel chan,
                       struct sensor_value *val);
int sensor_get_orientation(const struct device *dev, struct sensor_orientation *orient);

// 触发
int sensor_trigger_set(const struct device *dev,
                       const struct sensor_trigger *trig,
                       sensor_trigger_handler_t handler);

// 属性
int sensor_attr_set(const struct device *dev, enum sensor_channel chan,
                    enum sensor_attribute attr, const struct sensor_value *val);
int sensor_attr_get(const struct device *dev, enum sensor_channel chan,
                    enum sensor_attribute attr, struct sensor_value *val);

// 工具函数
double sensor_value_to_double(const struct sensor_value *val);
int64_t sensor_value_to_milli(const struct sensor_value *val);
