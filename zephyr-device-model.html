01 设备模型概览

02 device 结构体

// include/zephyr/device.h
struct device {
    const char *name;              // 设备名称
    const void *config;            // 编译时常量配置（来自设备树）
    const void *api;               // 驱动 API 函数表
    void *data;                    // 驱动运行时数据
    int state;                     // 设备状态

    // 依赖设备句柄（编译时构建）
    device_handle_t *deps;
    int8_t num_deps;

#ifdef CONFIG_PM_DEVICE
    struct pm_device *pm;          // 电源管理
#endif

#ifdef CONFIG_OBJ_CORE_DEVICE
    struct k_obj_core obj_core;    // 对象核心
#endif
};

// 设备句柄（紧凑表示）
typedef int16_t device_handle_t;

04 初始化流程

SYS_INIT 钩子

// 初始化级别（include/zephyr/init.h）
// SYS_INIT_EARLIEST  = -32768
// SYS_INIT_PRE_KERNEL_1 = -1000
// SYS_INIT_PRE_KERNEL_2 = -100
// SYS_INIT_POST_KERNEL = 0
// SYS_INIT_APPLICATION = 100
// SYS_INIT_LATE = 200

// 设备初始化级别：
// PRE_KERNEL_1: 架构初始化后，内核初始化前
// PRE_KERNEL_2: 内核初始化后，调度器启动前
// POST_KERNEL: 调度器启动后
// APPLICATION: 应用代码执行前
// LATE: 所有初始化完成后

依赖解析

// 设备依赖通过设备树自动推断

// 则 B 会在 A 之前初始化

//     spi-device = <&spi1>;  ← 依赖 spi1
// };

05 设备树绑定

设备树（Devicetree）在编译时描述硬件拓扑，生成常量配置数据。

设备树宏

// 设备树访问宏（include/zephyr/devicetree.h）

// 节点标识
DT_NODELABEL(uart0)              // 按标签引用
DT_ALIAS(gpio0)                  // 按别名引用
DT_INST(n, vendor_compatible)    // 按 compatible 实例引用

// 寄存器
DT_REG_ADDR(node_id)             // 基地址
DT_REG_SIZE(node_id)             // 地址范围大小

// 属性
DT_PROP(node_id, prop_name)      // 属性值
DT_NODE_HAS_PROP(node_id, prop)  // 是否有属性
DT_PROP_LEN(node_id, prop)       // 数组属性长度

// 中断
DT_INST_IRQN(n)                  // 中断号
DT_INST_IRQ(n, idx)              // 中断规格

// GPIO
DT_INST_GPIO_PIN(n, gpios, pin)  // GPIO 引脚
DT_INST_GPIO_FLAGS(n, gpios, pin)// GPIO 标志

设备树示例

// DTS 源文件（.dts）
/ {
    chosen {
        zephyr,console = &uart0;
    };

    gpio0: gpio@50000000 {
        compatible = "nordic,nrf-gpio";
        reg = <0x50000000 0x1000>;
        #gpio-cells = <2>;
    };

    i2c0: i2c@40003000 {
        compatible = "nordic,nrf-twi";
        reg = <0x40003000 0x1000>;
        interrupts = <3 0>;
        clock-frequency = ;

        bme280: bme280@76 {
            compatible = "bosch,bme280";
            reg = <0x76>;
        };
    };
};

07 GPIO 驱动示例

// GPIO 驱动注册（简化版）

// 1. 定义驱动配置（从设备树提取）
struct gpio_nrf_config {
    uint32_t base;
    int irq;
};

// 2. 定义驱动数据
struct gpio_nrf_data {
    gpio_callback_t cb;
};

// 3. 驱动 API 实现
static const struct gpio_driver_api gpio_nrf_api = {
    .pin_configure = gpio_nrf_pin_configure,
    .pin_set = gpio_nrf_pin_set,
    .pin_get = gpio_nrf_pin_get,
    .pin_interrupt_configure = gpio_nrf_pin_interrupt_configure,
};

// 4. 设备树绑定宏
#define GPIO_NRF_INIT(n) \
    static struct gpio_nrf_data gpio_nrf_data_##n; \
    DEVICE_DT_INST_DEFINE(n, gpio_nrf_init, NULL, \
                          &gpio_nrf_data_##n, \
                          &gpio_nrf_config_##n, \
                          POST_KERNEL, \
                          CONFIG_GPIO_INIT_PRIORITY, \
                          &gpio_nrf_api);

// 5. 为每个设备树实例生成代码
DT_INST_FOREACH_STATUS_OKAY(GPIO_NRF_INIT);

08 最佳实践

实践说明
使用设备树硬件配置通过设备树定义，不要硬编码
DEVICE_DT_INST_DEFINE基于设备树实例定义设备，支持多实例
设备就绪检查使用 device_is_ready() 检查设备状态
初始化级别选择正确的初始化级别，避免循环依赖
错误处理检查所有 API 返回值
ISR 安全检查 API 是否可在 ISR 中使用
阅读建议:
include/zephyr/device.h 设备模型核心定义。
kernel/device.c 设备初始化框架。
include/zephyr/devicetree.h 设备树访问宏。
drivers/gpio/ GPIO 驱动实现示例。
