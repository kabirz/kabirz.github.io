03 软件定时器

软件定时器在指定时间后执行回调函数，支持单次和周期性模式。

数据结构

// include/zephyr/kernel.h
struct k_timer {
    struct _timeout timeout;         // 超时节点
    k_timer_handler_t handler;       // 回调函数（ISR 上下文）
    void *user_data;                 // 用户数据
    struct k_work_delayable work;    // 可选：与 workqueue 集成
};

使用示例

// 定时器回调（在 ISR 上下文中执行）
void my_timer_handler(struct k_timer *timer)
{
    // 不能调用阻塞 API！
    led_toggle(led_dev);
}

// 定义定时器
K_TIMER_DEFINE(my_timer, my_timer_handler, NULL);

// 启动定时器（100ms 后首次触发，之后每 500ms 周期触发）
k_timer_start(&my_timer, K_MSEC(100), K_MSEC(500));

// 停止定时器
k_timer_stop(&my_timer);

// 单次定时器
K_TIMER_DEFINE(one_shot_timer, one_shot_handler, NULL);
k_timer_start(&one_shot_timer, K_MSEC(1000), K_FOREVER);
// K_FOREVER 表示不周期触发

04 超时队列

超时队列管理所有基于时间的事件，包括定时器、sleep、超时等待等。

// 超时队列内部结构（kernel/include/timeout_q.h）
struct _timeout {
    struct rbnode node;            // 红黑树节点（按截止时间排序）
    z_timer_handler_t fn;          // 到期回调
    void *handle;                  // 回调参数
    uint32_t start;                // 启动时的 tick
    uint32_t duration;             // 超时时长（tick）
};

// 超时队列使用红黑树按到期时间排序
// 插入/删除 O(log n)，查找最近超时 O(1)

// 核心操作
void z_add_timeout(struct _timeout *t, z_timer_handler_t fn, k_timeout_t timeout);
void z_abort_timeout(struct _timeout *t);
void z_remove_timeout(struct _timeout *t);

// 时间点计算
k_timepoint_t sys_timepoint_calc(k_timeout_t timeout);
bool sys_timepoint_expired(k_timepoint_t timepoint);

超时队列工作流程

// 系统时钟中断处理
void z_timer_handler(void)
{
    // 1. 更新 tick 计数
    sys_clock_ticks++;

    // 2. 检查超时队列
    while (!RB_EMPTY(&timeout_tree)) {
        struct _timeout *t = RB_MIN(&timeout_tree);
        if (t->start + t->duration > sys_clock_ticks) {
            break;  // 还没到期
        }

        // 3. 移除超时节点
        RB_REMOVE(&timeout_tree, t);

        // 4. 调用回调
        t->fn(t->handle);
    }

    // 5. 检查是否需要切换线程
    z_sched_expire();
}

05 Tickless 模式

Tickless 模式（CONFIG_TICKLESS_KERNEL）允许系统在空闲时停止周期性 tick 中断，按需唤醒，显著降低功耗。

工作原理

实现

// Tickless 核心逻辑
// 当系统空闲时：
// 1. 查找下一个超时事件
// 2. 计算需要等待的时间
// 3. 设置硬件定时器在该时间点唤醒
// 4. 进入深度休眠（如 WFI/WFE）

// 当有新事件插入超时队列时：
// 1. 如果新事件比当前定时器更早
// 2. 重新编程硬件定时器

// Kconfig 配置
// CONFIG_TICKLESS_KERNEL=y
// CONFIG_TICKLESS_LOW_POWER=y          ← 低功耗模式
// CONFIG_SYS_CLOCK_MIN_SLEEP_CHARS=10  ← 最小睡眠时间

06 高精度计时

Zephyr 支持高精度计时（timing/），用于性能测量和微秒级延迟。

// 高精度计时 API
#include <zephyr/timing/timing.h>

// 读取硬件周期计数器
timing_t start = timing_counter_get();

// 执行操作...
do_something();

timing_t end = timing_counter_get();

// 计算时间差
timing_t delta = timing_cycles_get(&start, &end);

// 转换为纳秒
uint64_t ns = timing_cycles_to_ns(delta);

// 忙等待指定微秒数
k_busy_wait(10);  // 等待 10 微秒

// 精确忙等待
timing_t target = timing_counter_get() + timing_ns_to_cycles(100);
while (timing_counter_get() < target) {
    // 忙等待
}

07 忙等待

// k_busy_wait：忙等待（不释放 CPU）
k_busy_wait(100);  // 等待 100 微秒

// k_sleep：休眠（释放 CPU）
k_sleep(K_MSEC(1));  // 休眠 1 毫秒

// k_msleep：毫秒休眠
k_msleep(10);

// k_usleep：微秒休眠
k_usleep(100);

// 选择指南：
// 等待 < 1ms → k_busy_wait()
// 等待 >= 1ms → k_sleep()
// 需要精确延时 → k_busy_wait()（但占用 CPU）
