01 概述

Cargo 是 Rust 官方构建工具，集成包管理、构建系统、测试运行、文档生成、发布。

主要功能：

命令用途
cargo new创建新项目
cargo build编译
cargo run编译并运行
cargo test运行测试
cargo check快速检查（不生成可执行）
cargo doc生成文档
cargo fmt格式化代码
cargo clippylint
cargo publish发布到 crates.io
cargo update更新依赖
cargo add添加依赖
cargo tree查看依赖树
cargo vendor离线依赖

项目类型：

  • binary：可执行（cargo new 默认）
  • library：库（cargo new --lib）
  • mixed：同时是 lib + 多个 bin
  • workspace：多 crate 共同管理

版本号语义化：

// 语义化版本 (SemVer):
// MAJOR.MINOR.PATCH
// 1.4.2
//  ↑ ↑ ↑
//  │ │ └── 兼容性修复
//  │ └──── 向后兼容的新功能
//  └────── 不兼容的变更

// Cargo 解析：
"1.0.0"    # 等价 ^1.0.0
"=1.0.0"   # 精确版本
">=1.0.0"  # 范围
"~1.4"     # >=1.4.0, <1.5.0
"*"        # 任意

// 预发布：
"1.0.0-beta.1"

// 来源：
[dependencies]
foo = "1.0"                                  # crates.io
foo = { git = "https://github.com/x/foo" }   # Git
foo = { path = "../foo" }                    # 本地路径
foo = { version = "1.0", registry = "myreg" } # 自定义 registry

03 依赖管理

// Cargo.lock：
//   锁定所有依赖的精确版本
//   保证构建可重现
//
// binary 项目：提交 Cargo.lock
// library 项目：不提交 Cargo.lock

// 依赖解析：
//   默认用 SemVer 兼容性
//   使用 PubGrub 算法（最近）

// Cargo.lock 内容：
[[package]]
name = "serde"
version = "1.0.197"
source = "registry+https://github.com/rust-lang/crates.io-index"
checksum = "3ac1..."

// 添加依赖：
cargo add serde
cargo add serde --features derive
cargo add tokio --features full
cargo add proptest --dev

// 升级：
cargo update                    # 升级所有（符合 SemVer）
cargo update -p serde           # 仅升级 serde
cargo update -p serde --precise 1.0.198

// 查看依赖树：
cargo tree
cargo tree -i serde   # 反向（谁依赖 serde）
cargo tree -e features   # 显示 features

// 检查过期依赖：
cargo install cargo-outdated
cargo outdated

依赖类型：

[dependencies]
# 普通依赖
serde = "1"

# 平台条件
[target.'cfg(unix)'.dependencies]
nix = "0.27"

[target.'cfg(windows)'.dependencies]
winapi = { version = "0.3", features = ["winuser"] }

# Optional 依赖（受 feature 控制）
[dependencies]
tokio = { version = "1", optional = true }

[features]
async = ["dep:tokio"]   # 1.60+ 语法

# workspace 共享
[dependencies]
serde.workspace = true

04 Feature Flags

// Features 是条件编译开关
[features]
default = ["std"]
std = []
no_std = []
tls = ["dep:rustls", "dep:tokio-rustls"]
async = ["dep:tokio", "dep:async-trait"]
json = ["dep:serde", "dep:serde_json"]

[dependencies]
rustls = { version = "0.21", optional = true }
tokio = { version = "1", optional = true }
serde = { version = "1", optional = true }

// 使用：
cargo build --no-default-features --features async,tls

// 条件编译：
#[cfg(feature = "tls")]
mod tls;

#[cfg(all(feature = "tls", feature = "async"))]
async fn fetch() { ... }

// Features 规则（Rust 1.60+）：
// 1. 不能命名冲突
// 2. dep: 前缀明确可选依赖
// 3. feature 累加（union）

// 反例（旧式，不推荐）：
[features]
tls = []   # 隐式启用 rustls（如果有 optional dep）

Feature Unification（重要）

// Cargo 把所有 features 合并
// 即使只在某依赖中用，最终都启用

// 例子：
// A 依赖 C with feature "x"
// B 依赖 C with feature "y"
// 你依赖 A + B
// → C 编译时启用 x + y

// 后果：
//   - 依赖图兼容性高
//   - 不能"按需编译"
//   - 不能在 lib 中区分 caller

// 替代方案：
//   - 把 C 拆成多个 crate
//   - 或用 generic + trait（运行期多态）

05 Workspace

// Workspace: 多个 crate 共享 Cargo.lock + target
// 
// myproject/
// ├── Cargo.toml          # workspace 根
// ├── crates/
// │   ├── core/
// │   │   └── Cargo.toml
// │   ├── api/
// │   │   └── Cargo.toml
// │   ├── web/
// │   │   └── Cargo.toml
// │   └── utils/
// │       └── Cargo.toml

// 根 Cargo.toml:
[workspace]
members = ["crates/*"]
resolver = "2"

[workspace.dependencies]
serde = "1"
tokio = "1"
anyhow = "1"

[workspace.package]
version = "0.1.0"
edition = "2021"
authors = ["Team"]
license = "MIT"

// 子 crate:
[package]
name = "myproject-core"
version.workspace = true
edition.workspace = true

[dependencies]
serde.workspace = true
myproject-utils = { path = "../utils" }

// 优势：
//   - 统一依赖版本
//   - 共享 target 目录
//   - cargo build 一次构建所有
//   - 跨 crate 重构方便

06 构建系统

// 编译流程：
// 1. 解析 Cargo.toml
// 2. 下载依赖到 ~/.cargo
// 3. 解析依赖图
// 4. 编译每个 crate（依赖顺序）
// 5. 链接

// 增量编译：
//   cargo 只重新编译修改的部分
//   fingerprint 机制（哈希每个文件）
//   metadata/.fingerprint/

// 并行编译：
//   依赖图无依赖的 crate 可并行
//   proc-macro 必须串行

// Build Script（build.rs）：
//   编译前执行
//   用途：
//     - 编译 C 代码
//     - 生成代码
//     - 检测环境

// build.rs:
fn main() {
    cc::Build::new()
        .file("src/native.c")
        .compile("native");
    
    println!("cargo:rustc-link-lib=ssl");
    println!("cargo:rerun-if-changed=src/native.c");
}

// 输出（环境变量）：
//   OUT_DIR - 生成代码目录
//   CARGO_FEATURE_* - features
//   TARGET - 目标三元组

// 生成代码：
fn main() {
    let out_dir = std::env::var("OUT_DIR").unwrap();
    let dest = format!("{}/generated.rs", out_dir);
    std::fs::write(&dest, "pub fn generated() {}").unwrap();
}

// 在 lib.rs：
include!(concat!(env!("OUT_DIR"), "/generated.rs"));

交叉编译：

// 目标三元组：arch-vendor-os-env
// 例：x86_64-unknown-linux-gnu

// 添加 target：
rustup target add aarch64-unknown-linux-gnu

// 编译：
cargo build --target aarch64-unknown-linux-gnu

// 需要正确 linker：
// ~/.cargo/config.toml:
[target.aarch64-unknown-linux-gnu]
linker = "aarch64-linux-gnu-gcc"

// 常见 target：
//   x86_64-unknown-linux-gnu   (Linux x86_64)
//   aarch64-apple-darwin       (Apple Silicon macOS)
//   x86_64-pc-windows-msvc     (Windows MSVC)
//   wasm32-unknown-unknown     (WebAssembly)
//   armv7-unknown-linux-gnueabihf (Raspberry Pi)

07 测试

// 1. 单元测试（同模块内）
pub fn add(a: i32, b: i32) -> i32 { a + b }

#[cfg(test)]
mod tests {
    use super::*;
    
    #[test]
    fn test_add() {
        assert_eq!(add(1, 2), 3);
    }
}

// 2. 集成测试（tests/ 目录）
// tests/integration.rs:
#[test]
fn it_works() {
    assert_eq!(2 + 2, 4);
}

// 3. 文档测试（doctest）
/// Adds two numbers.
/// 
/// # Example
/// 
/// ```
/// use mycrate::add;
/// assert_eq!(add(1, 2), 3);
/// ```
pub fn add(a: i32, b: i32) -> i32 { a + b }

// 运行：
cargo test                  # 所有测试
cargo test test_add         # 名字过滤
cargo test --test integration  # 仅某个集成测试文件
cargo test --doc            # 仅 doctest
cargo test -- --nocapture   # 显示 println!
cargo test -- --test-threads=4

// 测试组织：
#[cfg(test)]
mod tests {
    use super::*;
    
    mod happy_path {
        use super::*;
        #[test]
        fn basic() { ... }
    }
    
    mod error_cases { ... }
}

// Property-based testing
// [dev-dependencies]
// proptest = "1.0"

proptest! {
    #[test]
    fn test_add_commutative(a in -1000i32..1000, b in -1000i32..1000) {
        prop_assert_eq!(add(a, b), add(b, a));
    }
}

// 基准测试：
// 1. Rust 内置 #[bench] 已弃用
// 2. criterion（推荐）
//    [dev-dependencies]
//    criterion = { version = "0.5", features = ["html_reports"] }
//    [[bench]]
//    name = "my_benchmark"
//    harness = false

// benches/my_benchmark.rs:
use criterion::{black_box, criterion_group, criterion_main, Criterion};

fn benchmark(c: &mut Criterion) {
    c.bench_function("add", |b| b.iter(|| add(black_box(1), black_box(2))));
}

criterion_group!(benches, benchmark);
criterion_main!(benches);

// 运行：
cargo bench

08 发布

// 发布到 crates.io：
// 1. 注册：https://crates.io
// 2. 获取 API token
// 3. cargo login <token>

// 发布：
cargo publish

// 检查（dry-run）：
cargo publish --dry-run

// 流程：
// 1. 验证 Cargo.toml
// 2. 打包源码（git tracked files）
// 3. 上传到 crates.io
// 4. 自动索引
// 5. 文档自动生成（docs.rs）

// 版本撤销（5 个版本内）：
cargo yank --vers 1.0.0

// 注意：
// - 不可删除已发布版本
// - 仅可 yank（防止新项目用）
// - 已有的依赖仍可用

// 私有 registry：
// ~/.cargo/config.toml:
[registries]
myreg = { index = "https://myreg.com/git/index" }

// 使用：
[dependencies]
private-lib = { version = "0.1", registry = "myreg" }

// 发布到私有 registry：
cargo publish --registry myreg

// 替代 registry：
//  私有：Cloudsmith、JFrog、AWS CodeArtifact
//   自托管：katano、cenaproxy

09 Profiles

// Profile：控制构建配置
[profile.dev]
opt-level = 0
debug = true
incremental = true
overflow-checks = true

[profile.release]
opt-level = 3
debug = false
lto = true           # 链接时优化
panic = "abort"      # panic 不 unwind
codegen-units = 1    # 编译单元数（更少更优化）
strip = true         # 去符号

[profile.bench]
inherits = "release"

[profile.test]
inherits = "dev"

// 自定义 profile（1.57+）：
[profile.release-fast]
inherits = "release"
lto = false
codegen-units = 16
opt-level = 2

// 使用：
cargo build --profile release-fast

// LTO 详解：
//   lto = false (默认)    - 不 LTO
//   lto = "thin"          - 轻量 LTO（快）
//   lto = "fat" or true   - 完整 LTO（最慢，最小二进制）
//   适合：最终发布
//   不适合：开发

// strip：
//   strip = false  - 不去
//   strip = true   - 全去（symbol + debuginfo）
//   strip = "debuginfo"
//   strip = "symbols"

// 跨 profile 依赖：
// cargo build 时不同 profile 的 dep
[profile.dev.package."*"]
opt-level = 3   # 依赖在 dev 下用 opt 3（快运行）

条件编译：

// cfg 属性：
#[cfg(target_os = "linux")]
mod linux;

#[cfg(target_os = "windows")]
mod windows;

#[cfg(feature = "tls")]
use rustls;

#[cfg(debug_assertions)]
fn debug_only() {}

#[cfg(test)]
mod tests;

// 条件依赖：
[target.'cfg(target_os = "linux")'.dependencies]
nix = "0.27"

// cfg_if 宏（更可读）：
// [dependencies]
// cfg-if = "1"
cfg_if::cfg_if! {
    if #[cfg(target_os = "linux")] {
        mod linux;
    } else if #[cfg(target_os = "windows")] {
        mod windows;
    } else {
        mod other;
    }
}

10 高级用法

[source.crates-io]

// 5. Workspace 内 link
// 子 crate 间互相引用
[dependencies]
myproject-utils = { path = "../utils" }

// 6. examples/ 目录

cargo run --example my_example

Cargo 总结:
1. 一站式工具：build、test、doc、publish
2. SemVer：依赖管理基于语义版本
3. Workspace：多 crate 项目
4. Features：条件编译
5. Profiles：dev/release 配置分离
6. crates.io：全球公共 registry
7. 扩展丰富：cargo-edit、cargo-watch 等
Cargo 是 Rust 生态成功的关键之一，让依赖管理"开箱即用"。
