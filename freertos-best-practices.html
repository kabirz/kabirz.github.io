01 栈溢出

// 问题：栈空间不足导致数据损坏

// 解决方案 1：启用栈溢出检测
#define configCHECK_FOR_STACK_OVERFLOW  2

// 解决方案 2：增加栈大小
// 使用 uxTaskGetStackHighWaterMark() 检查栈使用情况
UBaseType_t uxHighWaterMark = uxTaskGetStackHighWaterMark( xTask );
// 返回值 = 栈剩余的最小字数
// 如果返回值很小（如 < 20），说明栈快溢出了

// 解决方案 3：使用静态分配（更可预测）
StaticTask_t xTaskBuffer;
StackType_t xStack[ 256 ];  // 增加栈大小
xTaskCreateStatic( ..., xStack, &xTaskBuffer );

02 优先级反转

// 问题：高优先级任务被低优先级任务间接阻塞

// 错误示例：使用二值信号量做互斥（无优先级继承）
// SemaphoreHandle_t xSem = xSemaphoreCreateBinary();
// xSemaphoreTake(xSem, ...);  // 高优先级任务等待
// xSemaphoreGive(xSem);       // 低优先级任务释放
// 问题：低优先级任务不会被提升优先级

// 正确示例：使用互斥锁（自动优先级继承）
SemaphoreHandle_t xMutex = xSemaphoreCreateMutex();
xSemaphoreTake( xMutex, portMAX_DELAY );   // 高优先级任务等待
xSemaphoreGive( xMutex );                   // 低优先级任务释放
// 优点：低优先级任务的优先级会临时提升

04 性能优化

// 优化 1：使用任务通知（最轻量的 IPC）
// 比队列/信号量更快
xTaskNotify( xTaskHandle, ulValue, eSetBits );
xTaskNotifyWait( 0, 0, &ulValue, timeout );

// 优化 2：使用 vTaskDelayUntil 实现精确周期
// vTaskDelay() 会补偿执行时间，不够精确
TickType_t xLastWakeTime = xTaskGetTickCount();
while ( 1 ) {
    vTaskDelayUntil( &xLastWakeTime, pdMS_TO_TICKS( 10 ) );
    // 精确每 10ms 执行一次
}

05 调试技巧

// 技巧 1：启用 configASSERT

// 技巧 2：使用 vTaskList() 查看任务状态

06 最佳实践

编码规范

实践说明
启用 configASSERT验证配置和运行时状态
使用静态分配避免堆碎片和分配失败
启用栈溢出检测configCHECK_FOR_STACK_OVERFLOW = 2
优先使用任务通知最轻量的 IPC 机制
ISR 中使用 FromISR API避免在 ISR 中使用阻塞 API
限制中断禁用时间保持系统响应性
使用 vTaskDelayUntil实现精确周期任务
优先使用互斥锁避免优先级反转
检查栈使用情况定期使用 uxTaskGetStackHighWaterMark()
使用适当的优先级时间关键任务使用高优先级

常见错误

错误后果解决方案
ISR 中调用阻塞 API系统崩溃使用 FromISR 版本
栈大小不足栈溢出增加栈大小或使用静态分配
使用二值信号量做互斥优先级反转使用互斥锁
临界区过长系统响应性差缩短临界区
删除正在运行的任务系统崩溃先挂起再删除
