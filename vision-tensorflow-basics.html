01 TensorFlow 简介与安装

TensorFlow 是一个端到端的开源机器学习平台，最初由 Google Brain 团队开发。TensorFlow 2.x 引入了 Eager Execution 模式，让用户可以像写普通 Python 代码一样构建和调试模型。Keras 作为其高级 API，提供简洁易用的模型构建接口。

TensorFlow 的生态系统包括 TensorFlow Lite（移动端）、TensorFlow.js（浏览器端）、TensorFlow Serving（生产部署）和 TensorFlow Extended（MLOps 流水线）。

02 Eager Execution 与 Tensor

Eager Execution 是 TensorFlow 2.x 的默认模式。每行代码立即执行并返回具体数值，无需先构建计算图再启动 Session。这使得调试更加直观。

03 tf.function 图编译

@tf.function 装饰器将 Python 函数编译为 TensorFlow 计算图，在保留 Eager 调试便利性的同时获得静态图的性能优势。AutoGraph 机制会自动将 Python 控制流转换为 TensorFlow 图操作。

@tf.function
def compute(a, b):
    """编译为计算图，加速重复调用"""
    c = tf.matmul(a, b)
    d = tf.nn.relu(c)
    return d

# 第一次调用触发图追踪（tracing），后续调用复用
result = compute(
    tf.random.normal([128, 256]),
    tf.random.normal([256, 64])
)

# 查看图
print(compute.experimental_get_tracing_count())  # 追踪次数
print(compute.get_concrete_function(
    tf.TensorSpec([None, 256], tf.float32),
    tf.TensorSpec([256, None], tf.float32)
).graph.as_graph_def())

AutoGraph 控制流

@tf.function
def dynamic_sum(x, n):
    """Python 控制流自动转为图操作"""
    total = tf.constant(0)
    for i in tf.range(n):  # 使用 tf.range 而非 python range
        total = total + x[i]
    return total

@tf.function
def conditional(x):
    if tf.reduce_mean(x) > 0:
        return x * 2
    else:
        return x - 1

# 注：tf.function 内避免使用 Python 原生 list/dict 突变

输入签名

@tf.function(
    input_signature=[
        tf.TensorSpec([None, 28, 28, 1], tf.float32),
        tf.TensorSpec([None], tf.int32),
    ]
)
def predict(model, x, y):
    return model(x, training=False)

# 签名限制了输入形状和类型，避免反复 tracing
# 使用 None 表示可变批量维度
限制:
@tf.function 内不建议使用 Python 原生的 print()（使用 tf.print()），副作用操作（如列表追加）需谨慎。每个不同的输入形状会触发一次新的 tracing，大量不同形状会导致性能下降。

Functional API 支持非顺序拓扑结构：多输入、多输出、残差连接、共享层。它适用于需要分支和拼接的复杂网络。

# 第二个卷积块

# 全连接

残差连接示例

def residual_block(x, filters, kernel_size=3):
    shortcut = x

    x = keras.layers.Conv2D(filters, kernel_size,
                            padding="same")(x)
    x = keras.layers.BatchNormalization()(x)
    x = keras.layers.ReLU()(x)
    x = keras.layers.Conv2D(filters, kernel_size,
                            padding="same")(x)
    x = keras.layers.BatchNormalization()(x)

    # 残差连接
    if shortcut.shape[-1] != filters:
        shortcut = keras.layers.Conv2D(
            filters, 1, padding="same"
        )(shortcut)

    x = keras.layers.add([x, shortcut])
    x = keras.layers.ReLU()(x)
    return x

inputs = keras.Input(shape=(32, 32, 3))
x = keras.layers.Conv2D(64, 7, strides=2, padding="same")(inputs)
x = keras.layers.BatchNormalization()(x)
x = keras.layers.ReLU()(x)

x = residual_block(x, 64)
x = residual_block(x, 128)
x = residual_block(x, 256)

x = keras.layers.GlobalAveragePooling2D()(x)
outputs = keras.layers.Dense(10, activation="softmax")(x)
model = keras.Model(inputs=inputs, outputs=outputs)

多输入多输出

# 两个输入
input_text = keras.Input(shape=(512,), name="text")

    inputs=[input_text, input_image],
    outputs=[output_class, output_reg],
)

    loss={"class": "sparse_categorical_crossentropy",

    metrics={"class": "accuracy"},
)

06 自定义层与模型

通过继承 keras.layers.Layer 和 keras.Model，可以实现完全自定义的层和训练逻辑。

自定义层

class MyDense(keras.layers.Layer):
    def __init__(self, units, activation=None, **kwargs):
        super().__init__(**kwargs)
        self.units = units
        self.activation = keras.activations.get(activation)

    def build(self, input_shape):
        # build 在第一次调用时执行，可通过 input_shape 动态创建权重
        self.w = self.add_weight(
            shape=(input_shape[-1], self.units),
            initializer="glorot_uniform",
            trainable=True,
            name="kernel",
        )
        self.b = self.add_weight(
            shape=(self.units,),
            initializer="zeros",
            trainable=True,
            name="bias",
        )
        super().build(input_shape)

    def call(self, inputs):
        y = tf.matmul(inputs, self.w) + self.b
        if self.activation:
            y = self.activation(y)
        return y

    def get_config(self):
        config = super().get_config()
        config.update({
            "units": self.units,
            "activation": keras.activations.serialize(
                self.activation
            ),
        })
        return config

# 使用自定义层
layer = MyDense(128, activation="relu")

自定义模型（子类化）

class MLP(keras.Model):
    def __init__(self, hidden_dims=[256, 128], num_classes=10, **kwargs):
        super().__init__(**kwargs)
        self.flatten = keras.layers.Flatten()
        self.dense1 = keras.layers.Dense(hidden_dims[0],
                                         activation="relu")
        self.dropout = keras.layers.Dropout(0.3)
        self.dense2 = keras.layers.Dense(hidden_dims[1],
                                         activation="relu")
        self.out = keras.layers.Dense(num_classes,
                                      activation="softmax")

    def call(self, inputs, training=False):
        x = self.flatten(inputs)
        x = self.dense1(x)
        x = self.dropout(x, training=training)
        x = self.dense2(x)
        return self.out(x)

# 使用
model = MLP()
model.compile(
    optimizer=keras.optimizers.Adam(1e-3),
    loss="sparse_categorical_crossentropy",
    metrics=["accuracy"],
)
model.fit(x_train, y_train, epochs=10)

自定义训练循环

model = MLP()
optimizer = keras.optimizers.Adam(1e-3)
loss_fn = keras.losses.SparseCategoricalCrossentropy()
train_acc = keras.metrics.SparseCategoricalAccuracy()

for epoch in range(10):
    for batch in range(num_batches):
        x_batch = x_train[batch * 32: (batch + 1) * 32]
        y_batch = y_train[batch * 32: (batch + 1) * 32]

        with tf.GradientTape() as tape:
            logits = model(x_batch, training=True)
            loss = loss_fn(y_batch, logits)

        grads = tape.gradient(loss, model.trainable_weights)
        optimizer.apply_gradients(
            zip(grads, model.trainable_weights)
        )
        train_acc.update_state(y_batch, logits)

    print(f"Epoch {epoch}: "
          f"Loss = {loss.numpy():.4f}, "
          f"Acc = {train_acc.result().numpy():.4f}")
    train_acc.reset_state()

def augment_image(image, label):
    """在 tf.data 流水线中进行数据增强"""
    image = tf.image.random_flip_left_right(image)
    image = tf.image.random_brightness(image, max_delta=0.2)
    image = tf.image.random_contrast(image, lower=0.8, upper=1.2)
    image = tf.image.random_saturation(image, lower=0.8, upper=1.2)

    # 归一化
    image = tf.cast(image, tf.float32) / 255.0
    return image, label

# 应用增强

自定义训练循环

@tf.function
def train_step(x, y):
    with tf.GradientTape() as tape:
        logits = model(x, training=True)
        loss = loss_fn(y, logits)
        loss += sum(model.losses)   # 加入正则化损失

    grads = tape.gradient(loss, model.trainable_weights)
    optimizer.apply_gradients(zip(grads, model.trainable_weights))

    train_acc.update_state(y, logits)
    return loss

@tf.function
def val_step(x, y):
    logits = model(x, training=False)
    val_acc.update_state(y, logits)
    return loss_fn(y, logits)

# 训练循环
for epoch in range(num_epochs):
    for x_batch, y_batch in train_ds:
        loss = train_step(x_batch, y_batch)

    for x_batch, y_batch in val_ds:
        val_loss = val_step(x_batch, y_batch)

    print(f"Epoch {epoch}: "
          f"train_acc={train_acc.result():.4f}, "
          f"val_acc={val_acc.result():.4f}")

    train_acc.reset_state()
    val_acc.reset_state()

09 回调函数

Keras 回调在训练的不同阶段（epoch 开始/结束、batch 结束）被调用，用于模型检查点、早停、学习率调整和日志记录。

callbacks = [
    # 模型检查点
    keras.callbacks.ModelCheckpoint(
        filepath="best_model.weights.h5",
        monitor="val_accuracy",
        save_best_only=True,
        save_weights_only=True,
        mode="max",
        verbose=1,
    ),

    # 早停
    keras.callbacks.EarlyStopping(
        monitor="val_loss",
        patience=10,
        restore_best_weights=True,
        min_delta=1e-4,
        mode="min",
    ),

    # 学习率衰减
    keras.callbacks.ReduceLROnPlateau(
        monitor="val_loss",
        factor=0.5,
        patience=5,
        min_lr=1e-6,
        verbose=1,
    ),

    # TensorBoard
    keras.callbacks.TensorBoard(
        log_dir="./logs",
        histogram_freq=1,       # 每 epoch 记录直方图
        write_graph=True,       # 写计算图
        write_images=True,      # 写权重可视化
        update_freq="epoch",    # "batch" 或 "epoch"
    ),

    # 学习率日志
    keras.callbacks.LearningRateScheduler(
        schedule=lambda epoch: 1e-3 * (0.95 ** epoch)
    ),

    # 自定义回调
    class CustomCallback(keras.callbacks.Callback):
        def on_epoch_end(self, epoch, logs=None):
            if logs["val_accuracy"] > 0.95:
                print(f"\nReached 95% accuracy at epoch {epoch}")
                self.model.stop_training = True

        def on_batch_end(self, batch, logs=None):
            if batch % 100 == 0:
                print(f"Batch {batch}: loss = {logs['loss']:.4f}")
]

history = model.fit(
    train_ds,
    validation_data=val_ds,
    epochs=100,
    callbacks=callbacks,
)

导出为 TensorFlow Lite

# 转换模型
converter = tf.lite.TFLiteConverter.from_saved_model(
    "saved_model/my_model"
)

# 量化优化
converter.optimizations = [tf.lite.Optimize.DEFAULT]

# 指定输入输出形状（可选）
converter.target_spec.supported_ops = [
    tf.lite.OpsSet.TFLITE_BUILTINS,
    tf.lite.OpsSet.SELECT_TF_OPS,
]

tflite_model = converter.convert()

# 保存 .tflite 文件
with open("model.tflite", "wb") as f:
    f.write(tflite_model)

# 运行 TFLite 推理
interpreter = tf.lite.Interpreter(model_content=tflite_model)
interpreter.allocate_tensors()
input_details = interpreter.get_input_details()
output_details = interpreter.get_output_details()

interpreter.set_tensor(input_details[0]["index"], input_data)
interpreter.invoke()
output = interpreter.get_tensor(output_details[0]["index"])
