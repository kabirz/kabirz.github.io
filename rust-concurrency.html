01 概述

Rust 的"无畏并发"（Fearless Concurrency）：编译期阻止数据竞争，让多线程编程不再可怕。

核心保证：

  • 无数据竞争（Send/Sync + 所有权规则）
  • 并发 bug 变成编译错误
  • 零开销抽象
  • 易组合（async/sync 互操作）

主要 API：

分类类型用途
线程std::threadOS 线程
共享状态Mutex<T>互斥锁
共享状态RwLock<T>读写锁
共享所有权Arc<T>原子引用计数
同步Condvar条件变量
同步Barrier, Once一次性同步
消息传递mpsc, crossbeam通道
原子AtomicBool/I32/Usize/Ptr无锁原子操作

02 Thread

use std::thread;
use std::time::Duration;

// 创建线程
let handle = thread::spawn(|| {
    for i in 1..10 {
        println!("hi {} from thread", i);
        thread::sleep(Duration::from_millis(1));
    }
});

for i in 1..5 {
    println!("hi {} from main", i);
    thread::sleep(Duration::from_millis(1));
}

// 等待线程结束
handle.join().unwrap();

// move 闭包（获取所有权）
let data = vec![1, 2, 3];
let handle = thread::spawn(move || {
    println!("{:?}", data);   // data 被 move 进线程
});
// println!("{:?}", data);   // ❌ data 已 move
handle.join().unwrap();

// 线程属性
use std::thread::Builder;
let handle = Builder::new()
    .name("worker-1".to_string())
    .stack_size(4 * 1024 * 1024)   // 4 MB
    .spawn(|| { /* ... */ })
    .unwrap();

// 当前线程信息
let name = thread::current().name().unwrap_or("unnamed");
let id = thread::current().id();

// park / unpark（轻量同步）
thread::park();   // 阻塞当前线程
// 另一线程
thread::current().unpark();   // 唤醒

// thread_local!
use std::cell::RefCell;
thread_local! {
    static COUNTER: RefCell<u32> = RefCell::new(1);
}
COUNTER.with(|c| {
    *c.borrow_mut() = 2;
});

03 Send/Sync 详解

// Send: 类型可以跨线程转移所有权
//   T: Send 表示 T 可以从一个线程 move 到另一个线程

// Sync: &T 可以跨线程共享
//   T: Sync 表示多线程可以同时持有 &T

// 关系：T: Sync ⟺ &T: Send

// 自动派生规则：
//   struct Foo { a: T, b: U }
//   Foo: Send ⟺ T: Send && U: Send
//   Foo: Sync ⟺ T: Sync && U: Sync

// 不是 Send/Sync 的类型：
//   Rc<T>          - 不是 Send（非原子）
//   Cell<T>        - 不是 Sync
//   RefCell<T>     - 不是 Sync
//   *const T/*mut T - 默认不 Send/Sync

// 强制实现（unsafe）：
struct MyPtr(*mut u8);
unsafe impl Send for MyPtr {}
unsafe impl Sync for MyPtr {}

// spawn 要求：F: Send + 'static, T: Send + 'static
let data = Rc::new(5);
thread::spawn(move || {
    println!("{}", data);   // ❌ Rc 不是 Send
});

let data = Arc::new(5);
thread::spawn(move || {
    println!("{}", data);   // ✅ Arc 是 Send
});

// 'static 限制：
//   不是说真的活一辈子
//   是说"借用没有非 'static 的引用"
//   实际：要么拥有数据，要么 borrow 'static

04 Mutex / RwLock

// Mutex<T> — 互斥锁
use std::sync::Mutex;

let m = Mutex::new(5);

{
    // lock() 返回 Result，防止 poison
    let mut num = m.lock().unwrap();
    *num = 6;
}   // 自动 unlock（RAII）

// 与 C/Rust lock 区别：
//   lock() 返回 MutexGuard，drop 时自动 unlock
//   不能忘记 unlock

// poison：
//   线程 panic 时持有 lock → Mutex 中毒
//   后续 lock 返回 Err
//   防止数据处于不一致状态

// RwLock<T> — 读写锁
use std::sync::RwLock;
let lock = RwLock::new(5);

// 多个 read
{
    let r1 = lock.read().unwrap();
    let r2 = lock.read().unwrap();
    println!("{}, {}", *r1, *r2);
}

// 独占 write
{
    let mut w = lock.write().unwrap();
    *w += 1;
}

// RwLock vs Mutex：
//   RwLock: 多读单写，读多写少场景
//   Mutex: 简单，写多场景
//   实际：现代 Mutex 性能已接近 RwLock

// 公平性：
//   std Mutex 不保证公平（写饥饿可能）
//   parking_lot Mutex 支持公平锁

// 第三方实现：
//   parking_lot - 性能更好，API 类似
//   spin        - 自旋锁（仅特定场景）

parking_lot 库：

// [dependencies]
// parking_lot = "0.12"

use parking_lot::Mutex;
let m = Mutex::new(5);
let mut g = m.lock();   // 不返回 Result（无 poison）
*g = 6;

// 优势：
//   - 性能更好（自旋 + park 组合）
//   - API 更简洁
//   - 无 poison 概念
//   - 支持 const fn new
//   - 支持 try_lock_for / lock_for

// 劣势：
//   - 第三方依赖
//   - 失去 poison 安全网

05 Arc 引用计数

// Arc<T> — 原子引用计数（多线程）
use std::sync::Arc;

let data = Arc::new(vec![1, 2, 3]);

// 在主线程：Arc 计数 = 1
let data_clone = Arc::clone(&data);
// 计数 = 2

thread::spawn(move || {
    println!("{:?}", data_clone);   // 计数 = 2 (this thread)
}).join().unwrap();
// 线程结束，计数 = 1

// 完整例子：多线程共享可变状态
let counter = Arc::new(Mutex::new(0));
let mut handles = vec![];

for _ in 0..10 {
    let counter = Arc::clone(&counter);
    let handle = thread::spawn(move || {
        let mut num = counter.lock().unwrap();
        *num += 1;
    });
    handles.push(handle);
}

for h in handles {
    h.join().unwrap();
}

println!("Result: {}", *counter.lock().unwrap());   // 10

// Arc<Mutex<T>> 是 Rust 中的"共享可变状态"标准模式
// Arc 提供共享所有权
// Mutex 提供内部可变性

// Weak<T> — 弱引用
use std::sync::Weak;
let strong = Arc::new(5);
let weak: Weak<_> = Arc::downgrade(&strong);

// weak 不增加强引用计数
// strong_count + weak_count 都为 0 时才释放

// 防止循环引用：
struct Node {
    parent: Weak<RefCell<Node>>,    // 弱引用，避免循环
    children: Vec<Arc<RefCell<Node>>>,
}

// upgrade() 返回 Option<Arc>
if let Some(strong_again) = weak.upgrade() {
    println!("{}", strong_again);
}

Arc vs Rc：

RcArc
引用计数非原子原子（atomic）
线程安全否是
性能略快略慢
跨线程❌✅

06 Condvar 条件变量

use std::sync::{Arc, Mutex, Condvar};
use std::thread;

let pair = Arc::new((Mutex::new(false), Condvar::new()));
let pair2 = Arc::clone(&pair);

// 等待线程
thread::spawn(move || {
    let (lock, cvar) = &*pair2;
    let mut started = lock.lock().unwrap();
    while !*started {
        started = cvar.wait(started).unwrap();
        // 自动 unlock + sleep，唤醒后重新 lock
    }
});

// 通知线程
thread::spawn(move || {
    let (lock, cvar) = &*pair;
    let mut started = lock.lock().unwrap();
    *started = true;
    cvar.notify_one();   // 唤醒一个等待者
    // cvar.notify_all() 唤醒所有
});

// 用法：
// 1. wait_while 简化循环
let guard = cvar.wait_while(lock, |g| !*g).unwrap();

// 2. wait_timeout
let (guard, timeout) = cvar.wait_timeout(guard, Duration::from_secs(1)).unwrap();

// Barrier — 同步点
use std::sync::Barrier;
let barrier = Arc::new(Barrier::new(3));
let mut handles = vec![];
for _ in 0..3 {
    let b = Arc::clone(&barrier);
    handles.push(thread::spawn(move || {
        // 做一些工作
        b.wait();   // 所有线程都到这里才继续
        // 后续工作
    }));
}
for h in handles { h.join().unwrap(); }

// Once — 一次性初始化
use std::sync::Once;
static INIT: Once = Once::new();
let val = INIT.call_once(|| {
    // 只执行一次
    expensive_init()
});

07 通道 (Channels)

// "Don't communicate by sharing memory; share memory by communicating"

use std::sync::mpsc;
use std::thread;

// 创建通道
let (tx, rx) = mpsc::channel();

// 发送端
thread::spawn(move || {
    let vals = vec!["hi", "from", "thread"];
    for v in vals {
        tx.send(v).unwrap();
    }
});

// 接收端（主线程）
for received in rx {
    println!("Got: {}", received);
}

// 多生产者
let (tx, rx) = mpsc::channel();
let tx1 = tx.clone();
thread::spawn(move || { tx1.send("from 1").unwrap(); });
thread::spawn(move || { tx.send("from 2").unwrap(); });

// 同步通道（有界，缓冲为 0）
let (tx, rx) = mpsc::sync_channel(0);
// send 会阻塞直到 recv

// sync_channel(N) — 容量 N 的有界通道
let (tx, rx) = mpsc::sync_channel(100);

// rx 是 Iterator
let (tx, rx) = mpsc::channel();
thread::spawn(move || {
    for i in 0..10 { tx.send(i).unwrap(); }
    // tx drop 时关闭
});
let collected: Vec<_> = rx.iter().collect();

// try_recv / recv_timeout
match rx.try_recv() {
    Ok(v) => println!("got {}", v),
    Err(_) => println!("empty"),
}

08 原子操作

use std::sync::atomic::{AtomicBool, AtomicI32, AtomicUsize, AtomicPtr, Ordering};

// 基本原子类型
let counter = AtomicUsize::new(0);
counter.fetch_add(1, Ordering::SeqCst);
println!("{}", counter.load(Ordering::SeqCst));

let flag = AtomicBool::new(false);
flag.store(true, Ordering::Release);
let old = flag.swap(false, Ordering::AcqRel);

let ptr = AtomicPtr::new(std::ptr::null_mut());

// 操作：
//   load / store
//   swap
//   compare_exchange / compare_exchange_weak（CAS）
//   fetch_add / fetch_sub / fetch_and / fetch_or / fetch_xor
//   fetch_max / fetch_min / fetch_update

// CAS（compare-and-swap）：
let v = AtomicI32::new(5);
match v.compare_exchange(5, 10, Ordering::SeqCst, Ordering::SeqCst) {
    Ok(old) => println!("swapped from {}", old),
    Err(actual) => println!("failed, was {}", actual),
}

// fetch_update（loop CAS）
v.fetch_update(SeqCst, SeqCst, |x| Some(x * 2));

// 无锁栈示例：
use std::sync::atomic::{AtomicPtr, Ordering};
struct Node<T> { data: T, next: *mut Node<T> }
struct Stack<T> { head: AtomicPtr<Node<T>> }

impl<T> Stack<T> {
    fn push(&self, data: T) {
        let node = Box::into_raw(Box::new(Node { data, next: std::ptr::null_mut() }));
        loop {
            let head = self.head.load(Ordering::Relaxed);
            unsafe { (*node).next = head; }
            if self.head.compare_exchange_weak(head, node, Ordering::Release, Ordering::Relaxed).is_ok() {
                break;
            }
        }
    }
}

09 内存序

// Ordering 控制原子操作周围的内存可见性

// 5 种 Ordering：
//   Relaxed       - 只保证本操作原子，无其他同步
//   Release       - 之前的读写对获取同一变量的线程可见
//   Acquire       - 看到 Release 写入后，所有之前的读写都可见
//   AcqRel        - Acquire + Release（read-modify-write）
//   SeqCst        - 全局顺序一致（最强、最贵）

// 用例：

// 1. 简单计数（Relaxed 足够）
let counter = AtomicUsize::new(0);
counter.fetch_add(1, Ordering::Relaxed);

// 2. 发布订阅（Release/Acquire 配对）
let data: AtomicBool = AtomicBool::new(false);
let ready = AtomicBool::new(false);

// 线程 A
data.store(true, Ordering::Relaxed);   // 写数据
ready.store(true, Ordering::Release);  // 发出"准备好了"

// 线程 B
while !ready.load(Ordering::Acquire) {}
// 此处能看到 data 是 true
println!("{}", data.load(Ordering::Relaxed));

// 3. 全局顺序（SeqCst）
// 罕见需要，多数 Release/Acquire 足够

// 选择指南：
//   简单计数 → Relaxed
//   标志位 / flag → Release（写）+ Acquire（读）
//   复杂场景 → 先用 SeqCst，profile 后降级
//   几乎不用 SeqCst（90% 用 Relaxed + Acquire/Release）
使用建议:
如果你不确定用哪个 Ordering：
用 SeqCst（最强但安全）。
99% 场景 Release + Acquire 够用。
Relaxed 仅用于不依赖顺序的简单计数。
错误的 Ordering 不会编译报错，但可能导致难以调试的并发 bug。

并发总结:
1. Send/Sync：编译期阻止数据竞争
2. Mutex / RwLock：共享可变状态（RAII）
3. Arc：跨线程共享所有权
4. Channels：消息传递替代共享内存
5. Atomic + Ordering：无锁编程
6. Scoped threads：1.63+，简化借用
Rust 的并发原语 + 类型系统让"无畏并发"成为可能。
