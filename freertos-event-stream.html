01 事件组

事件组允许任务等待多个事件位的组合，支持"任意"或"全部"匹配。

// 创建事件组
EventGroupHandle_t xEventGroup = xEventGroupCreate();

// 设置位
xEventGroupSetBits(xEventGroup, BIT_0 | BIT_2);

// 等待位（任意匹配）
EventBits_t uxBits = xEventGroupWaitBits(
    xEventGroup,
    BIT_0 | BIT_1,    // 等待的位
    pdFALSE,          // 退出时不清除
    pdFALSE,          // 任意位匹配
    pdMS_TO_TICKS(100)
);

// 等待位（全部匹配）
uxBits = xEventGroupWaitBits(xEventGroup, BIT_0 | BIT_1,
                              pdTRUE, pdTRUE, pdMS_TO_TICKS(100));

03 流缓冲

// 流缓冲 — 字节流，无消息边界
// 单生产者单消费者，零拷贝

StreamBufferHandle_t xStreamBuf = xStreamBufferCreate(256, 10);

// 发送（ISR 安全）
xStreamBufferSend(xStreamBuf, data, len, pdMS_TO_TICKS(100));

// 接收
size_t xReceivedBytes = xStreamBufferReceive(xStreamBuf, buf, bufSize,
                                              pdMS_TO_TICKS(100));

04 消息缓冲

// 消息缓冲 — 保留消息边界
// 每条消息前加 4 字节长度头

MessageBufferHandle_t xMsgBuf = xMessageBufferCreate(256);

// 发送消息
xMessageBufferSend(xMsgBuf, msg, msgLen, pdMS_TO_TICKS(100));

// 接收完整消息
size_t xReceived = xMessageBufferReceive(xMsgBuf, buf, bufSize,
                                          pdMS_TO_TICKS(100));
