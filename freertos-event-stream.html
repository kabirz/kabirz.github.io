02 事件设置

// event_groups.c 中的 xEventGroupSetBits()
EventBits_t xEventGroupSetBits( EventGroupHandle_t xEventGroup,
                                 const EventBits_t uxBitsToSet )
{
    EventGroup_t * pxEventBits = xEventGroup;
    List_t * pxTasksWaitingForBits = &( pxEventBits->xTasksWaitingForBits );
    BaseType_t xConstraintAlreadyMet = pdFALSE;

    taskENTER_CRITICAL();
    {
        // 1. 设置事件位
        pxEventBits->uxEventBits |= uxBitsToSet;

        // 2. 检查等待列表中的任务
        const ListItem_t * pxIterator = listGET_HEAD_ENTRY( pxTasksWaitingForBits );

        while( listGET_END( pxTasksWaitingForBits ) != pxIterator )
        {
            // 3. 获取等待的任务
            const tskTCB * pxTCB = listGET_LIST_ITEM_OWNER( pxIterator );
            EventBits_t uxBitsToCheck = listGET_LIST_ITEM_VALUE( pxIterator );

            // 4. 检查等待条件是否满足
            EventBits_t uxBitsWaitCondition = uxBitsToCheck & EventBitsCLEAR;
            EventBits_t uxBitsToSatisfyWaitFor = uxBitsToCheck & EventBitsWAIT_ALL;

            // 5. 如果满足，唤醒任务
            if( ( uxBitsWaitCondition & pxEventBits->uxEventBits ) == uxBitsToSatisfyWaitFor )
            {
                // 从等待列表中移除
                listCURRENT_LIST_ITEM_VALUE( pxIterator ) = 0;
                xTaskRemoveFromEventList( pxIterator );
            }

            pxIterator = listGET_NEXT( pxIterator );
        }
    }
    taskEXIT_CRITICAL();

    return pxEventBits->uxEventBits;
}

03 事件等待

// event_groups.c 中的 xEventGroupWaitBits()
EventBits_t xEventGroupWaitBits( EventGroupHandle_t xEventGroup,
                                  const EventBits_t uxBitsToWaitFor,
                                  const BaseType_t xClearOnExit,
                                  const BaseType_t xWaitForAllBits,
                                  TickType_t xTicksToWait )
{
    EventBits_t uxBitsToSet = 0;

    taskENTER_CRITICAL();
    {
        // 1. 检查事件位是否已满足
        const EventBits_t uxCurrentEventBits = xEventGroup->uxEventBits;

        if( xWaitForAllBits == pdFALSE )
        {
            // 任意位匹配
            if( ( uxCurrentEventBits & uxBitsToWaitFor ) != 0 )
            {
                uxBitsToSet = uxCurrentEventBits & uxBitsToWaitFor;
            }
        }
        else
        {
            // 全部位匹配
            if( ( uxCurrentEventBits & uxBitsToWaitFor ) == uxBitsToWaitFor )
            {
                uxBitsToSet = uxCurrentEventBits & uxBitsToWaitFor;
            }
        }

        if( uxBitsToSet != 0 )
        {
            // 2. 事件已满足，清除位
            if( xClearOnExit == pdTRUE )
            {
                xEventGroup->uxEventBits &= ~uxBitsToSet;
            }
        }
        else
        {
            // 3. 事件未满足，阻塞等待
            vTaskPlaceOnEventList( &( pxEventBits->xTasksWaitingForBits ),
                                   uxBitsToWaitFor | ( xWaitForAllBits << EventBitWaitAllBit ),
                                   xTicksToWait );
        }
    }
    taskEXIT_CRITICAL();

    return uxBitsToSet;
}

// stream_buffer.c 内部结构
typedef struct StreamBufferDef_t
{
    volatile size_t xTail;           // 读取位置
    volatile size_t xHead;           // 写入位置
    volatile size_t xMessagesWaiting; // 缓冲区中的字节数
    size_t xLength;                  // 缓冲区总长度
    size_t xTriggerLevel;            // 触发级别
    TaskHandle_t xTaskWaitingToReceive; // 等待接收的任务
    TaskHandle_t xTaskWaitingToSend;    // 等待发送的任务
    uint8_t *pucBuffer;              // 缓冲区指针
    struct StreamBufferDef_t *pxTaskWaitingToSend; // 发送等待任务
} StreamBuffer_t;

06 流缓冲操作

// stream_buffer.c 中的发送操作
size_t xStreamBufferSend( StreamBufferHandle_t xStreamBuffer,
                           const void *pvItemToWrite,
                           size_t xDataLengthBytes,
                           TickType_t xTicksToWait )
{
    StreamBuffer_t *pxStreamBuffer = xStreamBuffer;
    size_t xRequiredSpace, xSpace;

    // 1. 计算所需空间
    xRequiredSpace = xDataLengthBytes;

    // 2. 检查是否有足够空间
    xSpace = xStreamBufferSpacesAvailable( pxStreamBuffer );

    if( xSpace <= pxStreamBuffer->xLength - pxStreamBuffer->xMessagesWaiting )
    {
        // 3. 有空间：拷贝数据
        xDataLengthBytes = prvWriteBytesToBuffer( pxStreamBuffer, pvItemToWrite, xDataLengthBytes );
        pxStreamBuffer->xMessagesWaiting += xDataLengthBytes;

        // 4. 检查触发级别
        if( pxStreamBuffer->xMessagesWaiting >= pxStreamBuffer->xTriggerLevel )
        {
            // 5. 唤醒等待接收的任务
            xTaskNotifyGive( pxStreamBuffer->xTaskWaitingToReceive );
        }
    }
    else
    {
        // 6. 空间不足，阻塞等待
        if( xTicksToWait != 0 )
        {
            vTaskSuspendAll();
            vTaskPlaceOnEventList( &..., xTicksToWait );
            xTaskResumeAll();
        }
    }

    return xDataLengthBytes;
}

09 完整示例

// ===== FreeRTOS 事件组完整示例 =====

#include "FreeRTOS.h"
#include "task.h"
#include "event_groups.h"
#include <stdio.h>

// 定义事件位
#define BIT_SENSOR_READY   ( 1 << 0 )
#define BIT_NETWORK_READY  ( 1 << 1 )
#define BIT_ALL_READY      ( BIT_SENSOR_READY | BIT_NETWORK_READY )

// 事件组句柄
EventGroupHandle_t xSystemEventGroup;

// 传感器任务
void vSensorTask( void * pvParameters )
{
    // 初始化传感器...
    vTaskDelay( pdMS_TO_TICKS( 100 ) );

    // 设置传感器就绪事件
    xEventGroupSetBits( xSystemEventGroup, BIT_SENSOR_READY );
    printf( "Sensor ready\\n" );

    for ( ;; )
    {
        // 读取传感器数据...
        vTaskDelay( pdMS_TO_TICKS( 500 ) );
    }
}

// 网络任务
void vNetworkTask( void * pvParameters )
{
    // 初始化网络...
    vTaskDelay( pdMS_TO_TICKS( 200 ) );

    // 设置网络就绪事件
    xEventGroupSetBits( xSystemEventGroup, BIT_NETWORK_READY );
    printf( "Network ready\\n" );

    for ( ;; )
    {
        // 处理网络数据...
        vTaskDelay( pdMS_TO_TICKS( 500 ) );
    }
}

// 主任务（等待所有事件）
void vMainTask( void * pvParameters )
{
    // 等待传感器和网络都就绪
    EventBits_t uxBits = xEventGroupWaitBits(
        xSystemEventGroup,
        BIT_ALL_READY,    // 等待的位
        pdFALSE,          // 退出时不清除
        pdTRUE,           // 等待所有位
        portMAX_DELAY     // 无限等待
    );

    if( ( uxBits & BIT_ALL_READY ) == BIT_ALL_READY )
    {
        printf( "All systems ready!\\n" );
    }

    for ( ;; )
    {
        // 正常运行...
        vTaskDelay( pdMS_TO_TICKS( 1000 ) );
    }
}

int main( void )
{
    // 创建事件组
    xSystemEventGroup = xEventGroupCreate();

    // 创建任务
    xTaskCreate( vSensorTask, "Sensor", 128, NULL, 2, NULL );
    xTaskCreate( vNetworkTask, "Network", 128, NULL, 2, NULL );
    xTaskCreate( vMainTask, "Main", 128, NULL, 3, NULL );

    vTaskStartScheduler();

    for ( ;; ) {}
}
