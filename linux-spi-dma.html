1. 概述与背景

DMA（Direct Memory Access，直接内存访问）是一种允许硬件设备绕过 CPU 直接访问系统内存的技术。在 Linux 内核的 SPI 子系统中，DMA 传输是提高大数据量传输效率的关键机制。

本文以 NXP i.MX 系列 SoC 的 SPI 控制器驱动（drivers/spi/spi-imx.c）为实例，深入分析 Linux 内核中 SPI 设备驱动如何使用 DMA 框架进行高效数据传输。

为什么需要 DMA？
在没有 DMA 的情况下，CPU 需要逐字节地从内存读取数据并写入 SPI 控制器，或者从控制器读取数据写入内存。这种方式对于大数据量传输会占用大量 CPU 时间。DMA 允许 SPI 控制器直接从内存读取/写入数据，CPU 只需要设置好传输参数并等待传输完成即可。

2.2 DMA 在 SPI 中的角色

当使用 DMA 传输时，SPI 控制器驱动需要实现以下关键操作：

1DMA 通道请求：在驱动 probe 时通过 dma_request_chan() 请求 TX/RX 通道

2缓冲区映射：使用 dma_map_single() 将内核缓冲区映射为 DMA 可访问地址

3传输描述符准备：使用 dmaengine_prep_slave_single() 准备传输描述符

4传输提交：通过 dmaengine_submit() 提交传输，dma_async_issue_pending() 启动

5等待完成：通过 completion 等待 DMA 传输完成

6缓冲区取消映射：使用 dma_unmap_single() 释放 DMA 映射

为什么需要 dma_len 和 data_len 两个字段？
dma_len 是对齐到 4 字节边界的长度，用于 DMA 缓冲区分配和映射；data_len 是实际传输的数据长度。当数据长度不是 4 字节的倍数时，需要在首字节中调整数据位置。

4.2 DMA 通道初始化

DMA 通道的初始化在驱动 probe 阶段完成，核心函数为 spi_imx_sdma_init()：

static int spi_imx_sdma_init(struct device *dev,
                             struct spi_imx_data *spi_imx,
                             struct spi_controller *controller)
{
    int ret;

    // 计算 Watermark Level（FIFO 大小的一半）
    spi_imx->wml = spi_imx->devtype_data->fifo_size / 2;

    // 请求 TX DMA 通道
    controller->dma_tx = dma_request_chan(dev, "tx");
    if (IS_ERR(controller->dma_tx)) {
        ret = PTR_ERR(controller->dma_tx);
        dev_err_probe(dev, ret, "can't get the TX DMA channel!\n");
        controller->dma_tx = NULL;
        goto err;
    }

    // 请求 RX DMA 通道
    controller->dma_rx = dma_request_chan(dev, "rx");
    if (IS_ERR(controller->dma_rx)) {
        ret = PTR_ERR(controller->dma_rx);
        dev_err_probe(dev, ret, "can't get the RX DMA channel!\n");
        controller->dma_rx = NULL;
        goto err;
    }

    // 初始化完成量
    init_completion(&spi_imx->dma_rx_completion);
    init_completion(&spi_imx->dma_tx_completion);

    // 设置控制器标志：必须同时有 TX 和 RX
    spi_imx->controller->flags = SPI_CONTROLLER_MUST_RX |
                                 SPI_CONTROLLER_MUST_TX;

    return 0;
err:
    spi_imx_sdma_exit(spi_imx);
    return ret;
}

4.3 DMA 能力检测

驱动通过 spi_imx_can_dma() 函数判断是否应该使用 DMA 传输：

static bool spi_imx_can_dma(struct spi_controller *controller,
                            struct spi_device *spi,
                            struct spi_transfer *transfer)
{
    struct spi_imx_data *spi_imx = spi_controller_get_devdata(controller);

    // 1. 检查是否启用 DMA 且未设置回退标志
    if (!use_dma || controller->fallback)
        return false;

    // 2. 检查 DMA RX 通道是否可用
    if (!controller->dma_rx)
        return false;

    // 3. Target 模式下的传输长度限制
    if (transfer->len > MX53_MAX_TRANSFER_BYTES &&
        spi_imx->target_mode)
        return false;

    // 4. 传输长度必须大于 FIFO 大小
    if (transfer->len < spi_imx->devtype_data->fifo_size)
        return false;

    // 5. DMA 只能传输 8/16/32 位数据
    if (spi_imx->bits_per_word != 8 &&
        spi_imx->bits_per_word != 16 &&
        spi_imx->bits_per_word != 32)
        return false;

    // 6. 传输长度不能超过 SDMA BD 最大值（32KB）
    if (transfer->len >= MAX_SDMA_BD_BYTES)
        return false;

    // 7. DMA 模式下禁用 dynamic_burst
    spi_imx->dynamic_burst = 0;

    return true;
}

4.4 DMA 配置

DMA 配置函数负责设置 DMA 通道的从设备参数，包括传输方向、地址、总线宽度和突发长度。

4.4.1 DMA 从设备配置

static int spi_imx_dma_configure(struct spi_controller *controller,
                                 bool word_delay)
{
    struct spi_imx_data *spi_imx = spi_controller_get_devdata(controller);
    enum dma_slave_buswidth buswidth;
    struct dma_slave_config rx = {}, tx = {};

    // 根据 word_delay 决定总线宽度
    if (word_delay) {
        // Word delay 模式：使用实际字宽
        switch (spi_imx_bytes_per_word(spi_imx->bits_per_word)) {
        case 4:
            buswidth = DMA_SLAVE_BUSWIDTH_4_BYTES;
            break;
        case 2:
            buswidth = DMA_SLAVE_BUSWIDTH_2_BYTES;
            break;
        case 1:
            buswidth = DMA_SLAVE_BUSWIDTH_1_BYTE;
            break;
        default:
            return -EINVAL;
        }
    } else {
        // 普通模式：固定 4 字节宽度
        buswidth = DMA_SLAVE_BUSWIDTH_4_BYTES;
    }

    // 配置 TX DMA 通道
    tx.direction = DMA_MEM_TO_DEV;
    tx.dst_addr = spi_imx->base_phys + MXC_CSPITXDATA;  // TX FIFO 地址
    tx.dst_addr_width = buswidth;
    tx.dst_maxburst = spi_imx->wml;
    dmaengine_slave_config(controller->dma_tx, &tx);

    // 配置 RX DMA 通道
    rx.direction = DMA_DEV_TO_MEM;
    rx.src_addr = spi_imx->base_phys + MXC_CSPIRXDATA;  // RX FIFO 地址
    rx.src_addr_width = buswidth;
    rx.src_maxburst = spi_imx->wml;
    dmaengine_slave_config(controller->dma_rx, &rx);

    return 0;
}

4.4.2 Watermark Level 配置

Watermark Level（WML）决定了 DMA 传输的突发长度，影响传输效率：

static void mx51_setup_wml(struct spi_imx_data *spi_imx)
{
    u32 tx_wml = 0;

    // i.MX6UL 以后修复了 TX 毛刺，可以使用 WML
    if (spi_imx->devtype_data->tx_glitch_fixed)
        tx_wml = spi_imx->wml;

    // 配置 DMA 寄存器：设置水位线并启用 DMA 请求
    writel(MX51_ECSPI_DMA_RX_WML(spi_imx->wml - 1) |
           MX51_ECSPI_DMA_TX_WML(tx_wml) |
           MX51_ECSPI_DMA_RXT_WML(spi_imx->wml) |
           MX51_ECSPI_DMA_RXTDEN,
           spi_imx->base + MX51_ECSPI_DMA);
}

/* WML 计算函数：找到能整除 DMA 长度的最大 WML */
static void spi_imx_dma_max_wml_find(struct spi_imx_data *spi_imx,
                                     struct dma_data_package *dma_data,
                                     bool word_delay)
{
    unsigned int bytes_per_word = word_delay ?
        spi_imx_bytes_per_word(spi_imx->bits_per_word) :
        BYTES_PER_32BITS_WORD;
    unsigned int i;

    // 从 FIFO 大小的一半开始向下查找
    for (i = spi_imx->devtype_data->fifo_size / 2; i > 0; i--) {
        if (!(dma_data->dma_len % (i * bytes_per_word)))
            break;
    }
    // 如果没有找到合适的突发长度，使用 1
    if (i == 0)
        i = 1;

    spi_imx->wml = i;
}

4.5 DMA 传输流程

DMA 传输是 i.MX SPI 驱动的核心功能，涉及多个步骤的协调。

4.5.1 传输总入口

static int spi_imx_transfer_one(struct spi_controller *controller,
                                struct spi_device *spi,
                                struct spi_transfer *transfer)
{
    int ret;
    struct spi_imx_data *spi_imx = spi_controller_get_devdata(spi->controller);

    // 设置传输参数
    ret = spi_imx_setupxfer(spi, transfer);
    if (ret < 0)
        return ret;
    transfer->effective_speed_hz = spi_imx->spi_bus_clk;

    // 清空 RX FIFO
    while (spi_imx->devtype_data->rx_available(spi_imx))
        readl(spi_imx->base + MXC_CSPIRXDATA);

    // Target 模式使用 PIO
    if (spi_imx->target_mode && !spi_imx->usedma)
        return spi_imx_pio_transfer_target(spi, transfer);

    // 如果选择了 DMA，使用 DMA 传输
    if (spi_imx->usedma) {
        ret = spi_imx_dma_transfer(spi_imx, transfer);
        if (transfer->error & SPI_TRANS_FAIL_NO_START) {
            // DMA 失败，回退到 PIO
            controller->fallback = true;
            spi_imx_setupxfer(spi, transfer);
            return spi_imx_pio_transfer(spi, transfer);
        }
        return ret;
    }

    // 短传输使用轮询模式
    if (transfer->len == 1 || (polling_limit_us &&
        spi_imx_transfer_estimate_time_us(transfer) < polling_limit_us))
        return spi_imx_poll_transfer(spi, transfer);

    // 默认使用 PIO 中断模式
    return spi_imx_pio_transfer(spi, transfer);
}

4.5.2 DMA 数据准备

static int spi_imx_dma_data_prepare(struct spi_imx_data *spi_imx,
                                    struct spi_transfer *transfer,
                                    bool word_delay)
{
    u32 pre_bl, tail_bl;
    u32 ctrl;

    ctrl = readl(spi_imx->base + MX51_ECSPI_CTRL);

    if (word_delay) {
        // Word delay 模式：每帧一个字
        spi_imx->dma_package_num = 1;
        pre_bl = spi_imx->bits_per_word - 1;
    } else if (transfer->len <= MX51_ECSPI_CTRL_MAX_BURST) {
        // 单包传输
        spi_imx->dma_package_num = 1;
        pre_bl = transfer->len * BITS_PER_BYTE - 1;
    } else if (!(transfer->len % MX51_ECSPI_CTRL_MAX_BURST)) {
        // 长度是最大突发的倍数
        spi_imx->dma_package_num = 1;
        pre_bl = MX51_ECSPI_CTRL_MAX_BURST * BITS_PER_BYTE - 1;
    } else {
        // 需要分包：主包 + 尾包
        spi_imx->dma_package_num = 2;
        pre_bl = MX51_ECSPI_CTRL_MAX_BURST * BITS_PER_BYTE - 1;
        tail_bl = (transfer->len % MX51_ECSPI_CTRL_MAX_BURST) *
                  BITS_PER_BYTE - 1;
    }

    // 分配 DMA 数据包数组
    spi_imx->dma_data = kmalloc_objs(struct dma_data_package,
                                      spi_imx->dma_package_num,
                                      GFP_KERNEL | __GFP_ZERO);
    if (!spi_imx->dma_data)
        return -ENOMEM;

    // ... 初始化各数据包 ...

    return 0;
}

4.5.3 DMA 传输主函数

static int spi_imx_dma_transfer(struct spi_imx_data *spi_imx,
                                struct spi_transfer *transfer)
{
    bool word_delay = transfer->word_delay.value != 0 && !spi_imx->target_mode;
    int ret, i;

    // 准备 DMA 数据
    ret = spi_imx_dma_data_prepare(spi_imx, transfer, word_delay);
    if (ret < 0)
        goto fallback_pio;

    spi_imx->rx_offset = 0;

    // 逐个执行 DMA 包传输
    for (i = 0; i < spi_imx->dma_package_num; i++) {
        // 映射 DMA 缓冲区
        ret = spi_imx_dma_map(spi_imx, &spi_imx->dma_data[i]);
        if (ret < 0)
            break;

        // 更新控制寄存器的 BL 字段
        writel(spi_imx->dma_data[i].cmd_word,
               spi_imx->base + MX51_ECSPI_CTRL);

        // 执行 DMA 传输
        ret = spi_imx_dma_package_transfer(spi_imx,
               &spi_imx->dma_data[i], transfer, word_delay);

        // 取消映射（无论成功与否）
        spi_imx_dma_unmap(spi_imx, &spi_imx->dma_data[i]);

        if (ret < 0)
            break;
    }

    // 释放 DMA 缓冲区
    for (int j = 0; j < spi_imx->dma_package_num; j++) {
        kfree(spi_imx->dma_data[j].dma_tx_buf);
        kfree(spi_imx->dma_data[j].dma_rx_buf);
    }
    kfree(spi_imx->dma_data);

    return ret;
}

4.5.4 单包 DMA 传输

static int spi_imx_dma_package_transfer(struct spi_imx_data *spi_imx,
                                        struct dma_data_package *dma_data,
                                        struct spi_transfer *transfer,
                                        bool word_delay)
{
    struct spi_controller *controller = spi_imx->controller;
    int ret;

    // 查找最佳 WML
    spi_imx_dma_max_wml_find(spi_imx, dma_data, word_delay);

    // 配置 DMA 通道
    ret = spi_imx_dma_configure(controller, word_delay);
    if (ret)
        goto dma_failure_no_start;

    // 设置 WML 寄存器
    if (!spi_imx->devtype_data->setup_wml) {
        ret = -EINVAL;
        goto dma_failure_no_start;
    }
    spi_imx->devtype_data->setup_wml(spi_imx);

    // 提交 DMA 传输
    ret = spi_imx_dma_submit(spi_imx, dma_data, transfer);
    if (ret)
        return ret;

    // 同步 DMA 缓冲区到 CPU
    dma_sync_single_for_cpu(controller->dma_rx->device->dev,
                            dma_data->dma_rx_addr,
                            dma_data->dma_len,
                            DMA_FROM_DEVICE);

    // 处理接收到的数据
    spi_imx_dma_rx_data_handle(spi_imx, dma_data,
                               transfer->rx_buf + spi_imx->rx_offset,
                               word_delay);
    spi_imx->rx_offset += dma_data->data_len;

    return 0;
}

// DMA 提交和等待完成
static int spi_imx_dma_submit(struct spi_imx_data *spi_imx,
                              struct dma_data_package *dma_data,
                              struct spi_transfer *transfer)
{
    struct spi_controller *controller = spi_imx->controller;
    struct dma_async_tx_descriptor *desc_tx, *desc_rx;
    unsigned long transfer_timeout, time_left;
    dma_cookie_t cookie;

    // 1. 准备 RX 描述符（先配置 RX）
    desc_rx = dmaengine_prep_slave_single(
        controller->dma_rx,
        dma_data->dma_rx_addr,
        dma_data->dma_len,
        DMA_DEV_TO_MEM,
        DMA_PREP_INTERRUPT | DMA_CTRL_ACK);

    desc_rx->callback = spi_imx_dma_rx_callback;
    desc_rx->callback_param = (void *)spi_imx;

    cookie = dmaengine_submit(desc_rx);
    if (dma_submit_error(cookie))
        goto dmaengine_terminate_rx;

    reinit_completion(&spi_imx->dma_rx_completion);
    dma_async_issue_pending(controller->dma_rx);

    // 2. 准备 TX 描述符（TX 启动传输）
    desc_tx = dmaengine_prep_slave_single(
        controller->dma_tx,
        dma_data->dma_tx_addr,
        dma_data->dma_len,
        DMA_MEM_TO_DEV,
        DMA_PREP_INTERRUPT | DMA_CTRL_ACK);

    desc_tx->callback = spi_imx_dma_tx_callback;
    desc_tx->callback_param = (void *)spi_imx;

    cookie = dmaengine_submit(desc_tx);
    if (dma_submit_error(cookie))
        goto dmaengine_terminate_tx;

    reinit_completion(&spi_imx->dma_tx_completion);
    dma_async_issue_pending(controller->dma_tx);

    // 3. 触发传输
    spi_imx->devtype_data->trigger(spi_imx);

    // 4. 等待 TX 完成
    transfer_timeout = spi_imx_calculate_timeout(spi_imx, transfer->len);
    time_left = wait_for_completion_timeout(
        &spi_imx->dma_tx_completion, transfer_timeout);
    if (!time_left) {
        dmaengine_terminate_sync(controller->dma_tx);
        dmaengine_terminate_sync(controller->dma_rx);
        return -ETIMEDOUT;
    }

    // 5. 等待 RX 完成
    time_left = wait_for_completion_timeout(
        &spi_imx->dma_rx_completion, transfer_timeout);
    if (!time_left) {
        spi_imx->devtype_data->reset(spi_imx);
        dmaengine_terminate_sync(controller->dma_rx);
        return -ETIMEDOUT;
    }

    return 0;
}

5.1 缓冲区映射

static int spi_imx_dma_map(struct spi_imx_data *spi_imx,
                           struct dma_data_package *dma_data)
{
    struct spi_controller *controller = spi_imx->controller;
    struct device *tx_dev = controller->dma_tx->device->dev;
    struct device *rx_dev = controller->dma_rx->device->dev;
    int ret;

    // 映射 TX 缓冲区
    dma_data->dma_tx_addr = dma_map_single(tx_dev,
        dma_data->dma_tx_buf,
        DMA_CACHE_ALIGNED_LEN(dma_data->dma_len),
        DMA_TO_DEVICE);
    ret = dma_mapping_error(tx_dev, dma_data->dma_tx_addr);
    if (ret < 0) {
        dev_err(spi_imx->dev, "DMA TX map failed %d\n", ret);
        return ret;
    }

    // 映射 RX 缓冲区
    dma_data->dma_rx_addr = dma_map_single(rx_dev,
        dma_data->dma_rx_buf,
        DMA_CACHE_ALIGNED_LEN(dma_data->dma_len),
        DMA_FROM_DEVICE);
    ret = dma_mapping_error(rx_dev, dma_data->dma_rx_addr);
    if (ret < 0) {
        dma_unmap_single(tx_dev, dma_data->dma_tx_addr,
                         DMA_CACHE_ALIGNED_LEN(dma_data->dma_len),
                         DMA_TO_DEVICE);
        return ret;
    }

    return 0;
}

5.3 数据对齐与字节序处理

i.MX ECSPI 控制器以 32 位字为单位进行 DMA 传输，需要处理字节对齐和字节序问题：

/* 缓冲区长度对齐宏 */
#define DMA_CACHE_ALIGNED_LEN(x) ALIGN((x), dma_get_cache_alignment())

/* TX 数据处理：处理非 4 字节对齐的数据 */
static int spi_imx_dma_tx_data_handle(struct spi_imx_data *spi_imx,
                                      struct dma_data_package *dma_data,
                                      const void *tx_buf,
                                      bool word_delay)
{
    void *copy_ptr;
    int unaligned;

    if (word_delay) {
        dma_data->dma_len = dma_data->data_len;
    } else {
        // 对齐到 4 字节
        dma_data->dma_len = ALIGN(dma_data->data_len,
                                  BYTES_PER_32BITS_WORD);
    }

    // 分配 DMA 缓冲区
    dma_data->dma_tx_buf = kzalloc(dma_data->dma_len, GFP_KERNEL);
    if (!dma_data->dma_tx_buf)
        return -ENOMEM;

    dma_data->dma_rx_buf = kzalloc(dma_data->dma_len, GFP_KERNEL);
    if (!dma_data->dma_rx_buf) {
        kfree(dma_data->dma_tx_buf);
        return -ENOMEM;
    }

    // 处理非对齐数据
    if (dma_data->data_len % BYTES_PER_32BITS_WORD && !word_delay) {
        unaligned = dma_data->data_len % BYTES_PER_32BITS_WORD;
        // 数据放在对齐位置的后面
        copy_ptr = (u8 *)dma_data->dma_tx_buf +
                   BYTES_PER_32BITS_WORD - unaligned;
    } else {
        copy_ptr = dma_data->dma_tx_buf;
    }

    memcpy(copy_ptr, tx_buf, dma_data->data_len);

    // 字节序调整（小端系统）
#ifdef __LITTLE_ENDIAN
    unsigned int bytes_per_word = spi_imx_bytes_per_word(
        spi_imx->bits_per_word);
    u32 *temp = dma_data->dma_tx_buf;

    for (int i = 0; i < DIV_ROUND_UP(dma_data->dma_len,
                                      sizeof(*temp)); i++) {
        if (bytes_per_word == 1)
            swab32s(temp + i);      // 字节交换
        else if (bytes_per_word == 2)
            swahw32s(temp + i);     // 半字交换
    }
#endif

    return 0;
}

6.2 分包实现

static int spi_imx_dma_data_prepare(struct spi_imx_data *spi_imx,
                                    struct spi_transfer *transfer,
                                    bool word_delay)
{
    u32 pre_bl, tail_bl;
    u32 ctrl;

    ctrl = readl(spi_imx->base + MX51_ECSPI_CTRL);

    // 判断是否需要分包
    if (transfer->len <= MX51_ECSPI_CTRL_MAX_BURST) {
        // 单包
        spi_imx->dma_package_num = 1;
        pre_bl = transfer->len * BITS_PER_BYTE - 1;
    } else if (!(transfer->len % MX51_ECSPI_CTRL_MAX_BURST)) {
        // 长度是最大突发的整数倍
        spi_imx->dma_package_num = 1;
        pre_bl = MX51_ECSPI_CTRL_MAX_BURST * BITS_PER_BYTE - 1;
    } else {
        // 需要分包
        spi_imx->dma_package_num = 2;
        pre_bl = MX51_ECSPI_CTRL_MAX_BURST * BITS_PER_BYTE - 1;
        tail_bl = (transfer->len % MX51_ECSPI_CTRL_MAX_BURST) *
                  BITS_PER_BYTE - 1;
    }

    // 分配数据包数组
    spi_imx->dma_data = kmalloc_objs(struct dma_data_package,
                                      spi_imx->dma_package_num,
                                      GFP_KERNEL | __GFP_ZERO);

    if (spi_imx->dma_package_num == 1) {
        // 单包配置
        ctrl &= ~MX51_ECSPI_CTRL_BL_MASK;
        ctrl |= pre_bl << MX51_ECSPI_CTRL_BL_OFFSET;
        spi_imx->dma_data[0].cmd_word = ctrl;
        spi_imx->dma_data[0].data_len = transfer->len;
        spi_imx_dma_tx_data_handle(spi_imx, &spi_imx->dma_data[0],
                                   transfer->tx_buf, word_delay);
    } else {
        // 主包配置
        ctrl &= ~MX51_ECSPI_CTRL_BL_MASK;
        ctrl |= pre_bl << MX51_ECSPI_CTRL_BL_OFFSET;
        spi_imx->dma_data[0].cmd_word = ctrl;
        spi_imx->dma_data[0].data_len = round_down(transfer->len,
                                                    MX51_ECSPI_CTRL_MAX_BURST);
        spi_imx_dma_tx_data_handle(spi_imx, &spi_imx->dma_data[0],
                                   transfer->tx_buf, false);

        // 尾包配置
        ctrl &= ~MX51_ECSPI_CTRL_BL_MASK;
        ctrl |= tail_bl << MX51_ECSPI_CTRL_BL_OFFSET;
        spi_imx->dma_data[1].cmd_word = ctrl;
        spi_imx->dma_data[1].data_len = transfer->len %
                                         MX51_ECSPI_CTRL_MAX_BURST;
        spi_imx_dma_tx_data_handle(spi_imx, &spi_imx->dma_data[1],
                                   transfer->tx_buf +
                                   spi_imx->dma_data[0].data_len,
                                   false);
    }

    return 0;
}

8. DMA 使用最佳实践

8.1 缓冲区对齐

/* 正确：使用缓存行对齐 */
dma_addr_t addr = dma_map_single(dev, buf,
    DMA_CACHE_ALIGNED_LEN(len), DMA_TO_DEVICE);

/* 错误：直接使用原始长度 */
dma_addr_t addr = dma_map_single(dev, buf, len, DMA_TO_DEVICE);

8.2 传输顺序

重要：先配置 RX 再配置 TX
在 SPI DMA 传输中，必须先配置 RX DMA 通道再配置 TX DMA 通道。因为 TX DMA 传输会立即启动 SPI 控制器，如果 RX 未就绪会导致数据丢失。 
/* 正确顺序 */
desc_rx = dmaengine_prep_slave_single(dma_rx, ...);
dmaengine_submit(desc_rx);
dma_async_issue_pending(dma_rx);  // 先启动 RX

desc_tx = dmaengine_prep_slave_single(dma_tx, ...);
dmaengine_submit(desc_tx);
dma_async_issue_pending(dma_tx);  // 再启动 TX

8.3 超时处理

/* 设置合理的超时时间 */
static int spi_imx_calculate_timeout(struct spi_imx_data *spi_imx,
                                     int size)
{
    unsigned long timeout = 0;

    // 数据传输时间 + CS 切换延迟
    timeout = (8 + 4) * size / spi_imx->spi_bus_clk;

    // 调度器活动开销
    timeout += 1;

    // 双倍超时以留有余量
    return secs_to_jiffies(2 * timeout);
}

10. SDMA（Smart DMA）技术详解

i.MX 系列 SoC 使用的 SDMA（Smart DMA）是一种智能 DMA 控制器，与传统 DMA 相比具有更高的灵活性和效率。SDMA 拥有自己的微控制器核心，可以执行复杂的传输脚本，而不仅仅是简单的内存到设备的拷贝操作。

10.1 SDMA 架构特点

SDMA 控制器具有以下显著特点：

  • 独立的微控制器核心：SDMA 内置一个 32 位 RISC 处理器，可以执行存储在共享内存中的微代码（Microcode），这使得 SDMA 能够处理复杂的传输模式，而不仅仅是简单的线性传输。
  • 多通道并行：i.MX 系列 SoC 通常提供 32 个或更多的 SDMA 通道，每个通道可以独立配置和运行，支持同时处理多个外设的 DMA 请求。
  • 灵活的地址模式：支持线性地址、循环缓冲区、双缓冲区等多种地址管理模式，可以根据不同的应用场景选择最优的传输方式。
  • 事件驱动：SDMA 通过硬件事件（如 FIFO 水位线中断）触发传输，实现了与外设的紧密同步，减少了不必要的轮询操作。
  • 低功耗设计：SDMA 可以在 CPU 睡眠时继续工作，支持系统进入低功耗模式后仍然保持数据传输能力，这对于电池供电的嵌入式设备尤为重要。

10.2 SDMA 在 SPI 中的工作原理

在 SPI 传输中，SDMA 的工作流程如下：首先，SPI 控制器检测到 FIFO 达到预设的水位线（Watermark Level），然后向 SDMA 控制器发送请求信号。SDMA 控制器接收到请求后，根据预先配置的参数执行数据传输操作。对于发送方向，SDMA 从系统内存中读取数据并写入 SPI 控制器的 TX FIFO；对于接收方向，SDMA 从 SPI 控制器的 RX FIFO 中读取数据并写入系统内存。当传输完成或发生错误时，SDMA 通过中断通知 CPU 进行后续处理。

这种机制的关键优势在于 CPU 只需要在传输开始前配置好 SDMA 参数，然后就可以让出处理器资源去执行其他任务。只有在传输完成或出错时才需要 CPU 干预，大大提高了系统的整体效率。在大数据量传输场景下，这种方式可以将 CPU 占用率从几乎 100% 降低到接近零，同时还能保持较高的数据吞吐率。

11. DMA 与 CPU 的协作机制

DMA 传输并非完全独立于 CPU，两者之间需要精密的协作才能确保数据的正确性和系统的稳定性。理解这种协作机制对于编写高质量的驱动程序至关重要。

11.1 缓存一致性问题

现代处理器都使用缓存来提高内存访问速度，但 DMA 控制器通常直接访问物理内存，绕过了 CPU 的缓存系统。这就产生了一个严重的问题：当 CPU 通过缓存修改了某个内存位置的值，而 DMA 控制器从物理内存中读取的仍然是旧值；或者当 DMA 控制器将数据写入物理内存后，CPU 缓存中仍然是旧数据。这种不一致会导致数据错乱和程序错误。

为了解决这个问题，Linux 内核提供了两种主要的 DMA 映射方法：

第一种是一致性 DMA 映射（Consistent DMA Mapping），使用 dma_alloc_coherent() 分配内存。这种方式确保 CPU 和 DMA 控制器看到的始终是同一份数据，内核通过缓存禁用或缓存行填充等硬件机制来保证一致性。一致性映射的开销较大，适用于需要频繁同步的场景。

第二种是流式 DMA 映射（Streaming DMA Mapping），使用 dma_map_single() 或 dma_map_sg() 映射已分配的内存。这种方式性能更好，但需要程序员手动管理同步操作。在 DMA 传输开始前需要调用 dma_sync_single_for_device() 将 CPU 缓存刷新到内存，在 DMA 传输完成后需要调用 dma_sync_single_for_cpu() 使 CPU 缓存失效以读取最新数据。

i.MX SPI 驱动选择了流式 DMA 映射方式，这是因为 SPI 传输通常是突发性的，数据缓冲区在每次传输时都会重新分配和映射，流式映射的灵活性更适合这种使用模式。

11.2 中断处理与任务调度

DMA 传输完成后会触发中断通知 CPU，中断处理程序需要快速响应并进行必要的清理工作。在 SPI 驱动中，DMA 完成中断的处理相对简单：调用 complete() 函数唤醒等待传输完成的任务，让任务可以继续执行后续的处理逻辑。

这种使用 completion 机制的设计是 Linux 内核中异步操作的典型模式。驱动程序在提交 DMA 传输后调用 wait_for_completion_timeout() 阻塞等待，DMA 完成中断处理程序调用 complete() 唤醒等待的任务。超时机制确保即使 DMA 传输出现异常，系统也不会无限期地等待下去，而是会返回错误码让上层进行错误处理。

11.3 错误恢复机制

DMA 传输可能因为各种原因失败，包括硬件故障、超时、映射错误等。一个健壮的驱动程序必须能够正确处理这些错误情况，确保系统能够恢复正常运行。

i.MX SPI 驱动实现了多层错误恢复机制：在传输层面，如果 DMA 传输超时，驱动会调用 dmaengine_terminate_sync() 终止所有待处理的 DMA 传输，然后重置 SPI 控制器的硬件状态；在驱动层面，如果 DMA 初始化失败或传输反复失败，驱动会自动切换到 PIO（Programmed I/O）模式继续工作，确保功能可用性；在系统层面，驱动会设置 controller->fallback 标志，后续的传输将直接使用 PIO 模式，避免反复尝试失败的 DMA 传输。

12. DMA 性能优化策略

在实际应用中，DMA 性能的优化需要从多个维度考虑，包括缓冲区管理、突发长度配置、中断处理策略等。

12.1 缓冲区分配优化

缓冲区的分配位置和方式对 DMA 性能有显著影响。首先，缓冲区应该尽量分配在物理内存连续的区域，避免分散聚合（Scatter-Gather）带来的额外开销。在 i.MX 平台上，可以使用 CMA（Contiguous Memory Allocator）或 __GFP_COMP 标志来提高分配到连续内存的概率。

其次，缓冲区大小应该对齐到缓存行边界（通常为 64 字节），这样可以避免缓存行分裂（Cache Line Split）问题，提高内存访问效率。i.MX SPI 驱动使用 DMA_CACHE_ALIGNED_LEN() 宏来确保缓冲区长度的对齐。

此外，频繁的动态内存分配和释放会带来较大的开销。在高性能场景下，可以使用内存池（Memory Pool）或预分配的环形缓冲区来减少分配开销。虽然当前的 i.MX SPI 驱动没有实现这种优化，但在高频 SPI 传输的应用中，这是一个值得考虑的改进方向。

12.2 突发长度优化

DMA 突发长度（Burst Length）是指单次 DMA 传输中连续传输的数据量。突发长度的选择需要在传输效率和延迟之间取得平衡。

较长的突发长度可以提高总线利用率，减少 DMA 控制器和外设之间的握手次数，从而提高整体传输效率。但是，过长的突发长度会增加单次传输的延迟，并且可能需要更大的 FIFO 缓冲区来暂存数据。i.MX ECSPI 控制器的最大突发长度为 512 字节，这是一个在效率和延迟之间取得平衡的设计选择。

i.MX SPI 驱动通过 Watermark Level（WML）机制来优化突发长度。WML 决定了 FIFO 中数据达到多少时触发 DMA 请求，合适的 WML 设置可以确保 DMA 传输的连续性，避免 FIFO 溢出或空闲。驱动程序通过 spi_imx_dma_max_wml_find() 函数动态计算最优的 WML 值，确保突发长度能够整除 DMA 传输的总长度。

12.3 中断合并优化

频繁的 DMA 完成中断会消耗大量的 CPU 时间，特别是在高速 SPI 传输场景下。为了减少中断次数，可以使用中断合并（Interrupt Coalescing）技术，即等待多个 DMA 传输完成后才触发一次中断。

在 i.MX SPI 驱动中，虽然没有直接实现中断合并，但通过合理设置 WML 和突发长度，实际上已经达到了类似的效果。较大的突发长度意味着单次 DMA 传输的数据量更多，从而减少了总的传输次数和中断次数。在某些高性能场景下，还可以考虑使用延迟中断（Delayed Interrupt）机制，在数据量未达到阈值时延迟触发中断，进一步减少中断频率。

13. DMA 调试技术

DMA 相关的问题通常比较难以调试，因为涉及硬件行为、内存映射、中断处理等多个层面。掌握有效的调试技术对于快速定位和解决问题至关重要。

13.1 常用调试工具

Linux 内核提供了多种 DMA 调试工具：CONFIG_DMA_API_DEBUG 选项启用 DMA API 的调试检查，可以在开发阶段发现常见的 DMA 使用错误，如映射不平衡、缓存一致性问题等。CONFIG_DMABUF_HEAPS 提供 DMA 堆的调试信息，帮助追踪缓冲区的分配和释放。/sys/kernel/debug/dma/ 目录下的文件提供了 DMA 映射的运行时统计信息。

对于 i.MX 平台，还可以使用 devmem 或 devmem2 工具直接读取和修改 SPI 控制器的寄存器，这对于调试 DMA 寄存器配置问题非常有用。此外，ftrace 和 trace-cmd 工具可以跟踪 DMA 相关的内核函数调用，帮助分析 DMA 传输的时序和性能问题。

13.2 常见 DMA 问题排查

当遇到 DMA 传输失败时，可以按照以下步骤进行排查：

首先，检查 DMA 通道是否正确获取。通过 cat /proc/device-tree/soc/spi@*/dma-names 查看设备树中的 DMA 通道名称，通过 cat /sys/class/dma/*/in_use 检查通道是否被占用。如果通道获取失败，可能是设备树配置错误或 DMA 控制器驱动未正确加载。

其次，检查缓冲区映射是否成功。在驱动中添加 dma_mapping_error() 的错误检查日志，确认映射后的地址是否有效。如果映射失败，可能是内存不足或 IOMMU 配置问题。

最后，检查 DMA 寄存器配置。使用 devmem 读取 ECSPI_DMA 寄存器的值，确认 WML 和 DMA 使能位是否正确设置。如果寄存器配置正确但传输仍然失败，可能是硬件时序问题，需要使用逻辑分析仪或示波器检查 SPI 信号。

14. 实际应用案例分析

理解 DMA 的理论知识固然重要，但将其应用到实际项目中才是最终目标。本节将通过几个典型的应用场景，展示如何在实际项目中有效地使用 SPI DMA 技术。

14.1 SPI Flash 存储设备

SPI Flash 是嵌入式系统中最常见的存储设备之一，广泛用于存储固件、配置数据和用户数据。由于 Flash 的读写操作通常涉及大块数据的传输，使用 DMA 可以显著提高存储性能。

在读取操作中，CPU 首先发送读取命令和起始地址，然后等待数据从 Flash 返回。如果使用 PIO 模式，CPU 需要在每个时钟周期检查数据是否就绪，这会占用大量的处理器时间。而使用 DMA 模式，CPU 只需要配置好 DMA 传输参数，然后就可以让 SDMA 控制器自动完成整个数据接收过程，CPU 可以在此期间处理其他任务或进入低功耗状态。

在写入操作中，DMA 的优势更加明显。Flash 的写入速度通常较慢，如果使用 PIO 模式，CPU 会在等待 Flash 写入完成的过程中浪费大量时间。使用 DMA，CPU 可以一次性将整个数据块提交给 DMA 控制器，然后等待传输完成中断即可，大大提高了系统的整体效率。

对于 SPI Flash 的 DMA 优化，有几个关键点需要注意：首先，Flash 的页大小（通常为 256 字节或 512 字节）应该作为 DMA 传输的最小单位，这样可以减少 Flash 控制器的内部开销；其次，读取操作可以利用 Flash 的连续读取特性，设置较长的突发长度来提高总线利用率；最后，写入操作需要考虑 Flash 的编程时间，合理设置超时时间避免误判传输失败。

14.2 SPI 传感器数据采集

现代嵌入式系统通常配备多种 SPI 传感器，如加速度计、陀螺仪、磁力计、环境光传感器等。这些传感器通常以固定的采样率产生数据，需要及时读取并处理。

对于低速传感器（如温度传感器、气压传感器），采样率通常在几赫兹到几十赫兹之间，数据量很小，使用 PIO 模式即可满足需求。但对于高速传感器（如六轴惯性测量单元 IMU），采样率可能高达数百甚至数千赫兹，每次采样产生多个字节的数据，此时使用 DMA 可以显著降低 CPU 负担。

在传感器数据采集场景中，DMA 的配置需要特别注意实时性要求。由于传感器数据通常需要及时处理（如姿态控制、运动检测），DMA 传输的延迟必须控制在可接受的范围内。这通常意味着需要使用较小的缓冲区和较短的突发长度，以换取更低的传输延迟。同时，DMA 完成中断的优先级应该设置得较高，确保数据能够及时被处理。

另一个重要的考虑是电源管理。在电池供电的设备中，传感器通常是主要的功耗来源之一。通过合理配置 DMA 的唤醒机制，可以在传感器数据就绪时唤醒 CPU 进行处理，其余时间让系统保持低功耗状态，从而显著延长电池续航时间。

14.3 SPI 显示屏驱动

SPI 接口的小尺寸 TFT 显示屏在嵌入式设备中非常常见，如智能手表、工业仪表、医疗设备等。显示更新通常涉及大量像素数据的传输，是 DMA 应用的理想场景。

在显示屏驱动中，DMA 主要用于帧缓冲区到显示屏的数据传输。一帧图像数据通常按行存储在内存中，DMA 需要将这些数据逐行发送到显示屏控制器。对于常见的 240x240 像素、16 位色深的显示屏，一帧数据量约为 115KB，使用 DMA 传输可以在几十毫秒内完成，远快于 PIO 模式。

显示屏驱动的 DMA 实现需要考虑几个特殊因素：首先，显示屏通常需要持续刷新以维持显示内容，这意味着 DMA 传输需要周期性地进行，可以使用循环 DMA 模式来减少每次传输的配置开销；其次，显示屏的刷新率和 SPI 时钟频率需要匹配，避免出现画面撕裂或闪烁现象；最后，为了支持部分屏幕更新，DMA 驱动需要能够处理非连续的内存区域，这可以通过分散聚合（Scatter-Gather）DMA 来实现。

14.4 音频编解码器接口

虽然 I2S 是音频编解码器的主流接口，但一些低功耗或低成本的编解码器仍然使用 SPI 接口传输音频数据。在这种应用中，DMA 的实时性要求非常高，因为音频数据必须在严格的时间约束内传输，否则会出现音频断续或噪音。

音频 DMA 的典型实现使用双缓冲（Double Buffering）技术：当 DMA 正在传输一个缓冲区的数据时，CPU 可以填充另一个缓冲区，两个缓冲区交替使用，实现了传输和处理的并行化。这种方式可以显著降低音频延迟，同时避免因缓冲区欠载导致的音频问题。

在音频应用中，DMA 中断的延迟直接影响音频质量。如果中断处理延迟过大，可能导致缓冲区欠载，产生可闻的噪音或断续。因此，音频 DMA 驱动通常需要使用实时补丁（PREEMPT_RT）和高优先级中断，确保中断处理的及时性。此外，音频缓冲区的大小需要仔细调整，在延迟和功耗之间取得平衡。
