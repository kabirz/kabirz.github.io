03 守护任务

// timers.c 中的定时器守护任务
static portTASK_FUNCTION( prvTimerTask, pvParameters )
{
    TickType_t xNextExpireTime;

    for ( ;; )
    {
        // 1. 获取下一个定时器的到期时间
        xNextExpireTime = prvGetNextExpireTime();

        // 2. 阻塞等待命令或超时
        if( xNextExpireTime != tmrMAX_TIME_BEFORE_OVERFLOW )
        {
            // 等待直到下一个定时器到期或收到命令
            xQueueReceive( xTimerQueue, &xMessage, xNextExpireTime );
        }
        else
        {
            // 无活动定时器，无限等待命令
            xQueueReceive( xTimerQueue, &xMessage, portMAX_DELAY );
        }

        // 3. 处理收到的命令
        prvProcessReceivedCommands();

        // 4. 检查是否有定时器到期
        prvProcessExpiredTimers();
    }
}

04 创建定时器

// timers.c 中的 xTimerCreate()
TimerHandle_t xTimerCreate( const char * const pcTimerName,
                             const TickType_t xTimerPeriodInTicks,
                             const UBaseType_t uxAutoReload,
                             void * const pvTimerID,
                             TimerCallbackFunction_t pxCallbackFunction )
{
    Timer_t * pxNewTimer;

    // 1. 分配定时器结构体
    pxNewTimer = ( Timer_t * ) pvPortMalloc( sizeof( Timer_t ) );

    // 2. 初始化定时器
    if( pxNewTimer != NULL )
    {
        // 设置名称
        pxNewTimer->pcTimerName = pcTimerName;
        pxNewTimer->xTimerPeriodInTicks = xTimerPeriodInTicks;
        pxNewTimer->pvTimerID = pvTimerID;
        pxNewTimer->pxCallbackFunction = pxCallbackFunction;

        // 设置状态位
        pxNewTimer->ucStatus = 0x00;
        if( uxAutoReload != pdFALSE )
        {
            pxNewTimer->ucStatus |= tmrSTATUS_IS_AUTORELOAD;
        }

        // 初始化链表节点
        listINIT_LIST_ITEM( &( pxNewTimer->xTimerListItem ) );
    }

    return pxNewTimer;
}

07 完整示例

// ===== FreeRTOS 定时器完整示例 =====
// 演示：一次性、周期性、重载、修改周期

#include "FreeRTOS.h"
#include "task.h"
#include "timers.h"
#include <stdio.h>

// 定时器句柄
TimerHandle_t xOneShotTimer;
TimerHandle_t xPeriodicTimer;

// 一次性定时器回调
void vOneShotCallback( TimerHandle_t xTimer )
{
    printf( "One-shot timer expired!\\n" );
}

// 周期性定时器回调
void vPeriodicCallback( TimerHandle_t xTimer )
{
    static uint32_t ulCount = 0;
    ulCount++;
    printf( "Periodic timer expired: %lu\\n", ulCount );
}

// 控制任务
void vControlTask( void * pvParameters )
{
    // 启动一次性定时器（1秒后触发）
    xTimerStart( xOneShotTimer, pdMS_TO_TICKS( 100 ) );

    // 启动周期性定时器（每500ms触发）
    xTimerStart( xPeriodicTimer, pdMS_TO_TICKS( 100 ) );

    for ( ;; )
    {
        vTaskDelay( pdMS_TO_TICKS( 5000 ) );

        // 5秒后修改周期性定时器周期
        xTimerChangePeriod( xPeriodicTimer, pdMS_TO_TICKS( 1000 ), pdMS_TO_TICKS( 100 ) );
        printf( "Changed periodic timer period to 1000ms\\n" );

        // 重载一次性定时器
        xTimerReset( xOneShotTimer, pdMS_TO_TICKS( 100 ) );
        printf( "Reset one-shot timer\\n" );
    }
}

int main( void )
{
    // 创建一次性定时器
    xOneShotTimer = xTimerCreate( "OneShot",
                                   pdMS_TO_TICKS( 1000 ),
                                   pdFALSE,  // 一次性
                                   NULL,
                                   vOneShotCallback );

    // 创建周期性定时器
    xPeriodicTimer = xTimerCreate( "Periodic",
                                    pdMS_TO_TICKS( 500 ),
                                    pdTRUE,  // 周期性
                                    NULL,
                                    vPeriodicCallback );

    // 创建控制任务
    xTaskCreate( vControlTask, "Control", 256, NULL, 2, NULL );

    vTaskStartScheduler();

    for ( ;; ) {}
}
