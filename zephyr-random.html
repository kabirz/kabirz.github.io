01 随机数概览

Zephyr 提供多种随机数生成方案，从硬件 RNG 到软件伪随机数。

随机数源

类型Kconfig说明
硬件 RNGCONFIG_ENTROPY_HAS_DRIVER=y真随机数（硬件熵源）
软件 PRNGCONFIG_RANDOM_HAS_DRIVER=y伪随机数（Xorshift）
CTR-DRBGCONFIG_CTR_DRBG_CSPRNG=yNIST 推荐的 CSPRNG
Mbed TLSCONFIG_MBEDTLS_CTR_DRBG_C=yMbed TLS 随机数

02 硬件 RNG

// 硬件随机数生成器
#include <zephyr/random/random.h>

// 获取随机字节
uint8_t random_buf[32];
int ret = sys_rand_get(random_buf, sizeof(random_buf));

// 硬件熵源
// nRF52: AES-CTR-DRBG (基于热噪声)
// STM32: RNG 外设（硬件真随机）
// ESP32: eFuse 随机数

03 软件 PRNG

// 软件伪随机数生成器（Xorshift）
// CONFIG_XORSHIFT_RANDOM=y

// 初始化种子
// 使用硬件 RNG 或固定种子

// 注意：软件 PRNG 不是密码学安全的
// 适用于非安全场景（模拟、测试）

04 CTR-DRBG

// CTR-DRBG（Counter DRBG）
// NIST SP 800-90A 推荐的 CSPRNG
// CONFIG_CTR_DRBG_CSPRNG=y

// 使用 AES-CTR 模式生成随机数
// 基于硬件熵源重新播种

// 特点：
// - 密码学安全
// - 可预测性：否
// - 性能：中等

05 熵源

// 熵源（Entropy Source）
// CONFIG_ENTROPY_HAS_DRIVER=y

// 熵设备
const struct device *entropy_dev = DEVICE_DT_GET(DT_NODELABEL(rng));

// 获取熵
uint8_t entropy_buf[32];
int ret = entropy_get_reader(entropy_dev, entropy_buf, sizeof(entropy_buf));

// 熵质量
// - 硬件 RNG：高（真随机）
// - 软件 PRNG：低（伪随机）
// - CTR-DRBG：高（基于硬件熵）

06 加密随机数

// PSA Crypto 随机数
// CONFIG_PSA_CRYPTO=y

// 生成随机数
uint8_t random[32];
psa_status_t status = psa_generate_random(random, sizeof(random));

// Mbed TLS 随机数
// CONFIG_MBEDTLS_CTR_DRBG_C=y
mbedtls_ctr_drbg_random(&ctr_drbg, random, sizeof(random));
