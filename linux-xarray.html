01 概述

XArray（eXtensible Array）是 2017 年由 Matthew Wilcox 引入内核的数据结构，用于取代 radix tree。虽然底层仍然是多叉树，但它提供"数组"级别的心智模型——一个可以存储任意指针的、按 unsigned long 索引的稀疏数组。

struct xarray {
    spinlock_t xa_lock;      // 内置锁
    gfp_t xa_flags;          // 分配标志
    void __rcu *xa_head;     // 头结点（叶子指针或内部节点）
};
简单如数组
// 存、取、删 — 和使用普通数组一样简单
xa_init(&xa);
xa_store(&xa, 42, ptr, GFP_KERNEL);
ptr = xa_load(&xa, 42);
xa_erase(&xa, 42);
强大如树
// 但支持稀疏存储、标记、多索引范围
// ULONG_MAX 个 slot
// 内置 spinlock 保护
// RCU 安全的读路径
// 自动扩展/收缩

XArray 是 page cache（mapping->i_pages）、PID 分配器、文件描述符表等核心子系统的底层存储。

02 API 分层

XArray 提供三层 API，覆盖从简单到高级的所有需求。

第一层：简单 API（Simple）

xa_init(&xa);                        // 初始化
xa_store(&xa, index, entry, gfp);    // 存入，返回旧值
xa_load(&xa, index);                 // 读取
xa_erase(&xa, index);                // 删除，返回旧值
xa_cmpxchg(&xa, index, old, new, gfp); // CAS 语义

xa_for_each(&xa, index, entry) {     // 遍历全部
    // ...
}
xa_for_each_marked(&xa, index, entry, mark) {
    // 遍历打了标记的
}

第二层：高级 API（Advanced）

xa_reserve(&xa, index, gfp);         // 预留 slot（不会失败）
xa_alloc(&xa, &id, entry, limit, gfp); // 自动分配 ID
xa_alloc_cyclic(&xa, &id, entry, limit, gfp); // 循环分配
xa_get_mark(&xa, index, mark);      // 读取标记
xa_set_mark(&xa, index, mark);      // 设置标记
xa_clear_mark(&xa, index, mark);    // 清除标记

第三层：迭代器 API（Iterator / xas）

// 使用 xa_state 结构体，需要手动管理锁
XA_STATE(xas, &xa, first_index);
xas_lock(&xas);
xas_for_each(&xas, entry, max) {
    // 手动控制迭代过程
    if (xas_retry(&xas, entry))
        continue;
    // ...
}
xas_unlock(&xas);
什么时候用哪层?
- 简单 API：95% 的场景（page cache 的普通操作、ID 分配）
- 高级 API：需要避免分配失败的原子场景、IDA 模式
- 迭代器：需要批处理、在遍历中修改、需要 mmap 回写感知

03 内部数据结构

XArray 内部是一棵 N-叉前缀树。每个节点有 64 个 slot（6 bit 地址），树的高度取决于最大索引值。

// lib/xarray.c — 节点结构
struct xa_node {
    unsigned char shift;     // 当前层位移（决定覆盖范围）
    unsigned char offset;    // 在父节点中的位置
    unsigned char count;     // 非空 slot 数
    unsigned char nr_values; // 值类型的 slot 数
    struct xa_node *parent;  // 父节点
    struct rcu_head rcu_head;
    void __rcu *slots[64];   // 64 个 slot
    unsigned long tags[3];   // 3 组标记位图
};

// entry 编码（低 2 位标记类型）：
// - 00: 普通指针
// - 01: 内部节点
// - 10: 值条目（value entry）
// - 11: 兄弟条目（sibling，多索引用）

#define XA_ZERO_ENTRY     xa_mk_internal(257)
#define XA_RETRY_ENTRY    xa_mk_internal(256)

树的高度（shift 值）随着条目数量动态调整：

条目数范围shift树高叶子范围
0—0（空）—
1—0（单指针）—
2 到 64010-63
65 到 4096620-4095
4097 到 262k1230-262143
超出时继续增长+=6+1 层×64 每层

04 核心操作

xa_load — 查找

void *xa_load(struct xarray *xa, unsigned long index)
{
    XA_STATE(xas, xa, index);
    void *entry;

    rcu_read_lock();
    do {
        entry = xas_load(&xas);
        if (xas_retry(&xas, entry))
            continue;
        break;
    } while (true);
    rcu_read_unlock();
    return entry;
}

// 内部 xas_load 从根节点开始
// 按 shift 位提取逐级下沉，直到叶子节点
// slot == NULL → 未分配
// slot 指向内部节点 → 继续下沉
// slot 是值 → 返回

xa_store — 插入/更新

void *xa_store(struct xarray *xa, unsigned long index,
               void *entry, gfp_t gfp)
{
    XA_STATE(xas, xa, index);
    void *curr;

    // 1. 预留节点（如果需要分裂树）
    // 2. 获取锁
    xas_lock(&xas);
    // 3. 插入
    curr = xas_store(&xas, entry);
    xas_unlock(&xas);
    // 4. 释放旧值（如果有）
    return curr;
}

xa_store 内部自动处理：

  • 空 xarray → 直接设为单条目的叶子节点
  • 超出当前树范围 → 逐层增加内部节点
  • 替换已有条目 → 返回旧值
  • 覆盖 NULL → 插入
  • 内存不足 → 返回 ERR_PTR(-ENOMEM)

05 锁模型

API 层读写
简单 API xa_load: 内部 RCU 读锁 xa_store: 内部 spinlock
高级 API xa_get_mark: RCU 读锁 xa_set_mark: 内部 spinlock
迭代器 调用者必须自己管理 xas_lock/xas_unlock 
// 迭代器模式下，调用者负责锁
XA_STATE(xas, &xa, 0);
xas_lock(&xas);
xas_for_each(&xas, entry, ULONG_MAX) {
    // 在遍历中进行修改
    if (should_remove)
        xas_store(&xas, NULL);
}
xas_unlock(&xas);

// spinlock 变体：
xas_lock_irq(&xas);   // 关中断版
xas_lock_bh(&xas);    // 关底半部版
RCU 安全:
xa_load 只需要 RCU 读锁（不能睡眠）。
写入者通过 call_rcu 释放旧节点，确保所有读者退出后才释放内存。

06 标记系统

每个 XArray 条目有 3 个独立的标记位（XA_MARK_0 ~ XA_MARK_2），存储在内部节点的 tags 位图中。

// 标记是 per-slot 的位图
// 存在内部节点中：tags[mark][slot_index/64] & (1 << slot_index%64)

// API
xa_set_mark(&xa, index, xa_mark_t mark);
xa_clear_mark(&xa, index, xa_mark_t mark);
xa_get_mark(&xa, index, xa_mark_t mark);

// 遍历已标记条目
xa_for_each_marked(&xa, index, entry, XA_MARK_0) {
    // 只遍历设置了 XA_MARK_0 的条目
}

在 page cache 中的典型用途：

标记page cache 含义
XA_MARK_0PAGECACHE_TAG_DIRTY — 脏页
XA_MARK_1PAGECACHE_TAG_WRITEBACK — 写回中
XA_MARK_2PAGECACHE_TAG_TOWRITE — 待写回

07 多索引

XArray 支持用一个 entry 覆盖一段连续的索引范围——"多索引条目"，内部通过兄弟条目（sibling entry）实现。

// 示例：将 folio（4 页）存入索引 0-3
xa_store_range(&mapping->i_pages,
               0, 3, folio, GFP_KERNEL);

// 内部结构：
// slots[0] = folio_ptr    ← 主条目
// slots[1] = sibling(1)   ← 兄弟条目指向主条目
// slots[2] = sibling(2)
// slots[3] = sibling(3)
// 兄弟条目编码：低 2 位 = 11，偏移量编码在高位

// xa_load(&xa, 1) 自动解引用兄弟条目 → 返回 folio_ptr

兄弟条目的好处：

  • 查找单条目和查找多条目使用相同的 xa_load 路径
  • 不增加树的高度（兄弟条目不占用新节点）
  • 写时复制友好（大 folio 映射）

09 内核实战

1. Page Cache — 最大规模的 XArray 使用者

// mm/filemap.c
// address_space.i_pages 是一个 xarray
struct address_space {
    struct xarray     i_pages;   // 存储所有 folio
    // ...
};

// 读 page cache
struct folio *filemap_load_folio(struct address_space *mapping,
                                 pgoff_t index)
{
    // 使用简单的 xa_load
    struct folio *folio = xa_load(&mapping->i_pages, index);
    return folio;
}

// 遍历脏页回写
pgoff_t index;
struct folio *folio;
xa_for_each_marked(&mapping->i_pages, index, folio,
                   PAGECACHE_TAG_DIRTY) {
    writeback_folio(mapping, folio);
}

2. PID 分配 — xa_alloc 示例

// kernel/pid.c
static struct xarray pid_array;

int alloc_pid(struct pid_namespace *ns)
{
    struct pid *pid;
    int nr;

    // xa_alloc 自动找到第一个空闲 ID
    // 从 RESERVED_PIDS 到 PID_MAX_DEFAULT
    nr = xa_alloc(&pid_array, &nr, pid,
                  xa_limit_32b, GFP_KERNEL);
    // nr 被设为已分配的 ID
    return nr;
}

void free_pid(struct pid *pid)
{
    xa_erase(&pid_array, pid->nr);  // 删除条目，自动回收 ID
}
