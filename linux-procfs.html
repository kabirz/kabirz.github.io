01 概述

procfs 是 Linux 内核的伪文件系统（pseudo filesystem），挂载于 /proc。它不存储在磁盘上，而是由内核在内存中动态生成，为用户空间提供一个统一的接口来查看和控制内核运行时状态。

procfs 的核心作用
  • 进程信息：/proc/[pid]/ 下的 status、maps、fd 等
  • 系统信息：cpuinfo、meminfo、version 等
  • 内核配置：/proc/sys/ 下的可调参数
  • 硬件信息：interrupts、ioports、dma 等
  • 调试接口：kcore、slabinfo、vmstat 等
设计特点
  • 无持久化：纯内存实现，重启后消失
  • 按需生成：每个文件的数据在 read() 时实时生成
  • 可写控制：部分文件支持 write() 修改内核参数
  • 进程隔离：每个进程只看到自己的 /proc/[pid]
  • 零拷贝：通过 seq_file 避免大缓冲区分配
历史：procfs 最早由 Linus Torvalds 在 1991 年编写（fs/proc/inode.c 文件头注释）。最初用于替代 /dev/mem 提供进程信息，后来逐渐扩展为内核信息的主要接口。sysfs（/sys）和 debugfs（/sys/kernel/debug）后来分担了部分职责。

03 核心数据结构

struct proc_dir_entry（目录条目）

struct proc_dir_entry {
    atomic_t in_use;                 // 使用计数
    refcount_t refcnt;               // 引用计数
    struct list_head pde_openers;    // 谁打开了此文件（未 release）
    spinlock_t pde_unload_lock;
    struct completion *pde_unload_completion;

    const struct inode_operations *proc_iops;  // inode 操作
    union {
        const struct proc_ops *proc_ops;       // 文件操作（新接口）
        const struct file_operations *proc_dir_ops;  // 目录操作
    };
    union {
        const struct seq_operations *seq_ops;  // seq_file 操作
        int (*single_show)(struct seq_file *, void *);  // 单次 show
    };
    proc_write_t write;
    void *data;                        // 私有数据
    unsigned int state_size;
    unsigned int low_ino;              // inode 号
    nlink_t nlink;
    kuid_t uid;
    kgid_t gid;
    loff_t size;
    struct proc_dir_entry *parent;   // 父目录
    struct rb_root subdir;            // 子条目红黑树
    struct rb_node subdir_node;
    char *name;
    umode_t mode;
    u8 flags;
    u8 namelen;
    char inline_name[];               // 小名称内联存储
};
子目录管理：proc_dir_entry 使用红黑树（subdir）管理子条目，通过 pde_subdir_find() 进行 O(log n) 名称查找。子条目的 subdir_node 字段作为红黑树节点。

05 seq_file 接口

seq_file 是 procfs 的核心输出机制。它提供一个自动增长的内核缓冲区，解决了一次性 kmalloc(PAGE_SIZE) 输出大量信息的问题。

进程查找流程

// fs/proc/root.c:403-408
static struct dentry *proc_root_lookup(struct inode *dir, struct dentry *dentry,
                                        unsigned int flags)
{
    // 1. 先尝试查找 PID 目录
    if (!proc_pid_lookup(dentry, flags))
        return NULL;

    // 2. 再查找普通 proc 条目
    return proc_lookup(dir, dentry, flags);
}

/proc/[pid]/mem — 进程内存访问

安全注意：/proc/[pid]/mem 允许直接读写进程的虚拟内存。默认需要 CAP_SYS_PTRACE 权限，且受 ptrace 访问检查保护。proc_mem_force 枚举控制强制访问行为。

/proc/sys 与 sysctl 系统调用

双重访问：/proc/sys 可以通过文件 I/O（cat/echo）或 sysctl 系统调用访问。两者最终都调用 proc_sys_read() / proc_sys_write()。现代 Linux 推荐直接操作文件而非使用 sysctl 系统调用。

08 创建自定义条目

模式一：seq_file（推荐）

#include <linux/proc_fs.h>
#include <linux/seq_file.h>

static int my_show(struct seq_file *m, void *v)
{
    seq_printf(m, "status: %s\n", my_status);
    return 0;
}

static int my_open(struct inode *inode, struct file *file)
{
    return single_open(file, my_show, NULL);
}

static const struct proc_ops my_ops = {
    .proc_open    = my_open,
    .proc_read    = seq_read,
    .proc_lseek   = seq_lseek,
    .proc_release = single_release,
};

// 注册
proc_create("my_info", 0644, NULL, &my_ops);

模式二：最简方式

// 一行搞定
proc_create_single("my_info", 0644, NULL, my_show);

// 带私有数据
proc_create_single_data("my_info", 0644, NULL, my_show, my_priv_data);

// 多条目迭代
static const struct seq_operations my_seq_ops = {
    .start = my_start,
    .next  = my_next,
    .stop  = seq_stop,
    .show  = my_show,
};
proc_create_seq("my_list", 0644, NULL, &my_seq_ops);

模式三：带写入

static ssize_t my_write(struct file *file, const char __user *buf,
                          size_t count, loff_t *ppos)
{
    int val;
    if (kstrtoint_from_user(buf, count, 10, &val))
        return -EINVAL;
    my_value = val;
    return count;
}

static const struct proc_ops my_rw_ops = {
    .proc_open    = my_open,
    .proc_read    = seq_read,
    .proc_write   = my_write,
    .proc_lseek   = seq_lseek,
    .proc_release = single_release,
};

模式四：网络命名空间

// 为每个网络命名空间创建独立的 proc 条目
proc_create_net_single("snmp", 0444, net, snmp_seq_show, NULL);

// 带写入的网络 proc 条目
proc_create_net_single_write("stat", 0644, net,
                              stat_show, stat_write, NULL);
最佳实践：
• 优先使用 proc_create_single() 或 proc_create_seq()，内部已处理 open/read/lseek/release
• 写入操作必须验证用户输入，使用 kstrtoint_from_user() 等安全转换函数
• 敏感操作需要权限检查：!capable(CAP_SYS_ADMIN) 返回 -EPERM
• 模块卸载时必须调用 remove_proc_entry() 清理
