// 封装 UART/SPI 字节流

#include <zephyr/modem/pipe.h>

// 初始化 Pipe
struct modem_pipe pipe;
modem_pipe_init(&pipe, &uart_pipe_api, &uart_config);

// 打开/关闭
modem_pipe_open(&pipe);
modem_pipe_close(&pipe);

// 异步传输

04 Chat（AT 命令）

// Chat 模块处理 AT 命令交互

#include <zephyr/modem/chat.h>

// 定义 AT 命令匹配
MODEM_CHAT_MATCH(ok_match, "OK", "", NULL);
MODEM_CHAT_MATCH(err_match, "ERROR", "", NULL);

// 定义脚本
MODEM_CHAT_SCRIPT_CMDS_DEFINE(init_script_chats) = {
    MODEM_CHAT_SCRIPT_CMD("AT", ok_match, err_match),
    MODEM_CHAT_SCRIPT_CMD("AT+CFUN=1", ok_match, err_match),
    MODEM_CHAT_SCRIPT_CMD("AT+CGDCONT=1,\"IP\",\"internet\"",
                          ok_match, err_match),
};
MODEM_CHAT_SCRIPT_DEFINE(init_script, init_script_chats,
                         modem_chat_empty_matches, 5, NULL);

// 执行脚本
modem_chat_run_script(&chat, &init_script);

05 CMUX 复用

// CMUX（Concentrated MODEM eXtensions）
// 在单个 UART 上复用多个逻辑通道

// 通道 0: AT 命令
// 通道 1: PPP 数据
// 通道 2: 诊断

// CMUX 帧格式：
// +------+------+------+------+------+------+------+------+
// | F    | EA/C | len  | DLCI | data ...       | FCS  | F    |
// | 0xF9 | 1B   | 1-2B | 1B   | variable       | 1B   | 0xF9 |
// +------+------+------+------+------+------+------+------+

// Kconfig
// CONFIG_MODEM_CMUX=y

06 PPP 网络

// PPP（Point-to-Point Protocol）
// 通过 Modem 建立 IP 网络连接

// 配置
// CONFIG_MODEM_PPP=y
// CONFIG_NET_L2_PPP=y

// PPP 建立流程：
// 1. 发送 AT 命令激活 PDP 上下文
// 2. 启动 CMUX（可选）
// 3. 在逻辑通道上启动 PPP
// 4. PPP 协商（LCP → NCP）
// 5. 获取 IP 地址
// 6. 数据传输
