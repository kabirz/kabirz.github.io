01 文件系统概览

Zephyr 支持多种文件系统实现，从 FAT 到 LittleFS，适用于不同的存储需求。

文件系统对比

文件系统目录特点适用场景
FATsubsys/fs/fat_fs.c通用兼容SD 卡、USB 存储
LittleFSsubsys/fs/littlefs_fs.c掉电安全、磨损均衡Flash 存储
ext2subsys/fs/ext2/Linux 兼容大容量存储
NFSsubsys/fs/nfs/网络文件系统网络存储
FCBsubsys/fs/fcb/Flash 循环缓冲区日志存储
Settingssubsys/settings/键值对存储配置持久化

02 VFS 层

Zephyr VFS（Virtual File System）提供统一的文件操作接口。

// VFS 文件操作
#include <zephyr/fs/fs.h>

// 打开文件
struct fs_file_t file;
fs_file_t_init(&file);

int ret = fs_open(&file, "/littlefs/data.txt", FS_O_CREATE | FS_O_RDWR);
if (ret < 0) {
    printk("打开文件失败: %d\n", ret);
}

// 写入数据
char data[] = "Hello, Zephyr!";
fs_write(&file, data, sizeof(data));

// 读取数据
char buffer[64];
fs_seek(&file, 0, FS_SEEK_SET);
fs_read(&file, buffer, sizeof(buffer));

// 关闭文件
fs_close(&file);

// 目录操作
struct fs_dir_t dir;
fs_dir_t_init(&dir);
fs_opendir(&dir, "/littlefs");

struct fs_dirent entry;
while (fs_readdir(&dir, &entry) == 0) {
    if (entry.name[0] == '\0') break;
    printk("文件: %s\n", entry.name);
}

fs_closedir(&dir);

04 LittleFS

LittleFS 是专为 Flash 存储设计的文件系统，支持掉电安全和磨损均衡。

LittleFS 特点

特性说明
掉电安全崩溃后自动恢复
磨损均衡均匀使用 Flash 块
动态磨损管理运行时自动管理
块擦除支持 Flash 擦除特性
小 RAM 占用适合资源受限设备
// LittleFS 配置
// CONFIG_FILE_SYSTEM_LITTLEFS=y

// DTS 配置（Flash 分区）
// storage: storage@fd000 {
//     compatible = "zephyr,flash-disk";
//     partition-run = "storage-partition";
//     fat,sector-size = <512>;
//     fat,num-sectors = <128>;
// };

05 Flash 存储

直接操作 Flash 存储器，适用于原始数据存储。

// Flash API（include/zephyr/drivers/flash.h）

const struct device *flash_dev = DEVICE_DT_GET(DT_NODELABEL(flash0));

// 擦除 Flash 扇区
struct flash_pages_info info;
flash_get_page_info_by_addr(flash_dev, CONFIG_FLASH_BASE_ADDRESS, &info);
flash_erase(flash_dev, info.start_offset, info.size);

// 写入数据
uint8_t write_buf[256];
flash_write(flash_dev, CONFIG_FLASH_BASE_ADDRESS, write_buf, sizeof(write_buf));

// 读取数据
uint8_t read_buf[256];
flash_read(flash_dev, CONFIG_FLASH_BASE_ADDRESS, read_buf, sizeof(read_buf));

06 Flash 日志 (FCB)

FCB（Flash Circular Buffer）在 Flash 上实现循环缓冲区，适合日志存储。

// FCB 配置
// CONFIG_FCB=y

// FCB 使用
#include <zephyr/fs/fcb.h>

struct fcb fcb;
fcb.f_sectors = fcb_sectors;
fcb.fSectorCnt = ARRAY_SIZE(fcb_sectors);
fcb.fAlign = 1;

fcb_init(&fcb);

// 写入日志条目
struct fcb_entry entry;
uint8_t data[] = "Log entry data";
fcb_append(&fcb, sizeof(data), &entry);
fcb_write(&fcb, &entry, data, sizeof(data));

// 读取日志条目
uint8_t read_buf[64];
while (fcb_getnext(&fcb, &entry) == 0) {
    fcb_read(&fcb, &entry, read_buf, entry.fe_data_len);
}

07 Settings 存储

Settings 子系统提供键值对持久化存储。

// Settings 配置
// CONFIG_SETTINGS=y
// CONFIG_SETTINGS_FS=y

// 设置值
settings_save_one("my_module/device_name", "sensor_01", 11);

// 加载值
char name[32];
size_t len = sizeof(name);
settings_load_one("my_module/device_name", name, &len);

// 注册 handler
static int settings_set(const char *key, size_t len,
                        settings_read_cb read_cb, void *cb_arg)
{
    if (!strcmp(key, "device_name")) {
        read_cb(cb_arg, device_name, sizeof(device_name));
    }
    return 0;
}

SETTINGS_STATIC_HANDLER_DEFINE(my_module, "my_module",
                               NULL, settings_set, NULL, NULL);
