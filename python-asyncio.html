01 事件循环

事件循环是 asyncio 的核心调度器。它负责注册、执行和取消任务，处理 I/O 就绪通知，以及在协程之间切换上下文。

获取与设置事件循环

import asyncio

# 获取当前运行的事件循环（3.10+ 推荐方式）
loop = asyncio.get_running_loop()  # 无运行循环则抛出 RuntimeError

# 获取或创建事件循环（3.10 之前的方式，3.12 弃用）
loop = asyncio.get_event_loop()     # 没有则创建新的

# 创建新的事件循环
loop = asyncio.new_event_loop()
asyncio.set_event_loop(loop)

# 3.10+ 推荐：直接使用 asyncio.run()
async def main():
    print("Hello asyncio")

asyncio.run(main())  # 自动创建和关闭事件循环

run_until_complete 与 run_forever

# run_until_complete：运行协程直到完成
async def fetch_data():
    await asyncio.sleep(1)
    return {"data": "result"}

loop = asyncio.new_event_loop()
try:
    result = loop.run_until_complete(fetch_data())
    print(result)
finally:
    loop.close()

# run_forever：永久运行，需要手动停止
async def periodic_task():
    while True:
        print("Tick")
        await asyncio.sleep(1)

loop = asyncio.new_event_loop()
asyncio.ensure_future(periodic_task(), loop=loop)
try:
    loop.run_forever()  # 直到 loop.stop() 被调用
except KeyboardInterrupt:
    pass
finally:
    loop.close()

自定义事件循环策略

import asyncio
import platform

# 选择事件循环实现
if platform.system() == "Windows":
    # Windows 上使用 ProactorEventLoop（默认 3.8+）
    asyncio.set_event_loop_policy(asyncio.WindowsProactorEventLoopPolicy())
else:
    # Linux 上使用 SelectorEventLoop 或 uvloop

asyncio.gather 与协程并发

async def fetch_url(url: str) -> str:
    print(f"开始请求 {url}")
    await asyncio.sleep(1)  # 模拟网络延迟
    print(f"完成请求 {url}")
    return f""

async def concurrent_fetch():
    # 并行执行多个协程
    results = await asyncio.gather(
        fetch_url("https://api.example.com/1"),
        fetch_url("https://api.example.com/2"),

03 Task 与 Future

Task 的创建与管理

import asyncio

async def background_work(task_id: int, duration: float):
    print(f"任务 {task_id} 开始")
    await asyncio.sleep(duration)
    print(f"任务 {task_id} 完成")
    return f"结果 {task_id}"

async def main():
    # 使用 create_task 创建任务（立即调度）
    task1 = asyncio.create_task(background_work(1, 2.0))
    task2 = asyncio.create_task(background_work(2, 1.5))

    # 任务运行中可以做其他事情
    print("主协程继续工作...")

    # 等待任务完成
    result1 = await task1
    result2 = await task2

    print(result1, result2)

    # 检查任务状态
    print(f"task1 done: {task1.done()}")
    print(f"task1 cancelled: {task1.cancelled()}")

    # 任务取消
    task3 = asyncio.create_task(background_work(3, 10))
    task3.cancel()  # 在任务内部抛出 CancelledError
    try:
        await task3
    except asyncio.CancelledError:
        print("任务 3 被取消")

gather / wait / as_completed 对比

async def demo_gather():
    """gather：并发执行，返回有序结果列表"""
    results = await asyncio.gather(
        asyncio.sleep(1, result="A"),
        asyncio.sleep(2, result="B"),
        asyncio.sleep(3, result="C"),
    )
    print(results)  # ['A', 'B', 'C'] — 顺序与输入一致

async def demo_wait():
    """wait：细粒度控制，返回 (done, pending) 元组"""
    tasks = {
        asyncio.create_task(asyncio.sleep(1, result="A")),
        asyncio.create_task(asyncio.sleep(2, result="B")),
        asyncio.create_task(asyncio.sleep(3, result="C")),
    }
    done, pending = await asyncio.wait(
        tasks,
        timeout=2.5,           # 超时返回部分结果
        return_when=asyncio.FIRST_COMPLETED,  # 或 ALL_COMPLETED, FIRST_EXCEPTION
    )
    for task in done:
        print(task.result())   # A, B (C 仍在运行)
    for task in pending:
        task.cancel()          # 清理未完成的任务

async def demo_as_completed():
    """as_completed：按完成顺序逐个处理"""
    tasks = [
        asyncio.create_task(asyncio.sleep(2, result="慢速")),
        asyncio.create_task(asyncio.sleep(1, result="快速")),
        asyncio.create_task(asyncio.sleep(3, result="最慢")),
    ]
    for coro in asyncio.as_completed(tasks, timeout=3):
        earliest = await coro
        print(f"完成: {earliest}")  # 快速 → 慢速 → 最慢

shield — 保护任务不被取消

async def critical_operation():
    """不可中断的关键操作"""
    print("关键操作开始")
    await asyncio.sleep(2)
    print("关键操作完成")
    return "关键数据"

async def safe_operation():
    """使用 shield 保护子操作"""
    try:
        # shield 防止内部协程被外部取消传播
        result = await asyncio.shield(critical_operation())
        return result
    except asyncio.CancelledError:
        print("外部已取消，但 shield 确保内部完成")
        # 如果 shield 内部的协程还在运行，取消不会传播进去
        raise

async def main():
    task = asyncio.create_task(safe_operation())
    await asyncio.sleep(0.5)
    task.cancel()  # 尝试取消
    try:
        await task
    except asyncio.CancelledError:
        print("主协程捕获取消")

TaskGroup（3.11+ 结构化并发）

import asyncio

async def worker(n: int):
    await asyncio.sleep(n)
    if n == 2:
        raise ValueError(f"worker {n} 失败")
    return f"worker {n} 完成"

async def main():
    # TaskGroup 提供结构化并发
    # 任何一个任务抛出异常，所有任务被取消
    try:
        async with asyncio.TaskGroup() as tg:
            t1 = tg.create_task(worker(1))
            t2 = tg.create_task(worker(2))  # 将抛出异常
            t3 = tg.create_task(worker(3))
        # 所有任务正常完成才到达这里
    except ExceptionGroup as eg:
        print(f"捕获到异常组: {eg.exceptions}")
        # ExceptionGroup 封装所有子任务异常

    # 传统方式 vs TaskGroup
    # 传统：需要手动管理取消和异常聚合
    # TaskGroup：自动取消 + ExceptionGroup 聚合

04 异步上下文管理器

异步上下文管理器通过 __aenter__ 和 __aexit__ 协议管理异步资源生命周期。

import asyncio
from contextlib import asynccontextmanager

# 基于类的异步上下文管理器
class AsyncConnection:
    """模拟异步数据库连接"""
    async def connect(self):
        print("连接数据库...")
        await asyncio.sleep(0.5)
        self.connected = True
        print("数据库已连接")

    async def close(self):
        print("关闭连接...")
        await asyncio.sleep(0.3)
        self.connected = False
        print("连接已关闭")

    async def query(self, sql: str) -> str:
        if not self.connected:
            raise RuntimeError("未连接")
        await asyncio.sleep(0.2)
        return f"结果: {sql}"

    async def __aenter__(self):
        await self.connect()
        return self

    async def __aexit__(self, exc_type, exc_val, exc_tb):
        await self.close()
        # 返回 True 抑制异常
        return False  # 不抑制

async def use_connection():
    async with AsyncConnection() as conn:
        result = await conn.query("SELECT 1")
        print(result)
    # 退出时自动关闭连接

# 基于装饰器的异步上下文管理器
@asynccontextmanager
async def async_managed_resource(name: str):
    print(f"获取资源: {name}")
    resource = {"name": name, "ready": True}
    try:
        yield resource
    finally:
        print(f"释放资源: {name}")
        await asyncio.sleep(0.1)

async def use_resource():
    async with async_managed_resource("database") as res:
        print(f"使用资源: {res['name']}")

# 异步 ExitStack
from contextlib import AsyncExitStack

async def manage_multiple():
    """同时管理多个异步上下文"""
    async with AsyncExitStack() as stack:
        conn1 = await stack.enter_async_context(AsyncConnection())
        conn2 = await stack.enter_async_context(AsyncConnection())
        # 同时使用两个连接
        r1 = await conn1.query("SELECT 1")
        r2 = await conn2.query("SELECT 2")
        print(r1, r2)
    # 按逆序自动关闭所有连接
提示:
异步上下文管理器最常见的应用场景是数据库连接、HTTP 会话和文件操作。始终优先使用 async with 而非手动调用 __aenter__/__aexit__。AsyncExitStack 在动态管理不确定数量的异步资源时特别有用。

05 异步迭代器

异步迭代器通过 __aiter__ 和 __anext__ 协议支持在异步环境中逐个产生值。

import asyncio

# 基于类的异步迭代器
class AsyncCounter:
    """异步计数器，每秒递增"""
    def __init__(self, limit: int):
        self.limit = limit
        self.current = 0

    def __aiter__(self):
        return self

    async def __anext__(self):
        if self.current >= self.limit:
            raise StopAsyncIteration
        self.current += 1
        await asyncio.sleep(1)
        return self.current

async def use_async_iter():
    async for value in AsyncCounter(5):
        print(f"计数: {value}")

# 异步生成器（推荐方式）
async def async_counter(limit: int, delay: float = 1):
    """使用异步生成器产生值"""
    for i in range(limit):
        await asyncio.sleep(delay)
        yield i  # yield 在 async def 中创建异步生成器

async def use_async_gen():
    async for value in async_counter(5, 0.5):
        print(f"值: {value}")

# 实用场景：异步分页读取
async def paginated_fetch(url: str, page_size: int = 100):
    page = 1
    while True:
        data = await fetch_page(url, page, page_size)
        if not data:
            break
        yield data
        page += 1

async def process_all():
    async for page in paginated_fetch("https://api.example.com/items"):
        print(f"处理了 {len(page)} 条记录")

# 异步迭代器转换
async def transform_example():
    # 异步推导式（Python 3.6+）
    results = [x async for x in async_counter(5)]
    print(results)  # [0, 1, 2, 3, 4]

    # 异步列表推导 + 条件过滤
    even = [x async for x in async_counter(10) if x % 2 == 0]
    print(even)  # [0, 2, 4, 6, 8]

    # 异步字典推导
    mapping = {f"key-{x}": f"val-{x}" async for x in async_counter(5)}
    print(mapping)

06 异步队列

asyncio.Queue 是线程安全的异步 FIFO 队列，用于协程间通信和生产者-消费者模式。

import asyncio
import random

# 基础队列
async def producer(queue: asyncio.Queue, n: int):
    for i in range(n):
        item = f"item-{i}"
        await queue.put(item)
        print(f"生产: {item}")
        await asyncio.sleep(random.random())
    await queue.put(None)  # 信号结束

async def consumer(queue: asyncio.Queue, name: str):
    while True:
        item = await queue.get()
        if item is None:  # 收到结束信号
            queue.task_done()
            break
        print(f"消费者 {name} 处理: {item}")
        await asyncio.sleep(random.random() * 1.5)
        queue.task_done()

async def main():
    queue = asyncio.Queue(maxsize=5)  # 缓冲区大小
    # 启动生产者和多个消费者
    prod = asyncio.create_task(producer(queue, 10))
    consumers = [
        asyncio.create_task(consumer(queue, f"C{i}"))
        for i in range(3)
    ]
    await asyncio.gather(prod, *consumers)

# 队列类型
async def queue_types():
    # FIFO 队列（默认）
    fifo = asyncio.Queue()
    await fifo.put("first")
    await fifo.put("second")
    print(await fifo.get())  # first

    # LIFO 队列（栈）
    lifo = asyncio.LifoQueue()
    await lifo.put("first")
    await lifo.put("second")
    print(await lifo.get())  # second

    # 优先级队列
    pq = asyncio.PriorityQueue()
    await pq.put((2, "中等优先级"))
    await pq.put((1, "高优先级"))
    await pq.put((3, "低优先级"))

    while not pq.empty():
        priority, task = await pq.get()
        print(f"优先级 {priority}: {task}")

# queue.join() 与 task_done()
async def coordinated_work():
    q = asyncio.Queue()

    async def worker():
        while True:
            item = await q.get()
            try:
                await asyncio.sleep(0.5)
            finally:
                q.task_done()

    workers = [asyncio.create_task(worker()) for _ in range(5)]
    for i in range(20):
        await q.put(i)

    await q.join()  # 等待所有项目被处理
    print("所有任务完成")
提示:
Queue.join() 阻塞直到队列中所有项目都被 task_done() 标记完成。这是协调生产者和消费者的推荐方式。注意队列的 maxsize 参数控制背压，避免内存无限增长。

07 同步原语

asyncio 提供了与 threading 模块对应的异步同步原语，用于协程间的同步与互斥。

Lock（互斥锁）

import asyncio

shared_resource = 0

async def increment(lock: asyncio.Lock, n: int):
    global shared_resource
    for _ in range(n):
        async with lock:  # 异步获取锁
            current = shared_resource
            await asyncio.sleep(0.01)  # 模拟处理
            shared_resource = current + 1

async def test_lock():
    lock = asyncio.Lock()
    tasks = [asyncio.create_task(increment(lock, 100)) for _ in range(3)]
    await asyncio.gather(*tasks)
    print(f"最终值: {shared_resource}")  # 300（没有锁会小于 300）

Semaphore（信号量）

async def limited_request(sem: asyncio.Semaphore, url_id: int):
    """限制最多同时 3 个并发请求"""
    async with sem:
        print(f"请求 {url_id} 开始")
        await asyncio.sleep(1)
        print(f"请求 {url_id} 完成")

async def test_semaphore():
    sem = asyncio.Semaphore(3)  # 最多 3 个并发
    tasks = [limited_request(sem, i) for i in range(10)]
    await asyncio.gather(*tasks)
    # 总耗时 ~4s（10/3 向上取整）而不是 10s

Event（事件）

async def waiter(event: asyncio.Event, name: str):

    await event.wait()  # 阻塞直到 set()

async def setter(event: asyncio.Event):

    event.set()

async def test_event():

Condition（条件变量）

async def consumer(cv: asyncio.Condition, name: str, items: list):
    async with cv:
        while not items:
            print(f"{name} 等待项目...")
            await cv.wait()  # 释放锁并等待 notify
        item = items.pop(0)
        print(f"{name} 消费: {item}")
    return item

    async with cv:
        items.append("新项目")
        print("生产了新项目")
        cv.notify(1)  # 唤醒一个等待者
        # cv.notify_all() 唤醒所有

08 超时与取消

超时管理是异步编程的核心问题。asyncio 提供了多种超时和取消机制。

wait_for — 带超时的 await

import asyncio

async def slow_operation():
    await asyncio.sleep(10)
    return "done"

async def use_wait_for():
    try:
        result = await asyncio.wait_for(slow_operation(), timeout=2.0)
        print(result)
    except asyncio.TimeoutError:
        print("操作超时！")
        # slow_operation 内部被取消（CancelledError）

# wait_for 底层行为：超时后，内部协程被取消
# 这意味着如果协程有 finally 或 try/finally，它会被执行

async def cleanup_operation():
    try:
        await asyncio.sleep(10)
    finally:
        print("清理中...")
        await asyncio.sleep(0.5)

# 超时后，cleanup 的 finally 块会执行

timeout（3.11+）— 上下文管理器风格超时

async def work_with_timeout():
    """使用 asyncio.timeout 上下文管理器（3.11+）"""
    try:
        async with asyncio.timeout(2.0):
            await asyncio.sleep(10)  # 2 秒后超时
    except asyncio.TimeoutError:
        print("操作超时")

回调与性能分析

import asyncio
import time

# 自定义事件循环监控
class MonitoringLoop(asyncio.SelectorEventLoop):
    def __init__(self):
        super().__init__()
        self.task_times = {}

    def create_task(self, coro, **kwargs):
        task = super().create_task(coro, **kwargs)
        task_id = id(task)
        self.task_times[task_id] = time.monotonic()

        # 添加完成回调
        def on_done(t):
            if not t.cancelled() and not t.exception():
                elapsed = time.monotonic() - self.task_times[task_id]
                if elapsed > 0.5:
                    print(f"任务 {t.get_name()} 耗时 {elapsed:.2f}s")

        task.add_done_callback(on_done)
        return task

# 协程耗时分析工具
async def analyze_coroutines():
    """分析协程运行时间"""
    loop = asyncio.get_running_loop()

    async def timed_coro(name: str, delay: float):
        start = loop.time()
        await asyncio.sleep(delay)
        elapsed = loop.time() - start
        return {"name": name, "elapsed": elapsed}

    tasks = [
        timed_coro("fast", 0.1),
        timed_coro("medium", 0.5),
        timed_coro("slow", 1.0),
    ]

    results = await asyncio.gather(*tasks)
    for r in results:
        print(f"{r['name']}: {r['elapsed']:.2f}s")
