02 命令注册

// 定义命令
static BaseType_t vCommand( char *pcWriteBuffer,
                             size_t xWriteBufferLen,
                             const char *pcCommandString )
{
    // pcCommandString 包含完整命令行
    // pcWriteBuffer 用于输出结果
    // 返回 pdFALSE 表示输出完成

    strcpy( pcWriteBuffer, "Hello from command!\\r\\n" );
    return pdFALSE;
}

// 注册命令
static const CLI_CommandDefinition_t xCommand =
{
    "hello",                          // 命令字符串
    "hello\\r\\n  Print a greeting\\r\\n", // 帮助文本
    vCommand,                         // 回调函数
    -1                                // 参数数量（-1 = 无限制）
};

// 在 main 中注册
FreeRTOS_CLIRegisterCommand( &xCommand );

03 命令处理

// 初始化 CLI
FreeRTOS_CLIInit();

// 主循环处理输入
char cInputBuffer[ 50 ];
char cOutputBuffer[ 200 ];

while ( 1 )
{
    // 从串口读取一行
    if( xSerialGetChar( &cInputBuffer[ strlen( cInputBuffer ) ] ) )
    {
        // 处理命令
        while( FreeRTOS_CLIProcess( cInputBuffer, cOutputBuffer, 200 ) != pdFALSE )
        {
            // 输出多行结果
            printf( "%s", cOutputBuffer );
        }
    }
}

04 完整示例

// ===== FreeRTOS CLI 完整示例 =====

#include "FreeRTOS.h"
#include "FreeRTOS_CLI.h"
#include <stdio.h>

// help 命令
static BaseType_t vHelpCommand( char *pcWriteBuffer,
                                 size_t xWriteBufferLen,
                                 const char *pcCommandString )
{
    int i;
    for( i = 0; i < FreeRTOS_CLICommandRegistered(); i++ )
    {
        const CLI_CommandDefinition_t *pxCommand = FreeRTOS_CLIGetCommandDefinition( i );
        strcat( pcWriteBuffer, pxCommand->pcCommandString );
        strcat( pcWriteBuffer, "\\r\\n" );
    }
    return pdFALSE;
}

// status 命令
static BaseType_t vStatusCommand( char *pcWriteBuffer,
                                   size_t xWriteBufferLen,
                                   const char *pcCommandString )
{
    sprintf( pcWriteBuffer, "Tasks: %lu\\r\\nFree heap: %lu\\r\\n",
             uxTaskGetNumberOfTasks(), xPortGetFreeHeapSize() );
    return pdFALSE;
}

int main( void )
{
    // 注册命令
    const CLI_CommandDefinition_t xHelpCommand = {
        "help", "help: List all commands\\r\\n", vHelpCommand, -1
    };
    const CLI_CommandDefinition_t xStatusCommand = {
        "status", "status: Show system status\\r\\n", vStatusCommand, -1
    };

    FreeRTOS_CLIRegisterCommand( &xHelpCommand );
    FreeRTOS_CLIRegisterCommand( &xStatusCommand );

    // 初始化 CLI
    FreeRTOS_CLIInit();

    // 创建 CLI 任务
    xTaskCreate( vCLITask, "CLI", 512, NULL, 2, NULL );

    vTaskStartScheduler();

    for ( ;; ) {}
}
