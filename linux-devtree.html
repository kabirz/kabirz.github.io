01 概述

设备树（Device Tree）是一种硬件描述数据结构，以树形方式描述 SoC 和板级硬件信息。ARM 架构在 Linux 3.x 引入设备树后，替代了内核源码中的大量板级 C 代码，实现了硬件描述与驱动代码的分离。

设备树解决的问题
  • 消除板级代码：不再为每块板写 C 代码
  • 硬件可发现：内核无需硬编码硬件信息
  • 单一镜像：一个内核支持多块板
  • 供应商友好：厂商可独立维护 DT 文件
  • 动态修改：DT Overlay 支持运行时修改
文件格式
格式扩展名说明
DTS*.dts源码（文本）
DTSI*.dtsi包含文件（公共定义）
DTC*.dtb二进制（编译后）
DTSO*.dtsoOverlay（运行时修改）

02 DTB 格式

DTB 内存布局

┌─────────────────────────────┐  ← DTB 起始
│  Header (40 字节)            │  magic + totalsize + off_struct
├─────────────────────────────┤
│  Memory Reservation Block   │  独占内存区域描述
├─────────────────────────────┤
│  Structure Block            │  节点/属性/数据（核心）
├─────────────────────────────┤
│  Strings Block              │  属性名和字符串常量
└─────────────────────────────┘  ← DTB 结束

Header 结构

// include/linux/libfdt.h
struct fdt_header {
    uint32_t magic;              // 0xD00DFEED
    uint32_t totalsize;           // DTB 总大小
    uint32_t off_dt_struct;       // Structure Block 偏移
    uint32_t off_dt_strings;      // Strings Block 偏移
    uint32_t off_mem_rsvmap;      // Memory Reservation 偏移
    uint32_t version;             // DT 格式版本
    uint32_t last_comp_version;   // 最低兼容版本
    uint32_t boot_cpuid_phys;    // 引导 CPU 物理 ID
    uint32_t size_dt_strings;     // Strings Block 大小
    uint32_t size_dt_struct;     // Structure Block 大小
};

Structure Block token

Token值说明
FDT_BEGIN_NODE0x00000001节点开始
FDT_END_NODE0x00000002节点结束
FDT_PROP0x00000003属性（长度+名称偏移+值）
FDT_NOP0x00000004空操作
FDT_END0x00000009结构结束

03 语法基础

完整 DTS 示例

// 设备树源文件 (DTS)
/dts-v1/;
/plugin/;  // 可选：Overlay

// 包含公共定义
#include "soc-xxx.dtsi"

// 根节点
/ {
    model = "My Board Rev A";
    compatible = "vendor,my-board", "vendor,soc-xxx";
    #address-cells = <2>;
    #size-cells = <2>;

    // 内存节点
    memory@0x80000000 {
        device_type = "memory";
        reg = <0x0 0x80000000 0x0 0x40000000>;  // 1GB @ 0x80000000
    };

    // 选择节点
    chosen {
        bootargs = "console=ttyS0,115200 root=/dev/mmcblk0p2";
        stdout-path = "&uart0";
    };

    // SoC 节点
    soc@0x0 {
        compatible = "simple-bus";
        #address-cells = <1>;
        #size-cells = <1>;
        ranges;

        uart0: serial@0x10000000 {
            compatible = "ns16550a";
            reg = <0x10000000 0x1000>;
            interrupts = <0 1 IRQ_TYPE_LEVEL_HIGH>;
            clock-frequency = <24000000>;
        };
    };
};

数据类型

类型格式示例
字符串"string"compatible = "vendor,device"
整数<value>reg = <0x10000000 0x1000>
整数数组<v1 v2 ...>interrupts = <0 1 IRQ_TYPE_LEVEL_HIGH>
字符串列表"s1", "s2"compatible = "a,b", "c,d"
空（空）status; // 或 status = ""
句柄引用&labelpinctrl-0 = &uart0_pins

特殊属性

属性说明
compatible兼容字符串（驱动匹配用）
reg寄存器地址/大小对
reg-namesreg 各段的名称
interrupts中断号（父控制器 + 触发类型）
interrupt-parent中断控制器引用
ranges地址转换范围（子→父）
status"okay"（启用）/ "disabled"（禁用）
phandle节点的唯一标识符
#address-cells子节点 reg 中地址字段的 cell 数
#size-cells子节点 reg 中大小字段的 cell 数
#interrupt-cells子节点 interrupts 中的 cell 数
#clock-cells时钟标识的 cell 数
#gpio-cellsGPIO 标识的 cell 数

引用与覆盖

// 节点标签（label）
uart0: serial@0x10000000 { ... };

// 引用标签
pinctrl-0 = &uart0_pins;

// 属性覆盖（同一节点）
uart0: serial@0x10000000 {
    status = "okay";
    clock-frequency = <48000000>;  // 覆盖 dtsi 中的值
};

// 节点删除
/delete-node/ &uart1;
/delete-property/ &uart0.status;

常见 #cells 格式

类型格式说明
#address-cells1=32位, 2=64位reg 中的地址宽度
#size-cells1=32位, 2=64位reg 中的大小宽度
#interrupt-cells3（ARM GIC）[SPI 类型, SPI 号, 触发类型]
#clock-cells0 或 10=唯一标识, 1=带索引
#gpio-cells2[GPIO 号, 标志]
#phy-cells0-2PHY 标识
#thermal-sensor-cells0 或 1温度传感器

04 内核 API

节点查找

// 按完整路径查找
struct device_node *np = of_find_node_by_path("/soc/uart0");

// 按 compatible 查找
np = of_find_compatible_node(NULL, NULL, "ns16550a");

// 按设备类型查找
np = of_find_node_by_type(NULL, "serial");

// 按名称查找（匹配 @ 前的部分）
np = of_find_node_opts_by_name(NULL, "uart0", &opts);

// 按 phandle 查找
np = of_find_node_by_phandle(phandle);

// 遍历所有 compatible 匹配的节点
struct of_device_id table[] = {
    { .compatible = "vendor,device-v1" },
    { .compatible = "vendor,device-v2" },
    { }
};
for_each_matching_node(np, table) { ... }

属性读取

// 读取 u32 属性
u32 val;
of_property_read_u32(np, "clock-frequency", &val);

// 读取 u64 属性
u64 val64;
of_property_read_u64(np, "reg", &val64);

// 读取字符串属性
const char *str;
of_property_read_string(np, "status", &str);

// 读取字符串数组
const char *strs[4];

// 读取 GPIO

gpio = devm_gpiod_get_index(dev, "enable", 0, GPIOD_OUT_LOW);

// 读取中断

// 读取时钟

05 平台设备

设备树中的 SoC 内部设备（UART、I2C、SPI、GPIO 等）被映射为 platform_device，通过 of_platform_bus_type 自动匹配驱动。

自动注册流程

DT 节点（status="okay"）
↓ early_initcall
of_platform_populate()
遍历 DT 节点
↓
of_platform_device_create_pdata()
创建 platform_device
↓
device_add()
添加到设备模型
↓
driver_match_device()
compatible 匹配
↓
driver_probe_device()
调用 driver.probe()

手动创建

// 从 DT 节点创建 platform_device
struct platform_device *pdev = of_platform_device_create_pdata(np, NULL, &pdata, dev);

// 遍历所有子节点创建
of_platform_populate(np, NULL, NULL, dev);

驱动匹配

// 驱动中匹配设备树
static const struct of_device_id my_of_match[] = {
    { .compatible = "vendor,soc-uart", },
    { .compatible = "vendor,soc-uart-v2", },
    { /* sentinel */ }
};
MODULE_DEVICE_TABLE(of, my_of_match);

static struct platform_driver my_driver = {
    .probe = my_probe,
    .driver = {
        .name = "my-uart",
        .of_match_table = my_of_match,  // 优先使用 DT 匹配
        .acpi_match_table = ACPI_PTR(my_acpi_match),
    },
};

06 绑定规范

设备树绑定（Binding）定义了每个设备节点的属性格式，是设备树的"接口文档"。

绑定文档结构

// Documentation/devicetree/bindings/serial/ns16550.yaml
%YAML 1.2
---
$id: http://devicetree.org/schemas/serial/ns16550.yaml#
$schema: http://devicetree.org/meta/schemas/core.yaml#

title: National Semiconductor 16550A UART

properties:
  compatible:
    const: ns16550a

  reg:
    minItems: 2
    maxItems: 2
    description: Register base and size

  interrupts:
    maxItems: 1

  clock-frequency:
    default: <1843200>
    description: Reference clock frequency in Hz

  reg-shift:
    default: 0
    description: Register address shift

  reg-io-width:
    default: 1
    enum: [1, 2, 4]
    description: MMIO access width

required:
  - compatible
  - reg

Binding 规范

查找绑定：所有官方绑定文档在 Documentation/devicetree/bindings/ 目录下，使用 YAML 格式。新绑定必须通过 Linux 内核邮件列表审核。使用 make dt_binding_check 验证 YAML 语法。

07 DT Overlay

DT Overlay 允许在运行时动态修改设备树，广泛用于 Raspberry Pi HAT、FPGA 加载模块等场景。

Overlay 工作原理

基础 DT
SoC + 主板硬件
+
Overlay DT
HAT / 模块 / 配置
↓ 插桩 (overlay_apply)
合并后的 DT
完整硬件描述

Overlay DTSO 格式

// overlay.dtso — Raspberry Pi HAT 示例
/dts-v1/;
/plugin/;

#include "dt-bindings/leds/common.h"

/ {
    compatible = "brcm,bcm2835";
};

&i2c1 {
    status = "okay";

    sensor@48 {
        compatible = "vendor,my-sensor";
        reg = <0x48>;
    };
};

&gpio {
    my-led {
        compatible = "gpio-leds";
        led0 {
            gpios = &gpio 17 GPIO_ACTIVE_LOW;
            default-state = "on";
        };
    };
};

内核 API

// 应用 Overlay
struct of_overlay_fdt_arg arg = { .fdt = overlay_fdt };
struct of_overlay_id id;
of_overlay_fdt_apply(arg.fdt, arg.size, &id, NULL);

// 移除 Overlay
of_overlay_remove(&id);

// 查找 Overlay 状态
of_overlay_is_live(np);
Overlay 限制：并非所有属性都支持修改。某些内核内部状态（如中断路由）在 Overlay 后可能需要重新初始化。Raspberry Pi 的 config.txt 和 overlays/ 目录是 Overlay 的典型应用。
