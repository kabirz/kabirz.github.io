01 构建系统概览

02 West 元工具

West 是 Zephyr 的统一入口工具，管理多仓库、构建、调试和闪存。

常用命令

# 构建
west build -b  
west build -b stm32f4_disco samples/basic/blinky

# 闪存
west flash
west flash --runner jlink

# 调试
west debug
west attach

# 仓库管理
west init -l 
west update
west list

# 符号表
west build -t ram_report
west build -t rom_report

# 清理
west build -t clean
west build -t pristine

West 清单（west.yml）

# west.yml - 仓库清单
manifest:
  remotes:
    - name: zephyrproject
      url-base: https://github.com/zephyrproject

  projects:
    - name: zephyr
      remote: zephyrproject
      revision: main
      path: zephyr

    - name: hal_stm32
      remote: zephyrproject
      revision: main
      path: modules/hal/stm32
      groups:
        - hal

  self:
    path: application

CMake API

# 应用 CMakeLists.txt 示例
cmake_minimum_required(VERSION 3.20.0)

find_package(Zephyr REQUIRED HINTS $ENV{ZEPHYR_BASE})
project(my_application)

# 添加源文件
target_sources(app PRIVATE src/main.c src/sensor.c)

# 添加包含目录
target_include_directories(app PRIVATE src)

# 条件编译
target_sources_ifdef(CONFIG_SENSOR app PRIVATE src/sensor.c)
target_sources_ifdef(CONFIG_NETWORKING app PRIVATE src/network.c)

# 添加宏定义
target_compile_definitions(app PRIVATE MY_MACRO=1)

# 链接库
target_link_libraries(app PRIVATE some_library)

Zephyr CMake 函数

# 条件源文件
target_sources_ifdef(CONFIG_XXX target PRIVATE file.c)

# 条件包含
target_include_directories_ifdef(CONFIG_XXX target PRIVATE dir)

# 条件定义
target_compile_definitions_ifdef(CONFIG_XXX target PRIVATE DEF)

# 生成二进制
zephyr_executable(app)

# 添加设备树包含
zephyr_include_directories(${ZEPHYR_BASE}/dts)

# 嵌入二进制
zephyr_linker_sources(SECTIONS sections.ld)

04 Kconfig 配置

Kconfig 系统管理所有内核和驱动的配置选项。

Kconfig 文件

# Kconfig 配置示例
# drivers/Kconfig

menu "GPIO Driver"

config GPIO_NRF
    bool "Nordic nRF GPIO Driver"
    default y
    depends on HAS_GPIO
    help
      Enable the Nordic nRF GPIO driver.

config GPIO_NRF_INT_PIN
    int "GPIO interrupt pin"
    range 0 31
    depends on GPIO_NRF
    help
      GPIO pin number for interrupt.

endmenu

Kconfig 语法

语法说明
bool布尔选项（y/n）
int整数选项
string字符串选项
tristate三态选项（y/n/m）
depends on依赖条件
select自动选择
default默认值
help帮助文本

配置优先级

# 配置优先级（从高到低）：
# 1. application/prj.conf              ← 用户配置
# 2. boards/_defconfig           ← 板级默认
# 3. arch//Kconfig                ← 架构配置
# 4. subsys/Kconfig, drivers/Kconfig    ← 子系统默认
# 5. Kconfig.zephyr                     ← 内核默认

# menuconfig 界面
west build -t menuconfig

05 Devicetree 编译器

#define DT_REG_SIZE(DT_N_S_soc_S_uart_40002000)  0x1000

06 交叉编译

工具链配置

# Zephyr SDK 工具链（推荐）
# 安装后自动配置
# arm-zephyr-eabi-gcc
# riscv64-zephyr-elf-gcc
# x86_64-zephyr-elf-gcc

# 使用外部工具链
west build -b stm32f4_disco samples/basic/blinky \
  -- -DCMAKE_TOOLCHAIN_FILE=/toolchain.cmake

# 指定工具链变量
west build -b stm32f4_disco samples/basic/blinky \
  -- -DCROSS_COMPILE=arm-zephyr-eabi-

Board 配置

# 列出可用的 board
west boards

# Board 目录结构
boards/arm/stm32f4_disco/
├── stm32f4_disco.dts           # 设备树
├── stm32f4_disco_defconfig     # 默认配置
├── board.cmake                 # CMake 板级配置
├── pins.yaml                   # 引脚配置
└── README.rst                  # 说明

07 CMake API

# Zephyr 提供的 CMake 函数

target_compile_definitions_ifdef(CONFIG_XXX target PRIVATE DEF)
target_link_libraries_ifdef(CONFIG_XXX target PRIVATE lib)

zephyr_generate_cmake_file()

zephyr_linker_sources(SECTIONS linker.ld)

zephyr_dt_find_compatible()

zephyr_library()

zephyr_library_include_directories(include)

08 高级构建

构建选项

# 构建类型
west build -b   -DCMAKE_BUILD_TYPE=Debug
west build -b   -DCMAKE_BUILD_TYPE=Release

# 生成报告
west build -t ram_report    # RAM 使用报告
west build -t rom_report    # ROM 使用报告
west build -t initlevels    # 初始化级别报告

# 并行构建
west build -b   -- -GNinja -DCMAKE_BUILD_PARALLEL_LEVEL=8

# 增量构建
west build                  # 自动增量

# 完全重新构建
west build -t pristine
west build -b

自定义构建

# 自定义板级 CMake
# boards//board.cmake

board_runner_args(jlink "--device=STM32F407VG")
board_runner_args(stm32 "--freq=8000000")

include(${ZEPHYR_BASE}/boards/common/jlink.board.cmake)
include(${ZEPHYR_BASE}/boards/common/stm32.board.cmake)
阅读建议:
CMakeLists.txt（根目录）内核 CMake 入口。
Kconfig.zephyr 内核配置选项定义。
cmake/modules/ CMake 模块目录。
scripts/gen_defines.py Devicetree 代码生成器。
boards/ 板级支持定义。
