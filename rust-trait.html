01 概述

Trait 是 Rust 的接口抽象机制，灵感来自 Haskell 的 Type Classes。定义一组方法签名，由类型实现。

与其他语言对比：

语言接口机制特点
Java/C#interfacenominal，显式 implements
Gointerfacestructural，鸭子类型
C++虚函数 / concept模板 + CRTP
Pythonduck typing无接口
Rusttraitnominal + ad-hoc polymorphism

核心特征：

  • 类型类（Type Classes）：每个类型独立实现 trait
  • 零成本抽象：静态分发可内联
  • 组合优于继承：无类层次，用 trait 组合
  • 支持默认方法：trait 可提供默认实现
  • 支持泛型约束：trait bound

02 Trait 定义与实现

// 定义
pub trait Draw {
    fn draw(&self);
    
    // 默认方法
    fn draw_twice(&self) {
        self.draw();
        self.draw();
    }
}

// 实现
struct Circle { radius: f64 }
struct Square { side: f64 }

impl Draw for Circle {
    fn draw(&self) {
        println!("circle r={}", self.radius);
    }
}

impl Draw for Square {
    fn draw(&self) {
        println!("square side={}", self.side);
    }
}

// 使用
let shapes: Vec<Box<dyn Draw>> = vec![
    Box::new(Circle { radius: 1.0 }),
    Box::new(Square { side: 2.0 }),
];
for s in &shapes {
    s.draw();
}

// Trait 也可以包含关联常量、关联类型：
trait Container {
    type Item;          // 关联类型
    const MAX: usize;   // 关联常量
    
    fn first(&self) -> Option<&Self::Item>;
}

// 实现关联类型
impl Container for Vec<i32> {
    type Item = i32;
    const MAX: usize = 100;
    
    fn first(&self) -> Option<&i32> { self.first() }
}

// Trait 可继承（Supertrait）：
trait Animal: std::fmt::Debug {
    fn name(&self) -> &str;
}
trait Dog: Animal {
    fn bark(&self);
}
// Dog 要求实现 Animal

孤儿规则 (Orphan Rule)：

// 规则：impl Trait for Type 必须满足
//   - Trait 在当前 crate 定义，或
//   - Type 在当前 crate 定义

// ❌ 不能在 crate A 中：
impl Display for Vec<T> { ... }   // 都来自 std

// ✅ 在 crate A 中：
impl MyTrait for Vec<T> { ... }     // MyTrait 是 crate A 的
impl Display for MyType { ... }       // MyType 是 crate A 的

// 这防止 trait 实现冲突
// 保证全局一致的实现

03 Trait Bounds

// 泛型函数 + bound
fn print<T: Display>(x: T) {
    println!("{}", x);
}

// 多重 bound
fn debug_print<T: Display + Debug>(x: T) {
    println!("{}, {:?}", x, x);
}

// where 子句（更可读）
fn complex<T, U>(t: T, u: U)
where
    T: Display + Clone,
    U: Debug + Send,
{
    ...
}

// 结构体泛型 + bound
struct Pair<T: Clone> { a: T, b: T }

// impl 块中的 bound
impl<T: PartialOrd> Pair<T> {
    fn smaller(&self) -> &T {
        if self.a < self.b { &self.a } else { &self.b }
    }
}

// 关联类型 bound（更复杂）
trait Parse: Sized {
    fn parse(s: &str) -> Option<Self>;
}

fn parse_all<T: Parse>(items: &[&str]) -> Vec<T> {
    items.iter().filter_map(|&s| T::parse(s)).collect()
}

// lifetime bound
fn longest<'a, T>(x: &'a T, y: &'a T) -> &'a T
where
    T: PartialOrd,
{
    if x >= y { x } else { y }
}

impl Trait 语法（位置 trait）：

// 函数参数（匿名 trait bound）
fn print(x: impl Display) {
    println!("{}", x);
}

// 函数返回（编译期单态化）
fn make() -> impl Iterator<Item = i32> {
    (0..10).into_iter()
}

// 多个不同类型不能直接 impl Trait：
fn make(flag: bool) -> impl Iterator<Item=i32> {
    if flag {
        (0..10).into_iter()  // Range<i32>
    } else {
        vec![1, 2, 3].into_iter()  // std::vec::IntoIter<i32>
        // ❌ 不同类型
    }
}

// 解决：Box<dyn Trait> 或 enum dispatch

04 静态分发

泛型 + Trait bound = 静态分发（单态化），每个具体类型生成一份代码：

// 源代码：
fn print<T: Display>(x: T) {
    println!("{}", x);
}
print(5);              // T = i32
print("hi");           // T = &str

// 编译后（单态化）：
fn print_i32(x: i32) {
    println!("{}", x);  // 调用 i32::fmt
}
fn print_str(x: &str) {
    println!("{}", x);  // 调用 str::fmt
}

// 优势：
//   - 无运行时开销（直接调用，可内联）
//   - 编译器可深度优化
//   - 类型推断
// 劣势：
//   - 二进制变大（每种类型一份代码）
//   - 编译时间增长

// structofarrays vs trait object 优化：
fn process_all<T: Process>(items: &[T]) {
    for i in items { i.process(); }
}
// 编译器生成具体 T 的循环，可向量化

代码膨胀与去重：

// 共享泛型代码：所有 T 相同大小用同一份代码
// LLVM 通过 GVN 去重

// 实际策略：
//   - 小函数（如访问器）：完全内联
//   - 大函数：保留泛型版本，所有相同 layout 类型共享
//   - 调试 build：基本不去重（每类型一份）

// dylib vs rlib：
//   - dylib：导出泛型实例（避免下游重复编译）
//   - rlib：上游 metadata 包含单态化信息

05 动态分发 (dyn)

// dyn Trait 是 trait 对象（动态分发）
fn print(x: &dyn Display) {
    println!("{}", x);
}
let n = 5;
let s = "hi";
print(&n);    // &i32 → &dyn Display
print(&s);   // &&str → &dyn Display

// 内部实现：胖指针 (fat pointer)
//   struct FatPtr {
//       data: *const (),       // 数据指针
//       vtable: *const VTable, // 虚函数表
//   }
//   struct VTable {
//       drop: fn(*mut ()),
//       size: usize,
//       align: usize,
//       method1: fn(*const ()),
//       method2: fn(*const ()),
//       ...
//   }

// Box<dyn Trait> — 拥有的 trait 对象
let v: Vec<Box<dyn Draw>> = vec![
    Box::new(Circle { radius: 1.0 }),
    Box::new(Square { side: 2.0 }),
];

// Object Safety 规则：
// trait 必须是 object safe 才能作为 dyn 使用
//   1. 不能返回 Self
//   2. 不能有泛型方法
//   3. 第一个参数不能是 Self 类型
//   4. 不能有 Sized 作为 supertrait

// ❌ 不是 object safe：
trait Clone {
    fn clone(&self) -> Self;  // 返回 Self
}
trait Factory {
    fn make<T>() -> T;          // 泛型方法
}

// ✅ object safe：
trait Draw {
    fn draw(&self);
}

// dyn 的代价：
//   - 一次虚函数调用（不能内联）
//   - 失去具体类型信息
//   - 不能直接访问字段
//   - 适合异构集合、运行时多态

dyn 与泛型对比：

泛型 <T: Trait>dyn Trait
分发时机编译期运行期
调用开销0（可内联）1 次虚函数调用
代码体积变大（单态化）小（共享）
类型异构不行可以（Vec<Box<dyn>>）
Object Safety无要求必需
典型场景性能敏感、静态插件、动态加载

06 关联类型 vs 泛型

// 关联类型（推荐）
trait Iterator {
    type Item;
    fn next(&mut self) -> Option<Self::Item>;
}

impl Iterator for Counter {
    type Item = u32;
    fn next(&mut self) -> Option<u32> { ... }
}

// 使用：
let c = Counter::new();
for x in c { /* x 是 u32 */ }

// 泛型版本（不推荐）
trait Iterator<Item> {
    fn next(&mut self) -> Option<Item>;
}
// 问题：一个类型可以多次实现（不同 Item）
impl Iterator<u32> for Foo { ... }
impl Iterator<String> for Foo { ... }
// 调用方要明确类型

// 关联类型的优势：
//   1. 每个类型一种"输出类型"（更清晰）
//   2. 使用方不需写 Iterator<T, Item=u32>
//   3. trait bound 更简洁：T: Iterator<Item=u32>

// 高阶：关联类型 + bound
trait Graph {
    type Node: Copy;
    type Edge;
    fn nodes(&self) -> Vec<Self::Node>;
}

fn shortest_path<G: Graph>(g: &G, from: G::Node, to: G::Node)
    where G::Node: Eq + Hash
{
    ...
}

Generic Associated Types (GAT)：

// 1.65+ 稳定：trait 中的泛型关联类型
trait LendingIterator {
    type Item<'a> where Self: 'a;
    fn next(&mut self) -> Option<Self::Item<'_>>;
}

// 用途：返回值的 lifetime 与调用绑定
struct Buffer { data: Vec<u8>, pos: usize }
impl LendingIterator for Buffer {
    type Item<'a> = &'a [u8] where Self: 'a;
    
    fn next(&mut self) -> Option<&[u8]> {
        if self.pos < self.data.len() {
            let chunk = &self.data[self.pos..self.pos+10];
            self.pos += 10;
            Some(chunk)
        } else {
            None
        }
    }
}

07 Auto Trait 与 Send/Sync

// Auto Trait：编译器自动实现
// 不需要 impl 块，根据字段自动派生

// Send: 可在线程间转移所有权
// Sync: 可在线程间共享引用（&T 可跨线程）

// 自动实现规则：
//   Send: 所有字段都是 Send
//   Sync: 所有字段都是 Sync

// 例：
struct MyStruct {
    data: Vec<i32>,    // Send + Sync
    rc: Rc<u32>,        // 不是 Send（Rc 用非原子引用计数）
}
// MyStruct 不是 Send（因为 rc 字段）

// 强制实现（unsafe）：
struct MyUnsafe;
unsafe impl Send for MyUnsafe {}
unsafe impl Sync for MyUnsafe {}

// 实际用途：
//   - 包装非 Rust 库的指针
//   - 跨线程发送原始资源

// Unpin / Pin（async 相关）：
//   Unpin: 可以安全地 pin 后 unpin
//   !Unpin: 不能 unpin（如自引用结构）

// Sized（编译期大小已知）：
//   默认所有类型是 Sized
//   ?Sized: 允许 DST（如 [u8], str, dyn Trait）

08 Sized 与 ?Sized

// Sized 类型（编译期已知大小）：
//   i32, f64, struct, enum, Box, Vec
let x: i32 = 5;

// Unsized / DST 类型（编译期未知大小）：
//   [u8]（切片）, str（字符串切片）, dyn Trait
let s: &str = "hello";        // &str 是 fat pointer
let slice: &[u8] = &[1, 2, 3]; // &[u8] 也是 fat pointer

// 胖指针 (Fat Pointer)：
//   &str  = (ptr, length)
//   &[T]  = (ptr, length)
//   &dyn Trait = (ptr, vtable)

// ?Sized bound：允许 DST
struct MySlice<T: ?Sized> {
    data: T,
}
// 此时 MySlice<[u8]> 也成为 DST
// 不能直接用 let x = MySlice，必须通过引用

// 函数参数 ?Sized：
fn first<T>(slice: &[T]) -> Option<&T> {
    slice.first()
}
// 隐式 T: Sized
// 显式 ?Sized：
fn first_unsized<T: ?Sized>(slice: &T) { ... }

// trait Sized:
trait MyTrait {
    type Output: Sized;   // 必须是 Sized
}

// dyn Trait 是 ?Sized
fn use_trait(t: &dyn Trait) { ... }   // dyn Trait 不是 Sized

// !Sized（明确不是 Sized）：
trait UnsizedTrait: !Sized {
    fn method(&self);
}

09 Specialization（待稳定）

//#![feature(min_specialization)]

// Specialization：trait 实现可以针对具体类型覆盖通用实现
trait Example {
    fn call(&self);
}

// 默认实现（通用）
impl<T> Example for T {
    default fn call(&self) {
        println!("generic");
    }
}

// 特化实现（覆盖 i32）
impl Example for i32 {
    fn call(&self) {
        println!("i32: {}", self);
    }
}

// 使用：
5i32.call();       // "i32: 5"
"hi".to_string().call();  // "generic"

// 应用：
//   - 性能优化（针对具体类型用更快算法）
//   - 提供 default 实现 + 特殊化

// RFC 已稳定未完成（lint 中）
// 目前仅标准库内部使用 min_specialization

10 设计模式

// 1. Newtype 模式（封装外部类型）
struct Meters(u32);
struct Kilometers(u32);
// 不同类型，编译期防止混淆
fn add(a: Meters, b: Meters) -> Meters {
    Meters(a.0 + b.0)
}
// 实现 Deref 让外部 API 直接用：
impl Deref for Meters {
    type Target = u32;
    fn deref(&self) -> &u32 { &self.0 }
}

// 2. Builder 模式（链式构造）
#[derive(Default)]
struct ServerBuilder {
    port: Option<u16>,
    host: Option<String>,
}
impl ServerBuilder {
    fn port(mut self, p: u16) -> Self { self.port = Some(p); self }
    fn host(mut self, h: &str) -> Self { self.host = Some(h.into()); self }
    fn build(self) -> Server {
        Server {
            port: self.port.unwrap_or(80),
            host: self.host.unwrap_or_else(|| "0.0.0.0".into()),
        }
    }
}
let s = ServerBuilder::default().port(8080).host("localhost").build();

// 3. Type State（类型表达状态）
struct Locked;
struct Unlocked;
struct Door<State> { _state: State }
impl Door<Locked> {
    fn unlock(self) -> Door<Unlocked> {
        Door { _state: Unlocked }
    }
}
impl Door<Unlocked> {
    fn open(&self) { println!("opened"); }
    fn lock(self) -> Door<Locked> {
        Door { _state: Locked }
    }
}
// 编译期保证状态机正确

// 4. Extension Trait（扩展外部类型）
trait StringExt {
    fn shout(&self) -> String;
}
impl StringExt for String {
    fn shout(&self) -> String {
        format!("{}!!!", self.to_uppercase())
    }
}
"hi".to_string().shout();

// 5. Phantom Type（类型标记）
use std::marker::PhantomData;
struct Id<T>(u64, PhantomData<T>);
struct User;
struct Product;
// Id<User> 和 Id<Product> 编译期区分

// 6. Enum Dispatch（替代 dyn）
enum Shape {
    Circle(Circle),
    Square(Square),
}
impl Draw for Shape {
    fn draw(&self) {
        match self {
            Shape::Circle(c) => c.draw(),
            Shape::Square(s) => s.draw(),
        }
    }
}
// 比 dyn 快（无虚表），比单态化灵活
Trait 系统总结:
1. Type Classes：每个类型独立实现，无继承
2. 静态分发：泛型 + bound 可内联，零开销
3. 动态分发：dyn Trait 通过虚表，运行期多态
4. 关联类型：表达"类型依赖"
5. Auto Trait：Send/Sync 自动派生，并发安全
6. Orphan Rule：防止实现冲突
Trait 系统让 Rust 既有 OOP 的抽象能力，又有函数式编程的灵活性。
