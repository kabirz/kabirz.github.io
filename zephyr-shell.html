01 Shell 概览

Zephyr Shell 是一个功能丰富的交互式命令行框架，支持命令注册、自动补全、历史记录、通配符、多后端等特性。

核心特性

特性说明
命令注册静态/动态命令注册，支持子命令
自动补全Tab 键自动补全命令和参数
历史记录上下键浏览历史命令
通配符支持 * 和 ? 通配符
多后端UART/RTT/SSH/Telnet/MQTT/WebSocket
颜色支持VT100 终端颜色输出
别名命令别名支持
远程 CLI通过网络远程访问 Shell

02 架构设计

Shell 上下文结构

// include/zephyr/shell/shell.h
struct shell {
    const struct shell_backend *backend;   // 后端（UART/RTT 等）
    struct shell_ctx *ctx;                  // 运行时上下文
    const struct shell_transport *iface;   // 传输接口
    const struct shell_vt100_colors *vt100_color; // 颜色主题
};

// shell.c 中的上下文结构（简化）
struct shell_ctx {
    // 命令缓冲区
    uint8_t cmd_buff[CONFIG_SHELL_CMD_BUFF_SIZE];
    uint16_t cmd_buff_pos;       // 当前光标位置
    uint16_t cmd_buff_len;       // 命令长度

    // 历史记录
    struct shell_history *history;

    // 当前活动命令
    const struct shell_static_entry *active_cmd;

    // 状态机
    enum shell_state state;
    enum shell_receive_state receive_state;

    // Tab 补全
    struct shell_tab_completion tab_completion;
};

static int cmd_hello(const struct shell *sh, size_t argc, char **argv)
{
    shell_print(sh, "Hello, %s!", argv[1] ? argv[1] : "World");
    return 0;
}

// 注册单个命令
SHELL_CMD_REGISTER(hello, NULL, "Say hello", cmd_hello);

    SHELL_CMD(temperature, NULL, "Read temperature", cmd_temp),

    SHELL_CMD(accel, NULL, "Read accelerometer", cmd_accel),
    SHELL_SUBCMD_SET_END
);

05 历史记录

// shell.c 中的历史记录实现

// 配置
// CONFIG_SHELL_HISTORY=y
// CONFIG_SHELL_HISTORY_BUFFER_SIZE=512

// 历史记录操作
// 1. 上键：获取上一条历史命令
// 2. 下键：获取下一条历史命令
// 3. 输入新命令时保存到历史

// shell.c 中的 history_handle()
static void history_handle(const struct shell *sh, bool up)
{
    // 备份当前命令
    strcpy(sh->ctx->temp_buff, sh->ctx->cmd_buff);

    // 获取历史命令
    history_mode = z_shell_history_get(sh->history, up,
                                       sh->ctx->cmd_buff, &len);

    // 更新显示
    z_shell_op_cursor_home_move(sh);
    z_clear_eos(sh);
    z_shell_print_cmd(sh);
    sh->ctx->cmd_buff_pos = len;
    sh->ctx->cmd_buff_len = len;
}

08 完整示例

// ===== Zephyr Shell 完整示例 =====
// 演示：命令注册、子命令、参数解析、颜色输出

#include <zephyr/kernel.h>
#include <zephyr/shell/shell.h>

// LED 控制命令
static int cmd_led_on(const struct shell *sh, size_t argc, char **argv)
{
    shell_print(sh, "LED turned ON");
    // gpio_pin_set(led_dev, led_pin, 1);
    return 0;
}

static int cmd_led_off(const struct shell *sh, size_t argc, char **argv)
{
    shell_print(sh, "LED turned OFF");
    // gpio_pin_set(led_dev, led_pin, 0);
    return 0;
}

static int cmd_led_toggle(const struct shell *sh, size_t argc, char **argv)
{
    shell_print(sh, "LED toggled");
    return 0;
}

// LED 子命令
SHELL_STATIC_SUBCMD_SET_CREATE(sub_led,
    SHELL_CMD(on, NULL, "Turn LED on", cmd_led_on),
    SHELL_CMD(off, NULL, "Turn LED off", cmd_led_off),
    SHELL_CMD(toggle, NULL, "Toggle LED", cmd_led_toggle),
    SHELL_SUBCMD_SET_END
);
SHELL_CMD_REGISTER(led, &sub_led, "LED control commands", NULL);

// 传感器读取命令（带参数）
static int cmd_sensor_read(const struct shell *sh, size_t argc, char **argv)
{
    // argv[0] = "sensor"
    // argv[1] = "read"
    // argv[2] = 参数（如通道号）

    if (argc < 3) {
        shell_error(sh, "Usage: sensor read <channel>");
        return -EINVAL;
    }

    int channel = atoi(argv[2]);
    shell_print(sh, "Reading channel %d: value=25.5", channel);
    return 0;
}

SHELL_CMD_ARG(sensor_read, NULL, "Read sensor channel", cmd_sensor_read, 1, 1);
SHELL_CMD_REGISTER(sensor, NULL, "Sensor commands", NULL);

// 系统信息命令（颜色输出）
static int cmd_sys_info(const struct shell *sh, size_t argc, char **argv)
{
    shell_print(sh, "System Information:");
    shell_print(sh, "  Kernel: %s", kernel_version_get());
    shell_print(sh, "  Uptime: %d ms", k_uptime_get_32());
    shell_print(sh, "  Heap free: %d bytes", k_mem_heap_free(&k_sys_heap));

    // 使用颜色输出
    shell_print(sh, SHELL_GREEN "Status: OK" SHELL_RESET);
    return 0;
}

SHELL_CMD_REGISTER(sys, NULL, "System commands", cmd_sys_info);
