01 概述

UART（Universal Asynchronous Receiver/Transmitter）是 Linux 中最基础的异步串行通信接口。串口驱动是 Linux 最早的驱动之一（Linus 1991 年），至今仍是控制台、调试、嵌入式系统的核心。

核心用途
  • 系统控制台：内核打印、登录 shell
  • 调试：printk 输出、kgdb
  • 嵌入式通信：与 MCU、蓝牙模块等
  • GPS/GNSS：NMEA 数据接收
  • 工业控制：RS-232/RS-485
UART 信号

04 uart_ops

uart_ops 是 UART 驱动的核心回调表，定义了所有硬件操作。

关键回调

回调说明必须
tx_empty()检查发送 FIFO 是否为空必须
set_mctrl()设置调制解调器控制线（RTS/DTR 等）必须
get_mctrl()读取调制解调器状态（CTS/DSR/DCD/RI）必须
stop_tx()停止发送必须
start_tx()开始发送必须
start_rx()开始接收必须
throttle()通知驱动输入缓冲区接近满可选
unthrottle()通知驱动可以继续发送可选
send_xchar()发送高优先级字符（XON/XOFF）可选
startup()打开端口（使能时钟、FIFO 等）必须
shutdown()关闭端口必须
set_termios()设置波特率、数据位、停止位、校验必须

05 控制台

串口控制台是 Linux 内核最早期的输出方式，在系统启动早期即可使用。

配置方式

// 内核命令行参数
console=ttyS0,115200

// 内核配置（.config）
CONFIG_SERIAL_CORE_CONSOLE=y
CONFIG_SERIAL_8250_CONSOLE=y

控制台初始化

early_param("console", ...)
解析 console= 参数
↓
register_console()
注册控制台
↓
console_unlock() → printk()
输出内核消息
多控制台：Linux 支持同时注册多个控制台。第一个注册的控制台作为主控制台（preferred_console）。所有注册的控制台都会收到 printk 输出。

06 earlycon

earlycon 提供最早期的串口输出，在内核完全初始化之前即可使用。

配置方式

// 内核命令行
earlycon=uart8250,io,0x3f8,115200n8
earlycon=msm_serial,0x78b0000,115200n8

earlycon 注册

// 驱动中注册 earlycon
static void my_earlycon_init(struct earlycon_id *id,
                              unsigned long addr)
{
    // 最小化硬件初始化
    early_serial8250_setup(id, addr, &earlycon_port);
}
EARLYCON_DECLARE(my_uart, my_earlycon_init);
earlycon 限制：earlycon 在中断、DMA、时钟框架不可用时运行。只能做最简单的寄存器读写。真正的驱动在稍后初始化。

08 编写驱动

UART 驱动模板

#include <linux/serial_core.h>
#include <linux/module.h>
#include <linux/platform_device.h>

#define MY_UART_TX_FIFO  64

static void my_uart_stop_tx(struct uart_port *port)
{
    // 禁用 TX 中断
    writel(0, port->membase + MY_UART_IER);
}

static void my_uart_start_tx(struct uart_port *port)
{
    // 使能 TX 中断
    writel(MY_UART_IER_TX_READY, port->membase + MY_UART_IER);
}

static unsigned int my_uart_tx_empty(struct uart_port *port)
{
    u32 sr = readl(port->membase + MY_UART_SR);
    return (sr & MY_UART_SR_TEMT) ? TIOCSER_TEMT : 0;
}

static int my_uart_startup(struct uart_port *port)
{
    // 使能 FIFO、设置中断
    writel(MY_UART_FCR_FIFO_EN | MY_UART_FCR_RX_RST,
           port->membase + MY_UART_FCR);
    return request_irq(port->irq, my_uart_irq, 0,
                       "my-uart", port);
}

static void my_uart_shutdown(struct uart_port *port)
{
    free_irq(port->irq, port);
    writel(0, port->membase + MY_UART_IER);
}

static void my_uart_set_termios(struct uart_port *port,
                                  struct ktermios *new,
                                  const struct ktermios *old)
{
    u32 baud = uart_get_baud(port, old, new, 921600);
    u32 quot = port->uartclk / (16 * baud);
    writel(quot & 0xFF, port->membase + MY_UART_DLL);
    writel((quot >> 8) & 0xFF, port->membase + MY_UART_DLM);
}

static const struct uart_ops my_uart_ops = {
    .tx_empty     = my_uart_tx_empty,
    .set_mctrl    = uart_set_mctrl_default,
    .get_mctrl    = uart_get_mctrl_default,
    .stop_tx      = my_uart_stop_tx,
    .start_tx     = my_uart_start_tx,
    .startup      = my_uart_startup,
    .shutdown     = my_uart_shutdown,
    .set_termios  = my_uart_set_termios,
};

static struct uart_port my_uart_port;
static struct uart_driver my_uart_driver = {
    .driver_name = "my-uart",
    .dev_name    = "ttyMY",
    .major       = 204,
    .minor       = 64,
    .nr          = 1,
};

static int my_uart_probe(struct platform_device *pdev)
{
    my_uart_port.dev = &pdev->dev;
    my_uart_port.ops = &my_uart_ops;
    my_uart_port.type = PORT_MY_UART;
    my_uart_port.fifosize = MY_UART_TX_FIFO;
    my_uart_port.mapbase = platform_get_resource(pdev, IORESOURCE_MEM, 0)->start;
    my_uart_port.membase = devm_ioremap_resource(&pdev->dev, ...);
    my_uart_port.irq = platform_get_irq(pdev, 0);
    my_uart_port.uartclk = 24000000;

    uart_add_one_port(&my_uart_driver, &my_uart_port);
    return 0;
}

module_uart_driver(my_uart_driver);
