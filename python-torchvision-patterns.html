01 torchvision 与测试图形

torchvision 是 PyTorch 生态的计算机视觉工具库，提供数据集（datasets）、模型（models）、图像变换（transforms）和工具函数（utils）四大模块。其中 torchvision.utils 提供了 make_grid、save_image 等实用工具，配合 PyTorch 张量操作，可以灵活生成各类测试图形。

torchvision
datasets models transforms utils ★
PyTorch 张量操作
torch.zeros / ones / rand torch.linspace / meshgrid torch.sin / cos / clamp F.interpolate / resize
输出
save_image make_grid to_pil_image matplotlib
torchvision + PyTorch 张量操作构成测试图形生成的核心链路

为什么需要合成测试图形

  • 单元测试：验证图像处理流水线的正确性（尺寸/通道/数值范围）
  • 模型调试：用已知模式（彩条/棋盘格）验证网络是否能提取特定特征
  • 性能基准：在可控输入下测量前向/反向传播的耗时和内存
  • 数据增强验证：可视化 augment 前后的效果，确认参数配置正确

核心概念

PyTorch 图像张量规范:
  shape          = (C, H, W)        # 单张图像（C == 1 或 3）
  shape          = (N, C, H, W)     # 批处理图像
  dtype          = torch.float32    # 值域 [0, 1]
  layout         = CHW ( contiguous )
  color order    = RGB

彩条（Color Bars）

经典的 SMPTE 彩条模式，每列一种纯色，常用于视频流水线测试。

def color_bars(width=320, height=240):
    """生成 8 色垂直彩条 (3, H, W) float32"""
    colors = torch.tensor([
        [1.0, 1.0, 1.0],  # 白
        [1.0, 1.0, 0.0],  # 黄
        [0.0, 1.0, 1.0],  # 青
        [0.0, 1.0, 0.0],  # 绿
        [1.0, 0.0, 1.0],  # 紫
        [1.0, 0.0, 0.0],  # 红
        [0.0, 0.0, 1.0],  # 蓝
        [0.0, 0.0, 0.0],  # 黑
    ])  # (8, 3)

    img = torch.zeros((3, height, width))
    bar_w = width // len(colors)
    for i, c in enumerate(colors):
        img[:, :, i*bar_w:(i+1)*bar_w] = c.view(3, 1, 1)
    return img

渐变（Gradient）

水平/垂直/对角渐变，检测量化伪影和色调映射。

def h_gradient(width=320, height=240):
    """水平渐变 (3, H, W)"""
    ramp = torch.linspace(0, 1, width).view(1, 1, width)
    return ramp.expand(3, height, -1).contiguous()

def v_gradient(width=320, height=240):
    """垂直渐变"""
    ramp = torch.linspace(0, 1, height).view(1, height, 1)
    return ramp.expand(3, -1, width).contiguous()

def radial_gradient(size=256):
    """径向渐变（中心亮边缘暗）"""
    y, x = torch.meshgrid(
        torch.linspace(-1, 1, size),
        torch.linspace(-1, 1, size),
        indexing='ij'
    )
    dist = torch.sqrt(x**2 + y**2)
    grad = torch.clamp(1 - dist, 0, 1)
    return grad.unsqueeze(0).expand(3, -1, -1)

def color_checker(size=64):
    """24 色色卡 (3, H, W)"""
    # 经典的 24 个 Macbeth 色块 (sRGB 近似值)
    patches = torch.tensor([
        [0.4, 0.2, 0.1], [0.6, 0.4, 0.3], [0.3, 0.5, 0.2],
        [0.7, 0.5, 0.4], [0.2, 0.3, 0.4], [0.5, 0.2, 0.3],
        [0.9, 0.8, 0.7], [0.1, 0.1, 0.1], [0.2, 0.2, 0.8],
        [0.8, 0.2, 0.2], [0.2, 0.8, 0.2], [0.2, 0.2, 0.8],
        [0.8, 0.8, 0.2], [0.8, 0.2, 0.8], [0.2, 0.8, 0.8],
        [0.9, 0.6, 0.2], [0.6, 0.3, 0.1], [0.3, 0.6, 0.1],
        [0.7, 0.7, 0.7], [0.5, 0.5, 0.5], [0.3, 0.3, 0.3],
        [0.9, 0.9, 0.1], [0.1, 0.9, 0.1], [0.1, 0.1, 0.9],
    ])  # (24, 3)

    cols, rows = 6, 4
    img = torch.zeros((3, rows*size, cols*size))
    for idx, c in enumerate(patches):
        r, col = divmod(idx, cols)
        img[:, r*size:(r+1)*size, col*size:(col+1)*size] = c.view(3,1,1)
    return img  # (3, 4*size, 6*size)

棋盘格（Checkerboard）

def checkerboard(size=256, tiles=8):
    """棋盘格 (3, H, W)"""
    cell = size // tiles
    pattern = torch.zeros((size, size))
    for i in range(tiles):
        for j in range(tiles):
            if (i + j) % 2 == 0:
                pattern[i*cell:(i+1)*cell, j*cell:(j+1)*cell] = 1.0
    return pattern.unsqueeze(0).expand(3, -1, -1)

波带片（Zone Plate）

波带片是经典的频率响应测试图，包含从中心到边缘递增的空间频率。

def zone_plate(size=512):
    """Fresnel 波带片 (3, H, W)"""
    y, x = torch.meshgrid(
        torch.linspace(-1, 1, size),
        torch.linspace(-1, 1, size),
        indexing='ij'
    )
    r2 = x**2 + y**2
    pattern = (torch.cos(torch.pi * 40 * r2) + 1) / 2
    return pattern.unsqueeze(0).expand(3, -1, -1)

正弦光栅（Sine Grating）

def sine_grating(size=256, freq=8, angle=0):
    """正弦光栅，可指定频率和角度"""
    y, x = torch.meshgrid(
        torch.arange(size, dtype=torch.float32),
        torch.arange(size, dtype=torch.float32),
        indexing='ij'
    )
    theta = torch.deg2rad(torch.tensor(float(angle)))
    u = x * torch.cos(theta) + y * torch.sin(theta)
    pattern = (torch.sin(2 * torch.pi * freq * u / size) + 1) / 2
    return pattern.unsqueeze(0).expand(3, -1, -1)

# 不同频率组合
def multi_freq_grating(size=256):
    """多频率多方向光栅拼接"""
    configs = [(4,0), (8,0), (16,0), (32,0),
               (8,45), (8,90), (8,135), (4,90)]
    grids = []
    for freq, angle in configs:
        g = sine_grating(size//2, freq, angle)
        grids.append(g)
    return make_grid(grids, nrow=4, padding=4, pad_value=0.5)

分辨率靶（Resolution Chart）

def resolution_chart(size=512):
    """辐射状分辨率测试图"""
    y, x = torch.meshgrid(
        torch.linspace(-1, 1, size),
        torch.linspace(-1, 1, size),
        indexing='ij'
    )
    angle = torch.atan2(y, x)
    radius = torch.sqrt(x**2 + y**2)
    # 角度决定黑白条纹，半径决定频率
    n_cycles = 36  # 36 个黑白对
    pattern = (torch.cos(n_cycles * angle) + 1) / 2
    # 外围渐变为灰（避免边界硬切）
    mask = torch.clamp(1 - radius, 0, 1)
    pattern = pattern * mask + 0.5 * (1 - mask)
    return pattern.unsqueeze(0).expand(3, -1, -1)

圆环（Concentric Circles）

def concentric_circles(size=256, n_rings=12):
    """同心圆环"""
    y, x = torch.meshgrid(
        torch.linspace(-1, 1, size),
        torch.linspace(-1, 1, size),
        indexing='ij'
    )
    r = torch.sqrt(x**2 + y**2)
    pattern = (torch.sin(n_rings * torch.pi * r) + 1) / 2
    return pattern.unsqueeze(0).expand(3, -1, -1)

均匀噪声

def uniform_noise(h, w, low=0.0, high=1.0):
    """均匀分布噪声 (3, H, W)"""
    return torch.empty(3, h, w).uniform_(low, high)

高斯噪声

def gaussian_noise(h, w, mean=0.0, std=0.1):
    """高斯噪声，叠加到图像上"""
    noise = torch.empty(3, h, w).normal_(mean, std)
    return noise

def add_gaussian_noise(img, std=0.05):
    """向图像添加高斯噪声，保持 [0,1] 范围"""
    noise = torch.empty_like(img).normal_(0, std)
    return torch.clamp(img + noise, 0, 1)

椒盐噪声

def salt_pepper_noise(img, prob=0.01, salt_val=1.0, pepper_val=0.0):
    """椒盐噪声，随机像素置为纯白/纯黑"""
    noisy = img.clone()
    mask = torch.rand_like(img[:, 0:1, :])  # 所有通道同一 mask
    noisy[:, mask < prob/2] = pepper_val
    noisy[:, (mask >= prob/2) & (mask < prob)] = salt_val
    return noisy

Perlin 噪声（纹理）

def perlin_noise(size=128, scale=0.1):
    """简单 Perlin 噪声（上采样均匀噪声）"""
    small = torch.empty(1, size//4, size//4).uniform_(0, 1)
    # 双线性插值放大
    pattern = torch.nn.functional.interpolate(
        small.unsqueeze(0), size=(size, size),
        mode='bilinear', align_corners=False
    ).squeeze(0)
    return pattern.expand(3, -1, -1)

网格 + 噪声组合

def noisy_checkerboard(size=256, tiles=8, noise_std=0.05):
    """棋盘格 + 高斯噪声"""
    cb = checkerboard(size, tiles)
    noise = torch.empty_like(cb).normal_(0, noise_std)
    return torch.clamp(cb + noise, 0, 1)

自定义 Transform

class AddGaussianNoise(torch.nn.Module):
    def __init__(self, std=0.05):
        super().__init__()
        self.std = std

    def forward(self, img):
        noise = torch.randn_like(img) * self.std
        return torch.clamp(img + noise, 0, 1)

    def __repr__(self):
        return f"{self.__class__.__name__}(std={self.std})"

# 组合使用

from torchvision.utils import make_grid, save_image
import torch
import math

class TestPatternSuite:
    """综合测试图形生成套件"""

    def __init__(self, size=256):
        self.size = size

    def color_patterns(self):
        """色彩测试图形"""
        return [
            ("color_bars", color_bars(self.size, self.size)),
            ("h_gradient", h_gradient(self.size, self.size)),
            ("radial_gradient", radial_gradient(self.size)),
            ("color_checker", color_checker(self.size//4)),
        ]

    def spatial_patterns(self):
        """空间测试图形"""
        return [
            ("checkerboard", checkerboard(self.size, 8)),
            ("zone_plate", zone_plate(self.size)),
            ("resolution", resolution_chart(self.size)),
            ("concentric", concentric_circles(self.size)),
        ]

    def noise_patterns(self):
        """噪声测试图形"""
        return [
            ("uniform_noise", uniform_noise(self.size, self.size)),
            ("gaussian_noise", gaussian_noise(self.size, self.size, std=0.2)),
            ("perlin_noise", perlin_noise(self.size)),
        ]

    def generate_all(self, output_dir="test_patterns"):
        """生成所有测试图形"""
        import os
        os.makedirs(output_dir, exist_ok=True)

        all_patterns = []
        all_names = []

        for name, pat in (self.color_patterns()
                          + self.spatial_patterns()
                          + self.noise_patterns()):
            save_image(pat, f"{output_dir}/{name}.png")
            all_patterns.append(pat)
            all_names.append(name)
            print(f"  ✓ {name}.png")

        # 生成总览图
        grid = make_grid(all_patterns, nrow=4, padding=4, pad_value=1.0)
        save_image(grid, f"{output_dir}/_overview.png")
        print(f"\n总览图: {output_dir}/_overview.png")
        print(f"共 {len(all_patterns)} 个测试图形")

        return all_patterns

# 使用
suite = TestPatternSuite(size=256)

模型行为验证

用测试图形验证预训练模型的响应特性：

import torchvision.models as models

model = models.resnet18(weights='DEFAULT').eval()

def analyze_response(pattern, model):
    """分析模型对测试图形的响应"""
    with torch.no_grad():
        # pattern: (3, H, W) → (1, 3, H, W)
        batch = pattern.unsqueeze(0)
        # 如果尺寸不匹配，resize 到 224
        if batch.shape[-1] != 224:
            batch = F.resize(batch, (224, 224))
        output = model(batch)
        # 输出 top-5 概率
        probs = torch.nn.functional.softmax(output[0], dim=0)
        top5 = probs.topk(5)
        return list(zip(top5.indices.tolist(), top5.values.tolist()))

# 有趣：模型对波带片和棋盘格的响应差异
zp_response = analyze_response(zone_plate(512), model)
cb_response = analyze_response(checkerboard(256, 8), model)
print("Zone Plate top-5:", zp_response)
print("Checkerboard top-5:", cb_response)

  • 设备统一：生成时指定 device，避免后续多图推理时 CPU→GPU 拷贝
def generate_on_device(device='cuda', size=256):
    """在指定设备上生成测试图形"""
    def _cb():
        cell = size // 8
        img = torch.zeros(3, size, size, device=device)
        for i in range(8):
            for j in range(8):
                if (i + j) % 2 == 0:
                    img[:, i*cell:(i+1)*cell, j*cell:(j+1)*cell] = 1.0
        return img
    return _cb

# GPU 生成 + GPU 推理，全程无拷贝
pattern = generate_on_device('cuda')()
output = model(pattern.unsqueeze(0))

调试建议

  • 棋盘格 → 卷积测试：检查 conv 层的输出是否保持棋盘结构（验证 stride/dilation）
  • 波带片 → 频率响应：对 model.features 的输出做 FFT，观察频率衰减曲线
  • 彩条 → 颜色一致性：验证 RGB→BGR→RGB 转换是否还原，BatchNorm 的 running_mean 是否合理
  • 噪声 → 鲁棒性基准：在不同噪声水平下测量 accuracy drop，绘制鲁棒性曲线
def frequency_response(model, size=256):
    """用波带片测量模型的频率响应"""
    zp = zone_plate(size).unsqueeze(0)  # (1,3,size,size)
    with torch.no_grad():
        features = model(zp)
    # 对特征图做平均 FFT
    from torch.fft import fft2, fftshift
    feat = features[0].mean(dim=0)  # (H', W')
    spectrum = torch.log(fftshift(fft2(feat)).abs() + 1e-8)
    return spectrum

# 可视化频率响应
spec = frequency_response(model)
