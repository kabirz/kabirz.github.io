01 FastAPI 概述

FastAPI 是一个基于 Python 类型提示的现代 Web 框架，构建于 Starlette（Web 部分）和 Pydantic（数据部分）之上。

核心架构

# FastAPI 的层次架构
"""
┌─────────────────────────────────┐
│         FastAPI 应用            │
│  - 路由注册                      │
│  - 依赖注入容器                  │
│  - OpenAPI 自动生成              │
├─────────────────────────────────┤
│         Starlette               │
│  - ASGI 服务器接口               │
│  - 请求/响应处理                 │
│  - WebSocket 支持               │
│  - 中间件系统                    │
│  - 后台任务                      │
├─────────────────────────────────┤
│         Pydantic V2             │
│  - 数据模型定义                  │
│  - 请求体验证                   │
│  - 响应模型序列化               │
│  - JSON Schema 生成             │
├─────────────────────────────────┤
│     ASGI Server (Uvicorn)       │
│  - 异步 HTTP 服务器              │
│  - 支持 HTTP/1.1 + WebSocket    │
└─────────────────────────────────┘
"""

from fastapi import FastAPI

03 依赖注入

FastAPI 的依赖注入系统通过 Depends 函数实现，支持函数、类、生成器等多种形式。

基础依赖注入

from fastapi import FastAPI, Depends, HTTPException, status
from typing import Annotated

app = FastAPI()

# 函数作为依赖
async def common_params(
    skip: int = 0,
    limit: int = 100,
):
    return {"skip": skip, "limit": limit}

# 在路径操作中使用 Depends
@app.get("/items")
async def list_items(params: Annotated[dict, Depends(common_params)]):
    return params

@app.get("/users")
async def list_users(params: Annotated[dict, Depends(common_params)]):
    return params

# 更优雅的方式：直接解包
@app.get("/products")
async def list_products(
    skip: int = Depends(common_params),
    # 实际上 Depends 返回整个字典，但我们可以只取需要的字段
):
    pass

# 推荐的方式：类型安全的依赖
class Pagination:
    def __init__(self, skip: int = 0, limit: int = 100):
        self.skip = skip
        self.limit = limit

@app.get("/paginated")
async def paginated(p: Annotated[Pagination, Depends(Pagination)]):
    return {"skip": p.skip, "limit": p.limit}

可调用类与全局依赖

from fastapi import FastAPI, Depends, Request, HTTPException
from typing import Annotated

# 可调用类依赖
class DatabaseSession:
    def __init__(self):
        self.connected = False

    async def __call__(self):
        if not self.connected:
            await self.connect()
        return self

    async def connect(self):
        self.connected = True
        return self

    async def query(self, sql: str):
        return f"result: {sql}"

# 全局单例
db_session = DatabaseSession()

@app.get("/db-query")
async def query_db(
    db: Annotated[DatabaseSession, Depends(db_session)]
):
    return await db.query("SELECT 1")

# 生成器依赖（用于资源管理等）
from contextlib import asynccontextmanager

async def get_db_session():
    """提供数据库会话并自动关闭"""
    db = DatabaseSession()
    await db.connect()
    try:
        yield db
    finally:
        db.connected = False

@app.get("/safe-query")
async def safe_query(
    db: Annotated[DatabaseSession, Depends(get_db_session)]
):
    return await db.query("SELECT 2")

# 全局依赖
common_params_dep = Depends(common_params)
app = FastAPI(dependencies=[common_params_dep])
# 所有路由都会注入 common_params

依赖嵌套与优先级

from fastapi import FastAPI, Depends, HTTPException, status
from typing import Annotated

app = FastAPI()

# 多层嵌套依赖
async def verify_token(token: str = Header(...)):
    if token != "valid-token":
        raise HTTPException(status_code=401)
    return {"user_id": 1, "role": "admin"}

async def verify_admin(
    user: Annotated[dict, Depends(verify_token)]
):
    if user["role"] != "admin":
        raise HTTPException(status_code=403)
    return user

async def get_db():
    return {"connection": "ok"}

# 最终路由：依赖层层嵌套
@app.get("/admin/users")
async def admin_list_users(
    admin: Annotated[dict, Depends(verify_admin)],
    db: Annotated[dict, Depends(get_db)],
):
    # 先 verify_token → 再 verify_admin → 再 get_db
    # 依赖按深度优先遍历执行
    return {"admin": admin, "db": db}

# 依赖缓存：同一请求中相同依赖只执行一次
async def get_current_user(
    token: str = Header(...),
):
    return {"user_id": 1}

async def get_user_cart(
    user: Annotated[dict, Depends(get_current_user)],
):
    return {"items": [], "user_id": user["user_id"]}

@app.get("/cart")
async def read_cart(
    user: Annotated[dict, Depends(get_current_user)],  # 复用结果
    cart: Annotated[dict, Depends(get_user_cart)],
):
    # get_current_user 只执行一次
    return {"user": user, "cart": cart}

高级依赖模式

from fastapi import FastAPI, Depends
from typing import Annotated

app = FastAPI()

# Conditional dependencies
class UserRole:
    def __init__(self, min_role: str):
        self.min_role = min_role

    async def __call__(self, user: dict = Depends(get_current_user)):
        roles = {"admin": 3, "editor": 2, "viewer": 1}
        if roles.get(user.get("role", ""), 0) < roles.get(self.min_role, 0):
            raise HTTPException(status_code=403)
        return user

require_admin = Depends(UserRole("admin"))
require_editor = Depends(UserRole("editor"))

@app.get("/admin-panel")
async def admin_panel(user: Annotated[dict, require_admin]):
    return {"panel": "admin"}

@app.get("/editor-panel")
async def editor_panel(user: Annotated[dict, require_editor]):
    return {"panel": "editor"}

# 异步与同步依赖混合
def sync_dep():
    return "sync data"

async def async_dep(data: str = Depends(sync_dep)):
    return f"async {data}"

@app.get("/mixed")
async def mixed(data: str = Depends(async_dep)):
    return {"data": data}

    user_id: int
    items: list[OrderItem] = Field(..., min_length=1)
    shipping_address: Address
    coupon_code: str | None = None

    @field_validator("items")

05 中间件

中间件在请求被路由处理之前和响应返回之后执行，适合横切关注点。

基础中间件

from fastapi import FastAPI, Request
from starlette.middleware.base import BaseHTTPMiddleware
import time

app = FastAPI()

# 使用 @app.middleware 装饰器
@app.middleware("http")
async def add_process_time_header(request: Request, call_next):
    start_time = time.perf_counter()

    # 请求处理前
    response = await call_next(request)  # 调用下一个中间件/路由

    # 响应处理后
    process_time = time.perf_counter() - start_time
    response.headers["X-Process-Time"] = str(process_time)
    return response

# 基于类的中间件
class LoggingMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request: Request, call_next):
        # 请求进入
        print(f"→ {request.method} {request.url.path}")

        response = await call_next(request)

        # 响应出去
        print(f"← {response.status_code}")
        return response

app.add_middleware(LoggingMiddleware)

自定义中间件模式

from fastapi import FastAPI, Request, HTTPException
from starlette.middleware.base import BaseHTTPMiddleware
import time
import asyncio

app = FastAPI()

# 限流中间件
class RateLimitMiddleware(BaseHTTPMiddleware):
    def __init__(self, app, max_requests: int = 100, window: int = 60):
        super().__init__(app)
        self.max_requests = max_requests
        self.window = window
        self.requests: dict[str, list[float]] = {}

    async def dispatch(self, request: Request, call_next):
        client_ip = request.client.host if request.client else "unknown"
        now = time.time()

        # 清理过期记录
        self.requests[client_ip] = [
            t for t in self.requests.get(client_ip, [])
            if now - t < self.window
        ]

        # 检查是否超限
        if len(self.requests[client_ip]) >= self.max_requests:
            raise HTTPException(status_code=429, detail="请求过于频繁")

        self.requests[client_ip].append(now)
        response = await call_next(request)
        return response

app.add_middleware(RateLimitMiddleware, max_requests=60, window=60)

# 请求体日志中间件（谨慎使用）
class RequestLogMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request: Request, call_next):
        # 读取请求体（不会消耗原始流）
        body = await request.body()
        print(f"Body: {body[:1024]}...")  # 只记录前 1KB
        response = await call_next(request)
        return response

from fastapi import FastAPI, WebSocket, WebSocketDisconnect, WebSocketException
from typing import Annotated

app = FastAPI()

# 基础 WebSocket
@app.websocket("/ws")
async def websocket_endpoint(websocket: WebSocket):
    await websocket.accept()  # 接受连接
    try:
        while True:
            data = await websocket.receive_text()  # 接收文本
            await websocket.send_text(f"Echo: {data}")  # 发送文本
    except WebSocketDisconnect:
        print("客户端断开连接")
    except Exception as e:
        print(f"错误: {e}")

# 接收/发送多种类型

from fastapi import FastAPI, WebSocket, WebSocketDisconnect
from typing import Any

# 连接管理器
class ConnectionManager:
    def __init__(self):
        self.active_connections: dict[str, list[WebSocket]] = {}
        self.user_connections: dict[str, WebSocket] = {}

    async def connect(self, websocket: WebSocket, room: str, user_id: str = None):
        await websocket.accept()
        if room not in self.active_connections:
            self.active_connections[room] = []
        self.active_connections[room].append(websocket)
        if user_id:
            self.user_connections[user_id] = websocket

    def disconnect(self, websocket: WebSocket, room: str, user_id: str = None):
        if room in self.active_connections:
            if websocket in self.active_connections[room]:
                self.active_connections[room].remove(websocket)
        if user_id and user_id in self.user_connections:
            del self.user_connections[user_id]

    async def broadcast(self, room: str, message: Any):
        """广播消息到房间的所有连接"""
        if room not in self.active_connections:
            return
        dead_connections = []
        for conn in self.active_connections[room]:
            try:
                if isinstance(message, str):
                    await conn.send_text(message)
                else:
                    await conn.send_json(message)
            except WebSocketDisconnect:
                dead_connections.append(conn)

        # 清理断开的连接
        for conn in dead_connections:
            self.active_connections[room].remove(conn)

    async def send_to_user(self, user_id: str, message: Any):
        """向特定用户发送消息"""
        if user_id in self.user_connections:
            ws = self.user_connections[user_id]
            try:
                if isinstance(message, str):
                    await ws.send_text(message)
                else:
                    await ws.send_json(message)
            except WebSocketDisconnect:
                del self.user_connections[user_id]

manager = ConnectionManager()

@app.websocket("/ws/chat/{room_id}/{user_id}")
async def chat_websocket(websocket: WebSocket, room_id: str, user_id: str):
    await manager.connect(websocket, room_id, user_id)
    try:
        # 通知房间有新用户
        await manager.broadcast(
            room_id,
            {"type": "join", "user_id": user_id}
        )

        while True:
            data = await websocket.receive_text()
            message = json.loads(data)

            # 处理不同类型的消息
            if message.get("type") == "chat":
                await manager.broadcast(room_id, {
                    "type": "chat",
                    "user_id": user_id,
                    "content": message.get("content"),
                })
            elif message.get("type") == "typing":
                await manager.broadcast(room_id, {
                    "type": "typing",
                    "user_id": user_id,
                })
            elif message.get("type") == "private":
                await manager.send_to_user(
                    message.get("target_user"),
                    {"type": "private", "from": user_id, "content": message.get("content")}
                )

    except WebSocketDisconnect:
        manager.disconnect(websocket, room_id, user_id)
        await manager.broadcast(
            room_id,
            {"type": "leave", "user_id": user_id}
        )

# JWT 工具函数
def create_access_token(data: dict, expires_delta: timedelta = None):
    to_encode = data.copy()
    expire = datetime.utcnow() + (expires_delta or timedelta(minutes=15))
    to_encode.update({"exp": expire})
    return jwt.encode(to_encode, SECRET_KEY, algorithm=ALGORITHM)

def verify_token(token: str) -> TokenData:
    try:
        payload = jwt.decode(token, SECRET_KEY, algorithms=[ALGORITHM])
        username = payload.get("sub")
        role = payload.get("role")
        if username is None:
            raise HTTPException(status_code=401)
        return TokenData(username=username, role=role)
    except jwt.PyJWTError:
        raise HTTPException(status_code=401)

@app.post("/token", response_model=Token)

# BackgroundTasks 的限制：
# 1. 在响应返回后执行（不能保证完成）
# 2. 进程重启后丢失
# 3. 不支持分布式
# 4. 没有重试机制

# 适合：日志记录、通知推送、缓存预热

# Celery 对比
"""
特性            BackgroundTasks    Celery / RQ / Dramatiq
──────────────  ─────────────────  ──────────────────────
执行时机        响应后尽快执行       异步队列 worker
持久化          无                  支持（Redis/RabbitMQ）
重试            无                  支持超时重试
分布式          单进程              多 worker 多机器
任务调度        即时                定时/延迟/CRON
监控            无                  Flower/Dashboard
适用场景        简单后处理           生产级异步任务
"""
