02 DFU 架构

// Flash 分区布局
// +------------------+------------------+------------------+
// | Bootloader       | Slot 0 (主)      | Slot 1 (次)      |
// | (MCUboot)        | (当前固件)       | (新固件)          |
// | 0x0000 - 0x8000  | 0x8000 - 0x80000 | 0x80000 - 0xF8000|
// +------------------+------------------+------------------+
// | Scratch (擦写区) |                                        |
// | 0xF8000 - 0x100000                                         |
// +----------------------------------------------------------+

// OTA 升级流程：
// 1. 新固件写入 Slot 1
// 2. MCUboot 标记 Slot 1 为"测试"状态
// 3. 重启，MCUboot 从 Slot 1 启动
// 4. 应用确认：永久切换到 Slot 1
// 5. 未确认：超时回滚到 Slot 0

03 MCUboot 集成

// MCUboot 配置
// CONFIG_BOOTLOADER_MCUBOOT=y
// CONFIG_IMG_MANAGER=y
// CONFIG_STREAM_FLASH=y

// MCUboot 功能
// - 安全启动（签名验证）
// - 镜像加密
// - 回滚保护
// - 镜像压缩
// - 多槽位管理

// MCUboot Shell 命令
// mcuboot info        - 查看镜像信息
// mcuboot confirm     - 确认当前镜像
// mcuboot revert      - 回滚到上一版本
// mcuboot erase       - 擦除镜像

04 镜像管理

// 镜像管理 API
#include <zephyr/mgmt/mcumgr/img/img_mgmt.h>

// 查询镜像状态
struct image_mgmt_id id;
image_mgmt_id_get(0, &id);

// 确认镜像
image_mgmt_set_pending(0, 0);  // 0 = 不重启
image_mgmt_set_confirmed(0);

// 镜像状态标志
// IMAGE_FLAGS_CONFIRMED    - 已确认
// IMAGE_FLAGS_PENDING      - 待测试
// IMAGE_FLAGS_ACTIVE       - 活跃
// IMAGE_FLAGS_PENDING_TEST - 待确认测试

05 OTA 升级流程

// 完整 OTA 流程

// 1. 客户端上传新镜像
// mcumgr image upload -e firmware.bin

// 2. MCU 侧（mcumgr 自动处理）
// - 接收数据块
// - 写入 Flash (Slot 1)
// - 校验完整性

// 3. 客户端发送测试命令
// mcumgr image test <hash>

// 4. MCU 重启，MCUboot 从 Slot 1 启动

// 5. 应用确认
// mcumgr image confirm

// 6. 新镜像永久生效

06 回滚保护

// 回滚机制
// CONFIG_BOOT_BOOTUTIL_VALIDATE_SLOT0=y  ← 校验 Slot 0
// CONFIG_BOOT_MCUOOT_DOWNGRADE_PREVENTION=y ← 防止降级

// 回滚场景：
// 1. 新固件启动后崩溃
// 2. MCUboot 检测到启动失败
// 3. 自动回滚到 Slot 0
// 4. 标记 Slot 1 为无效

// 超时回滚
// CONFIG_BOOTUTIL_CONFIRM_TIMEOUT=10  ← 10秒确认超时

07 安全启动

// 安全启动配置
// CONFIG_MCUBOOT_SIGNATURE=y           ← RSA/ECDSA 签名
// CONFIG_MCUBOOT_ENCRYPTION=y          ← 镜像加密
// CONFIG_BOOT_SIGNATURE_TYPE_RSA=y     ← RSA-2048
// CONFIG_BOOT_SIGNATURE_TYPE_ECDSA_P256=y ← ECDSA-P256

// 签名密钥生成
// imgtool keygen -k signing-key.pem -t RSA-2048
// imgtool sign --key signing-key.pem firmware.bin signed.bin

// 启动时验证
// 1. 读取镜像头
// 2. 验证签名
// 3. 验证完整性（SHA-256）
// 4. 验证版本号（防降级）
