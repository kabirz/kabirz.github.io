01 概述

内核各子系统之间需要相互通知事件：网络设备 up/down、CPU 热插拔、系统重启、内核 panic、OOM 等。如果让每个消息发布者直接耦合所有订阅者，代码将不可维护。

Notifier Chain 是内核自带的发布-订阅（Pub-Sub）框架，约 300 行代码实现了完整的事件通知总线。

struct notifier_block
struct notifier_block {
    notifier_fn_t notifier_call;    // 回调函数
    struct notifier_block __rcu *next; // 链表
    int priority;                   // 优先级（高者先执行）
};
回调函数签名
typedef int (*notifier_fn_t)(
    struct notifier_block *nb,
    unsigned long action,   // 事件类型
    void *data              // 事件数据
);

发布者只需要调用 xxx_notifier_call_chain()，订阅者调用 xxx_notifier_chain_register()，双方通过 action 和 data 交换信息。

02 四类通知链

根据调用上下文不同，内核提供四种通知链：

类型锁调用上下文典型场景
ATOMIC_NOTIFIER spinlock 中断 / NMI / 原子上下文 panic、die、netlink
BLOCKING_NOTIFIER mutex 进程上下文（可睡眠） reboot、netdev、module
RAW_NOTIFIER 无锁 任意上下文（调用者自管理） 特殊架构需求
SRCU_NOTIFIER SRCU 进程上下文（读多写极少） 最通用，但开销较低
选择原则:
- 回调会在中断中执行 → ATOMIC（不可睡眠）
- 回调需要睡眠、读文件、调用内核 API → BLOCKING
- 性能敏感且发布极少 → SRCU
- 调用者自己处理锁 → RAW 
// 链头定义
struct atomic_notifier_head {
    spinlock_t lock;
    struct notifier_block __rcu *head;
};

struct blocking_notifier_head {
    struct rw_semaphore rwsem;
    struct notifier_block __rcu *head;
};

struct raw_notifier_head {
    struct notifier_block __rcu *head;
};

struct srcu_notifier_head {
    struct srcu_struct srcu;
    struct mutex mutex;
    struct notifier_block __rcu *head;
};

03 核心 API

四类通知链的 API 完全对称：

操作ATOMICBLOCKINGRAWSRCU
静态初始化 ATOMIC_NOTIFIER_HEAD(name) / BLOCKING_NOTIFIER_HEAD(name) …
注册 atomic_notifier_chain_register blocking_notifier_chain_register raw_notifier_chain_register srcu_notifier_chain_register
注销 …_chain_unregister()（对称命名）
通知 atomic_notifier_call_chain blocking_notifier_call_chain raw_notifier_call_chain srcu_notifier_call_chain 
// 通用通知函数签名
int atomic_notifier_call_chain(struct atomic_notifier_head *nh,
    unsigned long action, void *data);

// 返回值：最后一个 NOTIFY_OK / NOTIFY_STOP 的结果
// 或 NOTIFY_BAD（如果被任何回调拒绝）

帮助函数：

// 将 errno 编码为通知链返回值
#define notifier_from_errno(err) \
    (NOTIFY_STOP_MASK | (NOTIFY_OK - err))

// 从通知链返回值提取 errno
int notifier_to_errno(int ret)
{
    ret &= ~NOTIFY_STOP_MASK;
    return ret > NOTIFY_OK ? NOTIFY_OK - ret : 0;
}

04 返回值语义

返回值值含义
NOTIFY_DONE0x0000不关心，不反对
NOTIFY_OK0x0001处理成功，继续传播
NOTIFY_BAD0x8002处理失败，停止传播
NOTIFY_STOP0x8001停止传播（OK 但不要继续）
NOTIFY_STOP_MASK0x8000停止标志位掩码

传播规则：

  • 回调按 priority 从高到低依次调用
  • 如果某个回调返回 NOTIFY_BAD 或 NOTIFY_STOP（含 NOTIFY_STOP_MASK），传播立即停止
  • 所有回调返回后，取最后一个值作为整个 call_chain 的返回值
注意:
NOTIFY_BAD 和 NOTIFY_STOP 都设置了 NOTIFY_STOP_MASK 位（0x8000）。
区别：NOTIFY_STOP = STOP_MASK | OK，语义为"我处理了但到此为止"；
NOTIFY_BAD = STOP_MASK | 0x0002，语义为"我处理失败了"。

05 优先级排序

// 回调按 priority 从高到低排序
// 同一优先级按注册时间保持 FIFO（先注册先执行）

struct notifier_block nb1 = {
    .notifier_call = my_handler1,
    .priority = 10,   // 先执行
};
struct notifier_block nb2 = {
    .notifier_call = my_handler2,
    .priority = 0,    // 后执行
};

// 注册具有唯一优先级的回调（拒绝同优先级冲突）
int atomic_notifier_chain_register_unique_prio(
    struct atomic_notifier_head *nh,
    struct notifier_block *nb);

优先级典型取值：

  • INT_MAX — 最优先（如 watchdog 监听 panic）
  • 0 — 默认优先级
  • INT_MIN — 最后被通知

06 内核实战

1. panic_notifier_list — panic 事件

// kernel/panic.c
ATOMIC_NOTIFIER_HEAD(panic_notifier_list);

void panic(const char *fmt, ...)
{
    // ... 打印 panic 信息 ...
    atomic_notifier_call_chain(&panic_notifier_list,
                               PANIC_EVENT, buf);
    // ... kdump / 重启 ...
}

// 注册：在 panic 时执行自己的回调
static int my_panic_cb(struct notifier_block *nb,
                       unsigned long action, void *data)
{
    pr_emerg("my driver: system is panicking!\n");
    emergency_save_state();
    return NOTIFY_DONE;
}
static struct notifier_block my_panic_nb = {
    .notifier_call = my_panic_cb,
    .priority = INT_MAX,  // 最先执行
};
atomic_notifier_chain_register(&panic_notifier_list, &my_panic_nb);

2. netdev_chain — 网络设备事件

// net/core/dev.c
BLOCKING_NOTIFIER_HEAD(netdev_chain);

// 事件类型
enum {
    NETDEV_UP,       // 设备启用
    NETDEV_DOWN,     // 设备关闭
    NETDEV_REGISTER, // 设备注册
    NETDEV_UNREGISTER,
    NETDEV_CHANGENAME,
    NETDEV_CHANGEADDR,
    // ... 30+ 种事件
};

// 发布
call_netdevice_notifiers(NETDEV_UP, dev);

// 订阅
static int my_netdev_cb(struct notifier_block *nb,
                        unsigned long event, void *data)
{
    struct net_device *dev = data;
    if (event == NETDEV_UP)
        netdev_info(dev, "my: device came up\n");
    return NOTIFY_DONE;
}

07 实现原理

notifier_call_chain 是所有四种类型的公共内核：

// kernel/notifier.c — 核心遍历逻辑
static int notifier_call_chain(
    struct notifier_block **nl,
    unsigned long action, void *data,
    int nr_to_call, int *nr_calls)
{
    int ret = NOTIFY_DONE;
    struct notifier_block *nb, *next_nb;

    nb = rcu_dereference_raw(*nl);
    while (nb && nr_to_call) {
        next_nb = rcu_dereference_raw(nb->next);
        ret = nb->notifier_call(nb, action, data);
        if (nr_calls)
            (*nr_calls)++;
        if (ret & NOTIFY_STOP_MASK)
            break;  // 停止传播
        nb = next_nb;
        nr_to_call--;
    }
    return ret;
}

四类通知链的差异仅在锁策略：

类型注册/注销加锁通知加锁同步语义
ATOMICspinlockrcu_read_lock + spinlock通知时允许注册
BLOCKINGrwsem 写锁rwsem 读锁注册阻塞等待通知完成
RAW无锁无锁调用者自管理
SRCUmutexsrcu_read_lock注销后 synchronize_srcu
