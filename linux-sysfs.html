01 概述

sysfs 是 Linux 内核提供的虚拟文件系统，挂载在 /sys。它将内核设备模型（kobject、kset）以文件和目录的形式暴露给用户态，允许通过 cat/echo 读写内核参数。sysfs 是 udev 设备管理的基础，也是驱动开发调试的核心工具。

sysfs 核心功能
  • 设备属性：/sys/class/、/sys/devices/
  • 驱动绑定：bind/unbind 控制
  • 电源管理：/sys/power/、runtime PM
  • 模块参数：/sys/module/
  • 内核配置：/sys/kernel/
  • 虚拟设备：/sys/devices/virtual/

03 核心数据结构

// kobject + 属性 = sysfs 的基础

struct kobject {          // 内核对象基类
    const char      *name;     // 对象名称（成为目录名）
    struct kset      *kset;     // 所属 kset
    struct kobject   *parent;   // 父对象（成为父目录）
    struct kobj_type *ktype;    // 对象类型
    struct sysfs_dirent *sd;  // sysfs 条目
};

struct attribute {        // sysfs 属性（成为文件）
    const char     *name;     // 文件名（如 "mtu"）
    umode_t         mode;     // 文件权限
};

// 设备属性 — 每个属性对应一个 show/store 函数
struct device_attribute {
    struct attribute attr;
    ssize_t (*show)(struct device *, struct device_attribute *, char *);
    ssize_t (*store)(struct device *, struct device_attribute *,
                     const char *, size_t);
};

// 宏简化属性定义
#define DEVICE_ATTR(_name, _mode, _show, _store) \
    struct device_attribute dev_attr_##_name = \
        __ATTR(_name, _mode, _show, _store)
// 用法：DEVICE_ATTR(mtu, 0644, show_mtu, store_mtu);

04 属性读写流程

用户态读取属性（cat /sys/class/net/eth0/mtu）：

  cat /sys/class/net/eth0/mtu
        │
        ↓
  VFS → sysfs 文件操作
        │
        ↓
  sysfs_kf_seq_show()
        │
        ↓
  调用 device_attribute.show 回调
        │
        ├── 网络设备：netdev_show()
        │       → dev->mtu
        │       → 格式化为字符串 "1500"
        │
        └── 返回给用户态

用户态写入属性（echo 9000 > /sys/class/net/eth0/mtu）：

  echo 9000 > /sys/class/net/eth0/mtu
        │
        ↓
  VFS → sysfs 文件操作
        │
        ↓
  sysfs_kf_write()
        │
        ↓
  调用 device_attribute.store 回调
        │
        ├── 网络设备：netdev_store()
        │       → 解析 "9000"
        │       → dev_set_mtu(dev, 9000)
        │       → 调用ndo_change_mtu
        │
        └── 生效

05 应用场景

# 网络设备调试
$ cat /sys/class/net/eth0/mtu              # 查看 MTU
$ echo 9000 > /sys/class/net/eth0/mtu     # 设置 Jumbo Frame
$ cat /sys/class/net/eth0/statistics/rx_packets  # 收包计数

# CPU 频率调节
$ cat /sys/devices/system/cpu/cpu0/cpufreq/scaling_governor
$ echo performance > /sys/devices/system/cpu/cpu0/cpufreq/scaling_governor

# 电源管理
$ cat /sys/power/state          # 查看支持的休眠状态
$ echo mem > /sys/power/state   # 挂起到内存

# 内核模块参数
$ cat /sys/module/usbcore/parameters/authorized_default

# 块设备队列
$ cat /sys/block/sda/queue/scheduler        # I/O 调度器
$ cat /sys/block/sda/queue/nr_requests      # 队列深度

# 驱动绑定/解绑
$ echo "0000:00:1f.2" > /sys/bus/pci/drivers/ahci/unbind
$ echo "0000:00:1f.2" > /sys/bus/pci/drivers/ahci/bind
