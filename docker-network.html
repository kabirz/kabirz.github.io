01 жҰӮиҝ°

Docker йҖҡиҝҮ libnetworkпјҲзҺ°е·Іиў« libnetwork + containerd CRI plugin еҸ–д»ЈпјүжҸҗдҫӣеӨҡз§ҚзҪ‘з»ңй©ұеҠЁпјҡ

й©ұеҠЁйҖӮз”ЁеңәжҷҜйҡ”зҰ»зә§еҲ«
bridgeй»ҳи®ӨпјҢеҚ•жңәжҜҸе®№еҷЁзӢ¬з«Ӣ netnsпјҢйҖҡиҝҮ veth + bridge дә’иҒ”
hostж— йҡ”зҰ»пјҢй«ҳжҖ§иғҪе…ұдә«е®ҝдё» netns
noneе®Ңе…Ёйҡ”зҰ»ж— д»»дҪ•зҪ‘з»ңй…ҚзҪ®
overlayеӨҡжңәйӣҶзҫӨVXLAN йҡ§йҒ“и·Ёдё»жңә
macvlanжЎҘжҺҘеҲ°зү©зҗҶзҪ‘з»ңе®№еҷЁзӢ¬еҚ  MAC ең°еқҖ
ipvlanеҗҢ macvlanпјҢж—  MACе…ұдә« MACпјҢL2/L3 жЁЎејҸ
custom pluginз¬¬дёүж–№пјҲCalicoгҖҒCiliumпјүз”ұжҸ’д»¶е®ҡд№ү

еҹәжң¬жҰӮеҝөпјҡ

  • NetworkпјҡдёҖдёӘзҪ‘з»ңе®һдҫӢпјҲеҰӮ docker0гҖҒиҮӘе®ҡд№ү mynetпјү
  • Endpointпјҡе®№еҷЁеңЁжҹҗдёӘзҪ‘з»ңдёӯзҡ„жҺҘе…ҘзӮ№
  • ContainerпјҡйҖҡиҝҮ endpoint еҠ е…ҘзҪ‘з»ң
  • Sandboxпјҡе®№еҷЁзҡ„ network namespace

  --driver=bridge \
  --subnet=10.10.0.0/24 \
  --gateway=10.10.0.1 \

  -o "com.docker.network.bridge.name=br10" \

docker run --network=mynet -d nginx

  • еҗҢжңәе®№еҷЁеҸҜдә’йҖҡпјҢи·ЁжңәйңҖиҰҒ overlay жҲ–е…¶д»–
  • з«ҜеҸЈеҶІзӘҒпјҲhost:containerпјү

03 host зҪ‘з»ң

host жЁЎејҸдёӢе®№еҷЁдёҚеҲӣе»ә network namespaceпјҢзӣҙжҺҘдҪҝз”Ёе®ҝдё»зҪ‘з»ңж Ҳпјҡ

# дҪҝз”Ё
docker run --network=host nginx

# е®№еҷЁеҶ…пјҡ
$ docker exec -it <c> ip addr
# 1: lo
# 2: eth0  вҶҗ е®ҝдё»зҡ„зҪ‘еҚЎ
# 3: wlan0 вҶҗ е®ҝдё»зҡ„ WiFi
# е…ЁйғЁеҸҜи§Ғ

$ docker exec -it <c> ss -tlnp
# nginx зӣ‘еҗ¬еңЁ 80 з«ҜеҸЈпјҲе®ҝдё»зҡ„ 80пјү

# зү№зӮ№пјҡ
# - жҖ§иғҪжңҖдҪіпјҲж—  NATгҖҒж—  vethпјү
# - з«ҜеҸЈзӣҙжҺҘеҚ з”Ёе®ҝдё»
# - еӨҡе®№еҷЁж— жі•зӣ‘еҗ¬еҗҢдёҖз«ҜеҸЈ
# - жІЎжңүзҪ‘з»ңйҡ”зҰ»

е…ёеһӢз”ЁйҖ”пјҡ

  • жҖ§иғҪж•Ҹж„ҹеә”з”ЁпјҲеҰӮ packet captureгҖҒзҪ‘з»ңи®ҫеӨҮпјү
  • йңҖиҰҒи®ҝй—®е®ҝдё»зү№е®ҡзҪ‘з»ңж Ҳзҡ„еә”з”Ё
  • з®ҖеҢ–йғЁзҪІпјҲйҒҝе…Қз«ҜеҸЈжҳ е°„пјү

е®үе…ЁжіЁж„Ҹпјҡ

host зҪ‘з»ңзҡ„е®үе…ЁйЈҺйҷ©:
е®№еҷЁиҝӣзЁӢеҸҜзӣҙжҺҘзңӢеҲ°жүҖжңүе®ҝдё»зҪ‘з»ңжөҒйҮҸгҖӮ
иғҪзӣ‘еҗ¬е®ҝдё»жүҖжңүз«ҜеҸЈпјҢеҢ…жӢ¬е…¶д»–е®№еҷЁгҖӮ
иғҪз»‘е®ҡ 0.0.0.0пјҢеҜ№е…¬зҪ‘жҡҙйңІгҖӮ
е®һйҷ…дёҠзӯүеҗҢдәҺйқһе®№еҷЁеә”з”Ё вҖ” дё§еӨұе®№еҷЁйҡ”зҰ»д»·еҖјгҖӮ
е»әи®®пјҡд»…з”ЁдәҺеҸҜдҝЎгҖҒеҝ…йңҖзҡ„еңәжҷҜгҖӮ

04 none зҪ‘з»ң

none жЁЎејҸз»ҷе®№еҷЁзӢ¬з«Ӣзҡ„ net namespace дҪҶдёҚй…ҚзҪ®д»»дҪ•зҪ‘з»ңпјҡ

docker run --network=none alpine ip addr
# 1: lo  вҶҗ д»… loopback
# жІЎжңүд»»дҪ•е…¶д»–жҺҘеҸЈ

# з”ЁйҖ”пјҡ
# - й«ҳеәҰйҡ”зҰ»зҡ„жү№еӨ„зҗҶд»»еҠЎ
# - д»…жң¬ең° IPC йҖҡдҝЎ
# - жөӢиҜ•/и°ғиҜ•
# - иҮӘе®ҡд№үзҪ‘з»ңж ҲпјҲжүӢеҠЁй…ҚзҪ®пјү

# жүӢеҠЁй…ҚзҪ®зӨәдҫӢпјҡ
docker run --network=none -it alpine sh
# иҝӣе…Ҙе®№еҷЁеҗҺпјҡ

06 macvlan / ipvlan

macvlan и®©е®№еҷЁзӣҙжҺҘеҮәзҺ°еңЁзү©зҗҶзҪ‘з»ңдёӯпјҢжҜҸдёӘе®№еҷЁжңүзӢ¬з«Ӣ MAC ең°еқҖпјҡ

# еҲӣе»ә macvlan зҪ‘з»ң
docker network create -d macvlan \
  --subnet=192.168.1.0/24 \
  --gateway=192.168.1.1 \
  -o parent=eth0 \
  macvlan_net

# еҗҜеҠЁе®№еҷЁ
docker run --network=macvlan_net -d nginx

# зү№зӮ№пјҡ
# - е®№еҷЁжңүзӢ¬з«Ӣ MACпјҲи·Ҝз”ұеҷЁзңӢеҲ°зҡ„пјү
# - е®№еҷЁзӣҙжҺҘд»Һ DHCP иҺ·еҸ– IPпјҲеҰӮй…ҚзҪ®пјү
# - жІЎжңү NATгҖҒжҖ§иғҪй«ҳ
# - дёҚиғҪдёҺе®ҝдё»зӣҙжҺҘйҖҡдҝЎпјҲpromiscuous mode йҷҗеҲ¶пјү

# host vs bridge vs macvlanпјҡ
#   host      - е…ұдә«е®ҝдё»зҪ‘еҚЎпјҢж— зӢ¬з«Ӣ MAC
#   bridge    - йҖҡиҝҮ docker0 жЎҘжҺҘпјҢNAT
#   macvlan   - зӢ¬з«Ӣ MAC + IPпјҢзӣҙжҺҘеҮәзҺ°еңЁ L2 зҪ‘з»ң

ipvlanпјҲе…ұдә« MACпјүпјҡ

# ipvlan жңүдёӨз§ҚжЁЎејҸпјҡ
# L2 (й»ҳи®Ө): е®№еҷЁе…ұдә«зҲ¶жҺҘеҸЈ MACпјҢзӢ¬з«Ӣ IP
# L3: е®№еҷЁзӢ¬з«Ӣ IPпјҢи·Ҝз”ұеңЁеҶ…ж ёдёӯ

docker network create -d ipvlan \
  --subnet=192.168.1.0/24 \
  -o parent=eth0 \
  -o ipvlan_mode=l2 \
  ipvlan_net

# macvlan vs ipvlanпјҡ
#   macvlan:
#     - жҜҸе®№еҷЁзӢ¬з«Ӣ MAC
#     - йңҖиҰҒдәӨжҚўжңәе…Ғи®ёеӨҡ MAC
#     - жҖ§иғҪжңҖеҘҪ
#   
#   ipvlan:
#     - е…ұдә« MACпјҲжүҖжңүе®№еҷЁзңӢиө·жқҘжҳҜеҗҢдёҖдёӘи®ҫеӨҮпјү
#     - йҖӮеҗҲ MAC йҷҗеҲ¶дёҘж јзҡ„зҪ‘з»ң
#     - йңҖиҰҒи·Ҝз”ұеҷЁи·Ҝз”ұпјҲL3 жЁЎејҸпјү
macvlan зҡ„еёёи§ҒеңәжҷҜ:
е®¶еәӯзҪ‘з»ңпјҲзӣҙжҺҘиҺ·еҸ– LAN IPпјүгҖӮ
зү©иҒ”зҪ‘и®ҫеӨҮпјҲжҜҸе®№еҷЁзӢ¬з«Ӣиә«д»ҪпјүгҖӮ
жҖ§иғҪж•Ҹж„ҹеә”з”ЁпјҲйҒҝе…Қ NATпјүгҖӮ
дј з»ҹеә”з”ЁпјҲйңҖиҰҒе№ҝж’ӯгҖҒз»„ж’ӯпјүгҖӮ
йҷҗеҲ¶пјҡеӨ§еӨҡж•° WiFi дёҚж”ҜжҢҒ macvlanпјҲpromiscuous mode зҰҒз”ЁпјүпјӣдёҚиғҪдёҺе®ҝдё»дә’йҖҡгҖӮ

07 з«ҜеҸЈжҳ е°„

bridge зҪ‘з»ңдёӯпјҢеӨ–йғЁи®ҝй—®е®№еҷЁйңҖиҰҒз«ҜеҸЈжҳ е°„пјҡ

# 1. жҢҮе®ҡ host:container
docker run -p 8080:80 nginx
# е®ҝдё» 8080 вҶ’ е®№еҷЁ 80

# 2. д»… containerпјҲhost йҡҸжңәпјү
docker run -p 80 nginx

# 3. жҢҮе®ҡ IP
docker run -p 127.0.0.1:8080:80 nginx

# 4. жҢҮе®ҡеҚҸи®®
docker run -p 53:53/udp dns

# 5. иҢғеӣҙ
docker run -p 8000-8100:8000-8100 app

# 6. е…ЁйғЁжҡҙйңІпјҲEXPOSEпјү
docker run -P nginx
# -P еӨ§еҶҷпјҡжҳ е°„жүҖжңү EXPOSE зҡ„з«ҜеҸЈеҲ°йҡҸжңә host з«ҜеҸЈ

е®һзҺ°жңәеҲ¶пјҲдёӨз§Қпјүпјҡ

// ж–№ејҸ 1: iptables DNATпјҲй»ҳи®ӨпјҢжҖ§иғҪеҘҪпјү
iptables -t nat -A DOCKER \
  -p tcp --dport 8080 \
  -j DNAT --to-destination 172.17.0.2:80
iptables -t nat -A POSTROUTING \
  -s 172.17.0.2 -j MASQUERADE

// ж•°жҚ®жөҒпјҡ
//   еӨ–йғЁ вҶ’ host:8080 вҶ’ DNAT вҶ’ 172.17.0.2:80 вҶ’ е®№еҷЁ
//   е®№еҷЁ вҶ’ 172.17.0.2:80 вҶ’ еӣһеҢ… вҶ’ MASQUERADE вҶ’ host:8080 вҶ’ еӨ–йғЁ

// ж–№ејҸ 2: userland-proxyпјҲdocker-proxyпјү
//   еҗҜеҠЁдёҖдёӘ Go иҝӣзЁӢпјҢзӣ‘еҗ¬ host:8080
//   ж•°жҚ®жөҒз»Ҹ proxyпјҢеҶҚиҪ¬еҸ‘еҲ°е®№еҷЁ
//   зү№зӮ№пјҡе…је®№жҖ§жӣҙеҘҪпјҲж—  iptables д№ҹиғҪе·ҘдҪңпјү
//         дҪҶжҖ§иғҪе·®
//
// /etc/docker/daemon.json:
//   "userland-proxy": false   вҶҗ жҺЁиҚҗпјҲй»ҳи®Ө falseпјү

// е®№еҷЁеҶ…зӣ‘еҗ¬ 0.0.0.0
// еҗҰеҲҷз«ҜеҸЈжҳ е°„ж— ж•Ҳ
// nginx.conf:
//   listen 0.0.0.0:80;
