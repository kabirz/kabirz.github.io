// xBlockSize 的最高位用于标记分配状态
#define heapBLOCK_ALLOCATED_BITMASK  ( ( ( size_t ) 1 ) << ( ( sizeof( size_t ) * 8 ) - 1 ) )
#define heapBLOCK_IS_ALLOCATED( pxBlock )  ( ( ( pxBlock->xBlockSize ) & heapBLOCK_ALLOCATED_BITMASK ) != 0 )
#define heapALLOCATE_BLOCK( pxBlock )      ( ( pxBlock->xBlockSize ) |= heapBLOCK_ALLOCATED_BITMASK )
#define heapFREE_BLOCK( pxBlock )          ( ( pxBlock->xBlockSize ) &= ~heapBLOCK_ALLOCATED_BITMASK )

03 pvPortMalloc 实现

// heap_4.c 第 173-249+ 行
void * pvPortMalloc( size_t xWantedSize )
{
    BlockLink_t * pxBlock;
    BlockLink_t * pxPreviousBlock;
    void * pvReturn = NULL;

    // 1. 增加 BlockLink_t 头部大小
    xWantedSize += xHeapStructSize;

    // 2. 对齐到 portBYTE_ALIGNMENT
    if( ( xWantedSize & portBYTE_ALIGNMENT_MASK ) != 0x00 )
    {
        xWantedSize += portBYTE_ALIGNMENT - ( xWantedSize & portBYTE_ALIGNMENT_MASK );
    }

    // 3. 挂起调度器
    vTaskSuspendAll();
    {
        // 4. 首次调用时初始化堆
        if( pxEnd == NULL )
        {
            prvHeapInit();
        }

        // 5. 首次适配算法
        pxPreviousBlock = &xStart;
        pxBlock = xStart.pxNextFreeBlock;

        // 遍历空闲块链表，找到第一个足够大的块
        while( ( pxBlock->xBlockSize < xWantedSize ) && ( pxBlock->pxNextFreeBlock != NULL ) )
        {
            pxPreviousBlock = pxBlock;
            pxBlock = pxBlock->pxNextFreeBlock;
        }

        // 6. 找到足够大的块
        if( pxBlock != pxEnd )
        {
            pvReturn = ( void * ) ( ( ( uint8_t * ) pxPreviousBlock ) + xHeapStructSize );

            // 7. 从空闲链表中移除
            pxPreviousBlock->pxNextFreeBlock = pxBlock->pxNextFreeBlock;

            // 8. 如果剩余空间足够，分割块
            if( ( pxBlock->xBlockSize - xWantedSize ) > heapMINIMUM_BLOCK_SIZE )
            {
                // 创建新的空闲块
                pxNewBlockLink = ( void * ) ( ( ( uint8_t * ) pxBlock ) + xWantedSize );
                pxNewBlockLink->xBlockSize = pxBlock->xBlockSize - xWantedSize;
                pxBlock->xBlockSize = xWantedSize;

                // 插入空闲链表
                prvInsertBlockIntoFreeList( pxNewBlockLink );
            }

            // 9. 标记为已分配
            heapALLOCATE_BLOCK( pxBlock );
            xFreeBytesRemaining -= pxBlock->xBlockSize;
        }
    }
    xTaskResumeAll();

    return pvReturn;
}

04 vPortFree 实现

// heap_4.c 中的 vPortFree()
void vPortFree( void * pv )
{
    uint8_t * puc = ( uint8_t * ) pv;
    BlockLink_t * pxLink;

    // 1. 获取块头部
    puc -= xHeapStructSize;
    pxLink = ( BlockLink_t * ) puc;

    // 2. 验证块是否已分配
    configASSERT( heapBLOCK_IS_ALLOCATED( pxLink ) );

    // 3. 挂起调度器
    vTaskSuspendAll();
    {
        // 4. 清除分配标记
        heapFREE_BLOCK( pxLink );

        // 5. 插入空闲链表（自动合并相邻块）
        prvInsertBlockIntoFreeList( ( ( BlockLink_t * ) pxLink ) );

        // 6. 更新空闲字节数
        xFreeBytesRemaining += pxLink->xBlockSize;
    }
    xTaskResumeAll();
}

05 相邻块合并

// heap_4.c 中的 prvInsertBlockIntoFreeList()
// 这是 heap_4 的核心优势：自动合并相邻空闲块

static void prvInsertBlockIntoFreeList( BlockLink_t * pxBlockToInsert )
{
    BlockLink_t * pxIterator;
    uint8_t * puc;

    // 遍历空闲链表，按地址排序插入
    for( pxIterator = &xStart; pxIterator->pxNextFreeBlock != NULL; pxIterator = pxIterator->pxNextFreeBlock )
    {
        // 如果当前块地址在插入块之后，停止
        if( ( uint8_t * ) pxIterator->pxNextFreeBlock > ( uint8_t * ) pxBlockToInsert )
        {
            break;
        }
    }

    // 检查是否与前一个块相邻（地址连续）
    puc = ( uint8_t * ) pxIterator;
    if( ( puc + pxIterator->xBlockSize ) == ( uint8_t * ) pxBlockToInsert )
    {
        // 合并：扩大前一个块的大小
        pxIterator->xBlockSize += pxBlockToInsert->xBlockSize;
        pxBlockToInsert = pxIterator;
    }

    // 检查是否与后一个块相邻
    puc = ( uint8_t * ) pxBlockToInsert;
    if( ( puc + pxBlockToInsert->xBlockSize ) == ( uint8_t * ) pxIterator->pxNextFreeBlock )
    {
        // 合并：扩大当前块，移除后一个块
        pxBlockToInsert->xBlockSize += pxIterator->pxNextFreeBlock->xBlockSize;
        pxBlockToInsert->pxNextFreeBlock = pxIterator->pxNextFreeBlock->pxNextFreeBlock;
    }
    else
    {
        pxBlockToInsert->pxNextFreeBlock = pxIterator->pxNextFreeBlock;
    }

    // 插入链表
    if( pxIterator != pxBlockToInsert )
    {
        pxIterator->pxNextFreeBlock = pxBlockToInsert;
    }
}

09 完整示例

// ===== FreeRTOS 内存管理完整示例 =====
// 演示：堆配置、静态分配、栈溢出检测

#include "FreeRTOS.h"
#include "task.h"
#include "queue.h"
#include <stdio.h>

// 配置堆大小
// FreeRTOSConfig.h 中：
// #define configTOTAL_HEAP_SIZE  ((size_t)(32 * 1024))
// #define configSUPPORT_STATIC_ALLOCATION 1
// #define configSUPPORT_DYNAMIC_ALLOCATION 1

// 静态任务缓冲区
StaticTask_t xIdleTaskBuffer;
StackType_t xIdleStack[ configMINIMAL_STACK_SIZE ];

// 空闲任务内存回调（静态分配必需）
void vApplicationGetIdleTaskMemory( StaticTask_t ** ppxIdleTaskTCBBuffer,
                                    StackType_t ** ppxIdleTaskStackBuffer,
                                    uint32_t * pulIdleTaskStackSize )
{
    *ppxIdleTaskTCBBuffer = &xIdleTaskBuffer;
    *ppxIdleTaskStackBuffer = xIdleStack;
    *pulIdleTaskStackSize = configMINIMAL_STACK_SIZE;
}

// 栈溢出钩子
void vApplicationStackOverflowHook( TaskHandle_t xTask, char * pcTaskName )
{
    printf( "Stack overflow in: %s\\n", pcTaskName );
    for ( ;; ) {}
}

// 堆统计
void vPrintHeapStats( void )
{
    printf( "Free heap: %lu bytes\\n", xPortGetFreeHeapSize() );
    printf( "Min ever free: %lu bytes\\n", xPortGetMinimumEverFreeHeapSize() );
}

// 动态任务
void vDynamicTask( void * pvParameters )
{
    for ( ;; )
    {
        // 动态分配内存
        void * pv = pvPortMalloc( 100 );
        if( pv != NULL )
        {
            // 使用内存...
            vPortFree( pv );
        }
        vTaskDelay( pdMS_TO_TICKS( 1000 ) );
    }
}

// 静态任务
void vStaticTask( void * pvParameters )
{
    for ( ;; )
    {
        printf( "Static task running\\n" );
        vTaskDelay( pdMS_TO_TICKS( 2000 ) );
    }
}

int main( void )
{
    // 静态创建任务
    StaticTask_t xTaskBuffer;
    StackType_t xStack[ 128 ];
    xTaskCreateStatic( vStaticTask, "Static", 128, NULL, 1, xStack, &xTaskBuffer );

    // 动态创建任务
    xTaskCreate( vDynamicTask, "Dynamic", 128, NULL, 2, NULL );

    // 打印堆统计
    vPrintHeapStats();

    vTaskStartScheduler();

    for ( ;; ) {}
}
