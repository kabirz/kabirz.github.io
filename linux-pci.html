01 概述

PCI (Peripheral Component Interconnect) 及其后继 PCIe 是 x86/ARM 服务器的主流外设总线。Linux PCI 子系统负责枚举总线、绑定驱动、管理资源、处理中断。

03 总线枚举

BIOS/UEFI 完成硬件枚举（分配总线号、BAR 资源、IRQ）。Linux 启动时再次扫描：

// drivers/pci/probe.c

// 入口（架构相关）：
// x86: arch/x86/pci/common.c → pcibios_scan_specific_bus()
// ARM64: arch/arm64/kernel/pci.c → pci_acpi_scan_root()
// 通用: pci_host_probe() → pci_scan_bus()

// 关键函数：
// 1. 扫描每条总线
int pci_scan_bus(struct pci_bus *bus)
{
    // 对每个可能的 dev (0..31) × func (0..7)：
    //   - 读取 vendor ID
    //   - 如果 != 0xFFFF → 有效设备
    //   - pci_scan_device() 创建 pci_dev
    //   - 如果是 bridge → 递归扫描子总线
}

// 2. 创建 pci_dev
static struct pci_dev *pci_scan_device(struct pci_bus *bus, int devfn)
{
    u32 l;
    // 读配置 offset 0 (vendor + device)
    if (pci_bus_read_config_dword(bus, devfn, PCI_VENDOR_ID, &l))
        return NULL;
    if (l == 0xffffffff || l == 0x00000000 || l == 0x0000ffff)
        return NULL;
    
    struct pci_dev *dev = pci_alloc_dev(bus);
    dev->devfn = devfn;
    dev->vendor = l & 0xffff;
    dev->device = (l >> 16) & 0xffff;
    // ...
    
    // 初始化 capabilities
    pci_setup_device(dev);
    pci_init_capabilities(dev);
    
    return dev;
}

// 3. 初始化 capabilities
// drivers/pci/probe.c
static void pci_init_capabilities(struct pci_dev *dev)
{
    pci_msi_init(dev);            // MSI/MSI-X
    pci_msix_init(dev);
    pci_allocate_vc_save_buf(dev);
    pci_pm_init(dev);              // Power Management
    pci_ea_init(dev);              // Enhanced Allocation
    pci_bridge_resources_init(dev);
    pci_vpd_init(dev);             // Vital Product Data
    pci_hpbridge_init(dev);        // Hotplug bridge
    pci_hpmem_init(dev);
    pcie_aspm_init_link_state(dev); // PCIe ASPM
    pcie_bandwidth_notification(dev);
    pci_ptm_init(dev);              // Precision Time Measurement
    pci_aer_init(dev);              // Advanced Error Reporting
    pci_dpc_init(dev);              // Downstream Port Containment
    pci_doe_init(dev);              // Data Object Exchange
    pci_ats_init(dev);              // Address Translation Service
    pci_pri_init(dev);              // Page Request Interface
    pci_pasid_init(dev);            // PASID
    pci_iov_init(dev);              // SR-IOV
    pci_acs_init(dev);              // Access Control Services
    pci_ile_init(dev);
    // ...
}

04 驱动绑定

// drivers/pci/pci-driver.c

// 入口：注册 PCI 驱动
int __pci_register_driver(struct pci_driver *drv, struct module *owner)
{
    // 初始化 driver model
    drv->driver.bus = &pci_bus_type;
    drv->driver.owner = owner;
    drv->driver.mod_name = KBUILD_MODNAME;
    // ...
    return driver_register(&drv->driver);
}

// 设备 + 驱动匹配
static int pci_bus_match(struct device *dev, struct device_driver *drv)
{
    struct pci_dev *pci_dev = to_pci_dev(dev);
    struct pci_driver *pci_drv = to_pci_driver(drv);
    
    if (!pci_driver_for_dma(pci_drv, pci_dev))
        return false;
    
    // 通过 id_table 匹配
    return pci_match_device(pci_drv, pci_dev) != NULL;
}

// 匹配函数
const struct pci_device_id *pci_match_device(struct pci_driver *drv,
                                              struct pci_dev *dev)
{
    // 1. 遍历 drv->id_table 静态表
    for (id = drv->id_table; id->vendor; id++) {
        if ((id->vendor == PCI_ANY_ID || id->vendor == dev->vendor) &&
            (id->device == PCI_ANY_ID || id->device == dev->device) &&
            (id->subvendor == PCI_ANY_ID || id->subvendor == ...) &&
            (id->class_mask == 0 || ...))
            return id;
    }
    // 2. 检查动态 id 表（用户态通过 sysfs 添加）
    // 3. 检查 driver_override
    return NULL;
}

// 设备 probe（核心驱动入口）
static int pci_device_probe(struct device *dev)
{
    struct pci_dev *pci_dev = to_pci_dev(dev);
    struct pci_driver *drv = to_pci_driver(dev->driver);
    const struct pci_device_id *id;
    
    id = pci_match_device(drv, pci_dev);
    if (!id)
        return -ENODEV;
    
    // 1. 启用设备
    int error = pci_enable_device(pci_dev);
    
    // 2. 调用驱动 probe
    error = local_pci_probe(pci_dev, id, drv);
    
    return error;
}

// 驱动 probe 的典型代码（驱动开发者写的）：
static int my_driver_probe(struct pci_dev *pdev, const struct pci_device_id *ent)
{
    // 1. 启用设备（已由框架做）
    pci_enable_device_mem(pdev);
    
    // 2. 申请区域
    pci_request_regions(pdev, "my_driver");
    
    // 3. 映射 MMIO
    void __iomem *base = pci_iomap(pdev, 0, 0);
    
    // 4. 设置 DMA mask
    dma_set_mask_and_coherent(&pdev->dev, DMA_BIT_MASK(64));
    
    // 5. 注册 MSI-X 中断
    int nvec = pci_alloc_irq_vectors(pdev, 1, 32, PCI_IRQ_MSIX);
    int irq = pci_irq_vector(pdev, 0);
    request_irq(irq, handler, 0, "my_dev", pdev);
    
    // 6. 注册块设备/网卡/字符设备
    return 0;
}

06 MSI/MSI-X

MSI/MSI-X 是 PCIe 的中断机制（详见 通用中断子系统），通过写特定地址触发中断，无需专用引脚。

// drivers/pci/msi/

// MSI-X 描述符
struct msix_entry {
    u32    vector;          // Linux IRQ 号
    u16    entry;           // 设备内 entry 编号
};

// 启用 MSI-X 的典型流程：
int pci_alloc_irq_vectors(struct pci_dev *dev, unsigned int min_vecs,
                          unsigned int max_vecs, unsigned int flags)
{
    // flags = PCI_IRQ_MSIX | PCI_IRQ_MSI | PCI_IRQ_LEGACY
    // 优先尝试 MSI-X，失败则 MSI，最后 INTx
    
    if (flags & PCI_IRQ_MSIX)
        nvec = __pci_enable_msix_range(dev, NULL, min_vecs, max_vecs, NULL, 0);
    if (nvec < 0 && flags & PCI_IRQ_MSI)
        nvec = __pci_enable_msi_range(dev, min_vecs, max_vecs, NULL);
    // ...
    return nvec;
}

// 获取 IRQ 号
int pci_irq_vector(struct pci_dev *dev, unsigned int nr)
{
    // dev->irq 是 INTx IRQ 或第一个 MSI vector
    // nr > 0 时返回 MSI-X vector N
    if (dev->msix_enabled)
        return msi_get_virq(&dev->dev, nr);
    return dev->irq;
}

// MSI 与 MSI-X 对比：
//  ┌────────────┬────────────────────┬─────────────────────┐
//  │            │       MSI          │        MSI-X        │
//  ├────────────┼────────────────────┼─────────────────────┤
//  │ 最大数量   │ 32                 │ 2048                │
//  │ 地址/数据  │ 共享 1 组          │ 每 vector 独立       │
//  │ 寄存器位置 │ capability (0x05)  │ BAR 中的 table       │
//  │ Mask 位    │ 1 组 (32 bits)     │ 每 vector 独立       │
//  │ 灵活性     │ 低                  │ 高                  │
//  └────────────┴────────────────────┴─────────────────────┘

// MSI-X Table 通过 BAR 暴露给驱动
// 每项 16 字节：Message Address + Message Upper Addr + Message Data + Vector Control
// Mask Bit 在 Vector Control bit 0

使用模式：

// 单 vector（最常见）：
int nvec = pci_alloc_irq_vectors(pdev, 1, 1, PCI_IRQ_MSIX);
int irq = pci_irq_vector(pdev, 0);
request_irq(irq, handler, 0, name, pdev);

// 多 vector（NVMe 等高 IO 设备）：
int nvec = pci_alloc_irq_vectors(pdev, 1, 32, PCI_IRQ_MSIX);
for (i = 0; i < nvec; i++) {
    int irq = pci_irq_vector(pdev, i);
    request_irq(irq, handler_i, 0, name, ctx_i);
}

// Affinity 提示（per-CPU 中断）：
irq_set_affinity_hint(irq, cpumask_of(cpu));

07 PCIe 能力与 ASPM

// drivers/pci/pcie/aspm.c — Active State Power Management

// ASPM 链路状态：
//  L0: 全活跃
//  L0s: 部分省电（双向独立，进入快，恢复也快）
//  L1: 更深省电（必须双向同时）
//  L1.1 / L1.2: 极深省电（关闭 PLL，恢复慢）

// 链路状态切换通过：
//  - 硬件自动（基于链路 idle 计时器）
//  - 软件显式（驱动空闲时主动切换）

// 配置策略：
//  pcie_aspm=performance (默认): L0s 禁用，性能优先
//  pcie_aspm=powersave: 启用所有 ASPM
//  pcie_aspm=off: 完全禁用

// API：
//  pci_disable_link_state(pdev, PCIE_LINK_STATE_L0S | PCIE_LINK_STATE_L1);
//  pci_enable_link_state(pdev, ...);

// drivers/pci/pcie/aer.c — Advanced Error Reporting
// AER 提供更详细的错误信息：
//   - Correctable errors: bad TLP, bad DLLP, replay timeout
//   - Uncorrectable (non-fatal): malformed TLP, completion timeout
//   - Uncorrectable (fatal): DLP, surprise down

// 注册 AER 处理：
//  pci_dev->dev.driver->err_handler 提供：
//    error_detected() - 通知错误发生
//    mmio_enabled()   - MMIO 重新启用
//    slot_reset()     - 设备重置完成
//    resume()         - 完全恢复

// drivers/pci/iov.c — SR-IOV (Single Root I/O Virtualization)
//  - PF (Physical Function): 完整设备
//  - VF (Virtual Function): 轻量设备，用于虚拟化直通
//  - 一个 PF 可创建多个 VF
//  - VF 通过 PCI_EXT_CAP_ID_SRIOV capability 暴露

// 启用 SR-IOV：
//  echo N > /sys/bus/pci/devices/XXX/sriov_numvfs
//  → 调用 driver->sriov_configure(pdev, N)
//  → 创建 N 个 VF pci_dev
//  → VF 可独立分配给 VM (via vfio-pci)

int pci_set_power_state(struct pci_dev *dev, pci_power_t state);
//   - 写 PM capability Power State 寄存器
//   - D3hot: 关闭大部分时钟
//   - D3cold: 移除电源（需要 hotplug 支持）

// 启用/禁用设备：
int pci_enable_device(struct pci_dev *dev);  // 唤醒，分配资源
void pci_disable_device(struct pci_dev *dev);  // 禁用，释放

// 唤醒（用于 suspend）：
int pci_enable_wake(struct pci_dev *dev, pci_power_t state, bool enable);
//   - 配置 PME (Power Management Event)
//   - 设备可触发中断唤醒系统

// drivers/pci/hotplug/ — PCIe 热插拔框架
// 标准：
//   - PCIEHP (PCIe native hotplug)
//   - SHPC (Standard Hot-Plug Controller)
//   - ACPI based hotplug
//   - cPCI (CompactPCI)

// PCIEHP 流程：
// 1. Slot 状态变化（插入/拔出）触发中断
// 2. pciehp_isr() 读取 Slot Status
// 3. 调度 pciehp_handle_presence_or_link_change()
// 4. 调用 pci_hp_add_bridge() / pci_stop_and_remove_bus()
// 5. 通过 kobject_uevent 通知用户态
// 6. udev 触发驱动 probe/remove

// ACPI hotplug：
//  - ACPI 表定义 hotplug slot
//  - 通过 _EJ0 method 控制电源
//  - 用于服务器 ProLiant 等
