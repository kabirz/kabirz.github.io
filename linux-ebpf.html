01 概述

eBPF (extended Berkeley Packet Filter) 是 Linux 内核的可编程扩展框架。它允许在内核中安全地运行沙箱程序，而无需修改内核源码或加载内核模块。

核心思想：用户态提供字节码（BPF 指令），内核验证其安全性后 JIT 编译为本机指令执行。

内核模块 (KMOD) 方式
  • 任意 C 代码，可调用任意内核函数
  • 无安全验证，bug 即内核 panic
  • 需要 root 权限加载
  • 需匹配内核版本（kABI 不稳定）
  • 适合：驱动、文件系统
eBPF 方式
  • 受限指令集（64 位寄存器 R0-R10）
  • 验证器保证无越界、无死循环
  • 可通过 capability 授权非特权用户
  • 跨内核版本稳定（CO-RE）
  • 适合：网络、跟踪、安全、观测

02 BPF 程序类型

每个 BPF 程序属于一个类型，类型决定了它的挂载点和可调用的辅助函数：

    BPF_PROG_TYPE_KPROBE,           // 内核函数探针
    BPF_PROG_TYPE_SCHED_CLS,        // TC 流量分类
    BPF_PROG_TYPE_SCHED_ACT,        // TC 流量动作
    BPF_PROG_TYPE_TRACEPOINT,       // 静态 tracepoint
    BPF_PROG_TYPE_XDP,              // 网卡驱动 RX 路径
    BPF_PROG_TYPE_PERF_EVENT,       // perf 事件
    BPF_PROG_TYPE_CGROUP_SKB,       // cgroup 网络包
    BPF_PROG_TYPE_CGROUP_SOCK,      // cgroup 套接字
    BPF_PROG_TYPE_LWT_IN/OUT/XMIT,  // 轻量级隧道
    BPF_PROG_TYPE_SOCK_OPS,         // 套接字操作
    BPF_PROG_TYPE_SK_SKB,           // sockmap 流重定向
    BPF_PROG_TYPE_CGROUP_DEVICE,    // cgroup 设备访问

03 BPF 系统调用

eBPF 所有操作（加载程序、创建映射、附加、查询）都通过 bpf() 系统调用：

// kernel/bpf/syscall.c
SYSCALL_DEFINE3(bpf, int, cmd, union bpf_attr __user *, uattr,
                unsigned int, size)
{
    return __sys_bpf(cmd, USER_BPFPTR(uattr), size);
}

static int __sys_bpf(int cmd, bpfptr_t uattr, unsigned int size)
{
    // 1. 检查 size 字段
    // 2. copy_from_user 拷贝 attr
    // 3. 根据 cmd 分发到具体函数
    switch (cmd) {
    case BPF_MAP_CREATE:           return map_create(&attr);
    case BPF_MAP_LOOKUP_ELEM:      return map_lookup_elem(&attr);
    case BPF_MAP_UPDATE_ELEM:      return map_update_elem(&attr);
    case BPF_MAP_DELETE_ELEM:      return map_delete_elem(&attr);
    case BPF_PROG_LOAD:            return bpf_prog_load(&attr, uattr);
    case BPF_OBJ_PIN:              return obj_pin(&attr, uattr);
    case BPF_OBJ_GET:              return obj_get(&attr);
    case BPF_PROG_ATTACH:          return bpf_prog_attach(&attr);
    case BPF_PROG_DETACH:          return bpf_prog_detach(&attr);
    case BPF_LINK_CREATE:          return link_create(&attr, uattr);
    case BPF_LINK_UPDATE:          return link_update(&attr);
    case BPF_RAW_TRACEPOINT_OPEN:  return bpf_raw_tracepoint_open(&attr, uattr);
    case BPF_BTF_LOAD:             return bpf_btf_load(&attr, uattr);
    case BPF_TASK_FD_QUERY:        return bpf_task_fd_query(&attr, uattr);
    case BPF_ITER_CREATE:          return bpf_iter_create(&attr, uattr);
    // … 40+ cmd …
    }
}

主要命令分类：

分类命令用途
映射操作BPF_MAP_CREATE创建映射
BPF_MAP_LOOKUP_ELEM查询键
BPF_MAP_UPDATE_ELEM更新/插入键
BPF_MAP_DELETE_ELEM删除键
程序BPF_PROG_LOAD加载并验证 BPF 程序
BPF_PROG_ATTACH附加到 cgroup/netns 等
BPF_LINK_CREATE创建 link（新接口）
BTFBPF_BTF_LOAD加载 BPF Type Format
BPF_MAP_FREEZE冻结只读映射

程序加载流程 bpf_prog_load()：

bpf_prog_load(&attr)
  ├─ bpf_prog_alloc()       // 分配 struct bpf_prog
  ├─ bpf_prog_load_fixup_attach_type()
  ├─ find_prog_type()       // 根据 type 找到 bpf_prog_ops
  ├─ bpf_check()            // 验证器（核心！）
  │   ├─ 检查指令合法性
  │   ├─ 控制流图分析
  │   ├─ 寄存器状态跟踪
  │   └─ 辅助函数权限检查
  ├─ resolve_pseudo_ldimm64()  // 解析 map 引用
  ├─ bpf_int_jit_compile()     // JIT 编译
  ├─ bpf_prog_alloc_id()       // 分配 ID
  └─ bpf_prog_new_fd()         // 创建 fd 返回

04 验证器

BPF 验证器 (verifier) 是 eBPF 安全性的核心。它在不运行程序的情况下，静态分析程序是否安全。

验证器位于 kernel/bpf/verifier.c，入口函数 bpf_check()。验证分两个阶段：

// kernel/bpf/verifier.c
int bpf_check(struct bpf_prog **prog,
              union bpf_attr *attr,
              bpfptr_t uattr)
{
    struct bpf_verifier_env *env;
    env = kzalloc(sizeof(*env), GFP_KERNEL);
    env->prog = *prog;
    env->ops = bpf_verifier_ops[prog->type];
    env->insn_processed = 0;

    // 第一阶段：CFG 与抽象解释
    err = do_check_main(env);   // 核心分析

    // 第二阶段：指令改写（fixup）
    err = fixup_bpf_calls(env); // 替换伪指令为真实调用

    // 状态可达性分析
    while (env->cfg.env_queue) {
        // 处理每个可达状态
    }
    return err;
}

第一阶段：控制流图（CFG）构建

// 1. 反向扫描指令，构建 CFG
// 2. 检测不可达代码、死代码
// 3. 检测有环图（BPF 禁止有环）

// 关键规则：
// - 必须有退出指令 (BPF_EXIT)
// - 最多 100 万条指令 (BPF_COMPLEXITY_LIMIT_INSNS)
// - 函数调用深度限制

第二阶段：抽象解释（Abstract Interpretation）

// 这是验证器最复杂的部分
// 模拟执行每条指令，跟踪寄存器的可能取值

struct bpf_reg_state {
    enum bpf_reg_type type;       // 寄存器类型
    union {
        u32              range;   // PTR_TO_BUFFER 的可访问范围
        struct {
            u64          imm;     // SCALAR_VALUE 的常量值
            u64          tnum;    // 范围（已知位/未知位）
        };
    };
    s32              off;         // 偏移
    struct btf_reg   btf;
    // ...
};

// 寄存器类型：
//  UNKNOWN_VALUE     - 未知值
//  SCALAR_VALUE      - 已知值或范围
//  PTR_TO_CTX        - 指向程序上下文
//  PTR_TO_MAP        - 指向映射
//  PTR_TO_MAP_KEY    - 指向映射键
//  PTR_TO_MAP_VALUE  - 指向映射值
//  PTR_TO_STACK      - 指向栈
//  PTR_TO_PACKET     - 指向数据包
//  PTR_TO_PACKET_END - 数据包末尾

// 验证关键：
// - 每次数组/指针访问，检查 range 不越界
// - 函数返回值类型传播
// - 分支处状态合并（state merge）
// - 路径压缩（pruning）：相同状态节点复用结果

第三阶段：fixup_bpf_calls()

// 重写伪 BPF 指令为实际操作
// - BPF_LD_IMM64 with src_reg=BPF_PSEUDO_MAP_FD → 替换为 map 指针
// - BPF_EMIT_CALL with imm=helper_id → 解析为 helper 函数指针
// - 移除死代码
// - 插入 BPF_ST/STX 指令保存调用约定
验证器的限制:
最大指令数：100 万（unprivileged：4096）
最大栈空间：512 字节
最大寄存器状态：4096
不允许：无界循环（5.3+ 支持有界循环）、指针算术越界
不允许：未初始化寄存器访问

05 BPF 映射

BPF 映射 (map) 是 eBPF 程序之间、用户态与内核态之间共享状态的核心机制。

    BPF_MAP_TYPE_PERCPU_ARRAY,     // 每 CPU 数组
    BPF_MAP_TYPE_STACK_TRACE,      // 栈回溯
    BPF_MAP_TYPE_CGROUP_ARRAY,
    BPF_MAP_TYPE_LRU_HASH,         // LRU 哈希
    BPF_MAP_TYPE_LRU_PERCPU_HASH,
    BPF_MAP_TYPE_LPM_TRIE,         // 最长前缀匹配
    BPF_MAP_TYPE_ARRAY_OF_MAPS,
    BPF_MAP_TYPE_HASH_OF_MAPS,
    BPF_MAP_TYPE_RINGBUF,          // 环形缓冲（新）
    BPF_MAP_TYPE_INODE_STORAGE,
    BPF_MAP_TYPE_TASK_STORAGE,
    BPF_MAP_TYPE_BLOOM_FILTER,     // 布隆过滤器

06 运行时与 JIT

BPF 程序在加载后会被 JIT 编译为本机指令，性能接近原生代码。

// 1. 解释器（兜底，所有架构支持）
//    kernel/bpf/core.c
//    ___bpf_prog_run() — 解释执行 BPF 指令
//
//    特点：
//    - 64 位虚拟寄存器 R0-R10 通过 base + offset 映射到栈
//    - 通过 computed goto 优化跳转
//    - 通常比 JIT 慢 2-3 倍

// 2. JIT 编译器（每架构独立）
//    arch/x86/net/bpf_jit_comp.c        (x86_64)
//    arch/arm64/net/bpf_jit_comp.c      (ARM64)
//    arch/riscv/net/bpf_jit_comp.c      (RISC-V)
//
//    入口：bpf_int_jit_compile()
//    流程：
//      a. 遍历 BPF 指令，生成本机指令
//      b. 优化：常数折叠、寄存器分配
//      c. 生成可执行内存（bpf_jit_binary_alloc）
//      d. 设置 prog->bpf_func 指向 JIT 代码

// x86_64 寄存器映射：
//  BPF R0-R10  →  x86 rax, rdx, rcx, rsi, rdi, r8, r9, rbx, r13, r14
//  BPF R10 (FP) →  x86 rbp（帧指针，强制）

// 3. BPF Dispatcher（性能优化）
//    kernel/bpf/dispatcher.c
//
//    优化 BPF_PROG_TEST_RUN 和间接调用：
//    通过静态调用减少间接调用开销

JIT 优化示例：

// BPF 指令
BPF_MOV64_IMM(BPF_REG_0, 0)
BPF_EXIT_INSN()

// 解释器路径：
// 1. dispatch 到 BPF_ALU opcode
// 2. dispatch 到 BPF_MOV sub-opcode
// 3. 设置寄存器 R0
// 4. dispatch 到 BPF_JMP opcode
// 5. dispatch 到 BPF_EXIT
// 6. 返回 R0

// JIT 后（x86）：
//   xor eax, eax
//   ret
// 只有 2 条指令

07 辅助函数

BPF 程序不能直接调用内核函数，必须通过受控的辅助函数 (helper) 接口。

// 每个辅助函数由 struct bpf_func_proto 描述
struct bpf_func_proto {
    u32                  func_id;       // BPF_FUNC_*
    enum bpf_return_type ret_type;      // 返回类型
    union { struct bpf_arg_reg arg1, arg2, arg3, arg4, arg5; };
    bool                 gpl_only;
    bool                 might_sleep;
    bool                 allowed_in_nmi;
    bool                 allow_user_ptr;
    bool                 ret_btf_id;
    int                (*btf_id);       // 返回类型的 BTF id
};

// 辅助函数示例
const struct bpf_func_proto bpf_map_lookup_elem_proto = {
    .func           = bpf_map_lookup_elem,
    .gpl_only       = false,
    .ret_type       = RET_PTR_TO_MAP_VALUE_OR_NULL,
    .arg1_type      = ARG_CONST_MAP_PTR,
    .arg2_type      = ARG_PTR_TO_MAP_KEY,
};

// 主要辅助函数分类（200+ 个）：
// - map_lookup/update/delete_elem  — 映射操作
// - probe_read_kernel/user/str     — 安全内存读
// - get_current_pid_tgid/uid_gid   — 进程信息
// - get_prandom_u32                — 随机数
// - ktime_get_ns/boottime_ns       — 时间
// - trace_printk                   — 调试输出
// - perf_event_output              — 事件输出
// - skb_load_bytes/store_bytes     — 包操作
// - redirect/clone_redirect        — 包重定向
// - csum_diff                      — 校验和
// - xdp_adjust_head/tail/met       — XDP 调整包
// - get_socket_cookie              — 套接字标识
// - tcp_check_syncookie            — TCP 防护
// - skb_change_head/tail/proto     — 包修改
// - get_stackid                    — 栈标识

辅助函数按程序类型限制：

// kernel/bpf/helpers.c
const struct bpf_func_proto *
bpf_base_func_proto(enum bpf_func_id func_id)
{
    // 通用辅助函数（所有程序可用）
    switch (func_id) {
    case BPF_FUNC_map_lookup_elem:
        return &bpf_map_lookup_elem_proto;
    case BPF_FUNC_get_current_pid_tgid:
        return &bpf_get_current_pid_tgid_proto;
    // ...
    }
}

// 每种程序类型可以再扩展：
static const struct bpf_func_proto *
xdp_func_proto(enum bpf_func_id func_id)
{
    switch (func_id) {
    case BPF_FUNC_xdp_adjust_head:
        return &bpf_xdp_adjust_head_proto;
    case BPF_FUNC_xdp_adjust_meta:
        return &bpf_xdp_adjust_meta_proto;
    default:
        return bpf_sk_base_func_proto(func_id);
    }
}

08 Trampoline 与 fentry/fexit

BPF Trampoline 允许在不修改目标函数代码的情况下挂载 BPF 程序。它通过 ftrace 的动态 patch 机制实现。

// kernel/bpf/trampoline.c
struct bpf_trampoline {
    struct hlist_node       hlist;
    struct mod_node         node;
    u64                     key;            // 标识目标函数
    struct bpf_tramp_image *image;          // 生成的 trampoline 代码
    struct bpf_func_model   func_model;
    struct fentry_ops {
        struct bpf_prog   *prog;
        // ...
    } *fentries;                            // fentry 程序链
    int                     fentry_cnt;
    int                     fexit_cnt;
    struct bpf_tramp_links  extension_procs;
    void                   *hostname;
};

// Trampoline 生成的代码骨架：
// 1. 保存寄存器（按调用约定）
// 2. 调用所有 fentry BPF 程序（with args copy）
// 3. 调用原始函数
// 4. 调用所有 fexit BPF 程序（with ret value）
// 5. 恢复寄存器
// 6. 返回

// fentry：在函数入口处执行
// fexit：在函数返回处执行（包括异常返回）
// modify_return：可修改返回值

// 优势 vs kprobe：
// - 无 int3 异常开销
// - 直接调用，零开销
// - 性能比 kprobe 高 5-10 倍

Trampoline 还用于结构体 ops 修改（如 sched_ext）：内核通过 trampoline 替换某个函数指针，BPF 程序就接管了对应的 ops。

09 cgroup BPF

BPF 程序可以附加到 cgroup 上，对该 cgroup 内进程的网络行为统一治理：

// 主要 cgroup BPF 程序类型
BPF_PROG_TYPE_CGROUP_SKB        // 进/出包过滤
BPF_PROG_TYPE_CGROUP_SOCK       // 套接字创建/释放
BPF_PROG_TYPE_CGROUP_DEVICE     // 设备访问控制
BPF_PROG_TYPE_CGROUP_SOCK_ADDR  // connect/bind 等
BPF_PROG_TYPE_CGROUP_SOCKOPT    // getsockopt/setsockopt
BPF_PROG_TYPE_CGROUP_SYSCTL     /proc/sys 读写

// 附加方式（通过 bpf() 系统调用）：
// BPF_PROG_ATTACH with:
//   target_fd = cgroup_fd
//   attach_type = BPF_CGROUP_INET_INGRESS 等

// 关键 attach_type：
BPF_CGROUP_INET_INGRESS          // 入站包
BPF_CGROUP_INET_EGRESS           // 出站包
BPF_CGROUP_INET_SOCK_CREATE      // 套接字创建
BPF_CGROUP_INET_SOCK_RELEASE
BPF_CGROUP_INET4_POST_REDIRECT
BPF_CGROUP_INET4_BIND
BPF_CGROUP_INET4_CONNECT
BPF_CGROUP_INET6_GETPEERNAME
BPF_CGROUP_UDP4_RECVMSG
BPF_CGROUP_UDP6_SENDMSG
BPF_CGROUP_GETSOCKOPT
BPF_CGROUP_SETSOCKOPT
// ...

// 实现：kernel/bpf/cgroup.c
// 当进程触发网络操作时，遍历所属 cgroup 链
// 对每个附加的 BPF 程序调用，程序返回：
//   0     - 允许
//   -EPERM - 拒绝
//   -ENOMEM - 杀死套接字
