01 概述

03 Runtime PM

Runtime PM 让设备在空闲时自动进入低功耗状态，活动时再唤醒。核心是引用计数状态机：

// drivers/base/power/runtime.c

// 设备 PM 状态：
enum rpm_status {
    RPM_ACTIVE = 0,       // 完全活跃
    RPM_RESUMING,         // 正在唤醒
    RPM_SUSPENDED,        // 已挂起
    RPM_SUSPENDING,       // 正在挂起
};

// 每设备维护：
//   power.usage_count       - 主引用计数
//   power.child_count       - 子设备活跃数
//   power.disable_depth     - PM 禁用深度
//   power.runtime_status    - 当前状态
//   power.request_pending   - 等待处理
//   power.timer_expires     - autosuspend 超时

// 主要 API：
// include/linux/pm_runtime.h

int pm_runtime_get_sync(struct device *dev);
//   usage_count++
//   如果当前 SUSPENDED → runtime_resume()

int pm_runtime_put_sync(struct device *dev);
//   usage_count--
//   如果降到 0 → idle 回调 → 可能 suspend

int pm_runtime_put_sync_autosuspend(struct device *dev);
//   usage_count--
//   标记等待 autosuspend_delay 后 suspend

int pm_runtime_idle(struct device *dev);
//   调用 idle 回调
//   idle 回调决定是否触发 suspend

// 自动挂起延迟：
//   autosuspend_delay = 50ms (默认)
//   设备 last_busy 之后该时间内不活动才挂起
pm_runtime_mark_last_busy(dev);
pm_runtime_use_autosuspend(dev);
pm_runtime_set_autosuspend_delay(dev, 100);

状态转移图：

            get_sync
   ┌──────────────────────────┐
   │                          ▼
┌──┴───────┐               ┌─────────┐
│SUSPENDED │ ←─ suspend ── │ ACTIVE  │
└──┬───────┘               └────┬────┘
   │                          │
   │       resume             │
   │   ┌──────────────────────┘
   │   ▼
   └─►ACTIVE (从 SUSPENDED 经 RESUMING)
   
   ACTIVE → SUSPENDING → SUSPENDED
   SUSPENDED → RESUMING → ACTIVE

// 关键约定：
// 1. 驱动在每次访问设备前调用 pm_runtime_get_sync()
// 2. 访问结束后调用 pm_runtime_put_*()
// 3. 在 suspend/resume 回调中调用硬件寄存器配置
// 4. runtime_resume 必须能从 SUSPENDED 恢复到可用状态

04 系统休眠与唤醒

// kernel/power/suspend.c

// 主要入口：state_store() / sysfs /sys/power/state
// 用户态：echo mem > /sys/power/state

// 三种 sleep 状态：
const char *const pm_states[PM_SUSPEND_MAX] = {
    [PM_SUSPEND_TO_IDLE] = "freeze",
    [PM_SUSPEND_STANDBY] = "standby",
    [PM_SUSPEND_MEM]     = "mem",
};

// Suspend 流程：
int pm_suspend(suspend_state_t state)
{
    int error = enter_state(state);
    return error;
}

static int enter_state(suspend_state_t state)
{
    // 1. 同步文件系统
    pm_prepare_console();
    sys_sync();                  // 强制 sync
    
    // 2. 通知用户态（PM_SUSPEND_PREPARE）
    pm_notifier_call_chain(PM_SUSPEND_PREPARE);
    
    // 3. 冻结用户进程（停止 fork/exit）
    freeze_processes();
    
    // 4. 冻结内核线程
    freeze_kernel_threads();
    
    // 5. 关中断，进入 suspend
    error = suspend_devices_and_enter(state);
    
    // 6. 解冻
    thaw_processes();
    
    return error;
}

int suspend_devices_and_enter(suspend_state_t state)
{
    // 1. 调用每个设备的 dev_pm_ops.prepare
    dpm_prepare(NULL);
    
    // 2. 关闭外设 IRQ
    suspend_device_irqs();
    
    // 3. 调用 platform_suspend_ops->prepare
    //    例如 ACPI 设置 _PTS
    error = platform_suspend_begin(state);
    
    // 4. 调用所有设备的 dev_pm_ops.suspend(_late/_noirq)
    dpm_suspend_start(PMSG_SUSPEND);
    dpm_suspend_noirq(PMSG_SUSPEND);
    
    // 5. 调用 syscore_ops->suspend
    //    保存关键状态：clockevents, tick, hrtimers
    error = syscore_suspend();
    
    // 6. 调用 platform_suspend_ops->enter
    //    这一步 CPU 实际进入睡眠 (WFI/WFE)
    //    直到 wake source 触发
    error = suspend_ops->enter(state);
    
    // 7. 唤醒路径：反向操作
    //    syscore_resume → dpm_resume → platform_resume_done
    // ...
    
    return 0;
}

05 CPU 频率管理 (cpufreq)

cpufreq 框架管理 CPU 核心电压和频率的动态调整（DVFS）。

// drivers/cpufreq/cpufreq.c

// 驱动接口：
struct cpufreq_driver {
    char                   name[CPUFREQ_NAME_LEN];
    u16                    flags;
    void                  *driver_data;
    
    /* 必需回调 */
    int                  (*init)(struct cpufreq_policy *policy);
    int                  (*verify)(struct cpufreq_policy_data *policy);
    int                  (*setpolicy)(struct cpufreq_policy *policy);
    int                  (*target)(struct cpufreq_policy *policy,
                                    unsigned int target_freq,
                                    unsigned int relation);
    int                  (*target_index)(struct cpufreq_policy *policy,
                                          unsigned int index);
    unsigned int         (*get)(unsigned int cpu);
    int                  (*bios_limit)(int cpu, unsigned int *limit);
    int                  (*online)(struct cpufreq_policy *policy);
    int                  (*offline)(struct cpufreq_policy *policy);
    int                  (*exit)(struct cpufreq_policy *policy);
    void                 (*stop_cpu)(struct cpufreq_policy *policy);
    int                  (*suspend)(struct cpufreq_policy *policy);
    int                  (*resume)(struct cpufreq_policy *policy);
    struct freq_attr    **attr;
    
    /* fast switch (无锁，用于 schedutil) */

    struct freq_constraints   constraints;
    struct freq_qos          *qos;
    struct cpufreq_frequency_table *cached_freq_table;
    unsigned int              min;
    unsigned int              max;
    unsigned int              cur;
    unsigned int              restore_freq;
    unsigned int              suspend_freq;
    unsigned int              policy_cpu;
    unsigned int              last_cpu;
    unsigned int              cpu;
    unsigned int              last_freq_req;
    cpumask_var_t             related_cpus;
    cpumask_var_t             real_cpus;
    cpumask_var_t             cpus;
    // ... fast_switch_possible, shared_type, etc.
};

// 流程：
//   1. governor 决定目标频率（根据负载/利用率）
//   2. driver->target_index(policy, idx) 设置频率
//   3. driver 内部调用 SMC/MCR 调整硬件

主要 governors：

Governor策略用途
performance始终最高频率延迟敏感
powersave始终最低频率省电优先
userspace用户态指定细粒度控制
ondemand动态调整（按 idle/利用率）通用默认（旧）
conservative类似 ondemand，但更慢稳定性高
schedutil调度器集成（CFS/EEVDF）现代默认
// schedutil（最现代）：
// drivers/cpufreq/cpufreq_schedutil.c
// 利用调度器的 util_est 预测
// 在调度 tick 中通过 fast_switch 更新频率
// 无需 context switch，响应极快

// 流程：
//   sched_tick → cpu_util() → sugov_update_* → fast_switch
//   调度器决定 PELT 利用率，governor 直接应用

// intel_pstate：
//   Intel CPU 特定 driver
//   直接用 HWP（Hardware P-states）
//   遗留 mode（active / passive）影响 governor

06 CPU 空闲管理 (cpuidle)

cpuidle 管理 CPU 在 idle 时进入的不同深度睡眠状态（C-states）。

// drivers/cpuidle/cpuidle.c

struct cpuidle_driver {
    const char              *name;
    struct module           *owner;
    unsigned int            bctes;
    int                     rating;
    struct cpuidle_state     states[CPUIDLE_STATE_MAX];
    atomic_t                refcnt;
    struct cpumask          *cpumask;
    struct cpuidle_state0   *safe_state;
    struct cpuidle_state0   *stop_state;
    int                     state_count;
    int                     disabled;
    // ...
};

struct cpuidle_state {
    char                    name[CPUIDLE_NAME_LEN];
    char                    desc[CPUIDLE_DESC_LEN];
    u64                     flags;
    u64                     exit_latency;     // 退出延迟（ns）
    u64                     target_residency; // 推荐停留时间（ns）
    int                    (*enter)(struct cpuidle_device *dev,
                                     struct cpuidle_driver *drv,
                                     int index);
    int                    (*enter_dead)(struct cpuidle_device *dev, int index);
    int                     exit_latency_ns;
    // ... 计数、统计 ...
};

// CPU 进入 idle：
// kernel/sched/idle.c
void cpu_startup_entry(enum cpuhp_state state)
{
    cpuidle_loop();          // 进入 idle 循环
}

// 流程：
//   while (need_resched() == false):
//     1. 调用 cpuidle_select() 选 C-state
//        governor 根据历史 idle 时间预测
//     2. cpuidle_enter(dev, drv, idx)
//        调用 states[idx].enter(dev, drv, idx)
//     3. CPU 真正执行 mwait/halt 指令
//     4. 中断唤醒，从 enter 返回

// 关键决策：governor
// menu (默认): 基于 idle 历史预测是否值得进深 C-state
// ladder: 步进式，每次只下一级
// teo: Timer Events Oriented，基于定时器预测
// haltpoll: 简单 polling fallback

// 退出延迟 (exit_latency) vs 停留时间 (target_residency)
//   深度状态退出慢，但更省电
//   如果 idle 时间 < target_residency，进入反而亏
//   governor 的工作就是预测下一个中断何时来

典型 C-states：

状态退出延迟省电说明
C000%运行
C1 (Halt)~1μs少量停核心时钟
C1E~10μs更多降频 + 停时钟
C3 (Sleep)~60μs显著清空 L1/L2
C6 (Deep Sleep)~100μs更多保存状态，关电压
C7/C8/C9/C10~ms极多更深的省电

07 PM 域与 QoS

PM QoS (Quality of Service) 让需求方表达对延迟的约束，PM 框架据此决策。

// include/linux/pm_qos.h

enum pm_qos_flags_status {
    PM_QOS_FLAGS_UNDEFINED = -1,
    PM_QOS_FLAGS_NONE,
    PM_QOS_FLAGS_SOME,
    PM_QOS_FLAGS_ALL,
};

enum pm_qos_type {
    PM_QOS_UNITIALIZED,
    PM_QOS_MAX,            // 约束为"最多"
    PM_QOS_MIN,            // 约束为"至少"
    PM_QOS_SUM,            // 总和
};

enum pm_qos_req_action {
    PM_QOS_ADD_REQ,
    PM_QOS_UPDATE_REQ,
    PM_QOS_REMOVE_REQ,
};

// 约束分类：
#define PM_QOS_RESUME_LATENCY     1   // 唤醒延迟（us）
#define PM_QOS_LATENCY_TOLERANCE  2   // 容许延迟
#define PM_QOS_LATENCY_TOLERANCE_NO_C0  3
#define PM_QOS_MIN_FREQUENCY      4
#define PM_QOS_MAX_FREQUENCY      5
#define PM_QOS_FLAGS              6
#define PM_QOS_MAX_DEFAULT_VALUE  INT_MAX
#define PM_QOS_MIN_DEFAULT_VALUE  0

// API：

    // 嵌入到 per-device 约束链表
