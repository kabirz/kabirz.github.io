01 概览

READ_ONCE() 和 WRITE_ONCE() 是内核中最重要的并发原语之一，用于防止编译器对内存访问进行优化，确保每次读写都是真实的内存操作。

核心作用

宏作用
READ_ONCE(x)强制从内存读取一次，禁止编译器缓存/重排
WRITE_ONCE(x, val)强制写入内存一次，禁止编译器合并/重排

核心价值

两个主要用途：
1. **线程间通信** — 进程代码与 IRQ/NMI 处理程序在同一 CPU 上的通信
2. **防止编译器优化** — 防止编译器折叠、重排或以其他方式修改不依赖顺序或与显式内存屏障交互的访问

02 宏定义

定义在 include/asm-generic/rwonce.h 中。

READ_ONCE

// include/asm-generic/rwonce.h 第 47-51 行
#define READ_ONCE(x)                            \
({                                              \
    compiletime_assert_rwonce_type(x);          \
    __READ_ONCE(x);                             \
})

// 底层实现（第 44 行）
#define __READ_ONCE(x)  (*(const volatile __unqual_scalar_typeof(x) *)&(x))

WRITE_ONCE

// include/asm-generic/rwonce.h 第 58-62 行
#define WRITE_ONCE(x, val)                      \
do {                                            \
    compiletime_assert_rwonce_type(x);          \
    __WRITE_ONCE(x, val);                       \
} while (0)

// 底层实现（第 53-56 行）
#define __WRITE_ONCE(x, val)                    \
do {                                            \
    *(volatile typeof(x) *)&(x) = (val);        \
} while (0)

类型检查

// 编译时类型检查（第 35-37 行）
// 只允许 native word 或 long long 大小的类型
#define compiletime_assert_rwonce_type(t)                    \
    compiletime_assert(__native_word(t) || sizeof(t) == sizeof(long long), \
        "Unsupported access size for {READ,WRITE}_ONCE().")

03 为什么需要

不使用 WRITE_ONCE 的问题

// 错误示例：编译器可能优化掉写入
int global_flag = 0;

void producer(void) {
    global_flag = 1;  // 编译器可能优化为寄存器操作
}

void consumer(void) {
    while (global_flag == 0) {  // 编译器可能只读一次
        // 永远看不到更新
    }
}

使用 WRITE_ONCE 后

// 正确示例：强制内存访问
int global_flag = 0;

void producer(void) {
    WRITE_ONCE(global_flag, 1);  // 强制写入内存
}

void consumer(void) {
    while (READ_ONCE(global_flag) == 0) {  // 每次都从内存读取
        // 看到更新
    }
}

04 编译器行为

graph TD A["普通变量访问"] --> B["编译器优化"] B --> C["缓存到寄存器"] B --> D["重排序"] B --> E["合并读写"] F["READ_ONCE/WRITE_ONCE"] --> G["volatile 语义"] G --> H["每次访问真实内存"] G --> I["禁止重排"] G --> J["禁止合并"]

编译器优化示例

// 编译器可能做的优化：

// 1. 缓存到寄存器
// while (flag == 0);
// 编译器可能只读一次 flag 到寄存器，后续循环不读内存

// 2. 重排序
// a = 1;
// b = 2;
// 编译器可能先写 b 再写 a

// 3. 合并读写
// x = x + 1;
// 编译器可能：reg = x; reg++; x = reg;
// 而不是：reg = x; reg = reg + 1; x = reg;

WRITE_ONCE/READ_ONCE 如何解决

// volatile 语义确保：
// 1. 每次访问都从内存读取/写入
// 2. 禁止跨 READ_ONCE/WRITE_ONCE 重排
// 3. 不会合并多次读写

// 但注意：不保证原子性！
// 64 位平台上的 32 位访问通常是原子的
// 但不是所有平台都保证

05 典型用例

1. 跨 CPU 标志位通信

// 一个 CPU 设置标志，另一个 CPU 读取
static int data_ready = 0;

// CPU 0: 设置标志
void producer(void) {
    // 准备数据
    prepare_data();
    // 通知消费者
    WRITE_ONCE(data_ready, 1);
}

// CPU 1: 检查标志
void consumer(void) {
    while (READ_ONCE(data_ready) == 0) {
        cpu_relax();
    }
    // 处理数据
    process_data();
}

2. 任务状态更新

// 内核中常见模式
// task->state = TASK_RUNNING;
// 应该写为：
WRITE_ONCE(current->__state, TASK_RUNNING);

// 读取任务状态
int state = READ_ONCE(current->__state);

3. 防止编译器优化循环

// 错误：编译器可能优化掉循环
while (1) {
    if (some_flag) {
        break;
    }
}

// 正确：使用 READ_ONCE
while (1) {
    if (READ_ONCE(some_flag)) {
        break;
    }
}

4. 数据结构中的并发字段

// 结构体中的并发访问字段
struct my_struct {
    int data;
    volatile int refcount;  // 或使用 atomic_t
};

// 正确做法
struct my_struct {
    int data;
    int refcount;  // 不需要 volatile
};

// 读取时使用 READ_ONCE
int ref = READ_ONCE(obj->refcount);

// 写入时使用 WRITE_ONCE
WRITE_ONCE(obj->refcount, new_val);

06 vs 原子操作

特性READ_ONCE/WRITE_ONCEatomic_t
原子性不保证（但实际中通常是）保证
内存屏障不提供可选
类型任意标量类型仅 atomic_t
性能低开销略高
适用场景标志位、简单状态计数器、引用计数
// READ_ONCE/WRITE_ONCE：用于简单标志位
static int flag = 0;
WRITE_ONCE(flag, 1);
int val = READ_ONCE(flag);

// atomic_t：用于需要原子性的操作
atomic_t counter = ATOMIC_INIT(0);
atomic_inc(&counter);
int val = atomic_read(&counter);

// 选择原则：
// - 只是标志通知 → READ_ONCE/WRITE_ONCE
// - 需要原子递增/递减 → atomic_t
// - 需要内存屏障 → atomic_t + smp_mb()

07 vs volatile

特性READ_ONCE/WRITE_ONCEvolatile
类型限制标量类型任意类型
编译器屏障有部分有
KASAN/KCSAN 检查有无
内核推荐✅ 推荐❌ 不推荐
// 内核中禁止直接使用 volatile
// 应该使用 READ_ONCE/WRITE_ONCE

// ❌ 错误
volatile int *p = &some_var;
*p = 1;
int val = *p;

// ✅ 正确
WRITE_ONCE(some_var, 1);
int val = READ_ONCE(some_var);

08 常见陷阱

// 陷阱 1：READ_ONCE 不能保证原子性
// 在 32 位平台上读取 64 位值可能不是原子的
uint64_t val = READ_ONCE(my_64bit_var);  // 可能撕裂
// 如果需要原子性，使用 atomic64_t

// 陷阱 2：READ_ONCE 不提供内存屏障
// 如果需要顺序保证，需要额外的 smp_rmb()/smp_wmb()
READ_ONCE(flag);
smp_rmb();  // 确保 flag 读取在后续读取之前
int val = READ_ONCE(data);

// 陷阱 3：结构体不能整体 READ_ONCE/WRITE_ONCE
// ❌ struct foo f = READ_ONCE(some_struct);
// ✅ f.field1 = READ_ONCE(some_struct.field1);

// 陷阱 4：不要在 READ_ONCE 中使用指针
// ❌ p = READ_ONCE(ptr);
// ✅ p = READ_ONCE(*ptr);  或  p = READ_ONCE(ptr);
