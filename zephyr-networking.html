01 网络栈概览

Zephyr 内置完整的 TCP/IP 网络栈，支持 BSD Socket API、多种 L2 协议和高层应用协议。

03 网络接口 (net_if)

net_if 是 Zephyr 网络栈的核心抽象，代表一个网络接口。

// 网络接口结构体（简化）
struct net_if {
    const struct net_if_api *if_dev;   // 接口 API
    struct net_if_config *config;      // 接口配置
    struct net_l2 *l2;                 // L2 层
    struct net_if_addr *ipv4;          // IPv4 地址
    struct net_if_addr *ipv6;          // IPv6 地址
    // ...
};

// 获取网络接口
struct net_if *net_if_get_default(void);
struct net_if *net_if_get_by_index(int index);

// 接口管理
int net_if_up(struct net_if *iface);
int net_if_down(struct net_if *iface);

04 网络包 (net_pkt)

net_pkt 是网络栈中数据包的表示，管理多层协议头和数据。

// 网络包结构
struct net_pkt {
    struct net_buf *buffer;        // 数据缓冲区
    struct net_if *iface;          // 关联的网络接口
    uint16_t len;                  // 总长度
    // 各层偏移量（用于零拷贝协议头处理）
    uint16_t l2_offset;
    uint16_t l3_offset;
    uint16_t l4_offset;
    // ...
};

// 分配和释放
struct net_pkt *net_pkt_alloc(struct net_if *iface);
void net_pkt_unref(struct net_pkt *pkt);

// 数据访问
uint8_t *net_pkt_data(struct net_pkt *pkt);
int net_pkt_get_len(struct net_pkt *pkt);

05 BSD Socket API

Zephyr 提供兼容 POSIX 的 BSD Socket API，简化网络应用开发。

// 标准 BSD Socket API
#include <zephyr/net/socket.h>

// TCP 客户端示例
int sock = socket(AF_INET, SOCK_STREAM, IPPROTO_TCP);

struct sockaddr_in addr;
addr.sin_family = AF_INET;
addr.sin_port = htons(8080);
inet_pton(AF_INET, "192.168.1.100", &addr.sin_addr);

connect(sock, (struct sockaddr *)&addr, sizeof(addr));

char request[] = "GET / HTTP/1.1\r\nHost: example.com\r\n\r\n";
send(sock, request, sizeof(request), 0);

char response[512];
recv(sock, response, sizeof(response), 0);

close(sock);

// UDP 示例
int sock = socket(AF_INET, SOCK_DGRAM, IPPROTO_UDP);
bind(sock, (struct sockaddr *)&addr, sizeof(addr));
sendto(sock, data, len, 0, &dest_addr, sizeof(dest_addr));
recvfrom(sock, buffer, sizeof(buffer), 0, &src_addr, &addr_len);

Socket 选项

// 常用 Socket 选项
int opt = 1;
setsockopt(sock, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));

// Zephyr 特有选项
// SO_RCVTIMEO - 接收超时
// SO_SNDTIMEO - 发送超时
// SO_BINDTODEVICE - 绑定到特定接口

// MQTT 客户端示例
#include <zephyr/net/mqtt.h>

struct mqtt_client client;
mqtt_client_init(&client);

client.broker.type = MQTT_SOCKET_TCP;
client.broker.family = AF_INET;
inet_pton(AF_INET, "192.168.1.100", &client.broker.addr.ipv4);
client.broker.port = 1883;

client.rx_buf = rx_buffer;
client.rx_buf_size = sizeof(rx_buffer);
client.tx_buf = tx_buffer;
client.tx_buf_size = sizeof(tx_buffer);

mqtt_connect(&client);

// 订阅主题
struct mqtt_topic topics[] = {
    { .topic = { .topic = "sensors/temp", .len = 11 }, .qos = MQTT_QOS_0_AT_MOST_ONCE }
};
mqtt_subscribe(&client, topics, 1);
