01 概述

runc 是 OCI Runtime Spec 的参考实现。它读取 OCI config.json，把容器进程跑起来。是 Docker / containerd / Kubernetes 等所有上层系统的最底层依赖。

历史：

  • 2013 — Docker 内部的 libcontainer
  • 2015 — 捐给 OCI，命名 runc
  • 2016 — Runtime Spec v1.0 兼容
  • 现在 — 业界事实标准，月下载超 100 亿次

runc 是命令行工具，不是 daemon：

# 在 bundle 目录（包含 config.json + rootfs/）
$ runc create my-container
$ runc start my-container
$ runc list
ID               PID         STATUS      BUNDLE                         CREATED                      OWNER
my-container     12345       created     /path/to/bundle                2024-...                     root

$ runc kill my-container KILL
$ runc delete my-container

设计原则：

  • 一次性命令：每次 runc 调用完成后立即退出，不常驻
  • 最小依赖：仅依赖 Linux 内核特性，无第三方库
  • OCI 标准：严格遵循 spec，不增加额外功能
  • 无状态：状态持久化在文件系统，runc 重新读取

runc 的进程模型（关键的"两阶段"）：

// 启动容器涉及三个进程：

// 1. runc 主进程（用户调用）
$ runc create my-container
//    - fork 子进程进入新 namespace
//    - 子进程做 rootfs 准备
//    - 完成后立即退出

// 2. runc init 进程（在容器 namespace 内）
//    - /proc/self/exe with "init" 参数
//    - 完成 namespace 设置
//    - pivot_root
//    - 等待 runc 主进程的指令
//    - exec ENTRYPOINT

// 3. 容器进程（用户期望的）
//    - 实际运行的进程（如 nginx）

// 关键：runc init 通过 re-exec 自己（/proc/self/exe）
// 实现"在新 namespace 中执行"的效果

// 3. 创建 container 对象
// libcontainer/container_linux.go
func (c *linuxContainer) Start(process *Process) error {
    return c.start(process, ...)
}

func (c *linuxContainer) start(process *Process, isInit bool) error {
    // 关键：调用 runc init（在新 namespace 中）
    parent, err := c.newParentProcess(process, isInit)
    // parent 是个 exec.Cmd，目标是 /proc/self/exe
    
    if err := parent.start(); err != nil {
        return err
    }
    
    // 等待 init 完成
    c.created = time.Now()
    c.state = &createdState{c: c}
    return nil
}

// 4. newParentProcess
// libcontainer/process_linux.go
func (c *linuxContainer) newParentProcess(p *Process, isInit bool) (parent, error) {
    // 1. 准备 pipe 通信
    parentPipe, childPipe, _ := os.Pipe()
    
    // 2. 构造 cmd
    cmd := exec.Command("/proc/self/exe", "init")
    cmd.Args[0] = "runc"
    cmd.ExtraFiles = []*os.File{childPipe}
    cmd.Env = append(cmd.Env, "_LIBCONTAINER_INITPIPE=3")
    
    // 3. 设置 clone flags（新 namespace）
    cmd.SysProcAttr = &syscall.SysProcAttr{
        Cloneflags: getCloneFlags(c.config.Namespaces),
        Setpgid:    true,
    }
    
    // 4. 配置 IO
    cmd.Stdin = p.Stdin
    cmd.Stdout = p.Stdout
    cmd.Stderr = p.Stderr
    
    return p, nil
}

// 5. clone flags 转换
func getCloneFlags(namespaces []configs.Namespace) uintptr {
    var flag uintptr
    for _, ns := range namespaces {
        switch ns.Type {
        case configs.PID:
            flag |= syscall.CLONE_NEWPID
        case configs.NETWORK:
            flag |= syscall.CLONE_NEWNET
        case configs.MOUNT:
            flag |= syscall.CLONE_NEWNS
        case configs.NEWUSER:
            flag |= syscall.CLONE_NEWUSER
        case configs.IPC:
            flag |= syscall.CLONE_NEWIPC
        case configs.UTS:
            flag |= syscall.CLONE_NEWUTS
        case configs.CGROUP:
            flag |= syscall.CLONE_NEWCGROUP
        case configs.TIME:
            flag |= syscall.CLONE_NEWTIME
        }
    }
    return flag
}

// runc init 进程的代码：
// libcontainer/init_linux.go
func init() {
    if len(os.Args) > 1 && os.Args[1] == "init" {
        r := reexec.Init()
        os.Exit(r)
    }
}

// 详细的 init 函数：
// libcontainer/standard_init_linux.go
func (l *linuxStandardInit) Init() error {
    // 1. 设置 cgroup
    if err := setupCgroup(l.config); err != nil {
        return err
    }
    
    // 2. pivot_root（切换根目录）
    if err := prepareRootfs(l.pipe, l.config); err != nil {
        return err
    }
    
    // 3. setup 各 namespace
    if err := l.setupNetwork(); err != nil {
        return err
    }
    if err := l.setupRoute(); err != nil {
        return err
    }
    
    // 4. 设置 hostname
    if err := setHostname(l.config.Hostname); err != nil {
        return err
    }
    
    // 5. 应用 seccomp 规则
    if l.config.Seccomp != nil {
        if err := seccomp.InitSeccomp(l.config.Seccomp); err != nil {
            return err
        }
    }
    
    // 6. 应用 AppArmor/SELinux
    if err := apparmor.ApplyProfile(l.config.AppArmorProfile); err != nil {
        return err
    }
    
    // 7. 设置 capabilities
    if err := capabilities.Set(nil, l.config.Capabilities); err != nil {
        return err
    }
    
    // 8. 设置 rlimits、ulimit
    if err := rlimit.SetRlimit(l.config.Rlimits); err != nil {
        return err
    }
    
    // 9. 通知 runc 主进程：设置完成
    // 等待 start 信号
    
    // 10. execve 用户进程
    if err := syscall.Exec(l.config.Args[0], l.config.Args, os.Environ()); err != nil {
        return err
    }
    
    return nil
}

05 init 进程

runc init 是关键的过渡进程：

// runc init 必须做的事：
//   1. 在新 namespace 中
//   2. 设置 cgroup
//   3. 准备 rootfs（mount）
//   4. pivot_root 切换 rootfs
//   5. 配置网络
//   6. 应用安全策略
//   7. execve 用户进程

// 为什么用 /proc/self/exe re-exec？
//   - 直接 execve("/usr/bin/runc", "init") 也可以
//   - 但 runc binary 路径可能在 rootfs 之外
//   - 容器内可能没有 runc 二进制
//   - /proc/self/exe 是内核提供的引用，总是有效

// 不同 init 类型：
// libcontainer/factory_linux.go
type InitType int
const (
    InitStandard InitType = iota  // 标准容器
    InitSetns                      // 加入已存在 ns (exec)
)

// standard_init_linux.go - 新容器
// setns_init_linux.go - 已有容器中 exec

// Init 协议：
//   runc 主进程 <--> runc init 通过 socketpair 通信
//   
//   1. runc 启动 init (clone with CLONE_NEW*)
//   2. init 在新 ns 中执行 setupCgroup、mount、pivot_root
//   3. init 通过 socket 发送 "ready" 到 runc 主进程
//   4. runc 主进程通知 "start"
//   5. init 调用 execve(ENTRYPOINT)
//   6. ENTRYPOINT 取代 init，进程 ID 不变

pivot_root 关键操作：

// libcontainer/rootfs_linux.go
func pivotRoot(rootfs string, pivotBaseDir string) error {
    // pivot_root 系统调用：
    // 把当前 root 移到 put_old
    // 把 new_root 设为新的 /
    
    // 步骤：
    // 1. mount("", "/", "", MS_SLAVE|MS_REC, "")
    //    让 / 不再传播 mount 事件
    // 2. mount(rootfs, rootfs, "bind", MS_BIND|MS_REC, "")
    //    bind mount rootfs 到自己
    // 3. mkdir(rootfs + pivotBaseDir)
    // 4. syscall.PivotRoot(rootfs, rootfs + pivotBaseDir)
    //    - new_root = rootfs
    //    - put_old  = rootfs/oldroot
    // 5. chdir("/")
    // 6. umount2("/oldroot", MNT_DETACH)
    // 7. rmdir("/oldroot")
    
    return nil
}

// pivot_root vs chroot：
//   pivot_root：
//     - 整个 root 切换
//     - mount ns 也跟着变
//     - 旧 root 可被 umount
//     - 难以逃逸（更安全）
//   
//   chroot：
//     - 只改变路径解析起点
//     - 旧 root 仍可见（通过 fd）
//     - 容易被逃逸
//     - 不适合容器

06 cgroup 配置

// libcontainer/cgroups/fs2/fs2.go (cgroup v2)
// libcontainer/cgroups/fs/...   (cgroup v1)

// cgroup 路径：
//   spec.Linux.CgroupsPath = "myapp/container-1"
//   v2: /sys/fs/cgroup/myapp/container-1
//   v1: 每个子系统独立

// runc 创建 cgroup：
func (m *Manager) Apply(pid int) error {
    // 1. 创建 cgroup 目录
    if err := os.MkdirAll(m.Path, 0755); err != nil {
        return err
    }
    
    // 2. 写入 cgroup.procs
    if err := os.WriteFile(
        filepath.Join(m.Path, "cgroup.procs"),
        []byte(strconv.Itoa(pid)), 0644); err != nil {
        return err
    }
    
    // 3. 应用各 controller 配置
    if err := m.setResources(); err != nil {
        return err
    }
    
    return nil
}

// v2 资源设置（部分）：
func (m *Manager) setResources() error {
    // memory
    if r := m.Resources.Memory; r != nil {
        writeFile(path, "memory.max", strconv.FormatInt(r.Limit, 10))
        writeFile(path, "memory.swap.max", strconv.FormatInt(r.Swap, 10))
        // ...
    }
    
    // cpu
    if r := m.Resources.Cpu; r != nil {
        writeFile(path, "cpu.max", strconv.FormatInt(r.Quota, 10) +
                              " " + strconv.FormatInt(r.Period, 10))
        writeFile(path, "cpu.weight", strconv.FormatUint(uint64(r.Shares), 10))
        writeFile(path, "cpuset.cpus", r.Cpus)
    }
    
    // pids
    if r := m.Resources.Pids; r != nil {
        writeFile(path, "pids.max", strconv.FormatInt(r.Limit, 10))
    }
    
    // io
    if r := m.Resources.BlockIO; r != nil {
        writeFile(path, "io.weight", strconv.FormatUint(uint64(r.Weight), 10))
        // ...
    }
    
    // devices（通过 BPF）
    if len(m.Resources.Devices) > 0 {
        // 加载 eBPF 程序过滤 device access
        bpfProg := createDeviceBPFProgram(m.Resources.Devices)
        attachBPF(path, bpfProg)
    }
    
    return nil
}

// 读取 cgroup 版本：
// /sys/fs/cgroup/cgroup.controllers 存在 → v2
// /sys/fs/cgroup/memory 存在 → v1
device cgroup 在 v2 中的特殊性:
cgroup v2 没有 devices controller。
runc 通过 eBPF 程序替代。
attachBPF 到 cgroup，过滤 file open 的设备类型。
这是 cgroup v2 的设计哲学：cgroup 只管资源，安全用 BPF。

07 rootfs 准备

// libcontainer/rootfs_linux.go

func prepareRootfs(pipe *os.File, config *configs.Config) error {
    // 1. mount rootfs (来自 spec.rootfs)
    if err := mountRootfs(config.Rootfs, config.RootfsPropagation); err != nil {
        return err
    }
    
    // 2. 准备 mount points
    for _, m := range config.Mounts {
        // /proc, /sys, /dev, /dev/pts, ...
        if err := mountToRootfs(m, config.Rootfs); err != nil {
            return err
        }
    }
    
    // 3. 设置 maskedPaths
    //    这些路径被覆盖为 /dev/null
    //    "/proc/kcore" 等敏感文件
    for _, p := range config.MaskPaths {
        maskFile(filepath.Join(config.Rootfs, p))
    }
    
    // 4. 设置 readonlyPaths
    //    这些路径被 bind mount 为 readonly
    for _, p := range config.ReadonlyPaths {
        remountReadonly(filepath.Join(config.Rootfs, p))
    }
    
    return nil
}

08 kill / signal

// runc kill my-container [SIGTERM]

// libcontainer/container_linux.go
func (c *linuxContainer) Signal(process *Process, sig os.Signal, all bool) error {
    // 1. 找到容器主进程 PID
    pids, err := c.Processes()
    
    // 2. 发信号
    if all {
        // 给所有容器内进程发信号
        for _, pid := range pids {
            syscall.Kill(pid, sig.(syscall.Signal))
        }
    } else {
        // 只给 init (PID 1 in container) 发
        syscall.Kill(initPid, sig.(syscall.Signal))
    }
    
    return nil
}

// 关键问题：容器内 PID 1 不接收默认信号
//   - 如果 PID 1 是僵尸接收器（如 tini）
//     能正常处理 SIGTERM
//   - 如果 PID 1 是 nginx/java
//     可能没有信号处理 → SIGTERM 被丢弃
//     → docker stop 等待 10s 后 SIGKILL

// 最佳实践：
//   1. 让 ENTRYPOINT 是设计良好的 init (tini, dumb-init)
//   2. 或者 docker run --init（自动加 tini）
//   3. 容器内进程必须处理 SIGTERM
