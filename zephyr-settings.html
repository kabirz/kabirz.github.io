01 Settings 概览

Settings 子系统提供键值对持久化存储，适用于配置保存和跨重启状态传递。

核心特性

特性说明
键值对类似文件系统的层级键名
多后端Flash、NVS、文件系统、FD
运行时加载启动时自动加载所有配置
异步保存延迟写入减少 Flash 磨损

后端Kconfig说明

04 Handler 注册

// 注册 Settings Handler
#include <zephyr/settings/settings.h>

// 设置回调
static int my_settings_set(const char *key, size_t len,
                           settings_read_cb read_cb, void *cb_arg)
{
    if (!strcmp(key, "device_name")) {
        read_cb(cb_arg, device_name, sizeof(device_name));
        printk("加载设备名: %s\n", device_name);
    } else if (!strcmp(key, "baudrate")) {
        read_cb(cb_arg, &baudrate, sizeof(baudrate));
        printk("加载波特率: %d\n", baudrate);
    }
    return 0;
}

// 注册 Handler
SETTINGS_STATIC_HANDLER_DEFINE(my_module, "my_module",
                               NULL, my_settings_set, NULL, NULL);
// 参数：name, subtree, load, set, save, commit

05 运行时操作

// 保存值
settings_save_one("my_module/device_name", "sensor_01", 11);

int baudrate = 115200;
settings_save_one("my_module/baudrate", &baudrate, sizeof(baudrate));

// 加载值
char name[32];
size_t len = sizeof(name);
settings_load_one("my_module/device_name", name, &len);

// 加载所有配置
settings_load();

// 删除值
settings_delete("my_module/device_name");

// 持久化所有脏数据
settings_save();

08 Settings API
