01 概述

Linux 安全模块 (Linux Security Modules, LSM) 框架是一个可堆叠的钩子系统，允许不同的安全策略（MAC、能力、沙箱）共存并依次执行。

架构定位：

┌─────────────────────────────────────────┐
│  用户态：进程、文件操作                    │
└────────────────┬────────────────────────┘
                 │ syscall
┌────────────────▼────────────────────────┐
│  VFS、网络、内存等子系统                  │
│  (resource access)                       │
└────────────────┬────────────────────────┘
                 │ 访问决策点
┌────────────────▼────────────────────────┐
│  LSM 框架 (security/security.c)          │
│  security_*() 钩子分发器                 │
└────────────────┬────────────────────────┘
                 │ static call / list
       ┌─────────┴──────────┐
       │                    │
┌──────▼──────┐      ┌──────▼──────┐
│  Capabilities│      │  SELinux    │
│  (capability)│      │ (selinux)   │
└─────────────┘      └─────────────┘
┌─────────────┐      ┌─────────────┐
│  Yama       │      │ AppArmor    │
└─────────────┘      └─────────────┘
┌─────────────┐      ┌─────────────┐
│  LoadPin    │      │  Landlock   │
└─────────────┘      └─────────────┘
┌─────────────┐      ┌─────────────┐
│  Lockdown   │      │  BPF (LSM)  │
└─────────────┘      └─────────────┘

核心特征：

  • 不替代 DAC：DAC (Discretionary Access Control，传统权限检查) 先执行，LSM 在 DAC 通过后才被调用
  • 可堆叠：多个 LSM 可同时活跃（Linux 5.x+ 完全支持）
  • 策略独立：内核不规定策略，只提供 hook 接口
  • 不可旁路：所有资源访问都经过 LSM

典型 LSM 与策略：

LSM类型典型应用
capabilities能力分割root 权限细分
SELinuxMAC (Type Enforcement)服务器加固
AppArmorMAC (Path-based)Ubuntu/SUSE 默认
Yamaptrace 限制反调试
LoadPin模块来源限制Chrome OS
Lockdownkernel self-protectionUEFI Secure Boot
LandlockUnprivileged sandbox应用沙箱
BPF (LSM)可编程动态策略

// LSM_HOOK_INIT 宏：
#define LSM_HOOK_INIT(HEAD, HOOK) \
    { .head = HOOK, .hook = { .HEAD = HOOK } }

// struct security_hook_list
struct security_hook_list {
    struct hlist_node       list;
    struct hlist_head      *head;
    union security_list_options hook;
    enum lsm_id             lsmid;
} __randomize_layout;

// 每个 hook 点对应一个全局 hlist_head
// 注册时插入到对应 head 的链表

// lsm_info — 模块注册结构
struct lsm_info {

static struct lsm_info __lsm_info_selinux __used __section("lsm_info") = {

Blob (安全上下文数据)：

// 每个需要保存安全状态的内核对象（task、cred、file、inode...）
// 都有额外的 blob 数据，由各 LSM 共享空间
struct lsm_blob_sizes {
    int lbs_cred;
    int lbs_file;
    int lbs_inode;
    int lbs_sock;
    int lbs_ipc;
    int lbs_msg_msg;
    int lbs_task;
    int lbs_xattr_count;
    int lbs_superblock;
    int lbs_vm_area_struct;
    int lbs_nf_mark;
    int lbs_nf_conn:
    int lbs_nf_ct_data;
    int lbs_tun_dev;

// 访问自己 LSM 的 blob 数据：
void *selinux_cred(struct cred *cred) {
    return cred->security + selinux_blob_sizes.lbs_cred;
    // cred->security 是一个大 blob，各 LSM 按顺序存
}

03 初始化与 LSM 栈

// security/security.c

// 启动早期调用：
int __init security_init(void)
{
    struct lsm_info *lsm;
    
    // 1. 处理 early_lsms (capabilities, lockdown 等)
    hlist_for_each_entry(lsm, &early_lsm_info, list)
        initialize_lsm(lsm);
    
    ordered_lsms_parse();        // 解析 CONFIG_LSM= 排序
    
    // 2. 处理 ordered_lsms (按用户配置的顺序)
    list_for_each_entry(lsm, &ordered_lsm_head, order_entry) {
        // 调用每个 LSM 的 init()
        if (initialize_lsm(lsm) == 0) {
            // 启用
            lsm_set_enabled(lsm->name, NULL, LSM_FLAG_LEGACY_MAJOR);
        }
    }
    
    return 0;
}

// 排序：CONFIG_LSM= 配置 + 系统默认
//   lsm=lockdown,capabilities,yama,integrity,selinux,bpf,landlock
// 启动时通过 /sys/kernel/security/lsm 查看：
// $ cat /sys/kernel/security/lsm
// capability,yama,integrity,selinux,bpf

// Hook 注册：
static inline void __init lsm_early_cred(struct lsm_id *lsmid) { ... }
void __init security_add_hooks(struct security_hook_list *hooks, int count,
                                struct lsm_id *lsmid)
{
    int i;
    for (i = 0; i < count; i++) {
        hooks[i].lsmid = lsmid;
        hlist_add_tail_rcu(&hooks[i].list, hooks[i].head);
    }
}

// 现代 LSM 使用 static_call 优化：
//   call_void_hook(file_open, ...) 不是间接调用循环
//   而是被 static_call 替换为直接调用（如果只有一个 LSM）
//   有多个时仍是函数指针数组

启动时序：

// 1. early LSM init
//    (capabilities 必须 early，因为后续都需要它)
//
// 2. 内核主体初始化完成
//
// 3. security_init()
//    - 解析命令行 lsm= 参数
//    - 按 ordered_lsm_head 顺序调用 init
//
// 4. 各 LSM 加载策略
//    - SELinux: 从 /etc/selinux/policy 读取
//    - AppArmor: 从 /sys/kernel/security/apparmor
//    - IMA: 校验度量表

04 关键安全钩子

// security/security.c 中分发器的例子
int security_file_open(struct file *file, const struct cred *cred)
{
    int ret;
    
    // 1. 调用 file_open_alloc 钩子
    ret = call_int_hook(file_open, file, cred);
    if (ret)
        return ret;
    
    // 2. 调用 file_open_finished
    return fsnotify_perm(file, MAY_OPEN);
}

int security_mmap_file(unsigned long prot, unsigned long flags,
                        struct file *file, unsigned long addr)
{
    unsigned long prot_adj = mmap_prot(prot, flags);
    int ret;
    
    // CAP_SYS_RAWIO 等 capability 检查
    if (!(prot_adj & PROT_SAO))
        ret = cap_mmap_file(file, prot, flags);
    if (ret)
        return ret;
    
    // 调用所有 LSM 的 mmap_file
    return call_int_hook(mmap_file, prot, flags, file, addr);
}

int security_bprm_check(struct linux_binprm *bprm)
{
    // 执行程序前检查
    // SELinux/AppArmor 在这里检查是否允许执行
    return call_int_hook(bprm_check_security, bprm);
}

int security_sb_mount(const char *dev_name, const struct path *path,
                       const char *type, unsigned long flags, void *data)
{
    // mount syscall 入口检查
    return call_int_hook(sb_mount, dev_name, path, type, flags, data);
}

int security_socket_connect(struct socket *sock, struct sockaddr *address, int addrlen)
{
    // 网络连接检查
    return call_int_hook(socket_connect, sock, address, addrlen);
}

// call_int_hook / call_void_hook 宏：
#define call_void_hook(FUNC, ...)                       \
    do {                                                \
        struct security_hook_list *P;                   \
        hlist_for_each_entry(P, &security_hook_heads.FUNC, list) \
            P->hook.FUNC(__VA_ARGS__);                  \
    } while (0)

#define call_int_hook(FUNC, IRC, ...) ({                \
    int RC = IRC;                                       \
    do {                                                \
        struct security_hook_list *P;                   \
        hlist_for_each_entry(P, &security_hook_heads.FUNC, list) { \
            RC = P->hook.FUNC(__VA_ARGS__);             \
            if (RC != 0) break;                         \
        }                                               \
    } while (0);                                        \
    RC; })

分类代表 hooks
程序执行bprm_check_security, bprm_creds_for_exec
文件系统file_open, inode_permission, mmap_file
进程管理task_kill, task_setscheduler, ptrace_access_check
网络socket_connect, socket_bind, socket_accept
IPCmsg_queue_msgctl, sem_semctl

05 seccomp

seccomp (Secure Computing Mode) 限制进程可调用的系统调用。常用于沙箱（Docker、Chrome renderer、Android zygote）。

// kernel/seccomp.c

// 模式
#define SECCOMP_MODE_DISABLED    0
#define SECCOMP_MODE_STRICT      1   // 只能 read/write/exit/sigreturn
#define SECCOMP_MODE_FILTER      2   // BPF 过滤

struct seccomp {
    int                  mode;
    atomic_t             filter_count;
    struct seccomp_filter __rcu *filter;  // 过滤链
};

// seccomp_filter — BPF 过滤器
struct seccomp_filter {
    refcount_t            usage;
    bool                  log;
    bool                  wait_killable_recv;
    struct seccomp_filter *prev;       // 父过滤器
    struct bpf_prog       *prog;        // BPF 程序
    struct list_head      *notifier;
    struct notify_mm      *notif_mm;
    struct semaphore      notifier_lock;
};

// 设置：
// include/uapi/linux/seccomp.h
struct seccomp_data {
    int   nr;                // syscall number
    __u32 arch;              // AUDIT_ARCH_*
    __u64 instruction_pointer;
    __u64 args[6];           // syscall 参数
};

// seccomp() 系统调用：
// SECCOMP_SET_MODE_STRICT
// SECCOMP_SET_MODE_FILTER (用 sock_fprog 提供 BPF 程序)
// SECCOMP_GET_NOTIF_SIZES
// SECCOMP_GET_ACTION_AVAIL

// SECCOMP_SET_MODE_FILTER 流程：
// 1. 用户态提供 BPF 程序（通常用 libseccomp 生成）
// 2. 内核校验程序（限制 BPF_LD | BPF_W | BPF_ABS，只能读 seccomp_data）
// 3. 替换/合并到 task->seccomp.filter

// BPF 返回值：
#define SECCOMP_RET_KILL_PROCESS 0x80000000U   // 杀进程
#define SECCOMP_RET_KILL_THREAD  0x00000000U   // 杀线程
#define SECCOMP_RET_TRAP         0x00030000U   // SIGSYS
#define SECCOMP_RET_ERRNO        0x00050000U   // 返回错误码
#define SECCOMP_RET_USER_NOTIF   0x7fc00000U   // 通知用户态
#define SECCOMP_RET_TRACE        0x7ff00000U   // ptrace 处理
#define SECCOMP_RET_LOG          0x7ffc0000U   // 日志后允许
#define SECCOMP_RET_ALLOW        0x7fff0000U   // 允许

// 检查入口：
// syscall 入口处 do_syscall_64() 之前
// __secure_computing() 调用 __seccomp_filter()
// 1. 遍历 filter 链
// 2. 运行每个 BPF 程序
// 3. 根据返回值决定动作

int __secure_computing(const struct seccomp_data *sd)
{
    int this_syscall = sd ? sd->nr : 0;
    struct seccomp_filter *f;
    
    u32 action;
    
    switch (current->seccomp.mode) {
    case SECCOMP_MODE_STRICT:
        // 只允许 read/write/exit/sigreturn
        if (this_syscall < 4 && syscall_table_32[this_syscall])
            return 0;
        action = SECCOMP_RET_KILL_THREAD;
        break;
    
    case SECCOMP_MODE_FILTER:
        action = __seccomp_filter(this_syscall, sd, false);
        break;
    }
    
    switch (action) {
    case SECCOMP_RET_ALLOW:
        return 0;
    case SECCOMP_RET_KILL_THREAD:
    case SECCOMP_RET_KILL_PROCESS:
        // 发送 SIGSYS
        do_seccomp_longjmp();  // 不会返回
    case SECCOMP_RET_TRAP:
        force_sig(SIGSYS);
        return -1;
    case SECCOMP_RET_ERRNO:
        return -errno;
    case SECCOMP_RET_TRACE:
        return -1;  // ptrace 处理
    case SECCOMP_RET_USER_NOTIF:
        // 通过 /dev/seccomp 通知用户态
        return seccomp_do_user_notification(...);
    }
}

典型用法（应用开发者）：

// 1. 用 SECCOMP_SET_MODE_FILTER + BPF 程序
// 2. 推荐 libseccomp：
#include <seccomp.h>

seccomp_rule_add(ctx, SCMP_ACT_ALLOW, SCMP_SYS(read), 0);

06 SELinux

SELinux (Security-Enhanced Linux) 是 NSA 主导的强制访问控制 (MAC) 实现，采用 Type Enforcement 模型。

// security/selinux/hooks.c

    LSM_HOOK_INIT(bprm_committing_creds, selinux_bprm_committing_creds),
    LSM_HOOK_INIT(bprm_secureexec, selinux_bprm_secureexec),
    LSM_HOOK_INIT(sb_kern_mount, selinux_sb_kern_mount),
    LSM_HOOK_INIT(sb_statfs, selinux_sb_statfs),
    LSM_HOOK_INIT(file_open, selinux_file_open),
    LSM_HOOK_INIT(file_ioctl, selinux_file_ioctl),
    LSM_HOOK_INIT(file_lock, selinux_file_lock),
    LSM_HOOK_INIT(mmap_file, selinux_mmap_addr),
    LSM_HOOK_INIT(inode_permission, selinux_inode_permission),
    LSM_HOOK_INIT(inode_setxattr, selinux_inode_setxattr),
    LSM_HOOK_INIT(task_create, selinux_task_create),
    LSM_HOOK_INIT(task_kill, selinux_task_kill),
    LSM_HOOK_INIT(task_setscheduler, selinux_task_setscheduler),
    LSM_HOOK_INIT(socket_create, selinux_socket_create),
    LSM_HOOK_INIT(socket_connect, selinux_socket_connect),
    // ... 100+ hooks ...
    LSM_HOOK_INIT(getprocattr, selinux_getprocattr),
    LSM_HOOK_INIT(setprocattr, selinux_setprocattr),
    LSM_HOOK_INIT(audit_rule_init, selinux_audit_rule_init),
};

// 核心概念：
//   - Subject: 通常是进程 (domain)
//   - Object: 通常是文件/IPC (type)
//   - Permission: read/write/execute/...
//
//   格式：TE 规则
//     allow source_domain target_type : class { perms };
//   例：
//     allow httpd_t httpd_log_t : file { read write append };
//     deny  unconfined_t shadow_t : file { read };
//
//   Type：在 file_contexts 中定义路径到 type 的映射
//   Domain：在 policy 中定义 type 属性
//   Role: RBAC 层（user → role → domain）

// AVC (Access Vector Cache):
// security/selinux/avc.c
//   高频查询，需要缓存
//   cache_key = (subject_sid, target_sid, requested_class)

struct avc_node {
    struct hlist_node    list;
    struct avc_entry     ae;
    struct rcu_head      rhead;
};

struct avc_entry {
    u32           ssid;        // subject security id
    u32           tsid;        // target security id
    u16           tclass;      // target class
    struct av_decision avd;     // 决策
};

// 决策路径：
// 1. selinux_inode_permission() 被调用
// 2. 收集 (ssid, tsid, tclass, requested)
// 3. avc_has_perm() 查询缓存
// 4. 未命中 → security_compute_av() 读策略
// 5. 缓存结果

// Policy 加载：
// security/selinux/selinuxfs.c
//   /sys/fs/selinux/load 写入新 policy
//   security_load_policy() 解析
//   - TE rules (allow/deny/auditallow/...)
//   - type definitions
//   - role/user mappings
//   - constraints
//   - initial SID assignments

// 用户态工具：
//   sestatus, getenforce, setenforce
//   semanage (管理策略)
//   audit2allow (从 audit 日志生成规则)
//   checkmodule, semodule (编译/加载模块)

SELinux 模式：

模式行为
Disabled不启用
Permissive记录违规但不拒绝（调试）
Enforcing记录并拒绝违规（生产）

// 规则类型：
//   file rules:
//     /etc/passwd r,
//     /var/log/** rw,
//   capability rules:
//     capability net_admin,
//   network rules:
//     network inet stream,
//   rlimit rules:
//     set rlimit cpu <= 60,
//
// 模式：
//   enforce  - 违规拒绝
//   complain - 违规记录（学习模式）
//   audit    - 记录所有访问

// 用户态：
//   /sys/kernel/security/apparmor/profiles
//   /etc/apparmor.d/*  (profile 文件)
//   apparmor_parser
//   aa-status
//   aa-logprof (从日志学习生成 profile)

// Profile 示例（/etc/apparmor.d/usr.sbin.nginx）：
//   #include <tunables/global>
//   
//   profile nginx /usr/sbin/nginx {
//     #include <abstractions/base>
//     #include <abstractions/nameservice>
//     
//     capability net_bind_service,
//     capability setgid,
//     capability setuid,
//     
//     /usr/sbin/nginx mr,
//     /etc/nginx/** r,

08 Landlock

Landlock 是 Linux 5.13+ 引入的非特权沙箱机制，普通进程可以限制自己的访问权限。

// security/landlock/

// 与 SELinux/AppArmor 的区别：
//   - 不需要 root/CAP_SYS_ADMIN
//   - 进程自愿降权（不能提权）
//   - 用 ruleset 抽象，简单

// API：
// landlock_create_ruleset()
// landlock_add_rule()
// landlock_restrict_self()

// 1. 创建 ruleset：
struct landlock_ruleset_attr attr = {
    .handled_access_fs = LANDLOCK_ACCESS_FS_EXECUTE |
                          LANDLOCK_ACCESS_FS_WRITE_FILE |
                          LANDLOCK_ACCESS_FS_READ_FILE |
                          LANDLOCK_ACCESS_FS_REMOVE_FILE |
                          LANDLOCK_ACCESS_FS_MAKE_DIR |
                          // ...
};
int ruleset_fd = syscall(SYS_landlock_create_ruleset,
                          &attr, sizeof(attr), 0);

// 2. 添加规则（white-list）：
struct landlock_path_beneath_attr path_attr = {
    .allowed_access = LANDLOCK_ACCESS_FS_READ_FILE |
                       LANDLOCK_ACCESS_FS_READ_DIR,
    .parent_fd = open("/tmp", O_PATH | O_CLOEXEC),
};
syscall(SYS_landlock_add_rule, ruleset_fd,
         LANDLOCK_RULE_PATH_BENEATH, &path_attr, 0);

// 3. 应用到自己：
prctl(PR_SET_NO_NEW_PRIVS, 1, 0, 0, 0);
syscall(SYS_landlock_restrict_self, ruleset_fd, 0);

// 之后：
//   - 只能 read /tmp 下的文件
//   - 不能 write /tmp 下的文件
//   - 不能访问其他路径

// 数据结构：
struct landlock_ruleset {
    refcount_t             usage;
    bool                   living;
    u32                    num_layers;
    struct landlock_access_fs access_fs;
    struct landlock_rule   rules;
    struct rb_root         root;
    // ...
};

struct landlock_rule {
    struct rb_node         node;
    struct landlock_object *object;
    u32                    access;
};

// security/landlock/fs.c
// 实现 security_hook_list landlock_hooks[]:
//   hook_path_truncate
//   hook_file_open
//   hook_inode_permission
//   hook_path_link
//   hook_path_unlink
//   hook_path_rmdir
//   hook_path_mkdir
//   hook_path_mknod
//   hook_path_symlink
//   hook_path_rename
//   hook_path_chmod
//   hook_path_chown
//   hook_file_alloc
//   hook_file_ioctl
//   hook_file_truncate
//   ...

// 进程相关：
//   hook_cred_prepare  - 拷贝 ruleset 到新 cred
//   hook_task_kill     - 限制信号发送
//   hook_ptrace_access_check

Landlock vs seccomp：

seccompLandlock
对象系统调用文件系统访问
粒度syscall numberpath + access type
规则BPFdeclarative (path patterns)
需要特权无（PR_SET_NO_NEW_PRIVS）无（同上）
共存与 LSM 并行是 LSM 之一
典型应用场景:
应用自我沙箱（如浏览器渲染进程）。
容器运行时（runc 之外的额外限制）。
任意应用降权，无需 root 配置全局策略。
仍是 LSM 之一，所以与其他 LSM 同时生效。
