01 概述

Socket 层是用户态应用程序与内核协议栈之间的桥梁。当应用调用 socket()、bind()、connect()、send()、recv() 等系统调用时，请求经过 Socket 层分发到具体的协议实现（TCP/UDP/RAW）。

Socket 层的职责
  • 系统调用入口：接收用户态 socket 调用
  • fd → socket 映射：文件描述符关联到 socket 结构
  • 协议分发：根据 family/type/proto 选择协议实现
  • 缓冲区管理：发送/接收队列管理
  • 等待队列：阻塞/非阻塞 I/O 通知

两次多态分发

第一次分发：proto_ops（socket 级操作）

  socket(AF_INET, SOCK_STREAM, IPPROTO_TCP)
      │
      ↓
  pf->create() → inet_create()
      │
      ├── SOCK_STREAM → inet_stream_ops  → tcp_prot
      ├── SOCK_DGRAM  → inet_dgram_ops   → udp_prot
      └── SOCK_RAW    → inet_sockraw_ops → raw_prot

第二次分发：proto（协议级操作）

  send()/recv() → sock->ops->sendmsg()
      │
      ↓
  tcp_sendmsg() / udp_sendmsg() / raw_sendmsg()

03 核心数据结构

// include/linux/net.h:116 — BSD Socket（用户态可见）
struct socket {
    socket_state    state;      // SS_UNCONNECTED / SS_CONNECTING / SS_CONNECTED
    short           type;       // SOCK_STREAM / SOCK_DGRAM / SOCK_RAW
    unsigned long   flags;
    struct file     *file;      // 关联的 file 结构（VFS 集成）
    struct sock     *sk;        // 指向内核 sock 结构（核心！）
    const struct proto_ops *ops; // 协议族操作表（TCP/UDP 差异）
    struct socket_wq wq;      // 等待队列（poll/select/epoll）
};
// include/net/sock.h:1286 — 协议操作表（多态关键）
struct proto_ops {
    int family;
    int (*bind)(struct socket *sock, struct sockaddr *myaddr, int sockaddr_len);
    int (*connect)(struct socket *sock, struct sockaddr *vaddr, int sockaddr_len, int flags);
    int (*accept)(struct socket *sock, struct socket *newsock, struct proto_accept_arg *arg);
    ssize_t (*sendmsg)(struct socket *sock, struct msghdr *m, size_t total_len);
    ssize_t (*recvmsg)(struct socket *sock, struct msghdr *m, size_t total_len, int flags);
    __poll_t (*poll)(struct file *file, struct socket *sock, struct poll_table_struct *wait);
    int (*setsockopt)(struct socket *sock, int level, int optname, sockptr_t optval, unsigned int optlen);
    // ...
};

// inet_stream_ops — TCP/IPv4 的 proto_ops
// inet_dgram_ops  — UDP/IPv4 的 proto_ops
// include/net/sock.h:1286 — 协议操作表（协议级多态）
struct proto {
    void (*close)(struct sock *sk, long timeout);
    int  (*connect)(struct sock *sk, ...);
    struct sock *(*accept)(struct sock *sk, ...);
    int  (*sendmsg)(struct sock *sk, struct msghdr *msg, size_t len);
    int  (*recvmsg)(struct sock *sk, struct msghdr *msg, ...);
    int  (*bind)(struct sock *sk, ...);
    int  (*hash)(struct sock *sk);
    int  (*get_port)(struct sock *sk, unsigned short snum);
    // ...
    union {
        struct inet_hashinfo *hashinfo;  // TCP
        struct udp_table     *udp_table;  // UDP
        struct raw_hashinfo  *raw_hash;   // RAW
    } h;
};

// tcp_prot   — TCP 协议的 proto 实现
// udp_prot   — UDP 协议的 proto 实现
// raw_prot   — RAW 协议的 proto 实现

04 socket() 系统调用

socket(AF_INET, SOCK_STREAM, IPPROTO_TCP)
        │
        ↓
  __sys_socket() — net/socket.c
        │
        ├── 1. sock_create()
        │       │
        │       ├── 分配 struct socket
        │       ├── 查找 net_families[AF_INET] → inet_family_ops
        │       ├── 调用 pf->create(net, sock, protocol)
        │       │       │
        │       │       ↓
        │       │   inet_create() — net/ipv4/af_inet.c
        │       │       │
        │       │       ├── 遍历 inetsw[SOCK_STREAM] 找到匹配的协议
        │       │       │
        │       │       ├── SOCK_STREAM → inet_stream_ops + tcp_prot
        │       │       ├── SOCK_DGRAM  → inet_dgram_ops  + udp_prot
        │       │       └── SOCK_RAW    → inet_sockraw_ops + raw_prot
        │       │
        │       │       ├── 分配 struct sock（tcp_sock / udp_sock）
        │       │       ├── sk->sk_prot->init(sk)  ← TCP/UDP 初始化
        │       │       └── sock->ops = inet_stream_ops
        │       │
        │       └── 返回 struct socket
        │
        ├── 2. sock_map_fd()
        │       ├── 分配 file 结构
        │       ├── file->f_op = socket_file_ops
        │       ├── file->private_data = socket
        │       └── 返回 fd（如 3）
        │
        ↓
  返回 fd=3 给用户态
关键设计
① socket 结构通过 file->private_data 与 VFS fd 关联
② socket->sk 指向内核协议无关的 sock 结构
③ sock->sk_prot 指向具体协议的 proto 操作表
④ 用户态的 send()/recv() 最终调用 sock->sk_prot->sendmsg()/recvmsg()

05 send/recv 数据流

发送路径

  应用调用 send(fd, buf, len, 0)
        │
        ↓
  sock_sendmsg() — net/socket.c
        │
        ↓
  sock->ops->sendmsg(sock, msg, len)
        │
        ├── TCP: inet_stream_ops.sendmsg = inet_sendmsg()
        │           → sk->sk_prot->sendmsg = tcp_sendmsg()
        │
        ├── UDP: inet_dgram_ops.sendmsg = inet_sendmsg()
        │           → sk->sk_prot->sendmsg = udp_sendmsg()
        │
        └── RAW: inet_sockraw_ops.sendmsg = raw_sendmsg()

接收路径

  应用调用 recv(fd, buf, len, 0)
        │
        ↓
  sock_recvmsg() — net/socket.c
        │
        ↓
  sock->ops->recvmsg(sock, msg, len, flags)
        │
        ├── TCP: → tcp_recvmsg()
        │          从 sk_receive_queue 读取数据
        │          如果无数据 → 睡眠等待
        │
        ├── UDP: → udp_recvmsg()
        │          从 udp_table 查找数据包
        │          如果无数据 → 睡眠等待
        │
        └── RAW: → raw_recvmsg()

阻塞 vs 非阻塞

阻塞模式（默认）
// 数据未就绪时，进程睡眠等待
n = recv(fd, buf, len, 0);
// 进程在此处阻塞
// 直到 sk_data_ready() 唤醒
非阻塞模式
// 设置 O_NONBLOCK 后
fcntl(fd, F_SETFL, O_NONBLOCK);
n = recv(fd, buf, len, 0);
// 无数据时立即返回 -1
// errno = EAGAIN / EWOULDBLOCK

// net/ipv4/af_inet.c — 协议族注册
static struct list_head inetsw[SOCK_MAX];  // 按 socket 类型索引

static const struct inet_protosw inetsw_array[] = {
    {   // TCP
        .type = SOCK_STREAM,
        .protocol = IPPROTO_TCP,
        .ops = &inet_stream_ops,
        .prot = &tcp_prot,
    },
    {   // UDP
        .type = SOCK_DGRAM,
        .protocol = IPPROTO_UDP,
        .ops = &inet_dgram_ops,
        .prot = &udp_prot,
    },
    {   // RAW
        .type = SOCK_RAW,
        .protocol = IPPROTO_ICMP,
        .ops = &inet_sockraw_ops,
        .prot = &raw_prot,
    },
};

// socket() 调用时：
// inetsw[SOCK_STREAM] → 找到 TCP
// inetsw[SOCK_DGRAM]  → 找到 UDP
// inetsw[SOCK_RAW]    → 找到 RAW
