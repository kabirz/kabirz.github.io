01 zbus 概览

zbus 是 Zephyr 的发布-订阅消息总线，实现松耦合的线程间通信，编译时构建，零运行时开销。

02 通道 (Channel)

通道是 zbus 的核心抽象，代表一个消息主题。

定义通道

// 包含 zbus 头文件
#include <zephyr/zbus/zbus.h>

// 定义消息类型
struct sensor_msg {
    uint32_t timestamp;
    int32_t temperature;
    int32_t humidity;
};

// 定义通道
ZBUS_CHAN_DEFINE(sensor_chan,
    ZBUS_CHAN_NAME("sensor"),     // 通道名
    ZBUS_MSG_INIT(0),             // 初始消息
    &sensor_data,                 // 消息数据
    NULL,                         // 无用户态访问
    ZBUS_OBSERVERS_EMPTY,         // 初始无观察者
    ZBUS_CHAN_PREPARE_INIT(0)
);

// 或使用便利宏
ZBUS_CHAN_DEFINE_WITH_NAME(struct sensor_msg, sensor_chan,
                           "sensor", &sensor_data);

通道属性

属性说明
名称通道字符串标识
消息类型编译时类型检查
观察者列表订阅该通道的观察者
访问控制用户态/内核态权限

03 观察者 (Observer)

观察者是通道消息的消费者，当通道有新消息时被通知。

// 定义观察者处理函数
static void sensor_handler(const struct zbus_channel *chan)
{
    // 获取通道消息
    const struct sensor_msg *msg = zbus_chan_const_msg(chan);

    // 处理数据
    printk("温度: %d°C, 湿度: %d%%\n",
           msg->temperature, msg->humidity);
}

// 注册观察者
ZBUS_OBS_DEFINE(sensor_handler);
ZBUS_CHAN_ADD_OBS(sensor_chan, sensor_handler, 1);
// 参数：通道, 观察者, 优先级（0 最高）

观察者优先级

// 观察者按优先级排序执行
// 优先级 0：最先执行
// 优先级 1：次之
// ...

ZBUS_CHAN_ADD_OBS(sensor_chan, high_prio_observer, 0);  // 最高优先级
ZBUS_CHAN_ADD_OBS(sensor_chan, low_prio_observer, 5);   // 较低优先级

04 发布与订阅

发布消息

// 发布消息到通道
struct sensor_msg data = {
    .timestamp = k_uptime_ms(),
    .temperature = 25,
    .humidity = 60,
};

// 同步发布（等待所有观察者完成）
zbus_chan_pub(&sensor_chan, &data, K_FOREVER);

// 异步发布（立即返回）
zbus_chan_pub(&sensor_chan, &data, K_NO_WAIT);

// ISR 中发布
zbus_chan_pub(&sensor_chan, &data, K_NO_WAIT);

订阅消息

// 方式 1：观察者回调（推荐）
ZBUS_OBS_DEFINE(my_observer);
ZBUS_CHAN_ADD_OBS(sensor_chan, my_observer, 1);

// 方式 2：主动读取
struct sensor_msg msg;
zbus_chan_read(&sensor_chan, &msg, K_FOREVER);

// 方式 3：使用信号量等待
K_SEM_DEFINE(sensor_sem, 0, 1);

// 观察者中通知
static void sensor_notify(const struct zbus_channel *chan)
{
    k_sem_give(&sensor_sem);
}

// 线程中等待
void worker_thread(void)
{
    while (1) {
        k_sem_take(&sensor_sem, K_FOREVER);
        zbus_chan_read(&sensor_chan, &msg, K_NO_WAIT);
        // 处理消息...
    }
}

05 远程代理

zbus 远程代理支持跨设备通信，通过 UART 或网络传输通道消息。

// 远程代理配置
// CONFIG_ZBUS_PROXY_AGENT=y

// 代理通道
ZBUS_CHAN_DEFINE(remote_chan,
    ZBUS_CHAN_NAME("remote"),
    ZBUS_MSG_INIT(0),
    &remote_data,
    NULL,
    ZBUS_OBSERVERS(remote_proxy),
    ZBUS_CHAN_PREPARE_INIT(0)
);

06 内部实现

// zbus 内部使用迭代器段（ITERABLE_SECTION）
// 通道和观察者在编译时链接到特殊段
// 运行时通过段遍历访问

// 核心数据结构（subsys/zbus/zbus.c）
struct zbus_channel {
    const char *name;
    void *data;
    size_t data_size;
    struct zbus_observer_obs_desc *observers;
    uint8_t num_observers;
};

// 发布流程：
// 1. 拷贝消息数据到通道
// 2. 遍历观察者段
// 3. 按优先级调用观察者回调
// 4. 观察者可读取通道消息
