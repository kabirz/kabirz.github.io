01 概述

LED 子系统提供了统一的 LED 亮度控制框架，支持 GPIO LED、I2C/SPI LED 控制器、RGB LED、Flash LED 等。通过触发器（Trigger）机制，LED 可以响应内核事件（网络活动、磁盘 I/O、CPU 负载等）。

核心功能
  • 亮度控制：0-255 级亮度，PWM 调光
  • 闪烁控制：硬件/软件闪烁
  • 触发器：事件驱动 LED 行为
  • 多色 LED：RGB/RGBW 支持
  • Flash LED：相机闪光灯控制
  • sysfs 接口：用户空间控制
常见触发器
触发器说明
heartbeat心跳指示
disk-activity磁盘活动
netdev网络活动
cpuCPU 负载
panic内核 panic
default-on默认开启
timer定时闪烁
input输入事件

04 触发器

触发器是 LED 子系统的事件驱动机制。内核事件触发触发器，触发器控制 LED 亮度。

触发器工作流

事件发生
网络活动
↓
led_trigger_event(trigger, LED_FULL)
↓
遍历 trigger->led_cdevs
所有使用此触发器的 LED
↓
led1->brightness_set(FULL)
led2->brightness_set(FULL)
↓ 延迟
led_trigger_event(trigger, LED_OFF)

简单触发器 vs 复杂触发器

类型注册方式说明
简单触发器led_trigger_register_simple()仅设置亮度，无自定义属性
复杂触发器led_trigger_register()自定义 activate/deactivate + sysfs 属性

自定义触发器示例

// 注册自定义触发器
static struct led_trigger my_trigger = {
    .name = "my-event",
    .activate = my_activate,
    .deactivate = my_deactivate,
};

static int my_activate(struct led_classdev *led_cdev)
{
    // LED 被绑定到此触发器时调用
    return 0;
}

static void my_deactivate(struct led_classdev *led_cdev)
{
    // LED 解绑时调用
}

// 事件触发
led_trigger_event(&my_trigger, LED_FULL);  // 亮
led_trigger_event(&my_trigger, LED_OFF);   // 灭

// 闪烁触发
led_trigger_blink(&my_trigger, 500, 500);  // 500ms 亮/灭交替
触发器绑定：用户空间通过 /sys/class/leds/<name>/trigger 绑定触发器。驱动中可通过 default_trigger 字段或设备树 linux,default-trigger 属性设置默认触发器。

07 编写驱动

LED 驱动模板

#include <linux/leds.h>
#include <linux/module.h>
#include <linux/platform_device.h>

struct my_led_priv {
    struct led_classdev cdev;
    void __iomem *reg_base;
};

static void my_led_brightness_set(struct led_classdev *led_cdev,
                                  enum led_brightness value)
{
    struct my_led_priv *priv = container_of(led_cdev, struct my_led_priv, cdev);
    u32 val;

    if (value == LED_OFF)
        val = 0;
    else
        val = value;

    writel(val, priv->reg_base + MY_LED_PWM);
}

static int my_led_blink_set(struct led_classdev *led_cdev,
                           unsigned long *delay_on,
                           unsigned long *delay_off)
{
    struct my_led_priv *priv = container_of(led_cdev, struct my_led_priv, cdev);

    // 配置硬件闪烁定时器
    writel(*delay_on, priv->reg_base + MY_LED_BLINK_ON);
    writel(*delay_off, priv->reg_base + MY_LED_BLINK_OFF);
    writel(1, priv->reg_base + MY_LED_BLINK_EN);

    return 0;
}

static int my_led_probe(struct platform_device *pdev)
{
    struct my_led_priv *priv;
    struct led_init_data init_data = {};

    priv = devm_kzalloc(&pdev->dev, sizeof(*priv), GFP_KERNEL);

    // 初始化 LED
    priv->cdev.name = "myboard:status";
    priv->cdev.max_brightness = 255;
    priv->cdev.brightness_set = my_led_brightness_set;
    priv->cdev.blink_set = my_led_blink_set;
    priv->cdev.default_trigger = "heartbeat";
    priv->cdev.flags = LED_CORE_SUSPENDRESUME;

    // 注册（设备管理版本）
    init_data.fwnode = dev_fwnode(&pdev->dev);
    devm_led_classdev_register_ext(&pdev->dev, &priv->cdev, &init_data);

    dev_info(&pdev->dev, "LED registered\n");
    return 0;
}

static int my_led_remove(struct platform_device *pdev)
{
    return 0;  // devm 自动清理
}

static const struct of_device_id my_led_of_match[] = {
    { .compatible = "vendor,my-led" },
    { }
};
MODULE_DEVICE_TABLE(of, my_led_of_match);

static struct platform_driver my_led_driver = {
    .probe = my_led_probe,
    .remove = my_led_remove,
    .driver = {
        .name = "my-led",
        .of_match_table = my_led_of_match,
    },
};
module_platform_driver(my_led_driver);
MODULE_LICENSE("GPL");
