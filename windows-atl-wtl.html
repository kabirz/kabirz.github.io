01 概述

ATL（Active Template Library，活动模板库）是微软提供的轻量级 C++ 模板库，最初专为编写小型、快速的 COM 组件而设计。WTL（Windows Template Library）则是构建在 ATL 之上的GUI 扩展，由微软员工 Nenad Stefanovic 开发并开源，把窗口、控件、对话框、GDI、命令栏等 Win32 GUI 元素以模板方式优雅封装。两者组合后形成的轻量替代 MFC 方案在追求小体积、零运行时依赖、高启动性能的桌面项目中长期占据一席之地。

核心特性
  • 纯模板：编译期多态，几乎无虚函数开销
  • 零运行时 DLL：静态链接，分发体积仅 KB 级
  • 薄封装：对象布局常只是一个 HWND/HDC
  • 消息映射：CRTP 基类 + 宏表，避免巨型虚函数表
  • COM 友好：与 ATL COM 服务器无缝集成
  • 现代 C++：支持 C++17/20，可与 STL/智能指针混用
适用场景
场景选择理由
COM 服务器 / ActiveXATL 是黄金标准
Shell 扩展 / 命名空间体积敏感、需快速加载
嵌入式 UI / 瘦客户端无 CRT 扩展依赖
工具型小应用单一 exe 即可分发
IE/Edge 浏览器助手需暴露 COM 接口
系统级后台 UI避免大体积运行时
哲学：ATL/WTL 的设计准则是"用编译时间换取运行性能"。模板把绝大部分分发逻辑推到编译期，运行时几乎只剩 Win32 API 本身的开销。这与 MFC"虚函数+动态创建+运行时类型"的厚重架构形成鲜明对比。

02 ATL 基础与 COM

ATL 的根基是 CComObjectRootBase 与线程模型模板 CComObjectRootEx<ThreadModel>。它们为每个 COM 对象提供引用计数与接口映射表。所有非聚合对象的最终派生类由 CComObject<T> 包装，由它实现 IUnknown 的三个方法。

// IMath.h —— 简单 COM 对象
#include <atlbase.h>
#include <atlcom.h>

// IDL: [object, uuid(...)] interface IMath : IUnknown { HRESULT Add(...); }
class ATL_NO_VTABLE CMath :
    public CComObjectRootEx<CComSingleThreadModel>,  // 线程模型
    public CComCoClass<CMath, &CLSID_Math>,       // 类工厂
    public IMath                                   // 业务接口
{
public:
    CMath() {}

    BEGIN_COM_MAP(CMath)                       // 接口表：QueryInterface 查它
        COM_INTERFACE_ENTRY(IMath)
        COM_INTERFACE_ENTRY(IUnknown)
    END_COM_MAP()

    // IMath
    STDMETHOD(Add)(LONG a, LONG b, LONG* pResult) override {
        if (!pResult) return E_POINTER;
        *pResult = a + b;
        return S_OK;
    }

    DECLARE_REGISTRY_RESOURCEID(IDR_MATH)     // 注册表脚本
};

OBJECT_ENTRY_AUTO(CLSID_Math, CMath)             // 自动登记到对象映射表

服务器模块与 IUnknown

// DLL 服务器入口
class CMathModule : public CAtlDllModuleT<CMathModule> {
public:
    DECLARE_LIBID(LIBID_MathLib)
    DECLARE_REGISTRY_APPID_RESOURCEID(IDR_MATH, "{...AppID...}")
};
CMathModule _AtlModule;                        // 全局唯一模块对象

extern "C" BOOL WINAPI DllMain(HINSTANCE h, DWORD r, LPVOID lp) {
    return _AtlModule.DllMain(h, r, lp);
}
// EXE 服务器则派生 CAtlExeModuleT，自带命令行 /RegServer、消息循环

// IUnknown 三方法实现要点（由 CComObject<T> 与 CComObjectRootEx 提供）：
//   QueryInterface  →  COM_MAP 表查找 IID，返回接口指针 + AddRef
//   AddRef          →  InternalAddRef()（线程模型决定原子/非原子）
//   Release         →  InternalRelease()，归零时 delete this
线程模型选择：CComSingleThreadModel 用于非线程安全的对象（最快），CComMultiThreadModel 提供原子引用计数（COM 标准要求）。进程内多用 CComObjectThreadModel 自动适配，按 _ATL_FREE_THREADED 宏切换。

03 ATL 窗口支持

ATL 提供最小化的窗口封装。CWindow 直接包装一个 HWND（成员 m_hWnd），重载 operator HWND() 实现到句柄的隐式转换——这意味着任何 Win32 API 都能直接接受 CWindow 对象。CWindowImpl<T> 用 CRTP（奇异递归模板）让派生类把 WndProc 路由到自己的消息映射表。

#include <atlbase.h>
#include <atlwin.h>

// CWindow：薄到只有 m_hWnd 一个成员
class CWindow {
public:
    HWND m_hWnd;
    operator HWND() const { return m_hWnd; }
    BOOL Create(LPCTSTR cls, HWND parent, RECT* rc = NULL);
    void ShowWindow(int cmd) { ::ShowWindow(m_hWnd, cmd); }
    // ... 几百个对 Win32 API 的转发，全部非虚 ...
};

// CWindowImpl：CRTP + 消息映射
class CMyWindow : public CWindowImpl<CMyWindow> {
public:
    DECLARE_WND_CLASS(_T("MyWndClass"))     // 注册窗口类（含类名、光标、背景）

    BEGIN_MSG_MAP(CMyWindow)
        MESSAGE_HANDLER(WM_CREATE,  OnCreate)
        MESSAGE_HANDLER(WM_PAINT,   OnPaint)
        MESSAGE_HANDLER(WM_DESTROY, OnDestroy)
    END_MSG_MAP()

    LRESULT OnCreate(UINT, WPARAM, LPARAM, BOOL& bHandled) {
        bHandled = FALSE;                    // 让 DefWindowProc 继续处理
        return 0;
    }
    LRESULT OnPaint(UINT, WPARAM, LPARAM, BOOL& bHandled) {
        CPaintDC dc(m_hWnd);
        dc.TextOut(10, 10, _T("Hello ATL"));
        return 0;
    }
    LRESULT OnDestroy(UINT, WPARAM, LPARAM, BOOL&) {
        PostQuitMessage(0);
        return 0;
    }
};

int WINAPI wWinMain(HINSTANCE hInst, HINSTANCE, PWSTR, int) {
    CMyWindow wnd;
    wnd.Create(NULL, CWindow::rcDefault, _T("Demo"));
    wnd.ShowWindow(SW_SHOWNORMAL);

    MSG msg;
    while (GetMessage(&msg, NULL, 0, 0)) {
        TranslateMessage(&msg);
        DispatchMessage(&msg);
    }
    return (int)msg.wParam;
}
CRTP 优势：CWindowImpl<CMyWindow> 在编译期就知道派生类的具体类型，静态 WndProc 可以直接调用 CMyWindow::ProcessWindowMessage，无需虚函数表查找。第四个参数 BOOL& bHandled 设为 FALSE 表示"我处理过了但仍让 DefWindowProc 兜底"，灵活控制消息流向。

04 WTL 应用框架

WTL 在 ATL 之上扩展出完整的 GUI 框架。CAppModule（继承自 CMessageModule）管理全局消息循环，CFrameWindowImpl 提供带菜单/工具栏/状态栏的框架窗口，CMessageLoop 抽象出可被多线程共享的消息泵。CUpdateUI 实现菜单/工具栏命令的状态自动更新（启用、选中、复选）。

#include <atlbase.h>
#include <atlapp.h>          // CAppModule / CMessageLoop
#include <atlframe.h>        // CFrameWindowImpl
#include <atlctrls.h>        // 控件封装

CAppModule _Module;                           // 全局应用对象

class CMainFrame : public CFrameWindowImpl<CMainFrame>,
                     public CUpdateUI<CMainFrame>,
                     public CMessageFilter,
                     public CIdleHandler
{
public:
    DECLARE_FRAME_WND_CLASS(NULL, IDR_MAINFRAME)   // 关联资源 ID

    virtual BOOL PreTranslateMessage(MSG* pMsg) {
        if (CFrameWindowImpl::PreTranslateMessage(pMsg))
            return TRUE;
        return FALSE;
    }
    virtual BOOL OnIdle() {
        UIUpdateToolBar();              // 空闲时刷新工具栏状态
        return FALSE;
    }

    BEGIN_MSG_MAP(CMainFrame)
        MESSAGE_HANDLER(WM_CREATE, OnCreate)
        COMMAND_ID_HANDLER(ID_FILE_EXIT, OnFileExit)
        CHAIN_MSG_MAP(CUpdateUI<CMainFrame>)   // 转发到 UI 更新映射
        CHAIN_MSG_MAP(CFrameWindowImpl<CMainFrame>)
    END_MSG_MAP()

    BEGIN_UPDATE_UI_MAP(CMainFrame)            // 命令 UI 更新表
        UPDATE_ELEMENT(ID_VIEW_STATUSBAR, UPDUI_MENUPOPUP)
    END_UPDATE_UI_MAP()
};

int WINAPI wWinMain(HINSTANCE hInst, HINSTANCE, PWSTR, int nShow) {
    _Module.Init(NULL, hInst);                  // 初始化模块（含公共控件）

    CMessageLoop theLoop;
    _Module.AddMessageLoop(&theLoop);          // 注册主线程消息循环

    CMainFrame wndMain;
    if (wndMain.CreateEx() == NULL) {
        _Module.Term();
        return 0;
    }
    wndMain.ShowWindow(nShow);
    wndMain.UpdateWindow();

    int nRet = theLoop.Run();                  // 进入消息循环
    _Module.RemoveMessageLoop();
    _Module.Term();
    return nRet;
}
多消息循环设计：CAppModule 用线程局部存储为每个 UI 线程维护独立的 CMessageLoop。这意味着 WTL 天然支持多线程 GUI——只需在新线程里 AddMessageLoop，即可拥有自己的消息泵、PreTranslate 链、OnIdle 链。

05 WTL 消息映射宏

WTL 继承 ATL 的消息映射机制并扩展出更精细的宏。每条映射项是一个静态表项，由 ProcessWindowMessage 在运行时线性扫描匹配。设 bHandled=FALSE 可让消息继续向后查找。

宏用途典型场景
MESSAGE_HANDLER处理原始窗口消息WM_CREATE / WM_PAINT
COMMAND_HANDLER按 (id, code) 处理 WM_COMMAND按钮 BN_CLICKED
COMMAND_ID_HANDLER按 id 处理（忽略 code）菜单项
COMMAND_RANGE_HANDLER处理一段连续 id 范围动态生成的工具栏按钮
NOTIFY_HANDLER按 (id, code) 处理 WM_NOTIFYLVN_ITEMCHANGED
NOTIFY_CODE_HANDLER按通知码处理（任意控件）统一处理 NM_RCLICK
CHAIN_MSG_MAP整链转发到基类/成员复用基类处理
CHAIN_MSG_MAP_MEMBER转发到成员对象组合式窗口
DEFAULT_REFLECTION_HANDLER反射未处理消息给子控件自绘按钮
FORWARD_NOTIFICATIONS把 WM_NOTIFY 转发给父窗口子控件包装类
BEGIN_MSG_MAP(CMyView)
    // 窗口消息
    MESSAGE_HANDLER(WM_SIZE,    OnSize)
    MESSAGE_HANDLER(WM_ERASEBKGND, OnEraseBkgnd)
    // 命令：菜单 + 按钮
    COMMAND_ID_HANDLER(ID_FILE_NEW,    OnFileNew)
    COMMAND_ID_HANDLER(ID_FILE_OPEN,   OnFileOpen)
    COMMAND_RANGE_HANDLER(ID_VIEW_0, ID_VIEW_9, OnView)
    // 控件通知（来自 ListView m_list）
    NOTIFY_HANDLER(IDC_LIST, LVN_ITEMCHANGED, OnListItemChanged)
    NOTIFY_CODE_HANDLER(NM_RCLICK, OnAnyControlRClick)
    // 反射：把未处理消息回送给发件人（自绘控件必需）
    DEFAULT_REFLECTION_HANDLER()
    // 链式：本类未匹配则交基类
    CHAIN_MSG_MAP(CFrameWindowImpl<CMyView>)
END_MSG_MAP()
反射机制：传统 Win32 中，子控件的通知（WM_NOTIFY/WM_COMMAND）只送给父窗口。WTL 的 DEFAULT_REFLECTION_HANDLER 把这些消息"反射"回子控件本身——让自绘按钮的逻辑封装在按钮类内，而不是污染父窗口。这是把 Win32 转向 OO 设计的关键技巧。

06 WTL 控件封装

WTL 把 Win32 标准控件、公共控件、Shell 控件全部封装为薄模板类。所有控件类都派生自 CWindow（继承其 HWND 转换），并添加控件特有的方法。CCommandBarCtrl 是 WTL 标志性控件——它把菜单和工具栏合并为一个 CE 风格的命令栏，常见于 Windows Mobile/嵌入式。

        WS_CHILD | WS_VISIBLE | LVS_REPORT | LVS_SINGLESEL,

子类化（Subclassing）：WTL 通过 CWindowImpl::SubclassWindow(HWND) 把现有窗口的 WndProc 替换为派生类的 ProcessWindowMessage，是给对话框资源中的控件"加 OO 行为"的标准手段。派生类需用 DECLARE_WND_SUPERCLASS 或直接继承相应控件基类。

07 WTL 对话框

WTL 对话框由 CDialogImpl<T> 模板提供，每个派生类需声明一个 enum { IDD = IDD_XXX } 关联对话框资源。CSimpleDialog 是无消息映射的最简包装（一行代码弹消息框式对话框）。数据交换通过 CWinDataExchange 模板与 DDX_*/DDV_* 宏实现，API 与 MFC 高度一致但体积小得多。

class CSettingsDlg :
    public CDialogImpl<CSettingsDlg>,
    public CWinDataExchange<CSettingsDlg>
{
public:
    enum { IDD = IDD_SETTINGS };

    CString    m_name;
    int        m_port;
    CButton    m_ok;             // 控件包装（DDX_CONTROL 绑定）

    BEGIN_DDX_MAP(CSettingsDlg)
        DDX_TEXT(IDC_NAME, m_name)
        DDX_INT(IDC_PORT, m_port)
        DDX_CONTROL(IDOK, m_ok)
        DDV_MAXCHAR(m_name, 32)
        DDV_MINMAX_INT(m_port, 1, 65535)
    END_DDX_MAP()

    BEGIN_MSG_MAP(CSettingsDlg)
        MESSAGE_HANDLER(WM_INITDIALOG, OnInitDialog)
        COMMAND_ID_HANDLER(IDOK, OnOK)
        COMMAND_ID_HANDLER(IDCANCEL, OnCancel)
    END_MSG_MAP()

    LRESULT OnInitDialog(UINT, WPARAM, LPARAM, BOOL&) {
        DoDataExchange(FALSE);          // 成员 → 控件（FALSE=写入界面）
        return TRUE;                  // 设默认焦点为 IDC_NAME
    }
    LRESULT OnOK(UINT, WPARAM, LPARAM, BOOL& bHandled) {
        if (DoDataExchange(TRUE)) {      // 控件 → 成员 + 校验
            EndDialog(IDOK);            // 关闭模态对话框
        }
        bHandled = TRUE;
        return 0;
    }
};

// 使用：
CSettingsDlg dlg;
if (dlg.DoModal() == IDOK) {
    // dlg.m_name / dlg.m_port 已是用户输入
}

// 非模态：Create(hwndParent) 而非 DoModal
// 简单对话框（仅需要 IDOK/IDCANCEL）：CSimpleDialog<IDD_ABOUT>().DoModal();

08 WTL GDI 绘图

WTL 的 GDI 封装与 MFC 概念相似但更精简。CDC 包装 HDC，CGdiObject 是 GDI 对象基类，CPen/CBrush/CFont/CBitmap/CRegion 派生自它。所有这些类都是栈对象，析构时自动 DeleteObject。CClientDC、CWindowDC、CPaintDC 是作用域守卫——构造时获取 DC，析构时释放。

// 视图窗口的 OnPaint 绘制
LRESULT CMyView::OnPaint(UINT, WPARAM, LPARAM, BOOL& bHandled) {
    CPaintDC dc(m_hWnd);                    // 构造=BeginPaint，析构=EndPaint

    // 1. 创建 GDI 对象（栈对象）
    CPen   pen(PS_SOLID, 2, RGB(220,60,60));
    CBrush brush(RGB(80,160,240));
    CFont font;
    font.CreatePointFont(140, _T("Segoe UI"));

    // 2. 选入 DC，保存旧对象
    HPEN   oldPen   = dc.SelectPen(pen);
    HBRUSH oldBrush = dc.SelectBrush(brush);
    HFONT oldFont  = dc.SelectFont(font);

    // 3. 绘图（CDC 直接转发 Win32 API）
    dc.SetBkMode(TRANSPARENT);
    dc.Rectangle(20, 20, 220, 120);
    dc.MoveTo(20, 20);
    dc.LineTo(220, 120);
    dc.TextOut(30, 40, _T("WTL GDI Demo"));

    // 4. 恢复旧对象（必须！）
    dc.SelectPen(oldPen);
    dc.SelectBrush(oldBrush);
    dc.SelectFont(oldFont);
    return 0;
}

// 作用域 DC 守卫
void CMyView::DrawCross(HWND hwnd) {
    CClientDC dc(hwnd);                   // 客户区 DC，析构时 ReleaseDC
    CRect rc; CWindow(hwnd).GetClientRect(&rc);
    dc.MoveTo(rc.left,  rc.top);     dc.LineTo(rc.right, rc.bottom);
    dc.MoveTo(rc.right, rc.top);     dc.LineTo(rc.left,  rc.bottom);
}
资源卫生：WTL 的 CPen 等栈对象在析构时调用 DeleteObject，但如果此时它仍被 DC 选中，删除会失败并造成句柄泄漏。务必遵守"选入→保存旧→绘制→恢复旧→（让新对象析构）"的顺序。SelectPen/SelectBrush 等类型安全版本比 MFC 的 SelectObject 更难出错。

09 GDI+ 与现代特性

WTL 10 与 GDI+ 完全兼容——只需在 stdafx.h 包含 <gdiplus.h> 并链接 gdiplus.lib，即可在 WTL 窗口中混合使用 GDI 与 GDI+。UxTheme API 让控件呈现 Windows 视觉主题（Aero、Fluent）。WTL 也积极跟进 C++ 标准：C++11 的 override/final、C++14 的 make_unique、C++17 的 string_view、C++20 的 concepts 均可无缝集成。

#include <atlbase.h>
#include <gdiplus.h>
#pragma comment(lib, "gdiplus.lib")
using namespace Gdiplus;

struct GdiplusInit {
    ULONG_PTR token;
    GdiplusInit() { GdiplusStartupInput si; GdiplusStartup(&token, &si, NULL); }
    ~GdiplusInit() { GdiplusShutdown(token); }
};

LRESULT CMyView::OnPaint(UINT, WPARAM, LPARAM, BOOL&) {
    CPaintDC dc(m_hWnd);
    Graphics gfx(dc);                       // GDI+ 图形对象包装 HDC
    gfx.SetSmoothingMode(SmoothingModeAntiAlias);

    LinearGradientBrush brush(
        Rect(10,10,300,80),
        Color(255,107,63,160),                 // WTL 紫色
        Color(255,45,27,105),
        LinearGradientModeForwardDiagonal);
    gfx.FillRectangle(&brush, 10, 10, 300, 80);

    SolidBrush textBrush(Color(255,255,255,255));
    Font font(L"Segoe UI", 16);
    gfx.DrawString(L"GDI+ in WTL", -1, &font, PointF(20,25), &textBrush);
    return 0;
}

10 完整 SDI 示例

下面是一个最小可运行的 WTL SDI 应用骨架：应用对象 + 主框架 + 自定义视图 + 消息映射 + 菜单命令 + GDI 绘图。它体现了 WTL 各组件如何协作——典型工程结构如此，省略资源文件（.rc）。

// stdafx.h
#pragma once
#include <atlbase.h>
#include <atlstr.h>          // CString
#include <atlapp.h>          // CAppModule
#include <atlwin.h>          // CWindowImpl
#include <atlframe.h>        // CFrameWindowImpl
#include <atlctrls.h>        // 控件
#include <atlgdi.h>          // GDI
#include <atlmisc.h>         // CRect / CString
extern CAppModule _Module;

class CMainFrame : public CFrameWindowImpl<CMainFrame>,
                     public CUpdateUI<CMainFrame>,
                     public CMessageFilter {
public:
    DECLARE_FRAME_WND_CLASS(NULL, IDR_MAINFRAME)
    CMyView m_view;

    virtual BOOL PreTranslateMessage(MSG* pMsg) override {
        if (m_view.IsWindow() && m_view.PreTranslateMessage(pMsg))
            return TRUE;
        return CFrameWindowImpl::PreTranslateMessage(pMsg);
    }

    BEGIN_MSG_MAP(CMainFrame)
        MESSAGE_HANDLER(WM_CREATE,      OnCreate)
        COMMAND_ID_HANDLER(ID_FILE_EXIT,  OnFileExit)
        COMMAND_ID_HANDLER(ID_APP_ABOUT, OnAbout)
        CHAIN_MSG_MAP(CUpdateUI<CMainFrame>)
        CHAIN_MSG_MAP(CFrameWindowImpl<CMainFrame>)
    END_MSG_MAP()

    LRESULT OnCreate(UINT, WPARAM, LPARAM, BOOL& bHandled) {
        m_hWndClient = m_view.Create(m_hWnd, rcDefault, NULL,
            WS_CHILD | WS_VISIBLE | SS_NOTIFY);
        m_view.m_text = _T("Hello WTL!");
        bHandled = FALSE;                          // 让框架继续默认创建
        return 0;
    }

        PostMessage(WM_CLOSE);
        return 0;
    }
    LRESULT OnAbout(UINT, WPARAM, LPARAM, BOOL&) {

        dlg.DoModal();

int WINAPI wWinMain(HINSTANCE hInst, HINSTANCE, PWSTR, int nShow) {
    AtlInitCommonControls(ICC_BAR_CLASSES | ICC_LISTVIEW_CLASSES);
    _Module.Init(NULL, hInst);

    CMessageLoop theLoop;
    _Module.AddMessageLoop(&theLoop);

    CMainFrame wndMain;
    if (wndMain.CreateEx() == NULL) {
        _Module.Term();
        return 0;
    }
    wndMain.ShowWindow(nShow);
    wndMain.UpdateWindow();

    int nRet = theLoop.Run();
    _Module.RemoveMessageLoop();
    _Module.Term();
    return nRet;
}

11 对比 MFC

ATL/WTL 与 MFC 都封装 Win32，但设计哲学截然不同。下表从运行时依赖、体积、性能、开发体验、控件库、适用场景等维度全面对比。这决定了为何轻量型项目仍偏爱 WTL——尤其是在 Shell 扩展、嵌入式、需要原生控件外观的小工具中。

维度ATL/WTLMFC
运行时 DLL无（静态/动态 CRT 即可）需 mfc140u.dll（可静态但体积大）
分发体积典型 100~500 KB1~5 MB（含 MFC 运行时）
虚函数开销几乎为零（模板多态）大量虚函数（CObject/CWnd 层次）
启动速度极快（无全局对象初始化）有 theApp + 模块初始化
消息映射静态表 + CRTP静态表 + CWnd::WindowProc
开发效率需手写更多样板向导生成、类最多、文档最全
VS 向导需安装 WTL Wizard 插件VS 内置成熟向导
控件库原生 Win32 + 自绘需自己写含 CListCtrl/CTreeView 等丰富封装
Doc/View无（需手工组织）框架内建 Doc/View/Template
序列化无（用 STL/JSON）CArchive 强大但专有
COM 支持顶级（ATL 即为 COM 而生）有但不如 ATL 精致
许可证WTL 是开源（MS-PL/CPL）VS 商业授权
跨版本纯头文件，VS 2008~2022 通吃绑定 VS 版本的 DLL
选 WTL 当
  • 分发体积、启动性能是硬指标
  • 需要同时做 COM 服务器 + UI
  • Shell 扩展、IE/BHO、命名空间扩展
  • 嵌入式 Windows、瘦客户端
  • 团队愿接受更"裸"的 API
选 MFC 当
  • 维护大量 MFC 遗留代码
  • 重度依赖 Doc/View、Ribbon、文档模板
  • 需要 CArchive 二进制序列化生态
  • 用 VS 向导快速产出复杂界面
  • 商业控件（BCGSoft等）只支持 MFC
性能数据：在同等"空白 SDI 应用"基准下，WTL 编译产物通常比 MFC 小一个数量级（几百 KB vs 几 MB），冷启动延迟低 30%~50%。这是因为 MFC 在 CRT 启动阶段要构造大量全局对象（进程内对象映射、资源句柄表等），而 WTL 几乎为零开销。
陷阱：WTL 是社区维护项目（虽源自微软员工），微软官方不再为其背书或更新文档。新手遇到问题更多依赖 SourceForge 论坛、CodeProject 文章、经典书籍《ATL Internals》。商业项目评估时需把"文档稀缺、生态较小"计入风险。

atlctrls.h 所有 Win32 控件的薄封装，是日常开发引用最频繁的头文件。
atlgdi.h GDI 对象与作用域 DC 守卫，绘图必备。
