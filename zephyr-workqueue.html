01 工作队列概览

工作队列（Work Queue）是 Zephyr 中最重要的延迟执行框架，用于将任务从 ISR 或高优先级上下文延迟到线程上下文执行。

核心价值

场景问题workqueue 解决
ISR 中需要耗时操作ISR 不能阻塞提交 work 到队列，ISR 返回后执行
定时重试需要定时器+线程k_work_schedule 一步完成
解耦生产者/消费者直接调用阻塞work 异步执行
批量处理逐个处理开销大coalesce 合并多次请求

工作项类型

类型宏说明
静态工作K_WORK_DEFINE编译时定义，无延迟
可延迟工作K_WORK_DELAYABLE_DEFINE支持 schedule 延迟执行
用户工作k_work_user_init用户态可用

02 k_work 基础

数据结构

// include/zephyr/kernel.h
struct k_work {
    struct k_work_q *queue;    // 关联的工作队列
    k_work_handler_t handler;  // 工作处理函数
    atomic_t flags;            // 状态标志
    // 内部链表节点...
};

基本用法

// 1. 定义工作处理函数
static void my_work_handler(struct k_work *work)
{
    // 在线程上下文中执行（非 ISR）
    printk("工作执行中...\n");
    process_sensor_data();
}

// 2. 定义工作项
K_WORK_DEFINE(my_work, my_work_handler);

// 3. 提交工作
k_work_submit(&my_work);

// 4. 提交到特定工作队列
k_work_submit_to_queue(&my_work_q, &my_work);

// 5. ISR 中提交
void my_isr(void *arg, void *arg2)
{
    k_work_submit(&my_work);  // ISR 安全
}

03 可延迟工作

// 可延迟工作（支持 schedule）
K_WORK_DELAYABLE_DEFINE(my_delayable_work, my_work_handler);

// 延迟 100ms 执行
k_work_schedule(&my_delayable_work, K_MSEC(100));

// 周期性执行（每 500ms）
k_work_reschedule(&my_delayable_work, K_MSEC(500));

// 取消调度
k_work_cancel_delayable(&my_delayable_work);

// 典型场景：传感器定期采样
static void sensor_read_handler(struct k_work *work)
{
    struct k_work_delayable *dwork = k_work_delayable_from_work(work);

    // 读取传感器
    read_sensor();

    // 重新调度（周期性）
    k_work_reschedule(dwork, K_MSEC(1000));
}

K_WORK_DELAYABLE_DEFINE(sensor_work, sensor_read_handler);

// 启动周期采样
k_work_schedule(&sensor_work, K_MSEC(1000));

04 工作队列线程

// 工作队列结构
struct k_work_q {
    struct k_thread thread;   // 工作线程
    sys_slist_t pending;      // 待处理工作链表
};

// 静态定义工作队列
K_THREAD_STACK_DEFINE(my_work_stack, 2048);
struct k_work_q my_work_q;

// 初始化工作队列
k_work_queue_init(&my_work_q);
k_work_queue_start(&my_work_q, my_work_stack, 2048,
                   K_PRIO_PREEMPT(5), NULL);

// 默认工作队列
// Zephyr 内置 system_workq（CONFIG_SYSTEM_WORKQUEUE_PRIORITY）
k_work_submit_to_queue(&system_workq, &my_work);

// 工作队列配置
// CONFIG_SYSTEM_WORKQUEUE_STACK_SIZE=2048
// CONFIG_SYSTEM_WORKQUEUE_PRIORITY=-1  ← 协作式

多工作队列

// 可定义多个工作队列，不同优先级
// 高优先级队列：紧急任务
// 低优先级队列：后台任务

K_THREAD_STACK_DEFINE(low_pri_stack, 2048);
struct k_work_q low_pri_work_q;

k_work_queue_start(&low_pri_work_q, low_pri_stack, 2048,
                   K_PRIO_PREEMPT(10), NULL);

05 Flush 与 Cancel

// Flush：等待工作完成
// 阻塞直到当前提交的工作执行完毕
k_work_flush(&my_work, &flusher, K_FOREVER);

// 或非阻塞检查
int ret = k_work_flush(&my_work, &flusher, K_NO_WAIT);

// Cancel：取消工作
// 如果工作正在执行，等待执行完成
// 如果工作在队列中等待，移除
k_work_cancel(&my_work);

// Cancel + 等待完成
k_work_cancel_sync(&my_work, &cancel_sync);

// 延迟工作取消
k_work_cancel_delayable(&my_delayable_work);
k_work_cancel_delayable_sync(&my_delayable_work, &cancel_sync);

// 典型场景：系统关闭前确保所有工作完成
void system_shutdown(void)
{
    k_work_cancel_delayable_sync(&sensor_work, &cancel_sync);
    k_work_cancel_sync(&data_work, &cancel_sync);
    printk("所有工作已完成\n");
}

07 ISR 中使用

// 工作队列是 ISR 延迟处理的首选机制

// ISR：只提交工作，不做耗时操作
void button_isr(const struct device *dev, struct gpio_callback *cb, uint32_t pins)
{
    k_work_submit(&button_work);
}

// 工作处理：在线程上下文执行
static void button_work_handler(struct k_work *work)
{
    // 可以调用阻塞 API
    k_sem_give(&button_sem);
    printk("按钮按下\n");
}

K_WORK_DEFINE(button_work, button_work_handler);

// 可延迟版本（去抖动）
K_WORK_DELAYABLE_DEFINE(debounce_work, debounce_handler);

void button_isr(...)
{
    // 每次中断重新调度（去抖动）
    k_work_reschedule(&debounce_work, K_MSEC(50));
}
