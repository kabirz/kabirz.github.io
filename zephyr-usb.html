03 USB 类驱动

CDC-ACM（虚拟串口）

// CDC-ACM 使用
const struct device *uart_dev = DEVICE_DT_GET(DT_NODELABEL(zephyr_cdc_acm_uart));

// 配置波特率
struct uart_config cfg = { .baudrate = 115200 };
uart_configure(uart_dev, &cfg);

// 发送数据
uart_tx(uart_dev, "Hello USB\n", 10, K_FOREVER);

// 接收数据
uint8_t rx_buf[64];
uart_rx_enable(uart_dev, rx_buf, sizeof(rx_buf), K_FOREVER);
