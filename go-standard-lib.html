01 IO 接口体系

Go 的 io 包是标准库中最精巧的设计之一。它通过一系列小接口的组合，构建了整个 I/O 抽象层。其核心设计哲学是：通过组合小接口来构建复杂功能。

核心接口

接口方法说明
io.ReaderRead(p []byte) (n int, err error)从数据源读取到缓冲区
io.WriterWrite(p []byte) (n int, err error)从缓冲区写入到目标
io.CloserClose() error关闭资源
io.SeekerSeek(offset int64, whence int) (int64, error)移动读写位置
io.ReaderAtReadAt(p []byte, off int64) (n int, err error)从指定偏移量读取
io.WriterAtWriteAt(p []byte, off int64) (n int, err error)从指定偏移量写入

组合接口

// 标准库预定义的组合接口
type ReadWriter interface {
    Reader
    Writer
}

type ReadWriteCloser interface {
    Reader
    Writer
    Closer
}

type ReadWriteSeeker interface {
    Reader
    Writer
    Seeker
}

// 这些接口通过嵌入方式组合，零开销
// 任何实现了 Reader 和 Writer 的类型自动实现 ReadWriter

io.Copy 的实现原理

io.Copy 是 I/O 操作中最常用的函数，它将数据从 Reader 复制到 Writer，直到 EOF。其内部有值得一提的优化路径：

func Copy(dst Writer, src Reader) (int64, error) {
    // 优化路径 1: 如果 src 实现了 WriterTo，直接调用
    if wt, ok := src.(WriterTo); ok {
        return wt.WriteTo(dst)
    }
    
    // 优化路径 2: 如果 dst 实现了 ReaderFrom，直接调用
    if rt, ok := dst.(ReaderFrom); ok {
        return rt.ReadFrom(src)
    }
    
    // 默认路径: 使用 32KB 缓冲区复制
    buf := make([]byte, 32*1024)
    return copyBuffer(dst, src, buf)
}

// io.CopyN 限制复制的字节数
// io.CopyBuffer 允许自定义缓冲区大小

io.Pipe 管道

io.Pipe 创建一对 Reader/Writer，写入 Writer 的数据可以立即从 Reader 读出。它是内存中的管道，常用于连接不同的 I/O 处理链。

func pipeExample() {
    pr, pw := io.Pipe()
    
    // 写入端在 goroutine 中进行
    go func() {
        defer pw.Close()
        json.NewEncoder(pw).Encode(map[string]string{
            "message": "hello",
        })
    }()
    
    // 读取端同步读取
    var result map[string]string
    json.NewDecoder(pr).Decode(&result)
    fmt.Println(result)  // map[message:hello]
}

// io.Pipe 适用于:
// - 将同步 I/O 转为异步管道
// - 连接压缩/加密/编码链
// - 测试中的 mock 数据流

实用工具函数

// io.ReadAll - 读取所有数据到内存
data, err := io.ReadAll(resp.Body)

// io.ReadFull - 精确读取指定字节数
buf := make([]byte, 8)
_, err := io.ReadFull(r, buf)  // 要么读满 8 字节，要么返回错误

// io.MultiReader - 串联多个 Reader
r := io.MultiReader(bytes.NewReader([]byte("part1")),
                    bytes.NewReader([]byte("part2")))

// io.MultiWriter - 同时写入多个 Writer
var buf1, buf2 bytes.Buffer
w := io.MultiWriter(&buf1, &buf2)
w.Write([]byte("data"))  // 同时写入 buf1 和 buf2

// io.NopCloser - 为没有 Close 方法的 Reader 添加空关闭
rc := io.NopCloser(bytes.NewReader(data))
设计启示：io 包展示了 Go 接口设计的精髓：小接口、大组合。os.File、bytes.Buffer、strings.Reader、net.Conn 都实现了这些接口，使它们可以互换使用。

02 Context 包

Context 接口

context.Context 是 Go 中跨 API 和 goroutine 传递超时、取消信号和请求范围数据的标准方式。它自 Go 1.7 起成为标准库的一部分，现已成为 Go 并发编程的基础设施。

type Context interface {
    // Deadline 返回 context 被取消的时间
    Deadline() (deadline time.Time, ok bool)
    
    // Done 返回一个 channel，context 被取消时关闭
    Done() <-chan struct{}
    
    // Err 返回 context 被取消的原因
    Err() error
    
    // Value 返回 key 关联的值
    Value(key any) any
}

// 两个根 context 工厂函数
ctx := context.Background()   // 根 context，从未被取消
ctx := context.TODO()         // 占位符，表示将来要使用 context

取消链传播

Context 最强大的特性是取消信号的自动传播。当父 context 被取消，所有派生的子 context 都会被级联取消。

func cancellationChain() {
    ctx := context.Background()
    
    // 创建可取消的 context
    ctxA, cancelA := context.WithCancel(ctx)
    defer cancelA()
    
    // 从 ctxA 派生 - 取消链: 取消 A → 取消 B
    ctxB, cancelB := context.WithCancel(ctxA)
    defer cancelB()
    
    // 从 ctxB 派生 - 取消链: 取消 A → 取消 B → 取消 C
    ctxC, cancelC := context.WithCancel(ctxB)
    defer cancelC()
    
    // 取消 A → B 和 C 同时收到 Done 信号
    cancelA()
    
    println(ctxA.Err())  // context.Canceled
    println(ctxB.Err())  // context.Canceled
    println(ctxC.Err())  // context.Canceled
}

// 内部实现: 每个可取消的 context 维护一个 children 列表
// 当 cancel() 被调用时，遍历 children 并逐个取消

WithDeadline 与 WithTimeout

// 绝对时间截止
deadline := time.Now().Add(5 * time.Second)
ctx, cancel := context.WithDeadline(context.Background(), deadline)
defer cancel()

// 相对时间截止（自动转换为 Deadline）
ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
defer cancel()

// 监听取消
select {
case <-ctx.Done():
    switch {
    case errors.Is(ctx.Err(), context.DeadlineExceeded):
        log.Println("timeout")
    case errors.Is(ctx.Err(), context.Canceled):
        log.Println("cancelled")
    }
case result <-ch:
    // 正常处理结果
}

// WithTimeoutCause (Go 1.21+) - 自定义超时错误原因
ctx, cancel := context.WithTimeoutCause(
    context.Background(),
    5*time.Second,
    fmt.Errorf("database query timed out"),
)
defer cancel()
// ctx.Err() 返回时会被 context.Cause(ctx) 获取到自定义错误

WithValue 与键类型

// 使用自定义类型作为 key 避免冲突
type ctxKey string

const (
    UserIDKey    ctxKey = "user_id"
    TraceIDKey   ctxKey = "trace_id"
    RequestIDKey ctxKey = "request_id"
)

func WithUserID(ctx context.Context, uid string) context.Context {
    return context.WithValue(ctx, UserIDKey, uid)
}

func UserIDFromContext(ctx context.Context) (string, bool) {
    uid, ok := ctx.Value(UserIDKey).(string)
    return uid, ok
}

// context 的 Value 链式查找
// WithValue 返回的 context 存储了 (key, value, parent) 三元组
// Value() 方法沿着 parent 链向上查找，直到找到匹配的 key
context 使用守则：不要将 context 存储在结构体中。context 应作为函数的第一个参数传递。仅传递请求范围内的数据，不要使用 context 传递可选参数。

03 Encoding/JSON

Marshal 与 Unmarshal

encoding/json 是 Go 最常用的序列化库。它通过结构体标签（struct tag）控制 JSON 字段的映射行为。

type User struct {
    ID        int       `json:"id"`
    Name      string    `json:"name"`
    Email     string    `json:"email,omitempty"`
    CreatedAt time.Time `json:"created_at"`
    Password  string    `json:"-"`       // 忽略此字段
}

// 序列化
user := User{ID: 1, Name: "Alice", CreatedAt: time.Now()}
data, err := json.Marshal(user)   // 返回缩进格式
data, err := json.MarshalIndent(user, "", "  ")  // 带缩进

// 反序列化
var u User
err := json.Unmarshal(data, &u)

自定义 MarshalJSON / UnmarshalJSON

当默认的 JSON 编码行为不满足需求时，可以实现自定义的序列化方法。

type DateTime time.Time

func (dt DateTime) MarshalJSON() ([]byte, error) {
    t := time.Time(dt)
    formatted := t.Format("2006-01-02 15:04:05")
    return json.Marshal(formatted)
}

func (dt *DateTime) UnmarshalJSON(data []byte) error {
    var s string
    if err := json.Unmarshal(data, &s); err != nil {
        return err
    }
    t, err := time.Parse("2006-01-02 15:04:05", s)
    if err != nil {
        return err
    }
    *dt = DateTime(t)
    return nil
}

// 动态 JSON 编码 - 使用 map
func (u User) MarshalJSON() ([]byte, error) {
    type Alias User  // 避免递归
    return json.Marshal(struct{
        Alias
        Age int `json:"age"`  // 添加额外字段
    }{
        Alias: Alias(u),
        Age:   25,
    })
}

流式编码与解码

// Decoder - 从 Reader 流式读取 JSON
file, _ := os.Open("large.json")
defer file.Close()

decoder := json.NewDecoder(file)
for {
    var item Item
    err := decoder.Decode(&item)
    if errors.Is(err, io.EOF) {
        break
    }
    if err != nil {
        log.Fatal(err)
    }
    process(item)
}

// Encoder - 流式写入 JSON 到 Writer
w := os.Stdout
encoder := json.NewEncoder(w)
encoder.SetIndent("", "  ")
encoder.Encode(data)

// 流式编码可用于逐行输出的 JSON Lines 格式
for _, item := range items {
    json.NewEncoder(w).Encode(item)
}

Decoder 的高级配置

decoder := json.NewDecoder(reader)

// 拒绝未知字段（默认忽略）
decoder.DisallowUnknownFields()

// 不要求大数字使用 Number 类型
decoder.UseNumber()

// 限制 JSON 输入大小
decoder = json.NewDecoder(io.LimitReader(reader, 10<<20))  // 10MB 限制

// RawMessage - 延迟解析
type Response struct {
    Code int             `json:"code"`
    Data json.RawMessage  `json:"data"`  // 延迟到确定类型后再解析
}

// 先解析外层
var resp Response
json.Unmarshal(data, &resp)

// 根据条件再解析 Data
if resp.Code == 200 {
    var user User
    json.Unmarshal(resp.Data, &user)
}
性能提示：对大型 JSON 使用 Decoder 而非 Unmarshal。预分配结构体可以减少反射开销。使用 json.RawMessage 实现延迟解析可以避免不必要的类型映射。

04 Testing 表格驱动测试

基本表格测试

表格驱动测试（Table-Driven Tests）是 Go 社区最推崇的测试风格。它通过定义输入/期望值的数据表来减少重复代码。

// 被测试函数
func Divide(a, b int) (int, error) {
    if b == 0 {
        return 0, fmt.Errorf("division by zero")
    }
    return a / b, nil
}

// 表格驱动测试
func TestDivide(t *testing.T) {
    tests := []struct {
        name    string
        a, b    int
        want    int
        wantErr bool
    }{
        {name: "positive", a: 10, b: 2, want: 5},
        {name: "negative", a: -6, b: 3, want: -2},
        {name: "divide by one", a: 42, b: 1, want: 42},
        {name: "zero numerator", a: 0, b: 5, want: 0},
        {name: "division by zero", a: 10, b: 0, wantErr: true},
    }
    
    for _, tt := range tests {
        t.Run(tt.name, func(t *testing.T) {
            got, err := Divide(tt.a, tt.b)
            if (err != nil) != tt.wantErr {
                t.Errorf("Divide(%d,%d) error = %v, wantErr %v",
                    tt.a, tt.b, err, tt.wantErr)
                return
            }
            if got != tt.want {
                t.Errorf("Divide(%d,%d) = %d, want %d",
                    tt.a, tt.b, got, tt.want)
            }
        })
    }
}

子测试（Subtests）

t.Run() 创建的子测试具有独立的生命周期，可以单独运行。

// 只运行特定子测试
// $ go test -run TestDivide/positive
// $ go test -run TestDivide/negative

// 子测试支持并行
for _, tt := range tests {
    tt := tt  // 捕获循环变量
    t.Run(tt.name, func(t *testing.T) {
        t.Parallel()  // 并行执行
        got, err := Divide(tt.a, tt.b)
        // ... 断言
    })
}

// setup/teardown 每个子测试
t.Run("group", func(t *testing.T) {
    setup()   // 组级别前置
    defer teardown()  // 组级别后置
    
    t.Run("sub1", func(t *testing.T) { /* ... */ })
    t.Run("sub2", func(t *testing.T) { /* ... */ })
})

t.Helper 与测试辅助函数

// t.Helper() 标记辅助函数，错误报告时跳过该帧
func assertEqual(t *testing.T, got, want interface{}) {
    t.Helper()  // 错误行号会指向调用者而非本函数
    if got != want {
        t.Errorf("got %v, want %v", got, want)
    }
}

func assertNoError(t *testing.T, err error) {
    t.Helper()
    if err != nil {
        t.Fatalf("unexpected error: %v", err)
    }
}

func TestWithHelper(t *testing.T) {
    result, err := Divide(10, 2)
    assertNoError(t, err)        // 错误行号指向这里，而非 assertNoError 内部
    assertEqual(t, result, 5)  // 同上
}

基准测试

func BenchmarkDivide(b *testing.B) {

05 Testing 进阶

Fuzzing 测试

Go 1.18 引入了原生 fuzzing 支持。Fuzzing 通过自动生成随机输入来发现边界条件和安全漏洞。

func FuzzReverse(f *testing.F) {
    // 添加种子语料库
    testcases := []string{"Hello, world", " " , "!12345"}
    for _, tc := range testcases {
        f.Add(tc)  // 种子输入
    }
    
    f.Fuzz(func(t *testing.T, orig string) {
        rev := Reverse(orig)
        doubleRev := Reverse(rev)
        if orig != doubleRev {
            t.Errorf("Before: %q, after: %q", orig, doubleRev)
        }
        if utf8.ValidString(orig) && !utf8.ValidString(rev) {
            t.Errorf("Reverse produced invalid UTF-8: %q", rev)
        }
    })
}

// 运行 fuzzing:
// $ go test -fuzz=FuzzReverse
// $ go test -fuzz=FuzzReverse -fuzztime=30s

覆盖率

// 生成覆盖率报告
// $ go test -cover -coverprofile=coverage.out
// $ go tool cover -html=coverage.out -o coverage.html

// 覆盖率模式
// -covermode=set     每条语句是否执行
// -covermode=count   每条语句执行次数
// -covermode=atomic  并发安全的计数（racer 测试必须）

// 合并多个包的覆盖率
// $ go test -coverprofile=coverage.out ./...

// 仅运行集成测试
// $ go test -tags=integration ./...

Test Main 与测试生命周期

func TestMain(m *testing.M) {
    // 测试前的全局设置
    setup()
    
    // 运行所有测试
    code := m.Run()
    
    // 测试后的全局清理
    teardown()
    
    // 退出（必须）
    os.Exit(code)
}

// TestMain 常用于:
// - 初始化数据库连接
// - 启动测试 HTTP 服务
// - 创建临时文件目录
// - 设置环境变量
测试最佳实践：使用 go test -race 检测数据竞争。在 CI 中使用 go test -count=1 禁用测试缓存。对重要的包设定覆盖率阈值，低于阈值则 CI 失败。

06 Embed 嵌入

go:embed 基础用法

自 Go 1.16 起，//go:embed 指令允许在编译时将静态文件嵌入到 Go 二进制文件中。这消除了运行时将配置文件、模板等与二进制文件一起分发的问题。

import "embed"

// 嵌入单个文件
//go:embed config.yaml
var configYAML string  // 以字符串形式嵌入

//go:embed logo.png
var logoBytes []byte    // 以字节切片形式嵌入

//go:embed templates/index.html
var indexTemplate embed.FS  // 以文件系统形式嵌入（支持多个文件）

func main() {
    fmt.Println(configYAML)
    
    data, _ := indexTemplate.ReadFile("templates/index.html")
    fmt.Println(string(data))
}

模式匹配嵌入

// 嵌入整个目录
//go:embed static/*
var staticFS embed.FS

// 嵌入子目录
//go:embed templates/**/*.html
var templatesFS embed.FS

// 多个 embed 指令
//go:embed config.yaml
//go:embed secrets/*.enc
//go:embed migrations/*.sql
var appFS embed.FS

// 嵌入后的文件系统是只读的
func embedExample() {
    // 列出嵌入的文件
    entries, _ := appFS.ReadDir(".")
    for _, e := range entries {
        fmt.Println(e.Name())
    }
    
    // 读取子目录文件
    data, _ := appFS.ReadFile("migrations/001_init.sql")
    fmt.Println(string(data))
    
    // 注意: embed.FS 不支持 Walk 和绝对路径
    // 路径分割符始终是 /（即使是 Windows）
}

embed 的实际应用场景

// 场景 1: 嵌入 HTTP 静态文件
//go:embed static
var staticFiles embed.FS

func main() {
    mux := http.NewServeMux()
    mux.Handle("/static/", http.FileServer(
        http.FS(staticFiles),
    ))
    http.ListenAndServe(":8080", mux)
}

// 场景 2: 嵌入 SQL 迁移文件
//go:embed migrations/*.sql
var migrations embed.FS

func runMigrations(db *sql.DB) {
    entries, _ := migrations.ReadDir("migrations")
    for _, entry := range entries {
        data, _ := migrations.ReadFile("migrations/" + entry.Name())
        db.Exec(string(data))
    }
}

// 场景 3: 嵌入模板
//go:embed templates
var templateFS embed.FS

func renderTemplate(name string, data interface{}) (string, error) {
    tmpl := template.Must(template.ParseFS(templateFS,
        "templates/base.html", "templates/"+name))
    var buf strings.Builder
    err := tmpl.Execute(&buf, data)
    return buf.String(), err
}
约束条件：//go:embed 指令只能出现在包级别变量声明之上。嵌入路径不能包含 .. 或指向包目录之外。嵌入的文件必须在执行 go build 时存在。嵌入的 FS 是只读的，无法在运行时修改。

07 构建约束与条件编译

构建标签（Build Tags）

构建约束（Build Constraints）通过 //go:build 指令控制特定文件在哪些平台或条件下被编译。

// 仅在 Linux 平台编译
//go:build linux

package main

func platformInfo() string {
    return "Running on Linux"
}

// 文件: info_windows.go
//go:build windows
package main
func platformInfo() string {
    return "Running on Windows"
}

// 文件: info_darwin.go
//go:build darwin
package main
func platformInfo() string {
    return "Running on macOS"
}

构建标签表达式语法

表达式含义
//go:build linux仅在 Linux 上编译
//go:build !linux不在 Linux 上编译
//go:build linux && amd64仅在 Linux amd64 上编译
//go:build linux || darwin在 Linux 或 macOS 上编译
//go:build ignore永远不编译该文件
//go:build integration仅当指定 -tags=integration 时编译

文件名约定

Go 还支持通过文件名后缀自动推断构建约束，这种方式不需要显式的 //go:build 指令。

// 文件名约定（在 _test.go 之前匹配）
//
// *\_GOOS\_GOARCH  - 指定 OS 和架构
// *\_GOOS         - 指定 OS
// *\_GOARCH       - 指定架构
//
// 示例:
//   server_linux.go        - 仅在 Linux 编译
//   server_windows.go      - 仅在 Windows 编译
//   server_linux_amd64.go  - 仅在 Linux amd64 编译
//   server_test.go         - 不参与构建约束匹配

// 这些文件不需要 //go:build 指令
// 编译器自动从文件名推断构建约束

// 推荐使用文件名约定而非显式 build tag，因为:
// 1. 更简洁，不需要额外指令
// 2. 更直观，文件名一目了然
// 3. IDE 和工具支持更好

package util
func DefaultPath() string { return "/var/lib/data" }

// 文件: util_windows.go
//go:build windows
package util
import "os"
func DefaultPath() string {
    return os.Getenv("APPDATA") + "\\MyApp"
}

// 文件: util_darwin.go
//go:build darwin
package util
func DefaultPath() string { return "/usr/local/var/data" }

// 文件: util_other.go
//go:build !linux && !windows && !darwin
