04 完整示例

// ===== FreeRTOS MPU 完整示例 =====

#include "FreeRTOS.h"
#include "task.h"

// MPU 配置
// configENABLE_MPU = 1
// configENABLE_ACCESS_CONTROL_LIST = 1

// 特权任务（可访问所有资源）
void vPrivilegedTask( void * pvParameters )
{
    // 可以访问内核数据结构
    // 可以调用所有 API
    for ( ;; )
    {
        vTaskDelay( pdMS_TO_TICKS( 1000 ) );
    }
}

// 用户级任务（受限访问）
void vUserTask( void * pvParameters )
{
    // 只能访问自己的内存区域
    // 通过系统调用访问内核服务
    for ( ;; )
    {
        vTaskDelay( pdMS_TO_TICKS( 2000 ) );
    }
}

int main( void )
{
    // 创建特权任务
    StaticTask_t xTaskBuffer;
    StackType_t xStack[ 128 ];
    xTaskCreateStaticPrivileged( vPrivilegedTask, "Priv", 128, NULL, 2, xStack, &xTaskBuffer );

    // 创建用户级任务
    xTaskCreate( vUserTask, "User", 128, NULL, 1, NULL );

    vTaskStartScheduler();

    for ( ;; ) {}
}
