01 MCUboot 概览

MCUboot 是一个安全的引导加载程序，支持固件签名验证、加密、OTA 升级和回滚保护。

核心特性

特性说明
安全启动RSA/ECDSA 签名验证
固件加密AES-128-CTR 加密固件
OTA 升级支持多种传输方式
回滚保护升级失败自动回退
多槽位支持 Slot 0/1/2/3
RAM 加载固件从 Flash 加载到 RAM 执行
加密存储加密固件镜像
零拷贝直接在 Flash 上验证

│  Bootloader 分区                                │
│  MCUboot 引导加载程序                           │
├─────────────────────────────────────────────────┤
│  Slot 0（主分区）                               │
│  当前运行的应用固件                              │
├─────────────────────────────────────────────────┤
│  Slot 1（次分区）                               │
│  OTA 升级的新固件                               │
├─────────────────────────────────────────────────┤
│  Scratch（临时分区）                            │
│  升级过程中的临时数据                            │

│  持久化存储                                     │
└─────────────────────────────────────────────────┘

DTS Flash 分区定义

/* Flash 分区定义 */
&flash0 {
    /* Bootloader */
    partitions {
        compatible = "fixed-partitions";
        #address-cells = <1>;
        #size-cells = <1>;

        boot_partition: partition@0 {
            label = "mcuboot";
            reg = <0x00000000 0x8000>;
        };

        /* Slot 0 - 当前固件 */
        slot0_partition: partition@8000 {
            label = "image-0";
            reg = <0x00008000 0x60000>;
        };

        /* Slot 1 - 升级固件 */
        slot1_partition: partition@68000 {
            label = "image-1";
            reg = <0x00068000 0x60000>;
        };

        /* Scratch - 临时分区 */
        scratch_partition: partition@c8000 {
            label = "scratch";
            reg = <0x000c8000 0x4000>;
        };

        /* NVS - 持久化存储 */
        storage_partition: partition@cc000 {
            label = "storage";
            reg = <0x000cc000 0x34000>;
        };
    };
};

签名配置

// MCUboot 签名配置
// CONFIG_BOOT_SIGNATURE=y
// CONFIG_BOOT_SIGNATURE_TYPE_RSA=y
// CONFIG_BOOT_SIGNATURE_TYPE_ECDSA_P256=y

// 生成签名密钥
// imgtool keygen -k signing-key.pem -t RSA-2048
// imgtool sign --key signing-key.pem firmware.bin signed.bin

// 镜像头结构
struct image_header {
    uint32_t ih_magic;      // 魔数 0x96F3B83D
    uint32_t ih_load_addr;  // 加载地址
    uint16_t ih_hdr_size;   // 头部大小
    uint16_t ih_protect_tlv_size; // TLV 大小
    uint32_t ih_img_size;   // 镜像大小
    uint32_t ih_flags;      // 标志位
    struct image_version ih_ver; // 版本号
    uint32_t ih_pad1;
};

mcumgr 镜像管理命令

# 查看镜像信息
mcumgr image list

# 上传新镜像
mcumgr image upload -e firmware.bin

# 测试新镜像（标记为待确认）
mcumgr image test <hash>

# 确认镜像（永久生效）
mcumgr image confirm <hash>

# 回滚到上一版本
mcumgr image revert

# 删除镜像
mcumgr image erase <hash>

06 回滚保护

// 回滚机制
// CONFIG_BOOT_BOOTUTIL_VALIDATE_SLOT0=y  ← 校验 Slot 0
// CONFIG_BOOT_MCUOOT_DOWNGRADE_PREVENTION=y ← 防止降级

// 回滚场景：
// 1. 新固件启动后崩溃
// 2. MCUboot 检测到启动失败
// 3. 自动回滚到 Slot 0
// 4. 标记 Slot 1 为无效

// 超时回滚
// CONFIG_BOOTUTIL_CONFIRM_TIMEOUT=10  ← 10秒确认超时
// 超时后自动回滚

// 应用侧确认
// 在应用启动后调用 mcuboot_mark_app_confirmed()
// 或通过 mcumgr 命令确认

09 完整示例

// ===== Zephyr MCUboot 完整示例 =====
// 演示：固件确认、升级状态查询、回滚

#include <zephyr/kernel.h>
#include <zephyr/dfu/mcuboot.h>
#include <zephyr/mgmt/mcumgr/grp/img_mgmt/img_mgmt.h>
#include <stdio.h>

// 确认当前固件
void vConfirmFirmware(void)
{
    // 标记当前固件为已确认
    int ret = mcuboot_mark_app_confirmed();
    if (ret == 0) {
        printk("Firmware confirmed successfully\\n");
    } else {
        printk("Firmware confirmation failed: %d\\n", ret);
    }
}

// 查询升级状态
void vCheckUpgradeStatus(void)
{
    struct image_version ver;
    uint8_t slot;

    // 获取 Slot 0 版本
    if (img_mgmt_get_version(0, &ver) == 0) {
        printk("Slot 0: v%d.%d.%d\\n", ver.iv_major, ver.iv_minor, ver.iv_revision);
    }

    // 获取 Slot 1 版本
    if (img_mgmt_get_version(1, &ver) == 0) {
        printk("Slot 1: v%d.%d.%d\\n", ver.iv_major, ver.iv_minor, ver.iv_revision);
    }

    // 获取当前活动 Slot
    slot = img_mgmt_get_active_slot();
    printk("Active slot: %d\\n", slot);
}

// 监控任务
void vMonitorTask(void *pvParameters)
{
    // 启动后确认固件
    vConfirmFirmware();

    for (;;) {
        vCheckUpgradeStatus();
        k_msleep(30000);  // 每 30 秒检查一次
    }
}

// 应用任务
void vAppTask(void *pvParameters)
{
    for (;;) {
        printk("Application running\\n");
        k_msleep(5000);
    }
}
