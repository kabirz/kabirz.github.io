01 概览

内存屏障（Memory Barrier）是确保多核/多线程环境下内存访问顺序的硬件指令。Linux 内核提供多种屏障宏。

02 屏障类型

完整屏障 vs 单向屏障

类型宏阻止重排
完整屏障smp_mb()阻止屏障前后的所有读写重排
读屏障smp_rmb()阻止屏障前的读与屏障后的读重排
写屏障smp_wmb()阻止屏障前的写与屏障后的写重排
// 完整屏障：所有读写都不能跨越屏障重排
WRITE_ONCE(a, 1);
smp_mb();           // 完整屏障
READ_ONCE(b);

// 读屏障：只阻止读的重排
READ_ONCE(a);
smp_rmb();          // 读屏障
READ_ONCE(b);       // b 的读取一定在 a 之后

// 写屏障：只阻止写的重排
WRITE_ONCE(a, 1);
smp_wmb();          // 写屏障
WRITE_ONCE(b, 2);   // b 的写入一定在 a 之后

03 SMP 屏障

定义（barrier.h）

// SMP 模式下的屏障定义

// 完整内存屏障
#define smp_mb()    do { kcsan_mb(); __smp_mb(); } while (0)

// 读屏障
#define smp_rmb()   do { kcsan_rmb(); __smp_rmb(); } while (0)

// 写屏障
#define smp_wmb()   do { kcsan_wmb(); __smp_wmb(); } while (0)

// 非 SMP 模式下退化为编译器屏障
// #define smp_mb()   barrier()
// #define smp_rmb()  barrier()
// #define smp_wmb()  barrier()

底层实现

// barrier.h 第 127-128 行
// 写入 + 屏障的组合
#define smp_store_mb(var, value)  do { kcsan_mb(); __smp_store_mb(var, value); } while (0)

// __smp_store_mb 定义：
#define __smp_store_mb(var, value)  do { WRITE_ONCE(var, value); __smp_mb(); } while (0)

04 编译器屏障

// 编译器屏障（barrier()）
// 仅阻止编译器重排，不生成任何 CPU 指令

// 定义在 include/linux/compiler.h
// #define barrier() __asm__ __volatile__("" ::: "memory")

// 作用：
// 1. 防止编译器跨越屏障重排读写
// 2. 强制将内存值从寄存器刷新到内存
// 3. 不生成任何 CPU 指令

// 与 smp_mb() 的区别：
// barrier() — 仅编译器屏障
// smp_mb()  — 编译器屏障 + CPU 内存屏障

05 Acquire/Release

// barrier.h 第 171-177 行

// 释放语义：之前的读写不能重排到之后
#define smp_store_release(p, v) do { kcsan_release(); __smp_store_release(p, v); } while (0)

// 获取语义：之后的读写不能重排到之前
#define smp_load_acquire(p) __smp_load_acquire(p)

// 底层实现：
#define __smp_store_release(p, v)  do { \
    compiletime_assert_atomic_type(*p); \
    __smp_mb();                           \
    WRITE_ONCE(*p, v);                     \
} while (0)

#define __smp_load_acquire(p) ({           \
    typeof(*p) ___p1 = READ_ONCE(*p);     \
    compiletime_assert_atomic_type(*p);   \
    __smp_mb();                           \
    (typeof(*p))___p1;                     \
})

Acquire/Release vs 完整屏障

特性smp_mb()smp_store_release()
阻止重排前后所有操作仅之前的操作不能到之后
性能较高开销较低开销（某些架构优化）
适用场景通用同步发布/订阅模式

06 DMA 屏障

// barrier.h 第 41-51 行
// DMA 屏障用于设备 DMA 操作

// DMA 完整屏障
#define dma_mb()   do { kcsan_mb(); __dma_mb(); } while (0)

// DMA 读屏障
#define dma_rmb()  do { kcsan_rmb(); __dma_rmb(); } while (0)

// DMA 写屏障
#define dma_wmb()  do { kcsan_wmb(); __dma_wmb(); } while (0)

// 用途：
// - 确保 CPU 写入的数据对 DMA 控制器可见
// - 确保 DMA 写入的数据对 CPU 可见
// - 用于 PCI/PCIe 设备通信

07 典型用例

// 用例 1：生产者-消费者
// CPU 0 写数据，CPU 1 读数据

// 生产者
WRITE_ONCE(data, value);
smp_wmb();          // 确保数据先于标志写入
WRITE_ONCE(flag, 1);

// 消费者
while (READ_ONCE(flag) == 0) {
    cpu_relax();
}
smp_rmb();          // 确保先读标志再读数据
value = READ_ONCE(data);

// 用例 2：无锁队列
// 使用 smp_store_release 和 smp_load_acquire

// 生产者
smp_store_release(&queue->head, new_head);

// 消费者
int head = smp_load_acquire(&queue->head);

// 用例 3：原子操作前后
// 某些原子操作不包含内存屏障

// 原子递增前的屏障
smp_mb__before_atomic();
atomic_inc(&counter);

// 原子递增后的屏障
atomic_inc(&counter);
smp_mb__after_atomic();

08 常见陷阱

// 陷阱 1：混淆编译器屏障和内存屏障
// barrier() 只阻止编译器重排，不阻止 CPU 重排
// smp_mb() 同时阻止编译器和 CPU 重排

// 陷阱 2：在 UP 系统上忽略屏障
// UP 系统上 smp_mb() 退化为 barrier()
// 但与设备通信时仍需要 dma_mb()

// 陷阱 3：过度使用屏障
// 屏障有性能开销，应该只在需要时使用
// 优先使用 smp_store_release/smp_load_acquire

// 陷阱 4：在错误的位置使用屏障
// 屏障应该放在需要顺序保证的地方
// 而不是随意添加

// 陷阱 5：忘记 READ_ONCE/WRITE_ONCE
// 使用屏障时，被保护的变量也应该使用 READ_ONCE/WRITE_ONCE
WRITE_ONCE(a, 1);
smp_wmb();
WRITE_ONCE(b, 2);  // ✅ 正确
