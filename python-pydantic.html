01 BaseModel

BaseModel 是 Pydantic 的核心类。通过继承它并声明类型注解的字段，自动获得解析、验证和序列化能力。

基础字段定义与类型自动转换

from pydantic import BaseModel
from datetime import datetime
from uuid import UUID, uuid4
from decimal import Decimal

class User(BaseModel):
    id: UUID = uuid4()           # 自动生成 UUID
    name: str                     # 字符串
    age: int                      # 整数，自动转换
    height: float                 # 浮点数，自动转换
    is_active: bool               # 布尔值
    score: Decimal | None = None  # Decimal 精度控制
    created_at: datetime          # 自动解析日期字符串

# 类型自动转换
user = User(
    name="张三",
    age="25",                # 字符串 → int
    height="175.5",          # 字符串 → float
    is_active="yes",         # "yes"/"true"/"1" → True
    created_at="2024-01-15T10:30:00",
)

print(user.age)        # 25 (int)
print(type(user.age))  # 

# 验证失败时抛出 ValidationError
try:
    User(
        name="张三",
        age="not_a_number",  # 无法转换为 int
        height=175.5,
        is_active=True,
        created_at="invalid-date",
    )
except Exception as e:
    print(e)  # 显示详细的字段错误信息

Field 参数详解

from pydantic import BaseModel, Field
from typing import Annotated
from uuid import UUID, uuid4

class Product(BaseModel):
    # 基础约束
    id: UUID = Field(default_factory=uuid4)
    name: str = Field(
        min_length=1,
        max_length=100,
        strip_whitespace=True,
    )

    # 数值约束
    price: float = Field(
        gt=0,        # > 0
        le=100000,   # ≤ 100000
        multiple_of=0.01,  # 必须是 0.01 的倍数
    )

    quantity: int = Field(
        ge=0,        # ≥ 0
        lt=10000,    # < 10000
    )

    # 字符串模式
    sku: str = Field(
        pattern=r"^[A-Z]{3}-\d{4}$",
        description="格式: ABC-1234",
    )

    # 默认值
    discount: float = Field(
        default=0.0,
        ge=0.0,
        le=1.0,
    )

    # 使用 default_factory
    tags: list[str] = Field(default_factory=list)

    # 枚举验证
    status: str = Field(
        default="active",
        pattern=r"^(active|inactive|archived)$"
    )

    # 元数据
    created_by: str = Field(
        default="system",
        title="创建者",
        description="记录创建该商品的用户",
        examples=["admin", "张三"],
        json_schema_extra={"ui:widget": "hidden"},
    )

验证装饰器：@field_validator

from pydantic import BaseModel, Field, field_validator
from typing import Any

class Registration(BaseModel):
    username: str = Field(min_length=3, max_length=20)
    email: str
    password: str
    confirm_password: str
    age: int = Field(ge=0, le=150)

    # V2 的 field_validator（替代 V1 的 @validator）
    # mode="before"：在 Pydantic 内部验证之前执行
    @field_validator("email", mode="before")
    @classmethod
    def normalize_email(cls, v: Any) -> str:
        if isinstance(v, str):
            return v.strip().lower()
        return str(v).strip().lower()

    # 默认 mode="after"：在类型转换之后执行
    @field_validator("username")
    @classmethod
    def validate_username(cls, v: str) -> str:
        if not v.isalnum():
            raise ValueError("用户名只能包含字母和数字")
        return v

    # 验证多个字段
    @field_validator("password")
    @classmethod
    def validate_password(cls, v: str) -> str:
        if len(v) < 8:
            raise ValueError("密码至少 8 个字符")
        if not any(c.isdigit() for c in v):
            raise ValueError("密码必须包含数字")
        if not any(c.isupper() for c in v):
            raise ValueError("密码必须包含大写字母")
        return v

    # 字段间依赖验证
    @field_validator("confirm_password")
    @classmethod
    def passwords_match(cls, v: str, info) -> str:
        if "password" in info.data and v != info.data["password"]:
            raise ValueError("两次密码不一致")
        return v

# 验证单个字段的多个条件
class OrderItem(BaseModel):
    quantity: int
    unit_price: float

    @field_validator("quantity")
    @classmethod
    def check_quantity(cls, v: int) -> int:
        if v <= 0:
            raise ValueError("数量必须为正数")
        if v > 1000:
            raise ValueError("单次购买不能超过 1000 件")
        return v

@model_validator — 模型级别验证

from pydantic import BaseModel, model_validator
from typing import Any

class Order(BaseModel):
    items: list[dict]
    coupon_code: str | None = None
    shipping_address: str | None = None
    total: float = 0.0

    # 前置验证：在任何字段验证之前执行
    @model_validator(mode="before")
    @classmethod
    def check_raw_data(cls, data: Any) -> Any:
        if isinstance(data, dict):
            # 自动修正字段名
            if "items_count" in data:
                # 转换或处理
                pass
            # 设置默认值
            if "total" not in data:
                data["total"] = 0.0
        return data

    # 后置验证：在所有字段验证之后执行
    @model_validator(mode="after")
    def validate_order(self) -> "Order":
        errors = []

        # 检查购物车不为空
        if not self.items:
            errors.append("订单至少需要一个商品")

        # 检查总额
        computed_total = sum(
            item.get("price", 0) * item.get("quantity", 0) * (1 - item.get("discount", 0))
            for item in self.items
        )
        if abs(self.total - computed_total) > 0.01:
            errors.append(f"总额不匹配: 计算值 {computed_total:.2f}")

        # 根据优惠券代码验证
        if self.coupon_code and not self.shipping_address:
            errors.append("使用优惠券需要提供收货地址")

        if errors:
            raise ValueError("; ".join(errors))

        return self

# 使用示例
order_data = {
    "items": [{"name": "手机", "price": 2999, "quantity": 2}],
    "total": 5998.0,
}

try:
    order = Order(**order_data)
    print(f"订单验证通过: {order.total}")
except Exception as e:
    print(f"验证失败: {e}")

03 validator 进阶

@validator（V1 兼容，V2 仍支持）

from pydantic import BaseModel, validator
from typing import Any

# V1 风格的 @validator（V2 中仍可使用，但推荐 @field_validator）
class LegacyModel(BaseModel):
    name: str
    value: int

    @validator("name")
    @classmethod
    def name_not_empty(cls, v: str) -> str:
        if not v.strip():
            raise ValueError("名称不能为空")
        return v.strip()

    @validator("value", pre=True)  # pre=True = mode="before"
    @classmethod
    def coerce_value(cls, v: Any) -> int:
        if isinstance(v, str):
            return int(v.replace(",", ""))
        return v

    # 验证多个字段
    @validator("name", "value")
    @classmethod
    def validate_both(cls, v: Any, field) -> Any:
        """同一个验证器处理多个字段"""
        print(f"验证字段: {field.name}")
        return v

    # always=True：即使字段为 None 也执行验证
    @validator("value", always=True)
    @classmethod
    def check_value(cls, v: int | None) -> int | None:
        if v is not None and v < 0:
            raise ValueError("值不能为负数")
        return v

    # 字段依赖
    @validator("name")
    @classmethod
    def check_name_value(cls, v: str, values: dict) -> str:
        # values 包含已验证字段的字典
        val = values.get("value")
        if val is not None and len(v) > val:
            raise ValueError(f"名称长度不能超过 {val}")
        return v

根验证器（V1：@root_validator → V2：@model_validator）

# V1 的 root_validator 迁移到 V2 的 model_validator
# V1 方式（已弃用）：
# @root_validator(pre=True)
# def pre_process(cls, values):
#     ...

# V2 方式对比
from pydantic import BaseModel, model_validator

class UserRegistration(BaseModel):
    username: str
    email: str
    email_confirm: str

    # 等效于 V1 的 @root_validator(pre=True)
    @model_validator(mode="before")
    @classmethod
    def pre_process(cls, data: Any) -> Any:
        if isinstance(data, dict):
            # 统一邮箱格式
            if "email" in data:
                data["email"] = data["email"].strip().lower()
            if "email_confirm" in data:
                data["email_confirm"] = data["email_confirm"].strip().lower()
        return data

    # 等效于 V1 的 @root_validator（默认 pre=False）
    @model_validator(mode="after")
    def validate_email_match(self) -> "UserRegistration":
        if self.email != self.email_confirm:
            raise ValueError("两次输入的邮箱不一致")
        return self

    # 可以有多个 model_validator（按声明顺序执行）
    @model_validator(mode="after")
    def validate_username_rules(self) -> "UserRegistration":
        if self.username.lower() in self.email.lower():
            raise ValueError("用户名不应出现在邮箱中")
        return self

自定义错误与 ValidationInfo

from pydantic import BaseModel, Field, field_validator, ValidationInfo
from typing import Any

class HTTPConfig(BaseModel):
    host: str = "localhost"
    port: int = Field(default=8080, ge=1, le=65535)
    ssl: bool = False
    url_prefix: str = ""

    @field_validator("host")
    @classmethod
    def validate_host(cls, v: str, info: ValidationInfo) -> str:
        """使用 ValidationInfo 获取上下文"""
        # info.field_name: 当前字段名
        # info.data: 已验证字段的字典
        # info.config: 模型配置

        if v == "0.0.0.0" and not info.data.get("ssl"):
            raise ValueError("非 SSL 模式不建议监听 0.0.0.0")
        return v

    @field_validator("url_prefix")
    @classmethod
    def normalize_prefix(cls, v: str) -> str:
        if v and not v.startswith("/"):
            return "/" + v
        return v

class APIResponse(BaseModel):
    success: bool
    data: Any = None
    error: str | None = None

    @field_validator("error")
    @classmethod
    def check_error(cls, v: str | None, info: ValidationInfo) -> str | None:
        if info.data.get("success") and v:
            raise ValueError("成功的响应不应包含错误信息")
        if not info.data.get("success") and not v:
            raise ValueError("失败的响应必须包含错误信息")
        return v

配置分层与合并

from pydantic_settings import BaseSettings
from pydantic import BaseModel, Field
from typing import Optional

class DatabaseConfig(BaseModel):
    url: str = "sqlite:///./dev.db"
    pool_size: int = 5
    echo: bool = False

class LogConfig(BaseModel):
    level: str = "INFO"
    format: str = "%(asctime)s - %(name)s - %(levelname)s - %(message)s"
    file: Optional[str] = None

class DeployConfig(BaseSettings):
    """分层配置：环境 + 默认 + 覆盖"""
    env: str = Field(default="development", pattern="^(development|staging|production)$")
    db: DatabaseConfig = DatabaseConfig()
    logging: LogConfig = LogConfig()
    workers: int = 1
    host: str = "0.0.0.0"
    port: int = 8000

    model_config = {"env_nested_delimiter": "__"}

    @property
    def is_production(self) -> bool:
        return self.env == "production"

    @property
    def is_debug(self) -> bool:
        return self.env == "development"

# 根据环境加载不同配置
def load_config(env: str | None = None) -> DeployConfig:
    config = DeployConfig()  # 从环境变量加载
    if env:
        config.env = env

    # 环境特定调整
    if config.env == "production":
        config.db.pool_size = 20
        config.logging.level = "WARNING"
        config.workers = 4
    elif config.env == "staging":
        config.db.pool_size = 10
        config.db.echo = True

    return config

from pydantic import BaseModel, Field
from datetime import datetime
from typing import Literal
from enum import Enum

class Category(str, Enum):

    FOOD = "food"
    BOOKS = "books"

class Address(BaseModel):

class Item(BaseModel):

    created_at: datetime
    status: Literal["pending", "confirmed", "shipped", "delivered"]

from pydantic import BaseModel, Field
from typing import Any

class CustomSchema(BaseModel):
    # json_schema_extra：附加元数据
    name: str = Field(
        ...,
        json_schema_extra={
            "example": "张三",
            "ui:widget": "text",
            "ui:options": {"placeholder": "请输入姓名"},
        }
    )
    age: int = Field(
        ...,
        json_schema_extra={
            "example": 25,
            "ui:widget": "range",
            "ui:options": {"min": 0, "max": 150},
        }
    )

    # 自定义模型级别的 Schema 生成
    @classmethod
    def model_json_schema(cls, *args, **kwargs) -> dict:
        schema = super().model_json_schema(*args, **kwargs)
        # 添加自定义元数据到顶层
        schema["title"] = "用户注册表单"
        schema["description"] = "用于用户注册的 JSON Schema"
        schema["$id"] = "https://example.com/schemas/user-registration.json"
        schema["version"] = "1.0.0"
        return schema

from pydantic import BaseModel, Field
from fastapi import FastAPI
import json

# 与 FastAPI 集成（自动生成 OpenAPI）

# 将 Schema 用于前端验证
class ValidateForm(BaseModel):
    """前端表单验证 Schema"""
    username: str = Field(min_length=3, max_length=20, pattern=r"^\w+$")
    email: str = Field(pattern=r"^\S+@\S+\.\S+$")
    password: str = Field(min_length=8, max_length=128)
    confirm_password: str = Field(min_length=8)

# 生成前端验证规则
schema = ValidateForm.model_json_schema()
rules = {}

for field_name, field_schema in schema.get("properties", {}).items():
    rule = {}
    if "minLength" in field_schema:
        rule["minLength"] = field_schema["minLength"]
    if "maxLength" in field_schema:
        rule["maxLength"] = field_schema["maxLength"]
    if "pattern" in field_schema:
        rule["pattern"] = field_schema["pattern"]
    rules[field_name] = rule

print(json.dumps(rules, indent=2, ensure_ascii=False))

from datetime import datetime, date, time, timedelta
from typing import Annotated

class Event(BaseModel):
    name: str
    start_time: datetime
    end_time: datetime
    event_date: date
    duration: timedelta | None = None
    reminder_before: timedelta = timedelta(hours=1)

    # 自定义验证逻辑
    @field_validator("end_time")
    @classmethod
    def end_after_start(cls, v: datetime, info) -> datetime:

# 自定义类型
class PhoneNumber(str):
    """自定义电话号类型"""
    @classmethod
    def __get_pydantic_core_schema__(cls, source, handler):
        from pydantic_core import core_schema

        def validate_phone(v: str) -> str:
            if not v.startswith("+"):
                raise ValueError("电话号码必须以 + 开头")
            if not v[1:].isdigit():
                raise ValueError("电话号码只能包含数字")
            if not 8 <= len(v) <= 15:
                raise ValueError("电话号码长度不对")
            return v

        return core_schema.with_info_plain_validator_function(
            validate_phone,
            serialization=core_schema.to_string_ser_schema(),
        )

class Contact(BaseModel):
    name: str
    phone: PhoneNumber
    email: str

# Annotated 自定义验证器
def validate_port(v: int) -> int:
    if not 1 <= v <= 65535:
        raise ValueError("端口号必须在 1-65535 之间")
    return v

Port = Annotated[int, AfterValidator(validate_port)]

    host: str
    port: Port = 8080

server = Server(host="localhost", port="3000")

__get_validators__ 与自定义验证器

from pydantic import BaseModel
from typing import Any, Callable

# V1 风格的 __get_validators__（仍在 V2 中工作）
class PositiveInt(int):
    """仅在创建时验证的正整数"""
    @classmethod
    def __get_validators__(cls) -> Callable:
        yield cls.validate

    @classmethod
    def validate(cls, v: Any) -> "PositiveInt":
        if isinstance(v, str):
            v = int(v)
        if not isinstance(v, int):
            raise TypeError("必须是整数")
        if v <= 0:
            raise ValueError("必须是正数")
        return cls(v)

class ModelWithPositive(BaseModel):
    value: PositiveInt
    count: PositiveInt = PositiveInt(1)

# V2 推荐：使用 Annotated + BeforeValidator / AfterValidator
from pydantic.functional_validators import BeforeValidator

def coerce_float(v: Any) -> float:
    """将兼容类型转为 float"""
    if isinstance(v, (int, float)):
        return float(v)
    if isinstance(v, str):
        return float(v.strip("%").strip())
    raise ValueError(f"无法转换为 float: {v}")

FlexibleFloat = Annotated[float, BeforeValidator(coerce_float)]

class Product(BaseModel):
    price: FlexibleFloat
    discount: FlexibleFloat = 0.0

# 使用
p = Product(price="99.99", discount="10%")
print(p.price)     # 99.99
print(p.discount)  # 10.0

# PlainValidator — 不进行任何类型推断
from pydantic.functional_validators import PlainValidator

def parse_hex(v: Any) -> int:
    """解析十六进制字符串为整数"""
    if isinstance(v, int):
        return v
    if isinstance(v, str):
        return int(v, 16)
    raise ValueError("无效的十六进制输入")

HexInt = Annotated[int, PlainValidator(parse_hex)]

class Memory(BaseModel):
    address: HexInt

mem = Memory(address="0xFF")
print(mem.address)  # 255

核心架构变化

# 1. pydantic-core — Rust 重写的验证引擎

from pydantic import BaseModel, field_validator, model_validator
from typing import Any

class V2Model(BaseModel):
    name: str
    age: int

    # @validator → @field_validator
    # mode="after" 是默认值
    @field_validator("name")
    @classmethod
    def name_must_not_be_empty(cls, v: str) -> str:
        if not v.strip():
            raise ValueError("名称不能为空")
        return v

    # pre=True → mode="before"
    @field_validator("age", mode="before")
    @classmethod
    def coerce_age(cls, v: Any) -> int:
        if isinstance(v, str):
            return int(v)
        return v

    # @root_validator(pre=True) → @model_validator(mode="before")
    @model_validator(mode="before")
    @classmethod
    def pre_process(cls, data: Any) -> Any:
        if isinstance(data, dict):
            if "age" in data:
                data["age"] = int(data["age"])
        return data

    # @root_validator → @model_validator(mode="after")
    @model_validator(mode="after")
    def check_name_age(self) -> "V2Model":
        if self.name and self.age < 18:
            raise ValueError("未成年人")
        return self

from pydantic import BaseModel, field_validator
from typing import Any

# Wrap 验证器：装饰器模式
class ModelWithWrap(BaseModel):
    data: str

    @field_validator("data", mode="wrap")
    @classmethod
    def wrap_validator(cls, v: Any, handler) -> str:
        print(f"包装验证: {v!r}")
        # 可以修改值后调用内部验证
        if v is None:
            v = "default"
        result = handler(v)  # 调用内部验证逻辑
        print(f"验证结果: {result!r}")
        return result

# SerDe（序列化/反序列化）控制
from pydantic import BaseModel, Field
from datetime import datetime

class SerDeDemo(BaseModel):
    name: str
    password: str = Field(exclude=True)  # 序列化时排除
    internal_id: int = Field(exclude=True)
    created_at: datetime

    # 自定义序列化器
    @field_serializer("created_at")
    def serialize_dt(self, v: datetime) -> str:
        return v.isoformat()

    # 自定义反序列化器
    @field_validator("created_at", mode="before")
    @classmethod
    def parse_dt(cls, v: Any) -> datetime:
        if isinstance(v, str):
            return datetime.fromisoformat(v)
        return v
