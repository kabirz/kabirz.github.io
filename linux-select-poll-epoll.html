01 概述

Linux 的 I/O 多路复用经历了三代演进：select（1983 BSD）→ poll（1997 POSIX）→ epoll（2002 Linux 2.6）。核心问题只有一个：如何高效地同时监听多个文件描述符的可读/可写事件。

演进时间线
年份机制内核版本
1983selectBSD 4.2
1997pollLinux 2.0 (POSIX)
2002epollLinux 2.6
核心问题
  • 应用需要同时监听多个 socket（如 Web 服务器）
  • 每个 socket 可能有数据就绪、可写、异常等事件
  • 逐个轮询 → CPU 空转浪费
  • I/O 多路复用 → 一次调用等待多个 fd 事件

02 select 详解

用户态 API

// 头文件：<sys/select.h>
int select(int nfds,
           fd_set *readfds,     // 监听可读的 fd 集合
           fd_set *writefds,    // 监听可写的 fd 集合
           fd_set *exceptfds,   // 监听异常的 fd 集合
           struct timeval *timeout); // 超时时间

// 返回值：就绪的 fd 数量，0=超时，-1=出错
// 注意：select 返回后，fd_set 被修改为仅包含就绪的 fd

内核实现流程

用户调用 select(nfds, readfds, writefds, exceptfds, timeout)
        │
        ↓
  core_sys_select() — fs/select.c
        │
        ├── 1. copy_from_user: 从用户空间拷贝 fd_set 到内核
        │
        ├── 2. 为每个 fd 分配 poll_table_entry
        │       每个 entry 包含: file + wait_queue + key

        │
        ├─── 循环 0 ~ nfds：

        │       → 驱动的 file->f_op->poll()

        │
        ↓

核心数据结构

// fs/select.c:483 — do_select 核心逻辑
static noinline_for_stack int do_select(int n, fd_set_bits *fds,
                                        struct timespec64 *end_time)
{
    struct poll_wqueues table;        // 内嵌在栈上的 poll 等待队列
    poll_initwait(&table);            // 初始化，设置 _qproc 回调

    for (;;) {
        unsigned *rinp, *rout, *rexp;
        unsigned long __set;

        // 遍历每个 bit，对每个 fd 调用 vfs_poll()
        for (i = 0; i < n; ++rinp, ++rout, ++rexp, i++) {
            if (__FD_ISSET(i, inp))
                mask = vfs_poll(f.file, &pt);  // 调用驱动 poll
            // 检查 mask 是否有就绪事件
            // 有 → 设置对应 bit
        }

        if (res || signal_pending(current))
            break;      // 有就绪 fd 或收到信号 → 退出循环

        if (!poll_schedule_timeout(&table, ...))
            break;      // 超时 → 退出循环
    }

    poll_freewait(&table);    // 释放所有 poll_table_entry
    return res;
}

select 的局限

三大硬伤
① fd_set 位图上限：FD_SETSIZE 通常为 1024，无法监听更多 fd
② 每次调用都拷贝：fd_set 在用户态和内核态之间反复拷贝
③ O(n) 全量扫描：每次唤醒后都要遍历全部 n 个 fd，不管其中多少个是就绪的

03 poll 详解

用户态 API

// 头文件：<poll.h>
int poll(struct pollfd fds[], nfds_t nfds, int timeout);

struct pollfd {
    int   fd;         // 文件描述符
    short events;     // 请求的事件（POLLIN | POLLOUT）
    short revents;    // 返回的事件（内核填充）
};

与 select 的区别

改进之处
  • 无 fd 数量上限：使用 pollfd 数组而非位图
  • 参数不被修改：events 保留，revents 返回结果
  • events/revents 分离：用户可查看请求的事件
未解决的问题
  • 仍然 O(n) 扫描：每次调用都遍历所有 fd
  • 仍然需要拷贝：每次调用拷贝整个 pollfd 数组到内核
  • 仍然需要重新注册：唤醒后需要重新遍历设置等待

内核实现

// fs/compat.c / fs/select.c — sys_poll → do_poll
// do_poll 核心与 do_select 几乎相同：
// 1. 遍历所有 pollfd
// 2. 对每个 fd 调用 vfs_poll()
// 3. 如果没有就绪的 fd → 挂起
// 4. 被唤醒后 → 重新遍历
//
// poll_wait() 注册等待队列的回调机制与 select 完全相同

04 epoll 详解

用户态 API

// 1. 创建 epoll 实例
int epfd = epoll_create1(0);

// 2. 添加/修改/删除监听的 fd
struct epoll_event ev = {
    .events = EPOLLIN | EPOLLET,   // 边缘触发
    .data.fd = sockfd,
};
epoll_ctl(epfd, EPOLL_CTL_ADD, sockfd, &ev);

// 3. 等待事件
struct epoll_event events[MAX_EVENTS];
int n = epoll_wait(epfd, events, MAX_EVENTS, -1);
// 返回就绪的 fd 数量，events 数组填充就绪的事件

三大系统调用

调用作用执行频率
epoll_create1创建 epoll 实例（eventpoll 结构体）程序启动时一次
epoll_ctl(ADD/MOD/DEL)添加/修改/删除监听的 fdfd 变化时
epoll_wait阻塞等待就绪事件事件循环中反复调用

核心数据结构

// fs/eventpoll.c:179 — epoll 实例
struct eventpoll {
    struct mutex          mtx;         // 保护 epoll 的互斥锁
    struct wait_queue_head_t wq;      // epoll_wait 的等待队列
    struct wait_queue_head_t poll_wait;// file->poll() 的等待队列
    struct list_head      rdllist;     // 就绪 fd 链表（核心！）
    spinlock_t           lock;        // 保护 rdllist + ovflist
    struct rb_root_cached rbr;        // RB 红黑树（存储所有 epitem）
    struct epitem         *ovflist;    // 溢出链表（传递事件时使用）
    u64                    gen;        // 循环检测优化
    refcount_t             refcount;   // 引用计数
};

// fs/eventpoll.c:131 — 监听的 fd 条目
struct epitem {
    struct rb_node     rbn;        // 红黑树节点
    struct list_head   rdllink;    // 就绪链表节点
    struct epoll_filefd ffd;        // fd + file 引用
    struct eppoll_entry *pwqlist;  // 等待队列条目
    struct eventpoll   *ep;        // 所属 eventpoll
    struct epoll_event  event;     // 用户注册的事件
};

epoll_ctl (ADD) 流程

epoll_ctl(epfd, EPOLL_CTL_ADD, sockfd, &ev)
        │
        ↓
  1. 分配 epitem（从 slab 缓存）
        │
        ↓
  2. RB 红黑树插入
     epitem 插入 eventpoll->rbr（O(log n)）
     同时设置 file->f_ep 指向 epitems_head
        │
        ↓
  3. 注册等待队列回调
     调用 vfs_poll(file, &pt)
     → file->f_op->poll() 被调用
     → 驱动内部调用 poll_wait() 注册回调
     → 回调函数 = ep_poll_callback()
        │
        ↓
  4. 检查 fd 当前是否已就绪
     如果已就绪 → 直接加入 rdllist
     epoll_ctl 返回，没有拷贝，没有扫描！

epoll_wait 流程

epoll_wait(epfd, events, maxevents, timeout)
        │
        ↓
  ep_poll() — fs/eventpoll.c:1956
        │
        ├── 1. 检查 rdllist 是否非空
        │       非空 → 直接跳到步骤 4（零拷贝就绪！）
        │
        ├── 2. rdllist 为空
        │       把当前进程挂到 eventpoll->wq 等待队列
        │
        ├── 3. 睡眠等待
        │       被以下情况唤醒：
        │       - 某个 fd 就绪 → ep_poll_callback() 唤醒
        │       - 超时
        │       - 信号
        │
        ↓
  4. 从 rdllist 取出就绪的 epitem
        │
        ↓

        │
        ├── 遍历 rdllist
        │   对每个 epitem 调用 vfs_poll() 确认事件
        │   将 (events, data) 拷贝到用户态 events 数组
        │
        ↓
返回就绪的 fd 数量，用户直接处理

epoll 回调机制

// 当驱动/协议栈有数据就绪时，调用 wake_up()
// → 依次调用每个 wait_queue_entry 的 func
// → 也就是 ep_poll_callback()

// ep_poll_callback() 核心逻辑：
static int ep_poll_callback(struct wait_queue_entry *wentry, ...,
                           int mode, int sync, void *key)
{
    struct epitem *epi = container_of(wentry, ...);
    struct eventpoll *ep = epi->ep;

    // 检查事件是否匹配用户注册的 events
    if (!(epi->event.events & revents))
        return 1;  // 事件不匹配，忽略

    // 加入就绪链表
    list_add_tail(&epi->rdllink, &ep->rdllist);

    // 唤醒 epoll_wait
    if (waitqueue_active(&ep->wq))
        wake_up(&ep->wq);

    return 1;
}

LT vs ET 触发模式

模式全称行为注意事项
LTLevel Triggered只要 fd 处于就绪状态，每次 epoll_wait 都会返回默认模式，编程简单
ETEdge Triggered仅在 fd 状态变化时通知一次必须一次性读完所有数据，必须使用非阻塞 fd
LT（水平触发）— 只要有数据就绪就一直通知
     ┌──────────┐
     │ 数据到达  │ → epoll_wait 返回 ✓
     │ 数据未读完│ → epoll_wait 再次返回 ✓
     │ 数据读完  │ → epoll_wait 不再返回
     └──────────┘

ET（边缘触发）— 只在状态变化时通知一次
     ┌──────────┐
     │ 数据到达  │ → epoll_wait 返回 ✓
     │ 数据未读完│ → epoll_wait 不再返回 ✗  ← 必须读完！
     │ 新数据到达│ → epoll_wait 再次返回 ✓
     └──────────┘

为什么 epoll 是 O(1)

epoll 的高效来自于三个关键设计：

  • RB 红黑树（eventpoll.rbr）：存储所有监听的 fd，epoll_ctl ADD/DEL 为 O(log n)
  • 回调注册：epoll_ctl 时通过 poll_wait 注册 ep_poll_callback，fd 就绪时直接将 epitem 加入 rdllist
  • 就绪链表（eventpoll.rdllist）：epoll_wait 只遍历就绪的 fd，不扫描全部
// 对比：
// select/poll: epoll_wait → O(n) 扫描全部 fd
// epoll:       epoll_wait → O(1) 取 rdllist 首元素
//
// select/poll: 每次调用都需要拷贝 fd 集合到内核
// epoll:       fd 集合通过 epoll_ctl 持久保存在红黑树中

06 常见陷阱

1. epoll ET 模式未读完数据
// 错误：只读了一部分数据就返回
n = read(fd, buf, 1024);  // 可能只读了一部分

// 正确：循环读取直到 EAGAIN
while ((n = read(fd, buf, 1024)) > 0) { ... }
if (n == -1 && errno == EAGAIN) break;  // 读完了
2. ET 模式使用阻塞 fd
// 错误：阻塞 fd 在 ET 模式下 read 可能永久阻塞
n = read(fd, buf, 1024);  // 如果没有数据，永远阻塞

// 正确：ET 模式必须使用非阻塞 fd
int flags = fcntl(fd, F_GETFL);
fcntl(fd, F_SETFL, flags | O_NONBLOCK);
3. fd 被 close 后 epoll 自动删除
// epoll 会自动从红黑树中移除被 close 的 fd
// 但 epoll_ctl(EPOLL_CTL_DEL) 是更好的实践
close(fd);                   // 内核自动清理
// 或者：
epoll_ctl(epfd, EPOLL_CTL_DEL, fd, NULL);  // 显式删除
close(fd);
4. signal handler 中的 EINTR
// select/poll/epoll_wait 可能被信号中断
while (1) {
    n = epoll_wait(epfd, events, max, timeout);
    if (n == -1) {
        if (errno == EINTR) continue;  // 被信号中断，重试
        perror("epoll_wait");
    }
}
