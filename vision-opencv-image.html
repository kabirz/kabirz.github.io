01 图像读写与显示

1.1 读取图像 imread

cv2.imread() 是 OpenCV 中最基本的图像读取函数。它返回一个 numpy ndarray，形状为 (height, width, channels)，通道顺序为 BGR（而非常见的 RGB）。

import cv2

# 读取方式标志
# cv2.IMREAD_COLOR      = 1  — 彩色模式（默认），忽略 alpha 通道
# cv2.IMREAD_GRAYSCALE  = 0  — 灰度模式
# cv2.IMREAD_UNCHANGED  = -1 — 原样读取（含 alpha 通道）

img_color  = cv2.imread("photo.jpg", cv2.IMREAD_COLOR)
img_gray   = cv2.imread("photo.jpg", cv2.IMREAD_GRAYSCALE)
img_alpha  = cv2.imread("photo.png", cv2.IMREAD_UNCHANGED)

# 读取失败检查
if img_color is None:
    print("图像文件不存在或路径错误")

# 查看图像属性
print(img_color.shape)     # (480, 640, 3)
print(img_color.dtype)     # uint8
print(img_color.size)      # 921600 = 480 * 640 * 3
imread 注意事项:
OpenCV 默认使用 BGR 通道顺序，与常见的 RGB 相反。显示时如果使用 matplotlib，需要先用 cv2.cvtColor(img, cv2.COLOR_BGR2RGB) 转换。imread 不支持读取中文路径（在部分平台上），建议先使用 np.fromfile() 绕过此限制。

1.2 写入图像 imwrite

cv2.imwrite() 根据文件扩展名自动选择合适的编码格式：

# 基本写入
cv2.imwrite("output.jpg", img_color)

# JPEG 质量参数（0–100，默认 95）
cv2.imwrite("high_quality.jpg", img_color,
            [cv2.IMWRITE_JPEG_QUALITY, 95])

# PNG 压缩级别（0–9，默认 3）
cv2.imwrite("lossless.png", img_color,
            [cv2.IMWRITE_PNG_COMPRESSION, 0])

# WebP 质量参数（0–100）
cv2.imwrite("output.webp", img_color,
            [cv2.IMWRITE_WEBP_QUALITY, 80])

1.3 显示图像 imshow

# 创建窗口并显示
cv2.imshow("Window Title", img_color)

02 色彩空间转换

色彩空间转换是图像处理中最基础也最重要的操作之一。cv2.cvtColor() 是 OpenCV 中用于色彩空间转换的函数，支持 150 种以上的转换方式。

2.1 常用色彩空间

色彩空间表示方式典型用途
BGR蓝、绿、红三通道（OpenCV 默认）相机/图像文件原始输出
GRAY单通道灰度（0–255）边缘检测、特征提取、降低计算量
HSV色相(H 0–180)、饱和度(S 0–255)、明度(V 0–255)颜色分割、对象跟踪（对光照变化鲁棒）
Lab亮度(L)、a(绿–红)、b(蓝–黄)颜色差异度量、色彩校正
YCrCb亮度(Y)、红色差(Cr)、蓝色差(Cb)JPEG 压缩、肤色检测
XYZCIE 1931 标准三刺激值颜色管理、跨设备色彩匹配

2.2 代码示例

import cv2
import numpy as np

img = cv2.imread("photo.jpg")

# BGR → Gray
gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

# BGR → HSV
hsv = cv2.cvtColor(img, cv2.COLOR_BGR2HSV)

# BGR → Lab
lab = cv2.cvtColor(img, cv2.COLOR_BGR2Lab)

# BGR → YCrCb
ycr = cv2.cvtColor(img, cv2.COLOR_BGR2YCrCb)

# BGR → RGB（用于 matplotlib 显示）
rgb = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)

2.3 HSV 颜色范围与物体分割

HSV 空间广泛用于基于颜色的对象分割。色相 H 在 OpenCV 中的范围是 0–180（而不是 0–360），饱和度 S 和明度 V 的范围是 0–255。

# 在 HSV 空间中提取绿色物体
hsv = cv2.cvtColor(img, cv2.COLOR_BGR2HSV)

# 绿色的 HSV 范围（需根据具体光照微调）
lower_green = np.array([40, 40, 40])
upper_green = np.array([80, 255, 255])

# 二值掩码
mask = cv2.inRange(hsv, lower_green, upper_green)

# 与原图进行按位与，提取绿色区域
result = cv2.bitwise_and(img, img, mask=mask)

cv2.imshow("Mask", mask)
cv2.imshow("Result", result)
HSV 调参技巧:
使用 cv2.createTrackbar() 创建滑块，在实际图像上实时调节 H/S/V 上下限，是确定颜色范围最有效的方法。常见颜色的 H 基准：红色 (0/170)、黄色 (30)、绿色 (60)、青色 (90)、蓝色 (120)、紫色 (150)。

03 几何变换

3.1 缩放 resize

cv2.resize() 是图像缩放的核心函数，支持多种插值方法以适应不同场景：

插值方法适用场景计算速度
INTER_NEAREST最近邻（像素放大、缩略图）极快
INTER_LINEAR双线性（默认，通用缩放）快
INTER_CUBIC双三次（放大图像，质量较好）较慢
INTER_LANCZOS4Lanczos（缩小图像，抗锯齿最优）最慢
INTER_AREA区域插值（缩小图像，避免摩尔纹）中等
import cv2

img = cv2.imread("photo.jpg")
h, w = img.shape[:2]

# 指定尺寸
resized = cv2.resize(img, (224, 224), interpolation=cv2.INTER_LINEAR)

# 按比例缩放（宽度减半）
scale = 0.5
resized_scale = cv2.resize(img, None, fx=scale, fy=scale,
                            interpolation=cv2.INTER_AREA)

# 自适应长边缩放（保持宽高比）
max_size = 640
if max(h, w) > max_size:
    scale = max_size / max(h, w)
    new_w, new_h = int(w * scale), int(h * scale)
    resized_aspect = cv2.resize(img, (new_w, new_h))

3.2 平移与仿射变换 warpAffine

仿射变换是线性变换（旋转、缩放、剪切）加平移的组合，变换后直线仍然是直线，平行线仍然平行。使用 2x3 矩阵表示。

import cv2
import numpy as np

img = cv2.imread("photo.jpg")
h, w = img.shape[:2]

# 平移（tx=100, ty=50）
M_trans = np.float32([[1, 0, 100],
                       [0, 1,  50]])
translated = cv2.warpAffine(img, M_trans, (w, h))

# 旋转（绕中心旋转 45 度，缩放 0.8）
center = (w // 2, h // 2)
M_rot = cv2.getRotationMatrix2D(center, angle=45, scale=0.8)
rotated = cv2.warpAffine(img, M_rot, (w, h))

# 仿射变换（三点确定）
pts_src = np.float32([[50, 50], [200, 50], [50, 200]])
pts_dst = np.float32([[10, 100], [200, 50], [100, 250]])
M_affine = cv2.getAffineTransform(pts_src, pts_dst)
affined = cv2.warpAffine(img, M_affine, (w, h))

3.3 透视变换 warpPerspective

透视变换（投影变换）使用 3x3 矩阵，可以将矩形投影为任意四边形。常用于文档矫正、视角调整。

# 透视变换 — 四点确定
# 假设图像中有个矩形区域需要矫正
pts_src = np.float32([[141, 131], [480, 159],
                       [493, 630], [64, 601]])
pts_dst = np.float32([[0, 0], [600, 0],
                       [600, 800], [0, 800]])

M_persp = cv2.getPerspectiveTransform(pts_src, pts_dst)
warped = cv2.warpPerspective(img, M_persp, (600, 800))

cv2.imshow("Original", img)
cv2.imshow("Corrected", warped)

3.4 翻转与裁剪

# 翻转
flip_h = cv2.flip(img, 1)     # 水平翻转
flip_v = cv2.flip(img, 0)     # 垂直翻转
flip_hv = cv2.flip(img, -1)   # 同时翻转

# 裁剪（numpy 切片）

# 旋转 90/180/270
r90  = cv2.rotate(img, cv2.ROTATE_90_CLOCKWISE)

r270 = cv2.rotate(img, cv2.ROTATE_90_COUNTERCLOCKWISE)

04 图像滤波

图像滤波是图像预处理的核心步骤，用于去噪、平滑和增强。OpenCV 提供多种滤波方法，分为线性滤波和非线性滤波两大类。

4.1 均值滤波 blur

均值滤波使用核内所有像素的平均值替换中心像素，是最简单的平滑滤波器。核越大，平滑效果越强，但也会丢失更多细节。

import cv2
import numpy as np

img = cv2.imread("noisy.jpg")

# 均值滤波，核大小 5x5
blur_3  = cv2.blur(img, (3, 3))
blur_5  = cv2.blur(img, (5, 5))
blur_15 = cv2.blur(img, (15, 15))

4.2 高斯滤波 GaussianBlur

高斯滤波使用高斯分布的权重对邻域进行加权平均，核中心权重最大，边缘权重递减。它能更好地保留边缘信息，是实际项目中最常用的平滑方法。

# 高斯滤波
# 参数： (src, ksize, sigmaX, sigmaY=sigmaX)
gauss_3  = cv2.GaussianBlur(img, (3, 3),   sigmaX=1.0)
gauss_5  = cv2.GaussianBlur(img, (5, 5),   sigmaX=1.5)
gauss_9  = cv2.GaussianBlur(img, (9, 9),   sigmaX=3.0)

# sigma 自动计算（sigma = 0.3 * ((ksize-1)*0.5 - 1) + 0.8）
gauss_auto = cv2.GaussianBlur(img, (5, 5), sigmaX=0)

4.3 中值滤波 medianBlur

中值滤波用核内所有像素的中位数替换中心像素，对椒盐噪声（salt-and-pepper noise）有极好的去除效果。核大小必须为奇数。

# 中值滤波（核大小必须为奇数）
median_3  = cv2.medianBlur(img, 3)
median_5  = cv2.medianBlur(img, 5)
median_9  = cv2.medianBlur(img, 9)

# 中值滤波特别适合去除椒盐噪声
# 与均值滤波对比：中值不会产生模糊的"鬼影"

4.4 双边滤波 bilateralFilter

双边滤波在平滑的同时能够保留边缘，因为它结合了空间距离权重和像素值相似度权重。但计算量较大，不适合实时应用。

# 双边滤波
# 参数： (src, d, sigmaColor, sigmaSpace)
# d      — 邻域直径
# sigmaColor — 颜色空间的标准差（越大，越多的颜色被混合）
# sigmaSpace — 坐标空间的标准差（越大，越远的像素被影响）
bilateral = cv2.bilateralFilter(
    img, d=9, sigmaColor=75, sigmaSpace=75
)

# 双边滤波的效果类似美颜滤镜

4.5 自定义卷积 filter2D

OpenCV 允许通过 filter2D() 使用任意的卷积核进行滤波，适用于自定义的锐化、浮雕等效果。

# 锐化核
kernel_sharpen = np.array([
    [ 0, -1,  0],
    [-1,  5, -1],
    [ 0, -1,  0]
], dtype=np.float32)

# 浮雕核
kernel_emboss = np.array([
    [-2, -1,  0],
    [-1,  1,  1],
    [ 0,  1,  2]
], dtype=np.float32)

# 应用自定义卷积
sharpened = cv2.filter2D(img, -1, kernel_sharpen)
embossed  = cv2.filter2D(img, -1, kernel_emboss)

# 第三个参数 ddepth = -1 表示输出与输入相同的深度
滤波选型指南:
去高斯噪声 → 高斯滤波；去椒盐噪声 → 中值滤波；需要美颜/保留边缘 → 双边滤波；最快的大面积模糊 → 均值滤波。在实时视频处理中，高斯滤波 + 降采样是最常用的组合。

05 图像梯度

图像梯度测量像素值的变化率和变化方向，是边缘检测和特征提取的基础。OpenCV 提供 Sobel、Scharr 和 Laplacian 三种主要梯度算子。

5.1 Sobel 算子

Sobel 算子结合了高斯平滑和差分运算，对噪声有一定的抑制能力。它计算水平方向（x）和垂直方向（y）的梯度。

import cv2
import numpy as np

img = cv2.imread("photo.jpg", cv2.IMREAD_GRAYSCALE)

# Sobel 梯度（ddepth 使用 CV_64F 避免负值被截断）
sobel_x = cv2.Sobel(img, ddepth=cv2.CV_64F, dx=1, dy=0, ksize=3)
sobel_y = cv2.Sobel(img, ddepth=cv2.CV_64F, dx=0, dy=1, ksize=3)

# 取绝对值并转换为 uint8
sobel_x_abs = cv2.convertScaleAbs(sobel_x)
sobel_y_abs = cv2.convertScaleAbs(sobel_y)

# 梯度幅值（近似计算）
sobel_combined = cv2.addWeighted(sobel_x_abs, 0.5,
                                  sobel_y_abs, 0.5, 0)

5.2 Scharr 算子

Scharr 算子是 Sobel 在 ksize=3 时的优化版本，提供更高的梯度精度。当 Sobel 的 ksize=3 时，Scharr 给出的结果更准确。

# Scharr 算子（固定 3x3 核，精度比 Sobel 高）
scharr_x = cv2.Scharr(img, ddepth=cv2.CV_64F, dx=1, dy=0)
scharr_y = cv2.Scharr(img, ddepth=cv2.CV_64F, dx=0, dy=1)

scharr_x_abs = cv2.convertScaleAbs(scharr_x)
scharr_y_abs = cv2.convertScaleAbs(scharr_y)
scharr_combined = cv2.addWeighted(scharr_x_abs, 0.5,
                                    scharr_y_abs, 0.5, 0)

# Scharr 核系数（x 方向）
# [[ -3, 0,  3],
#  [-10, 0, 10],
#  [ -3, 0,  3]]

5.3 Laplacian 算子

Laplacian 算子是二阶微分算子，对噪声非常敏感。它检测图像强度的二阶变化，在边缘处产生零交叉（zero-crossing）。

# Laplacian 算子（二阶梯度）
laplacian = cv2.Laplacian(img, ddepth=cv2.CV_64F, ksize=3)
laplacian_abs = cv2.convertScaleAbs(laplacian)

# Laplacian 对噪声敏感，通常先做高斯平滑
blurred = cv2.GaussianBlur(img, (5, 5), 1.0)
laplacian_blur = cv2.Laplacian(blurred, ddepth=cv2.CV_64F, ksize=3)
laplacian_blur_abs = cv2.convertScaleAbs(laplacian_blur)

5.4 梯度计算对比

算子阶数对噪声敏感度精度适用场景
Sobel一阶中等中等通用梯度计算
Scharr一阶中等较高需要精确梯度方向
Laplacian二阶高中等边缘定位、零交叉

06 边缘检测

6.1 Canny 算法原理

Canny 边缘检测是业界最广泛使用的边缘检测算法，由 John F. Canny 在 1986 年提出。它包含五个步骤：

  1. 高斯滤波去噪 — 减少噪声对梯度计算的影响
  2. 计算梯度幅值和方向 — 使用 Sobel 算子
  3. 非极大值抑制（NMS） — 沿梯度方向只保留局部最大值，细化为单像素边缘
  4. 双阈值检测 — 高阈值确定强边缘，低阈值保留弱边缘
  5. 滞后边缘跟踪 — 弱边缘只有与强边缘相连时保留
import cv2
import numpy as np

img = cv2.imread("photo.jpg", cv2.IMREAD_GRAYSCALE)

# Canny 边缘检测
# 参数： (src, threshold1=50, threshold2=150)
# threshold1 — 低阈值（弱边缘）
# threshold2 — 高阈值（强边缘）

edges_default = cv2.Canny(img, 50, 150)

# 先做高斯滤波再 Canny（糖葫芦策略）
blurred = cv2.GaussianBlur(img, (5, 5), 1.4)
edges_better = cv2.Canny(blurred, 30, 100)

# 自适应阈值（基于中位数）
median_val = np.median(img)
lower = int(max(0, 0.7 * median_val))
upper = int(min(255, 1.3 * median_val))
edges_adaptive = cv2.Canny(img, lower, upper)

# L2 梯度计算（更精确，默认 L1）
edges_l2 = cv2.Canny(img, 50, 150, L2gradient=True)
双阈值设定经验:
高阈值：低阈值 ≈ 2:1 到 3:1。高阈值越高，检测到的边缘越少（只保留最强边缘）；低阈值越低，保留的细节越多（但也可能引入噪声）。推荐先使用 cv2.Canny(img, 50, 150) 作为起点，再根据结果微调。

07 形态学操作

形态学操作基于图像形状（形态）进行处理，常用于二值图像处理。操作的核心是一个称为结构元素（kernel/structuring element）的小矩阵。

7.1 腐蚀 erode 与膨胀 dilate

腐蚀使前景区域"收缩"，可以去除小的白色噪点；膨胀使前景区域"扩张"，可以填补物体内部的空洞。

import cv2
import numpy as np

# 创建二值图像（假设已经过阈值处理）
_, binary = cv2.threshold(gray, 127, 255, cv2.THRESH_BINARY)

# 创建结构元素
kernel = np.ones((5, 5), dtype=np.uint8)

# 腐蚀 — 前景缩小，去除小噪点
eroded = cv2.erode(binary, kernel, iterations=1)

# 膨胀 — 前景扩大，填补空洞
dilated = cv2.dilate(binary, kernel, iterations=1)

7.2 开运算与闭运算

开运算 = 先腐蚀后膨胀，用于去除小噪点并保持物体大小不变。闭运算 = 先膨胀后腐蚀，用于填补小空洞并保持物体大小不变。

# 开运算 — 去除小白点（噪点）
opening = cv2.morphologyEx(binary, cv2.MORPH_OPEN, kernel)

# 闭运算 — 填补小黑洞
closing = cv2.morphologyEx(binary, cv2.MORPH_CLOSE, kernel)

# 可以用不同形状的结构元素
kernel_ellipse = cv2.getStructuringElement(cv2.MORPH_ELLIPSE, (5, 5))
kernel_cross  = cv2.getStructuringElement(cv2.MORPH_CROSS, (5, 5))

opening_ellipse = cv2.morphologyEx(
    binary, cv2.MORPH_OPEN, kernel_ellipse
)

7.3 形态学梯度

形态学梯度 = 膨胀 − 腐蚀，可以得到物体的轮廓：

# 形态学梯度 — 物体的外轮廓
gradient = cv2.morphologyEx(binary, cv2.MORPH_GRADIENT, kernel)

# 顶帽 = 原图 − 开运算（提取亮区域中的小亮点）
tophat = cv2.morphologyEx(binary, cv2.MORPH_TOPHAT, kernel)

# 黑帽 = 闭运算 − 原图（提取暗区域中的小暗点）
blackhat = cv2.morphologyEx(binary, cv2.MORPH_BLACKHAT, kernel)

形态学操作在文档图像处理、车牌定位、细胞计数等领域有广泛应用：

# 实际案例 — 去除文字周围的噪点
# 先开运算去噪，再闭运算连接断开的笔画
cleaned = cv2.morphologyEx(
    binary, cv2.MORPH_OPEN,
    cv2.getStructuringElement(cv2.MORPH_ELLIPSE, (3, 3))
)
cleaned = cv2.morphologyEx(
    cleaned, cv2.MORPH_CLOSE,
    cv2.getStructuringElement(cv2.MORPH_ELLIPSE, (5, 5))
)

08 直方图

8.1 直方图计算 calcHist

图像直方图统计每个像素值（0–255）出现的频次，是图像亮度分布的可视化表示。它在对比度调整、图像分割和特征提取中有重要用途。

import cv2
import numpy as np
import matplotlib.pyplot as plt

img = cv2.imread("photo.jpg")
gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

# 计算灰度直方图
# 参数：(图像, 通道, 掩码, bins范围, 像素值范围)
hist = cv2.calcHist([gray], [0], None, [256], [0, 256])

# 绘制直方图
plt.figure(figsize=(10, 4))
plt.plot(hist, color='black', linewidth=1.5)
plt.title("Grayscale Histogram")
plt.xlabel("Pixel Value")
plt.ylabel("Frequency")
plt.xlim([0, 255])
plt.grid(alpha=0.3)
plt.show()

# 彩色直方图（三通道）
colors = ('b', 'g', 'r')
for i, color in enumerate(colors):
    hist_color = cv2.calcHist([img], [i], None, [256], [0, 256])
    plt.plot(hist_color, color=color, linewidth=1.5)

plt.title("Color Histogram (BGR)")
plt.show()

8.2 直方图均衡化 equalizeHist

直方图均衡化通过将像素值分布拉伸到全范围，自动提升图像对比度。尤其适用于过暗或过曝光图像的增强。

# 灰度图直方图均衡化
equ = cv2.equalizeHist(gray)

# 对比显示
result = np.hstack([gray, equ])
cv2.imshow("Original vs Equalized", result)

# 彩色图像的 CLAHE 自适应均衡化
# 对亮度通道做均衡，避免色彩失真
lab = cv2.cvtColor(img, cv2.COLOR_BGR2Lab)
l, a, b = cv2.split(lab)

# CLAHE（对比度限制的自适应直方图均衡化）
# clipLimit — 对比度限制（防止噪声放大）
# tileGridSize — 分块大小
clahe = cv2.createCLAHE(clipLimit=3.0, tileGridSize=(8, 8))
l_eq = clahe.apply(l)

# 合并并转回 BGR
lab_eq = cv2.merge([l_eq, a, b])
img_clahe = cv2.cvtColor(lab_eq, cv2.COLOR_Lab2BGR)

8.3 直方图比较

OpenCV 提供了多种直方图相似度比较方法，用于图像检索和匹配：

# 计算直方图并归一化
hist1 = cv2.calcHist([gray1], [0], None, [256], [0, 256])
hist2 = cv2.calcHist([gray2], [0], None, [256], [0, 256])

cv2.normalize(hist1, hist1, 0, 1, cv2.NORM_MINMAX)
cv2.normalize(hist2, hist2, 0, 1, cv2.NORM_MINMAX)

# 比较方法
compare_methods = [
    (cv2.HISTCMP_CORREL,      "相关性 (1=最相似)"),
    (cv2.HISTCMP_CHISQR,      "卡方 (0=最相似)"),
    (cv2.HISTCMP_INTERSECT,   "交集 (高=相似)"),
    (cv2.HISTCMP_BHATTACHARYYA, "巴氏距离 (0=最相似)")
]

for method, name in compare_methods:
    score = cv2.compareHist(hist1, hist2, method)
    print(f"{name}: {score:.4f}")

8.4 反向投影与模板匹配

直方图反向投影可用于在图像中定位特定颜色或纹理区域：

# 直方图反向投影
# 在图像中寻找与目标直方图相似的颜色区域
roi = cv2.imread("template.jpg")  # 目标区域
hsv_roi = cv2.cvtColor(roi, cv2.COLOR_BGR2HSV)
hsv_img = cv2.cvtColor(img, cv2.COLOR_BGR2HSV)

# 计算 ROI 直方图
roi_hist = cv2.calcHist([hsv_roi], [0, 1], None,
                        [180, 256], [0, 180, 0, 256])
cv2.normalize(roi_hist, roi_hist, 0, 255, cv2.NORM_MINMAX)

# 反向投影
mask = cv2.calcBackProject([hsv_img], [0, 1],
                            roi_hist, [0, 180, 0, 256], 1)

# 使用圆盘核将分散的高亮区域连接
disc = cv2.getStructuringElement(cv2.MORPH_ELLIPSE, (5, 5))
cv2.filter2D(mask, -1, disc, mask)
直方图应用总结:
直方图均衡化适合全局对比度偏低的图像；CLAHE 适合局部光照不均匀的图像；直方图比较适合图像检索和重复检测；反向投影适合颜色跟踪和肤色检测。在医学影像和遥感图像中，直方图处理是必备的预处理步骤。
