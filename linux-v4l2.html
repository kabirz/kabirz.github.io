01 概述与架构

V4L2（Video for Linux 2）是 Linux 内核中视频设备驱动的标准框架，定义于 1999 年，取代了早期的 V4L1 API。它提供了统一的接口用于视频采集、输出、VBI（Vertical Blanking Interval）、radio、SDR（Software Defined Radio）和 touch 设备。

V4L2 采用 字符设备模型，主设备号 81，次设备号动态分配。设备节点通常位于 /dev/videoX、/dev/vbiX、/dev/radioX、/dev/swradioX 等。用户空间通过 ioctl() 系统调用与驱动交互。

02 核心数据结构

定义于 include/media/v4l2-dev.h，是 V4L2 框架中最核心的结构体，代表一个 V4L2 设备节点：

struct video_device {

    int index;               // 设备编号
    int minor;               // 次设备号
    u32 num;                 // 视频节点编号

    /* 标准 ioctl 缓存 */
    struct v4l2_capability capability;
    struct v4l2_audio vin;
    struct v4l2_audioout vaud;
    struct v4l2_tuner tuner;
    struct v4l2_modulator modulator;
    struct v4l2_output output;
    struct v4l2_input input;

    /* 队列 */
    spinlock_t fh_lock;      // 文件句柄锁

03 设备注册与初始化

V4L2 设备的注册流程包含三个层次：v4l2_device 注册、video_device 注册和 v4l2_subdev 注册。

v4l2_device 注册

v4l2_device_register() 在 v4l2-device.c（约 295 行）中实现：

int v4l2_device_register(struct device *dev, struct v4l2_device *v4l2_dev)
{
    if (v4l2_dev == NULL)
        return -EINVAL;
    INIT_LIST_HEAD(&v4l2_dev->subdevs);
    spin_lock_init(&v4l2_dev->lock);
    v4l2_dev->dev = dev;
    if (dev == NULL) {
        /* 如果 dev 为 NULL，使用内核虚拟设备 */
        if (v4l2_device_register_empty)
            return -EBUSY;
        v4l2_device_register_empty = v4l2_dev;
        return 0;
    }
    dev_set_drvdata(dev, v4l2_dev);
    return 0;
}
EXPORT_SYMBOL_GPL(v4l2_device_register);

video_device 注册

__video_register_device() 在 v4l2-dev.c（约 1258 行）中实现，流程如下：

  1. 分配次设备号——video_register_device() 调用 ida_alloc() 从 video_device_nr[] 池分配，或使用指定编号
  2. 初始化字符设备——cdev_init(&vdev->cdev, &v4l2_fops) 绑定 v4l2_fops
  3. 创建设备——device_register() 在 /dev 下创建设备节点
  4. 设置 vdev->minor 和 vdev->index
  5. 调用 media_device_register_entity() 注册媒体实体（如果启用了媒体控制器）
static int __video_register_device(struct video_device *vdev,
    enum vcd_device_type type, int nr, int warn_on_nr_alloc,
    struct module *owner)
{
    int ret = 0;
    int minor_offset = 0;
    int minor_cnt = VIDEO_NUM_DEVICES;
    const char *name_base;

    switch (type) {
    case VFL_TYPE_VIDEO:
        name_base = "video";
        break;
    case VFL_TYPE_VBI:
        name_base = "vbi";
        break;
    case VFL_TYPE_RADIO:
        name_base = "radio";
        break;
    case VFL_TYPE_SDR:
        name_base = "swradio";
        break;
    case VFL_TYPE_TOUCH:
        name_base = "touch";
        break;
    case VFL_TYPE_SUBDEV:
        name_base = "v4l-subdev";
        break;
    }
    /* ... */
    vdev->cdev = cdev_alloc();
    vdev->cdev->ops = &v4l2_fops;
    ret = cdev_add(vdev->cdev, MKDEV(VIDEO_MAJOR, vdev->minor), 1);
    if (ret < 0) {
        pr_err("%s: cdev_add failed\n", __func__);
        goto cleanup;
    }

    /* 向 udev 发送 UEVID 事件 */
    dev = device_create(&video_device_class, vdev->dev_parent,
            MKDEV(VIDEO_MAJOR, vdev->minor), vdev, "%s%d",
            name_base, vdev->num);
    if (IS_ERR(dev)) {
        ret = PTR_ERR(dev);
        goto cleanup;
    }

    /* 注册媒体实体 */
    if (vdev->entity)
        media_device_register_entity(vdev->v4l2_dev->mdev,
                      vdev->entity);
    return 0;
}

04 ioctl 分发机制

v4l2-ioctl.c（约 3534 行）是 V4L2 框架中最大的源文件，实现了 ioctl 的标准处理逻辑。核心入口是 video_ioctl2()：

long video_ioctl2(struct file *file, unsigned int cmd, unsigned long arg)
{
    struct video_device *vdev = video_devdata(file);
    const struct v4l2_ioctl_ops *ops = vdev->ioctl_ops;
    struct v4l2_fh *fh = file->private_data;
    int ret = -ENOTTY;

    if (ops == NULL)
        return ret;

    /* 查找并调用对应的 ioctl 处理函数 */
    ret = v4l2_ioctl(file, cmd, arg);
    return ret;
}
EXPORT_SYMBOL_GPL(video_ioctl2);

ioctl 描述符表

V4L2 使用 v4l2_ioctl_info 描述符表来管理所有标准 ioctl。每个描述符包含 ioctl 编号、优先级、标志和处理函数指针：

struct v4l2_ioctl_info {
    unsigned int ioctl;               // ioctl 命令字
    u32 flags;                         // 标志
    const char *name;                  // 名称（调试用）
    int (*func)(const struct v4l2_ioctl_ops *ops,
                struct file *file, void *arg);
    /* 辅助函数 */
    void (*fill)(const struct v4l2_ioctl_ops *ops,
                 struct file *file, void *arg);
    int (*req)(struct media_request *req, struct file *file,
               void *arg);
};

该框架自动处理以下通用逻辑：

  • 信息查询——QUERYCAP、ENUMINPUT、ENUMOUTPUT、ENUMSTD、ENUM_DV_TIMINGS
  • 格式协商——FMT、S_FMT、TRY_FMT、ENUM_FMT、ENUM_FRAMESIZES、ENUM_FRAMEINTERVALS
  • 缓冲管理——REQBUFS、QUERYBUF、QBUF、DQBUF、EXPBUF、STREAMON、STREAMOFF
  • 控制——QUERYCTRL、G_CTRL、S_CTRL、G_EXT_CTRLS、S_EXT_CTRLS、TRY_EXT_CTRLS
  • 裁剪与选择——CROPCAP、G_CROP、S_CROP、G_SELECTION、S_SELECTION
  • 参数与流控制——G_PARM、S_PARM、G_PRIORITY、S_PRIORITY

核心 ioctl 处理流程

/* v4l2_ioctl() 的主 switch 分发 */
static long v4l2_ioctl(struct file *file, unsigned int cmd, unsigned long arg)
{
    struct video_device *vdev = video_devdata(file);
    const struct v4l2_ioctl_ops *ops = vdev->ioctl_ops;
    void *parg = (void __user *)arg;
    long ret = -ENOTTY;

    /* 1. 通过 ioctl 描述符表查找命令 */
    const struct v4l2_ioctl_info *info = v4l2_ioctl_find(cmd);
    if (info == NULL)
        return ret;

    /* 2. 检查访问权限和优先级 */
    ret = v4l2_ioctl_check_prio(vdev, cmd, file);
    if (ret < 0)
        return ret;

    /* 3. 复制用户空间参数 */
    ret = v4l2_ioctl_alloc_and_copy(info, arg, true);

    /* 4. 调用驱动实现 */
    ret = info->func(ops, file, parg);

    /* 5. 复制结果回用户空间 */
    v4l2_ioctl_alloc_and_copy(info, arg, false);

    return ret;
}

05 videobuf2 缓冲框架

videobuf2（VB2）是 V4L2 的缓冲管理框架，位于 drivers/media/common/videobuf2/。它取代了旧的 videobuf，提供了统一的内存管理接口，支持多种内存访问模式。

核心结构

struct vb2_queue {
    enum v4l2_buf_type            type;       // 缓冲类型
    unsigned int                  io_modes;   // 支持的 I/O 模式
    unsigned int                  buf_struct_size; // 驱动私有数据大小
    const struct vb2_ops          *ops;       // 操作函数表
    const struct vb2_mem_ops      *mem_ops;   // 内存操作函数表

    /* 缓冲数组 */
    struct vb2_buffer             **bufs;
    unsigned int                  num_buffers;

    /* 缓冲队列 */
    struct list_head              queued_list;    // 已入队
    struct list_head              done_list;      // 已完成
    spinlock_t                    done_lock;
    wait_queue_head_t             done_wq;

    /* 文件句柄 */
    struct vb2_fileio_data        *fileio;

    /* 状态 */
    unsigned int                  streaming:1;
    unsigned int                  start_streaming_called:1;
    unsigned int                  error:1;
    unsigned int                  waiting_for_buffers:1;
    unsigned int                  is_multiplanar:1;

    /* 驱动回调 */
    void (*buf_queue)(struct vb2_buffer *vb);
    int (*start_streaming)(struct vb2_queue *q, unsigned int count);
    void (*stop_streaming)(struct vb2_queue *q);
    int (*wait_prepare)(struct vb2_queue *q);
    void (*wait_finish)(struct vb2_queue *q);

    /* 内存相关 */
    struct device                  *dev;
    struct vb2_mem_ops             *mem_ops;
    struct vb2_plane               planes[VB2_MAX_PLANES];
};

四步缓冲生命周期

  1. REQBUFS——请求分配缓冲。调用 vb2_core_reqbufs()，经过 __vb2_queue_alloc() 分配 vb2_buffer 对象，通过 mem_ops->alloc() 分配实际内存。
  2. QUERYBUF——查询缓冲信息。返回缓冲的长度、偏移、内存地址等信息给用户空间。
  3. QBUF——将缓冲入队。调用 vb2_core_qbuf() → __enqueue_in_driver() → 驱动 buf_queue() 回调。
  4. DQBUF——从已完成队列出队。调用 vb2_core_dqbuf()，等待 done_wq 条件变量。

三种 I/O 模式

模式标志说明
MMAPV4L2_MEMORY_MMAP驱动分配内存，通过 mmap() 映射到用户空间。最常用，性能最好。
USERPTRV4L2_MEMORY_USERPTR用户空间分配内存，通过用户指针传入内核。驱动使用 get_user_pages() 锁定页面。
DMABUFV4L2_MEMORY_DMABUF通过 DMA-BUF 文件描述符导入/导出缓冲。用于跨设备零拷贝。

内存后端

后端说明适用场景
vb2_dma_contig_memopsDMA 连续内存，通过 dma_alloc_coherent() 分配大多数摄像头、ISP
vb2_dma_sg_memopsDMA 分散-聚集，通过 dma_map_sg() 映射PCIe 设备、支持 S/G 的硬件

驱动实现这些回调来执行实际的硬件操作。典型的数据流：

/* vivid 驱动中的典型 vb2_ops 实现 */
static int vivid_queue_setup(struct vb2_queue *vq,
    unsigned int *nbuffers, unsigned int *nplanes,
    unsigned int sizes[], struct device *alloc_devs[])
{
    struct vivid_dev *dev = vb2_get_drv_priv(vq);
    /* 计算所需的缓冲大小 */
    sizes[0] = dev->fmt_rect.width * dev->fmt_rect.height * 2;
    *nplanes = 1;
    return 0;
}

static void vivid_buf_queue(struct vb2_buffer *vb)
{
    struct vivid_dev *dev = vb2_get_drv_priv(vb->vb2_queue);
    struct vivid_buffer *buf = container_of(vb, struct vivid_buffer, vb);
    spin_lock(&dev->slock);
    list_add_tail(&buf->list, &dev->vidcap_active);
    spin_unlock(&dev->slock);
}

static int vivid_start_streaming(struct vb2_queue *vq, unsigned int count)
{
    struct vivid_dev *dev = vb2_get_drv_priv(vq);
    dev->streaming = true;
    /* 启动硬件 DMA */
    schedule_work(&dev->capture_work);
    return 0;
}

06 控制框架

V4L2 控制框架（Control Framework）是驱动和用户空间之间交换参数的标准机制，实现于 v4l2-ctrls-core.c（约 2830 行）、v4l2-ctrls-api.c、v4l2-ctrls-defs.c 和 v4l2-ctrls-request.c 中。

核心数据结构

struct v4l2_ctrl {
    struct list_head    node;        // 链入 handler 的链表
    struct v4l2_ctrl_handler *handler; // 所属 handler
    struct v4l2_ctrl    **cluster;  // 簇数组（用于复合控制）
    unsigned int        ncontrols;  // 簇大小
    u32                 id;         // 控制 ID（如 V4L2_CID_BRIGHTNESS）
    enum v4l2_ctrl_type type;       // 类型（整数、布尔、菜单等）

    union {
        s64 val;                    // 当前值（整数类）
        char *string;               // 当前值（字符串类）
        u32 val_u32;                // 当前值（u32）
    };

    s64 minimum;      // 最小值
    s64 maximum;      // 最大值
    s64 default_value; // 默认值
    u64 step;          // 步长
    u32 flags;         // V4L2_CTRL_FLAG_*

    const char * const *qmenu;  // 菜单项

控制处理器的生命周期

  1. 初始化——v4l2_ctrl_handler_init() 分配哈希桶和引用计数
  2. 添加控制——v4l2_ctrl_new_std() 添加标准控制、v4l2_ctrl_new_custom() 添加自定义控制
  3. 设置处理函数——handler->notify 回调通知驱动值变化
  4. 挂载到设备——vdev->ctrl_handler = handler 或 sd->ctrl_handler = handler

控制簇（Cluster）

V4L2 控制框架支持将多个控制组合为一个"簇"，实现原子性的同时更新。例如自动曝光/手动曝光模式下，曝光时间和增益可能互斥，通过簇机制确保一致性：

/* 创建 3 个控制的簇 */

           V4L2_CID_EXPOSURE_AUTO, ...);
v4l2_ctrl_auto_cluster(3, &ctrl, 0, true);
/* 簇: [AUTO_EXPOSURE, EXPOSURE_ABSOLUTE, EXPOSURE_GAIN] */

类操作函数

驱动通过 struct v4l2_ctrl_ops 提供控制读写实现：

struct v4l2_ctrl_ops {
    int (*g_volatile_ctrl)(struct v4l2_ctrl *ctrl);
    int (*try_ctrl)(struct v4l2_ctrl *ctrl);
    int (*s_ctrl)(struct v4l2_ctrl *ctrl);
};

s_ctrl 在一个控制值变化时被调用。try_ctrl 用于验证值是否合法但不实际应用。g_volatile_ctrl 用于读取动态变化的值（如自动曝光计算出的实际曝光时间）。

int v4l2_mc_create_media_entity(struct video_device *vdev)
{
    struct media_device *mdev = vdev->v4l2_dev->mdev;
    if (mdev == NULL)
        return 0;

    /* 创建视频设备对应的媒体实体 */

    vdev->entity->function = MEDIA_ENT_F_IO_V4L;

        vdev->pad.flags = MEDIA_PAD_FL_SOURCE;

                   vdev->entity->num_pads,
                   vdev->entity->pads);

请求（Request）机制

媒体请求允许将多个 subdev 控制的配置更改原子性地同时应用。实现于 mc-request.c（约 539 行）：

    struct media_device *mdev;

    struct media_request_object *completed;
};

请求的使用流程：

  1. 用户空间创建请求——MEDIA_IOC_REQUEST_ALLOC
  2. 将多个 SET_CTRL 操作关联到同一个请求——通过 V4L2_EXT_CTRL_FLAG_REQUEST 标志
  3. 将 QBUF 关联到请求——通过 V4L2_BUF_FLAG_REQUEST_FD 标志
  4. 提交请求——MEDIA_REQUEST_IOC_QUEUE
  5. 内核按顺序原子性地应用请求中的所有控制

V4L2 事件机制允许驱动向用户空间发送异步事件通知，实现于 v4l2-event.c（约 373 行）。事件机制基于 struct v4l2_fh，每个文件句柄可以订阅和接收事件。

11 Async 异步 subdev 注册

V4L2 异步框架（v4l2-async.c，约 993 行）解决了摄像头模组中各子设备（传感器、闪光灯、ISP）探测顺序不确定的问题。当 subdev 的驱动在不同时间加载时，需要异步绑定机制。

核心结构

struct v4l2_async_notifier {
    struct list_head list;
    struct v4l2_device *v4l2_dev;
    struct v4l2_async_notifier *parent;
    unsigned int num_subdevs;
    unsigned int num_entries;
    struct list_head done_list;
    struct list_head waiting_list;
    const struct v4l2_async_notifier_operations *ops;
};

struct v4l2_async_connection {
    struct list_head list;
    struct v4l2_async_notifier *notifier;
    enum v4l2_async_match_type match_type;
    union {
        struct fwnode_handle *fwnode;
        const char *device_name;
        struct device_node *of_node;
    };
};

异步绑定流程

  1. 创建 notifier——主设备驱动调用 v4l2_async_nf_init() 初始化 notifier，并设置 .ops（bound()、unbind()、complete()回调）
  2. 添加 subdev——v4l2_async_nf_add_fwnode() / v4l2_async_nf_add_i2c() 添加等待绑定的 subdev
  3. 注册 notifier——v4l2_async_nf_register() 将 notifier 注册到系统中
  4. 匹配绑定——当 subdev 驱动探测时调用 v4l2_async_register_subdev()，框架遍历所有 notifier 的 waiting_list，匹配 fwnode 或 device_name
  5. 绑定完成——匹配成功后调用 notifier->ops->bound()，所有 subdev 绑定时调用 notifier->ops->complete()

{
    /* subdev 端：注册到异步框架 */
    v4l2_async_register_subdev(sd);
}

/* 主控端 (如 isp) */
static int isp_probe(struct platform_device *pdev)
{
    struct v4l2_async_notifier *nf = &isp->notifier;
    v4l2_async_nf_init(nf, &isp->v4l2_dev);
    v4l2_async_nf_add_fwnode(nf, sensor_fwnode,

12 Memory-to-Memory（M2M）

V4L2 M2M 框架（v4l2-mem2mem.c，约 1643 行）专为此类设备设计：设备从输入缓冲读取数据，处理后写入输出缓冲。典型设备包括硬件编码器、解码器、图像缩放器和格式转换器。

核心结构

struct v4l2_m2m_dev {
    struct list_head queues;           // M2M 队列列表
    spinlock_t lock;
    struct list_head running_jobs;     // 运行中的作业
    void (*job_ready)(void *priv);
    void (*job_abort)(void *priv);
};

struct v4l2_m2m_ctx {
    struct v4l2_m2m_dev *m2m_dev;
    struct vb2_queue out_q;            // 输出队列（SOURCE）
    struct vb2_queue cap_q;            // 采集队列（SINK）
    struct list_head rdy_src_queue;    // 就绪的源缓冲
    struct list_head rdy_dst_queue;    // 就绪的目旳缓冲
    struct v4l2_m2m_queue_ctx cap_q_ctx;
    struct v4l2_m2m_queue_ctx out_q_ctx;
};

M2M 工作流

  1. 用户空间在 OUTPUT 队列 QBUF 填入原始数据（如 YUV 帧）
  2. 用户空间在 CAPTURE 队列 QBUF 填入空缓冲
  3. 调用 STREAMON 启动两个队列
  4. 驱动收到 OUTPUT 缓冲后，通过 v4l2_m2m_buf_queue() 排队
  5. 当两个队列都有就绪缓冲时，v4l2_m2m_job_ready() 返回 true
  6. 调用驱动注册的 device_run() 回调处理数据
  7. 处理完成后调用 v4l2_m2m_buf_done() 标记完成
  8. 用户空间从 CAPTURE 队列 DQBUF 取回处理后的数据

驱动模板

/* M2M 驱动典型结构 */
static const struct v4l2_m2m_ops my_m2m_ops = {
    .device_run = my_device_run,       // 启动硬件处理
    .job_ready  = my_job_ready,        // 检查是否具备处理条件
    .job_abort  = my_job_abort,        // 中止当前作业
};

static void my_device_run(void *priv)
{
    struct my_dev *dev = priv;
    struct vb2_buffer *src, *dst;

    /* 获取源和目旳缓冲 */
    src = v4l2_m2m_next_src_buf(dev->m2m_ctx);
    dst = v4l2_m2m_next_dst_buf(dev->m2m_ctx);

    /* 启动硬件 DMA */
    my_hw_process(dev,
                  vb2_plane_vaddr(src, 0),  // 源数据
                  vb2_plane_vaddr(dst, 0),  // 目旳缓冲
                  dev->fmt_size);

    /* 标记完成（通常在硬件中断中调用） */
    v4l2_m2m_buf_done(src, VB2_BUF_STATE_DONE);
    v4l2_m2m_buf_done(dst, VB2_BUF_STATE_DONE);
    v4l2_m2m_job_finish(dev->m2m_dev, dev->m2m_ctx);
}

/* 文件打开时创建 M2M 上下文 */
static int my_open(struct file *file)
{
    struct my_dev *dev = video_drvdata(file);
    struct v4l2_m2m_ctx *ctx;
    ctx = v4l2_m2m_ctx_init(dev->m2m_dev, dev,
                            &my_queue_init);
    file->private_data = ctx;
}

        struct v4l2_fwnode_bus_mipi_csi2 mipi_csi2;

设备树绑定示例

&i2c5 {
    imx219: camera@10 {
        compatible = "sony,imx219";
        reg = <0x10>;
        clocks = <&cam_clk>;
        VANA-supply  = <&cam_vana>;
        VDIG-supply  = <&cam_vdig>;
        VDDL-supply  = <&cam_vddl>;

        port {
            imx219_0: endpoint {
                remote-endpoint = <&csi1_ep>;
                data-lanes = <1 2>;
                clock-noncontinuous;
                link-frequencies = /bits/ 64 <456000000>;
            };
        };
    };
};

驱动通过 v4l2_fwnode_endpoint_parse() 解析上述设备树节点，自动获取总线类型（MIPI CSI-2）、数据通道数、时钟频率等信息：

int imx219_probe(struct i2c_client *client)
{
    struct v4l2_fwnode_endpoint ep = { .bus_type = 0 };
    struct fwnode_handle *ep_h;

    ep_h = fwnode_graph_get_next_endpoint(dev_fwnode(&client->dev), NULL);
    if (!ep_h)
        return -EPROBE_DEFER;

    ret = v4l2_fwnode_endpoint_parse(ep_h, &ep);
    fwnode_handle_put(ep_h);

    if (ret == 0) {
        /* 解析成功，获取 MIPI 配置 */
        dev_info(&client->dev, "lanes: %d, freq: %llu\n",
                 ep.bus.mipi_csi2.num_data_lanes,
                 ep.link_frequencies[0]);
    }
}

15 file_operations

V4L2 框架通过 struct v4l2_file_operations 将字符设备文件操作映射到驱动实现。驱动可以选择性地实现这些回调：

struct v4l2_file_operations {
    struct module *owner;
    int (*open)(struct file *file);
    int (*release)(struct file *file);
    __poll_t (*poll)(struct file *file,
                     struct poll_table_struct *wait);
    long (*ioctl)(struct file *file, unsigned int cmd,
                  unsigned long arg);
    int (*mmap)(struct file *file, struct vm_area_struct *vma);
    int (*get_unmapped_area)(struct file *file,
                             unsigned long addr,
                             unsigned long len,
                             unsigned long pgoff,
                             unsigned long flags);
    ssize_t (*read)(struct file *file, char __user *buf,
                    size_t count, loff_t *ppos);
    ssize_t (*write)(struct file *file, const char __user *buf,
                     size_t count, loff_t *ppos);
};

默认 v4l2_fops

框架提供默认文件操作表 v4l2_fops，定义了通用的 open、release、ioctl 等实现：

static const struct v4l2_file_operations v4l2_fops = {
    .owner   = THIS_MODULE,
    .open    = v4l2_open,
    .release = v4l2_release,
    .poll    = v4l2_poll,
    .unlocked_ioctl = video_ioctl2,
    .mmap    = v4l2_mmap,
    .read    = v4l2_read,
    .write   = v4l2_write,
};

v4l2_open()（在 v4l2-dev.c 中）：

static int v4l2_open(struct inode *inode, struct file *file)
{
    struct video_device *vdev;
    int ret;

    vdev = video_devdata(file);
    if (vdev == NULL)
        return -ENODEV;

    /* 检查设备是否已注册 */
    if (!video_is_registered(vdev))
        return -ENODEV;

    /* 检查设备类型权限 */
    if (vdev->fops->open) {
        ret = vdev->fops->open(file);
    } else {
        /* 默认行为：分配 v4l2_fh */
        if (vdev->ioctl_ops) {
            ret = v4l2_fh_open(file);
        }
    }
    return ret;
}

16 驱动实例分析

vivid（虚拟视频驱动）

vivid 是 V4L2 框架的参考实现，位于 drivers/media/test-drivers/vivid/。它模拟了完整的视频设备管线，包括采集、输出、VBI、radio、SDR 和触摸设备。vivid 展示了 V4L2 框架的全部核心机制：

  • 多 video_device 注册（vidcap、viddir、vbi 等）
  • vb2_queue 管理
  • TPG 测试图形生成
  • 完整的控制框架实现（亮度、对比度、饱和度等）
  • DV 时序支持（HDMI 模拟）
  • 事件生成（VSYNC、SOURCE_CHANGE）
  • M2M 功能（视频缩放/裁剪）
  • 媒体控制器拓扑

UVC（USB Video Class）

UVC 驱动位于 drivers/media/usb/uvc/，是 USB 摄像头标准驱动的实现。它展示了：

  • USB isochronous 传输与 videobuf2 的集成
  • 动态格式协商（通过 USB 探测）
  • 自动曝光/白平衡等控制
  • JPEG/MJPEG 硬件压缩处理

驱动编写最小模板

#include <media/v4l2-dev.h>
#include <media/v4l2-device.h>
#include <media/v4l2-ioctl.h>
#include <media/videobuf2-v4l2.h>

struct my_dev {
    struct v4l2_device v4l2_dev;
    struct video_device vdev;
    struct vb2_queue queue;
    spinlock_t slock;
};

static const struct v4l2_file_operations my_fops = {
    .owner   = THIS_MODULE,
    .open    = v4l2_fh_open,
    .release = v4l2_fh_release,
    .poll    = vb2_fop_poll,
    .unlocked_ioctl = video_ioctl2,
    .mmap    = vb2_fop_mmap,
};

static const struct v4l2_ioctl_ops my_ioctl_ops = {
    .vidioc_querycap     = my_querycap,
    .vidioc_enum_fmt_vid_cap = my_enum_fmt,
    .vidioc_try_fmt_vid_cap  = my_try_fmt,
    .vidioc_s_fmt_vid_cap    = my_s_fmt,
    .vidioc_reqbufs      = vb2_ioctl_reqbufs,
    .vidioc_querybuf     = vb2_ioctl_querybuf,
    .vidioc_qbuf         = vb2_ioctl_qbuf,
    .vidioc_dqbuf        = vb2_ioctl_dqbuf,
    .vidioc_streamon     = vb2_ioctl_streamon,
    .vidioc_streamoff    = vb2_ioctl_streamoff,
};

static int my_probe(struct platform_device *pdev)
{
    struct my_dev *dev;
    int ret;

    dev = devm_kzalloc(&pdev->dev, sizeof(*dev), GFP_KERNEL);

    /* 1. 注册 v4l2_device */
    ret = v4l2_device_register(&pdev->dev, &dev->v4l2_dev);

    /* 2. 初始化 vb2_queue */
    dev->queue.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
    dev->queue.io_modes = VB2_MMAP | VB2_USERPTR;
    dev->queue.ops = &my_queue_ops;
    dev->queue.mem_ops = &vb2_vmalloc_memops;
    dev->queue.drv_priv = dev;
    ret = vb2_queue_init(&dev->queue);

    /* 3. 初始化 video_device */
    dev->vdev.queue = &dev->queue;
    dev->vdev.v4l2_dev = &dev->v4l2_dev;
    dev->vdev.fops = &my_fops;
    dev->vdev.ioctl_ops = &my_ioctl_ops;
    dev->vdev.release = video_device_release_empty;
    dev->vdev.device_caps = V4L2_CAP_VIDEO_CAPTURE | V4L2_CAP_STREAMING;
    video_set_drvdata(&dev->vdev, dev);

    /* 4. 注册 video_device */
    ret = video_register_device(&dev->vdev, VFL_TYPE_VIDEO, -1);

    return 0;
}
